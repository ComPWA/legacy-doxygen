238 std::shared_ptr<ComPWA::FunctionTree::FunctionTree>
239 IntensityBuilderXML::createIntegrationStrategyFT(
240  std::shared_ptr<ComPWA::FunctionTree::FunctionTree> UnnormalizedIntensity,
241  std::string IntegratorClassName) {
242  LOG(TRACE) << "creating IntegrationStrategy ...";
243 
244  if (PhspSample.size() == 0)
245  LOG(FATAL) << "IntensityBuilderXML::createIntegrationStrategyFT(): phsp "
246  "sample is not set!";
247 
248  using namespace ComPWA::FunctionTree;
249  std::shared_ptr<ComPWA::FunctionTree::FunctionTree> tr;
250 
251  if (IntegratorClassName == "MCIntegrationStrategy") {
252  if (PhspSample.size() == 0)
253  LOG(FATAL) << "IntensityBuilderXML::createIntegrationStrategyFT(): phsp "
254  "sample is not set!";
255 
256  // update the PhspData
257  updateDataContainerState();
258  auto PhspDataSet =
259  ComPWA::Data::convertEventsToDataSet(PhspSample, Kinematic);
260  ComPWA::FunctionTree::updateDataContainers(CurrentIntensityState.ActiveData,
261  PhspDataSet.Data);
262 
263  if (!PhspWeights) {
264  PhspWeights = MDouble("Weight", PhspDataSet.Weights);
265  }
266  double PhspWeightSum(std::accumulate(PhspWeights->values().begin(),
267  PhspWeights->values().end(), 0.0));
268 
269  std::string NodeName =
270  "Normalization(" + UnnormalizedIntensity->Head->name() + ")";
271  tr = std::make_shared<ComPWA::FunctionTree::FunctionTree>(
272  NodeName, ValueFactory(ParType::DOUBLE),
273  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)));
274  tr->createNode("Integral",
275  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)),
276  NodeName);
277  // normTree->createLeaf("PhspVolume", PhspVolume, "Integral");
278  tr->createLeaf("InverseSampleWeights", 1.0 / PhspWeightSum, "Integral");
279  tr->createNode("Sum",
280  std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)),
281  "Integral");
282  tr->createNode("WeightedIntensities",
283  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)),
284  "Sum");
285 
286  if (PhspWeights)
287  tr->createLeaf("EventWeight", PhspWeights, "WeightedIntensities");
288  tr->insertTree(UnnormalizedIntensity, "WeightedIntensities");
289 
290  tr->parameter();
291  } else {
292  LOG(WARNING) << "IntensityBuilderXML::createIntegrationStrategyFT(): "
293  "IntegrationStrategy type "
294  << IntegratorClassName << " unknown!";
295  }
296 
297  return tr;
298 }

337  IntegratorClassName = OptionalIntegratorName.get();

356

360  NodeName, MComplex("", 0), std::make_shared<MultAll>(ParType::MCOMPLEX));

377  FTPhspData->Head->name() + "_AbsSquared");

380  createIntegrationStrategyFT(FTPhspDataAbsSquared, IntegratorClassName);
381 
382  auto normtree = std::make_shared<ComPWA::FunctionTree::FunctionTree>(
383  normtreesquared->Head->name() + "_sqrt", ValueFactory(ParType::DOUBLE),
384  std::make_shared<SquareRoot>(ParType::DOUBLE));
385 
386  normtree->insertTree(normtreesquared,
387  normtreesquared->Head->name() + "_sqrt");
388 
389  if (CurrentIntensityState.IsDataActive) {
390  CurrentIntensityState.PhspData = CurrentIntensityState.ActiveData;
391  CurrentIntensityState.ActiveData = CurrentIntensityState.Data;
392  }
393  NormalizedFT->insertTree(normtree, NodeName);
394 
395  return NormalizedFT;
396 }

451 std::shared_ptr<ComPWA::FunctionTree::FunctionTree>
452 IntensityBuilderXML::createSequentialAmplitudeFT(
453  const boost::property_tree::ptree &pt, std::string suffix) {
454  LOG(TRACE) << "constructing SequentialAmplitude ...";
455 
456  std::string ampname;
457 
458  std::vector<std::shared_ptr<ComPWA::FunctionTree::FunctionTree>> Amplitudes;
459  auto PreFactor = std::complex<double>(1, 0);
460  for (const auto &v : pt) {
461  if (v.first == "Amplitude") {
462  std::shared_ptr<ComPWA::FunctionTree::FunctionTree> AmpTree =
463  createAmplitudeFT(v.second, suffix);
464 
465  AmpTree->parameter();
466  ampname += AmpTree->Head->name();
467  Amplitudes.push_back(AmpTree);
468  } else if (v.first == "PreFactor") {
469  boost::optional<double> optr =
470  v.second.get_optional<double>("<xmlattr>.Magnitude");
471  if (optr.is_initialized()) {
472  double r(optr.value());
473  if (r < 0.0)
474  throw BadConfig("IntensityBuilderXML::createSequentialAmplitude(): "
475  "PreFactor Magnitude below zero!");
476  double p(0.0);
477  boost::optional<double> optp =
478  v.second.get_optional<double>("<xmlattr>.Phase");
479  if (optp.is_initialized())
480  p = optp.value();
481  PreFactor = std::polar(r, p);
482  } else {
483  double real = v.second.get<double>("<xmlattr>.Real");
484  double im(0.0);
485  boost::optional<double> optim =
486  v.second.get_optional<double>("<xmlattr>.Imaginary");
487  if (optim.is_initialized())
488  im = optim.value();
489  PreFactor = std::complex<double>(real, im);
490  }
491  } else if (v.first != "<xmlattr>") {
492  throw BadConfig("SequentialAmplitude::createSequentialAmplitude() | "
493  "Unknown tag " +
494  v.first + "!");
495  }
496  }
497 
498  auto NodeName = "SequentialPartialAmplitude(" + ampname + ")" + suffix;
499 
500  using namespace ComPWA::FunctionTree;
501  auto tr = std::make_shared<ComPWA::FunctionTree::FunctionTree>(
502  NodeName, MComplex("", 0), std::make_shared<MultAll>(ParType::MCOMPLEX));
503 
504  for (auto x : Amplitudes) {
505  tr->insertTree(x, NodeName);
506  }
507  tr->createLeaf("Prefactor", PreFactor, NodeName);
508 
509  return tr;
510 }

512 std::shared_ptr<ComPWA::FunctionTree::FunctionTree>
513 IntensityBuilderXML::createHelicityDecayFT(
514  const boost::property_tree::ptree &pt, std::string suffix) {
515  LOG(TRACE) << "IntensityBuilderXML::createHelicityDecayFT(): ";
516  auto &kin = dynamic_cast<HelicityFormalism::HelicityKinematics &>(Kinematic);
517  unsigned int SubSystemIndex(kin.addSubSystem(SubSystem(pt)));
518  unsigned int DataPosition = 3 * SubSystemIndex;
519 
520  updateDataContainerState();
521 
522  auto ampname = pt.get<std::string>("<xmlattr>.Name") + "_" +
523  std::to_string(DataPosition) + ";";
524 
525  std::string name = pt.get<std::string>("DecayParticle.<xmlattr>.Name");
526  auto partProp = ComPWA::findParticle(PartList, name);
527 
528  double J = partProp.getQuantumNumber<double>("Spin");
529  double mu(pt.get<double>("DecayParticle.<xmlattr>.Helicity"));
530  // if the node OrbitalAngularMomentum does not exist, set it to spin J as
531  // default value
532  unsigned int orbitL(J);
533 
534  auto Mass = std::make_shared<FunctionTree::FitParameter>(
535  partProp.getMass().Name, partProp.getMass().Value,
536  partProp.getMass().Error.first);
537  Mass->fixParameter(partProp.getMass().IsFixed);
538  Mass = CurrentIntensityState.Parameters.addUniqueParameter(Mass);
539 
540  double PreFactor(1.0);
541  const auto &canoSum = pt.get_child_optional("CanonicalSum");
542  if (canoSum) {
543  const auto &sumTree = canoSum.get();
544  orbitL = sumTree.get<unsigned int>("<xmlattr>.L");
545  double coef = std::sqrt((2.0 * orbitL + 1) / (2 * J + 1));
546  for (const auto &cg : sumTree.get_child("")) {
547  if (cg.first != "ClebschGordan")
548  continue;
549  double j1 = cg.second.get<double>("<xmlattr>.j1");
550  double m1 = cg.second.get<double>("<xmlattr>.m1");
551  double j2 = cg.second.get<double>("<xmlattr>.j2");
552  double m2 = cg.second.get<double>("<xmlattr>.m2");
553  double J = cg.second.get<double>("<xmlattr>.J");
554  double M = cg.second.get<double>("<xmlattr>.M");
555  coef *= ComPWA::QFT::Clebsch(j1, m1, j2, m2, J, M);
556  }
557  PreFactor *= coef;
558  }
559 
560  // Read name and helicities from decay products
561  auto decayProducts = pt.get_child("DecayProducts");
562  if (decayProducts.size() != 2)
563  throw boost::property_tree::ptree_error(
564  "IntensityBuilderXML::createHelicityDecayFT(): Expect exactly two "
565  "decay products (" +
566  std::to_string(decayProducts.size()) + " given)!");
567 
568  std::pair<std::string, std::string> DecayProducts;
569  std::pair<double, double> DecayHelicities;
570 
571  auto p = decayProducts.begin();
572  DecayProducts.first = p->second.get<std::string>("<xmlattr>.Name");
573  DecayHelicities.first = p->second.get<double>("<xmlattr>.Helicity");
574  ++p;
575  DecayProducts.second = p->second.get<std::string>("<xmlattr>.Name");
576  DecayHelicities.second = p->second.get<double>("<xmlattr>.Helicity");
577 
578  auto parMass1 = std::make_shared<FitParameter>(
579  ComPWA::findParticle(PartList, DecayProducts.first).getMass().Name,
580  ComPWA::findParticle(PartList, DecayProducts.first).getMass().Value);
581  auto parMass2 = std::make_shared<FitParameter>(
582  ComPWA::findParticle(PartList, DecayProducts.second).getMass().Name,
583  ComPWA::findParticle(PartList, DecayProducts.second).getMass().Value);
584  parMass1->fixParameter(
585  ComPWA::findParticle(PartList, DecayProducts.first).getMass().IsFixed);
586  parMass2->fixParameter(
587  ComPWA::findParticle(PartList, DecayProducts.second).getMass().IsFixed);
588  parMass1 = CurrentIntensityState.Parameters.addUniqueParameter(parMass1);
589  parMass2 = CurrentIntensityState.Parameters.addUniqueParameter(parMass2);
590  auto decayInfo = partProp.getDecayInfo();
591  Dynamics::FormFactorType ffType = Dynamics::FormFactorType::noFormFactor;
592  std::shared_ptr<FitParameter> parRadius;
593  std::shared_ptr<FitParameter> Width;
594  for (const auto &node : decayInfo.get_child("")) {
595  if (node.first == "FormFactor") {
596  auto FFTypeInt = node.second.get<int>("<xmlattr>.Type");
597  if (Dynamics::FormFactorType::BlattWeisskopf == FFTypeInt)
598  ffType = Dynamics::FormFactorType::BlattWeisskopf;
599  else if (Dynamics::FormFactorType::CrystalBarrel == FFTypeInt)
600  ffType = Dynamics::FormFactorType::CrystalBarrel;
601  } else if (node.first == "Parameter") {
602  std::string parType = node.second.get<std::string>("<xmlattr>.Type");
603  if (parType == "Width") {
604  Width = std::make_shared<FitParameter>(node.second);
605  Width = CurrentIntensityState.Parameters.addUniqueParameter(Width);
606  } else if (parType == "MesonRadius") {
607  parRadius = std::make_shared<FitParameter>(node.second);
608  parRadius =
609  CurrentIntensityState.Parameters.addUniqueParameter(parRadius);
610  }
611  }
612  }
613 
614  std::string decayType = partProp.getDecayType();
615 
616  std::shared_ptr<ComPWA::FunctionTree::FunctionTree> DynamicFunctionFT(
617  nullptr);
618 
619  if (decayType == "stable") {
620  throw std::runtime_error(
621  "IntensityBuilderXML::createHelicityDecayFT(): Stable particle is "
622  "given as mother particle of a decay. Makes no sense!");
623  } else if (decayType == "relativisticBreitWigner") {
624  using namespace ComPWA::Physics::Dynamics::RelativisticBreitWigner;
625  InputInfo RBW;
626  RBW.Mass = Mass;
627  RBW.Width = Width;
628  RBW.MesonRadius = parRadius;
629  RBW.DaughterMasses = std::make_pair(parMass1, parMass2);
630  RBW.FFType = ffType;
631  RBW.L = (unsigned int)orbitL;
632  DynamicFunctionFT = createFunctionTree(
633  RBW, CurrentIntensityState.ActiveData, DataPosition, suffix);
634  } else if (decayType == "flatte") {
635  ComPWA::Physics::Dynamics::Flatte::InputInfo FlatteInfo;
636  FlatteInfo.Mass = Mass;
637  FlatteInfo.MesonRadius = parRadius;
638  FlatteInfo.DaughterMasses = std::make_pair(parMass1, parMass2);
639  FlatteInfo.FFType = ffType;
640  FlatteInfo.L = (unsigned int)orbitL;
641  std::vector<Dynamics::Coupling> couplings;
642  // Read parameters
643  for (const auto &v : decayInfo.get_child("")) {
644  if (v.first != "Parameter")
645  continue;
646  std::string type = v.second.get<std::string>("<xmlattr>.Type");
647  if (type == "Coupling") {
648  auto c = Dynamics::Coupling(PartList, v.second);
649  c.G = CurrentIntensityState.Parameters.addUniqueParameter(c.G);
650 
651  if ((c.MassA->value() == parMass1->value() &&
652  c.MassB->value() == parMass2->value()) ||
653  (c.MassB->value() == parMass1->value() &&
654  c.MassA->value() == parMass2->value())) {
655  FlatteInfo.G = c.G;
656  } else {
657  c.MassA =
658  CurrentIntensityState.Parameters.addUniqueParameter(c.MassA);
659  c.MassB =
660  CurrentIntensityState.Parameters.addUniqueParameter(c.MassB);
661  couplings.push_back(c);
662  }
663  }
664  }
665  FlatteInfo.HiddenCouplings = couplings;
666  DynamicFunctionFT = createFunctionTree(
667  FlatteInfo, CurrentIntensityState.ActiveData, DataPosition, suffix);
668  } else if (decayType == "voigt") {
669  using namespace ComPWA::Physics::Dynamics::Voigtian;
670  InputInfo VoigtInfo;
671  VoigtInfo.Mass = Mass;
672  VoigtInfo.Width = Width;
673  VoigtInfo.MesonRadius = parRadius;
674  VoigtInfo.DaughterMasses = std::make_pair(parMass1, parMass2);
675  VoigtInfo.FFType = ffType;
676  VoigtInfo.L = (unsigned int)orbitL;
677  VoigtInfo.Sigma = decayInfo.get<double>("Resolution.<xmlattr>.Sigma");
678  DynamicFunctionFT = createFunctionTree(
679  VoigtInfo, CurrentIntensityState.ActiveData, DataPosition, suffix);
680  } else if (decayType == "virtual" || decayType == "nonResonant") {
681  DynamicFunctionFT = Dynamics::NonResonant::createFunctionTree(
682  CurrentIntensityState.ActiveData, DataPosition, suffix);
683  } else {
684  throw std::runtime_error("HelicityDecay::Factory() | Unknown decay type " +
685  decayType + "!");
686  }
687 
688  auto AngularFunction =
689  ComPWA::Physics::HelicityFormalism::WignerD::createFunctionTree(
690  J, mu, DecayHelicities.first - DecayHelicities.second,
691  CurrentIntensityState.ActiveData, DataPosition + 1, DataPosition + 2);
692 
693  std::string nodeName = "PartialAmplitude(" + ampname + ")" + suffix;
694 
695  using namespace ComPWA::FunctionTree;
696  auto tr = std::make_shared<ComPWA::FunctionTree::FunctionTree>(
697  nodeName, MComplex("", 0), std::make_shared<MultAll>(ParType::MCOMPLEX));
698  tr->createLeaf("PreFactor", PreFactor, nodeName);
699  tr->insertTree(AngularFunction, nodeName);
700  tr->insertTree(DynamicFunctionFT, nodeName);
701 
702  // set production formfactor
703  if (ffType != ComPWA::Physics::Dynamics::FormFactorType::noFormFactor &&
704  ((unsigned int)orbitL > 0)) {
705  if (parRadius == nullptr) {
706  throw std::runtime_error("IntensityBuilderXML::createHelicityDecayFT(): "
707  "No MesonRadius given for amplitude " +
708  name +
709  "! It is needed to calculate the form factor!");
710  }
711 
712  std::shared_ptr<ComPWA::FunctionTree::FunctionTree> ProductionFormFactorFT =
713  Dynamics::createFunctionTree(
714  name, parMass1, parMass2, parRadius, orbitL, ffType,
715  CurrentIntensityState.ActiveData, DataPosition, suffix);
716 
717  tr->insertTree(ProductionFormFactorFT, nodeName);
718  }
719 
720  tr->parameter();
721 
722  return tr;
723 }

758  auto initialS = pt.get_child("InitialState");

762  std::string name = i.second.get<std::string>("<xmlattr>.Name");
763  auto partP = ComPWA::findParticle(PartList, name);
764  unsigned int pos(counter++);

766  i.second.get_optional<unsigned int>("<xmlattr>.PositionIndex");

769  InitialState.at(pos) = partP.getId();
770  }

774  auto FinalStateEventPositionMapping =
775  std::vector<unsigned int>(finalS.size());

777  for (auto i : finalS) {
778  std::string name = i.second.get<std::string>("<xmlattr>.Name");
779  auto partP = ComPWA::findParticle(PartList, name);
780  unsigned int id = i.second.get<unsigned int>("<xmlattr>.Id");
781  unsigned int pos(counter++);
782  boost::optional<unsigned int> opt_pos =
783  i.second.get_optional<unsigned int>("<xmlattr>.PositionIndex");
784  if (opt_pos)
785  pos = opt_pos.get();
786  FinalState.at(pos) = partP.getId();
787  FinalStateEventPositionMapping.at(pos) = id;
788  }

790  if (pt.find("InitialFourMomentum") != pt.not_found()) {

794  InitialState, FinalState, PartList, InitialStateP4,
