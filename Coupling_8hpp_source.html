86 inline std::complex<double> widthToCoupling(double sqrtS, double mR,
87  double width, double ma, double mb,
88  double orbitL, double mesonRadius,
89  FormFactorType type) {
90 
91  // Calculate normalized vertex function gammaA(s_R) (see PDG2014,
92  // Chapter 47.2)
93  std::complex<double> gammaA(1, 0); // orbitL==0
94  std::complex<double> qV;
95  if (orbitL > 0) {
96  qV = qValue(mR, ma, mb);
97  double ffR = FormFactor(qV, orbitL, mesonRadius, type);
98  std::complex<double> qR = std::pow(qV, orbitL);
99  gammaA = ffR * qR;
100  }
101 
102  // calculate phsp factor
103  std::complex<double> rho = phspFactor(sqrtS, ma, mb);
104 
105  return widthToCoupling(mR, width, gammaA, rho);
106 }
