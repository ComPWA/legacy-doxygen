53 bool DalitzKinematics::isWithinPhsp(const DataPoint &point) const {
54  double mA = point.KinematicVariableList[0];
55  double mB = point.KinematicVariableList[1];
56  double mC = point.KinematicVariableList[2];
57  double qAB = point.KinematicVariableList[3];
58  double qBC = point.KinematicVariableList[4];
59  double qCA = point.KinematicVariableList[5];
60 
61  double s2 = (qAB + qBC + qCA - mA * mA - mB * mB - mC * mC);
62 
63  if (s2 < M2)
64  return true;
65 
66  return false;
67 }
