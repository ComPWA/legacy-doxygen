89 // These are the resources that depend on the device.
90 void DirectXBase::CreateDeviceResources()
91 {
92  // This flag adds support for surfaces with a different color channel ordering than the API default.
93  // It is recommended usage, and is required for compatibility with Direct2D.
94  UINT creationFlags = D3D11_CREATE_DEVICE_BGRA_SUPPORT;
95  ComPtr<IDXGIDevice> dxgiDevice;
96 
97 #if defined(_DEBUG)
98  // If the project is in a debug build, enable debugging via SDK Layers with this flag.
99  creationFlags |= D3D11_CREATE_DEVICE_DEBUG;
100 #endif
101 
102  // This array defines the set of DirectX hardware feature levels this app will support.
103  // Note the ordering should be preserved.
104  // Don't forget to declare your application's minimum required feature level in its
105  // description. All applications are assumed to support 9.1 unless otherwise stated.
106  D3D_FEATURE_LEVEL featureLevels[] =
107  {
108  D3D_FEATURE_LEVEL_11_1,
109  D3D_FEATURE_LEVEL_11_0,
110  D3D_FEATURE_LEVEL_10_1,
111  D3D_FEATURE_LEVEL_10_0,
112  D3D_FEATURE_LEVEL_9_3,
113  D3D_FEATURE_LEVEL_9_2,
114  D3D_FEATURE_LEVEL_9_1
115  };
116 
117  // Create the DX11 API device object, and get a corresponding context.
118  ComPtr<ID3D11Device> device;
119  ComPtr<ID3D11DeviceContext> context;
120  DX::ThrowIfFailed(
121  D3D11CreateDevice(
122  nullptr, // specify null to use the default adapter
123  D3D_DRIVER_TYPE_HARDWARE,
124  0, // leave as 0 unless software device
125  creationFlags, // optionally set debug and Direct2D compatibility flags
126  featureLevels, // list of feature levels this app can support
127  ARRAYSIZE(featureLevels), // number of entries in above list
128  D3D11_SDK_VERSION, // always set this to D3D11_SDK_VERSION for Metro style apps
129  &device, // returns the Direct3D device created
130  &m_featureLevel, // returns feature level of device created
131  &context // returns the device immediate context
132  )
133  );
134  // Get the DirectX11.1 device by QI off the DirectX11 one.
135  DX::ThrowIfFailed(
136  device.As(&m_d3dDevice)
137  );
138 
139  // And get the corresponding device context in the same way.
140  DX::ThrowIfFailed(
141  context.As(&m_d3dContext)
142  );
143 
144  // Obtain the underlying DXGI device of the Direct3D11.1 device.
145  DX::ThrowIfFailed(
146  m_d3dDevice.As(&dxgiDevice)
147  );
148 
149  // Obtain the Direct2D device for 2-D rendering.
150  DX::ThrowIfFailed(
151  m_d2dFactory->CreateDevice(dxgiDevice.Get(), &m_d2dDevice)
152  );
153 
154  // And get its corresponding device context object.
155  DX::ThrowIfFailed(
156  m_d2dDevice->CreateDeviceContext(
157  D2D1_DEVICE_CONTEXT_OPTIONS_NONE,
158  &m_d2dContext
159  )
160  );
161 
162  // Release the swap chain (if it exists) as it will be incompatible with the new device.
163  m_swapChain = nullptr;
164 }

168 void DirectXBase::SetDpi(float dpi)
169 {
170  if (dpi != m_dpi)
171  {
172  // Save the DPI of this display in our class.
173  m_dpi = dpi;
174 
175  // Update Direct2D's stored DPI.
176  m_d2dContext->SetDpi(m_dpi, m_dpi);
177 
178  // Often a DPI change implies a window size change. In some cases Windows will issues
179  // both a size changed event and a DPI changed event. In this case, the resulting bounds
180  // will not change, and the window resize code will only be executed once.
181  UpdateForWindowSizeChange();
182  }

185 // This routine is called in the event handler for the view SizeChanged event.
186 void DirectXBase::UpdateForWindowSizeChange()
187 {
188  // Only handle window size changed if there is no pending DPI change.
189  if (m_dpi != DisplayProperties::LogicalDpi)
190  return;
191 
192  if (m_window->Bounds.Width != m_windowBounds.Width ||
193  m_window->Bounds.Height != m_windowBounds.Height)
194  {
195  m_d2dContext->SetTarget(nullptr);
196  m_d2dTargetBitmap = nullptr;
197  m_renderTargetView = nullptr;
198  m_depthStencilView = nullptr;
199  CreateWindowSizeDependentResources();
200  }

203 // Allocate all memory resources that change on a window SizeChanged event.
204 void DirectXBase::CreateWindowSizeDependentResources()
205 {
206  // Store the window bounds so the next time we get a SizeChanged event we can
207  // avoid rebuilding everything if the size is identical.
208  m_windowBounds = m_window->Bounds;
209 
210  // Calculate the necessary swap chain and render target size in pixels.
211  m_renderTargetSize.Width = ConvertDipsToPixels(m_windowBounds.Width);
212  m_renderTargetSize.Height = ConvertDipsToPixels(m_windowBounds.Height);
213 
214  // If the swap chain already exists, resize it.
215  if (m_swapChain != nullptr)
216  {
217  DX::ThrowIfFailed(
218  m_swapChain->ResizeBuffers(
219  2,
220  static_cast<UINT>(m_renderTargetSize.Width),
221  static_cast<UINT>(m_renderTargetSize.Height),
222  DXGI_FORMAT_B8G8R8A8_UNORM,
223  0
224  )
225  );
226  }
227  // Otherwise, create a new one.
228  else
229  {
230  // Allocate a descriptor.
231  DXGI_SWAP_CHAIN_DESC1 swapChainDesc = {0};
232  swapChainDesc.Width = static_cast<UINT>(m_renderTargetSize.Width); // Match the size of the windowm.
233  swapChainDesc.Height = static_cast<UINT>(m_renderTargetSize.Height);
234  swapChainDesc.Format = DXGI_FORMAT_B8G8R8A8_UNORM; // This is the most common swapchain format.
235  swapChainDesc.Stereo = false;
236  swapChainDesc.SampleDesc.Count = 1; // Don't use multi-sampling.
237  swapChainDesc.SampleDesc.Quality = 0;
238  swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
239  swapChainDesc.BufferCount = 2; // Use double buffering to enable flip.
240  swapChainDesc.Scaling = DXGI_SCALING_STRETCH;
241  swapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_SEQUENTIAL; // All Metro style apps must use this SwapEffect.
242  swapChainDesc.Flags = 0;
243 
244  // Once the desired swap chain description is configured, it must be created on the same adapter as our D3D Device.
245 
246  // First, retrieve the underlying DXGI Device from the D3D Device.
247  ComPtr<IDXGIDevice1> dxgiDevice;
248  DX::ThrowIfFailed(
249  m_d3dDevice.As(&dxgiDevice)
250  );
251 
252  // Identify the physical adapter (GPU or card) this device is running on.
253  ComPtr<IDXGIAdapter> dxgiAdapter;
254  DX::ThrowIfFailed(
255  dxgiDevice->GetAdapter(&dxgiAdapter)
256  );
257 
258  // And obtain the factory object that created it.
259  ComPtr<IDXGIFactory2> dxgiFactory;
260  DX::ThrowIfFailed(
261  dxgiAdapter->GetParent(IID_PPV_ARGS(&dxgiFactory))
262  );
263 
264  // Create the swap chain and then associate it with the SwapChainBackgroundPanel.
265  DX::ThrowIfFailed(
266  dxgiFactory->CreateSwapChainForComposition(
267  m_d3dDevice.Get(),
268  &swapChainDesc,
269  nullptr,
270  &m_swapChain
271  )
272  );
273 
274  ComPtr<ISwapChainBackgroundPanelNative> panelNative;
275  DX::ThrowIfFailed(
276  reinterpret_cast<IUnknown*>(m_panel)->QueryInterface(IID_PPV_ARGS(&panelNative))
277  );
278 
279  DX::ThrowIfFailed(
280  panelNative->SetSwapChain(m_swapChain.Get())
281  );
282 
283  // Ensure that DXGI does not queue more than one frame at a time. This both reduces
284  // latency and ensures that the application will only render after each VSync, minimizing
285  // power consumption.
286  DX::ThrowIfFailed(
287  dxgiDevice->SetMaximumFrameLatency(1)
288  );
289  }
290 
291  // Obtain the backbuffer for this window which will be the final 3D rendertarget.
292  ComPtr<ID3D11Texture2D> backBuffer;
293  DX::ThrowIfFailed(
294  m_swapChain->GetBuffer(0, IID_PPV_ARGS(&backBuffer))
295  );
296 
297  // Create a view interface on the rendertarget to use on bind.
298  DX::ThrowIfFailed(
299  m_d3dDevice->CreateRenderTargetView(
300  backBuffer.Get(),
301  nullptr,
302  &m_renderTargetView
303  )
304  );
305 
306  // Create a descriptor for the depth/stencil buffer.
307  CD3D11_TEXTURE2D_DESC depthStencilDesc(
308  DXGI_FORMAT_D24_UNORM_S8_UINT,
309  static_cast<UINT>(m_renderTargetSize.Width),
310  static_cast<UINT>(m_renderTargetSize.Height),
311  1,
312  1,
313  D3D11_BIND_DEPTH_STENCIL
314  );
315 
316  // Allocate a 2-D surface as the depth/stencil buffer.
317  ComPtr<ID3D11Texture2D> depthStencil;
318  DX::ThrowIfFailed(
319  m_d3dDevice->CreateTexture2D(
320  &depthStencilDesc,
321  nullptr,
322  &depthStencil
323  )
324  );
325 
326  // Create a DepthStencil view on this surface to use on bind.
327  CD3D11_DEPTH_STENCIL_VIEW_DESC depthStencilViewDesc(D3D11_DSV_DIMENSION_TEXTURE2D);
328  DX::ThrowIfFailed(
329  m_d3dDevice->CreateDepthStencilView(
330  depthStencil.Get(),
331  &depthStencilViewDesc,
332  &m_depthStencilView
333  )
334  );
335 
336  // Create a viewport descriptor of the full window size.
337  CD3D11_VIEWPORT viewport(
338  0.0f,
339  0.0f,
340  m_renderTargetSize.Width,
341  m_renderTargetSize.Height
342  );
343 
344  // Set the current viewport using the descriptor.
345  m_d3dContext->RSSetViewports(1, &viewport);
346 
347  // Now we set up the Direct2D render target bitmap linked to the swapchain.
348  // Whenever we render to this bitmap, it will be directly rendered to the
349  // swapchain associated with the window.
350  D2D1_BITMAP_PROPERTIES1 bitmapProperties =
351  BitmapProperties1(
352  D2D1_BITMAP_OPTIONS_TARGET | D2D1_BITMAP_OPTIONS_CANNOT_DRAW,
353  PixelFormat(DXGI_FORMAT_B8G8R8A8_UNORM, D2D1_ALPHA_MODE_PREMULTIPLIED),
354  m_dpi,
355  m_dpi
356  );
357 
358  // Direct2D needs the DXGI version of the backbuffer surface pointer.
359  ComPtr<IDXGISurface> dxgiBackBuffer;
360  DX::ThrowIfFailed(
361  m_swapChain->GetBuffer(0, IID_PPV_ARGS(&dxgiBackBuffer))
362  );
363 
364  // Get a D2D surface from the DXGI back buffer to use as the D2D render target.
365  DX::ThrowIfFailed(
366  m_d2dContext->CreateBitmapFromDxgiSurface(
367  dxgiBackBuffer.Get(),
368  &bitmapProperties,
369  &m_d2dTargetBitmap
370  )
371  );
372 
373  // So now we can set the Direct2D render target.
374  m_d2dContext->SetTarget(m_d2dTargetBitmap.Get());
375 
376  // Set D2D text anti-alias mode to Grayscale to ensure proper rendering of text on intermediate surfaces.
377  m_d2dContext->SetTextAntialiasMode(D2D1_TEXT_ANTIALIAS_MODE_GRAYSCALE);
378 }

380 // Method to deliver the final image to the display.
381 void DirectXBase::Present()
382 {
383  // The application may optionally specify "dirty" or "scroll" rects to improve efficiency
384  // in certain scenarios.
385  DXGI_PRESENT_PARAMETERS parameters = {0};
386  parameters.DirtyRectsCount = 0;
387  parameters.pDirtyRects = nullptr;
388  parameters.pScrollRect = nullptr;
389  parameters.pScrollOffset = nullptr;
390 
391  // The first argument instructs DXGI to block until VSync, putting the application
392  // to sleep until the next VSync. This ensures we don't waste any cycles rendering
393  // frames that will never be displayed to the screen.
394  HRESULT hr = m_swapChain->Present1(1, 0, &parameters);
395 
396  // If the device was removed either by a disconnect or a driver upgrade, we
397  // must completely reinitialize the renderer.
398  if (hr == DXGI_ERROR_DEVICE_REMOVED || hr == DXGI_ERROR_DEVICE_RESET)
399  {
400  Initialize(m_window.Get(), m_panel, m_dpi);
401  }
402  else
403  {
404  DX::ThrowIfFailed(hr);
405  }
406 }

409 float DirectXBase::ConvertDipsToPixels(float dips)
410 {
411  static const float dipsPerInch = 96.0f;
412  return floor(dips * m_dpi / dipsPerInch + 0.5f); // Round to nearest integer.
