55 // Blatt-Weisskopf form factors
56 // see e.g. hep-ex/0011065
57 // Dalitz Analysis of the Decay D0->K-pi+pi0 (CLEO)
58 //
59 // p - momentum of either daugher in the meson rest frame,

62 // the mass of the candidate is used

64 //
65 // In the CLEO paper R=5 GeV-1 for D0, R=1.5 for intermediate resonances

67 double EvtBlattWeisskopf::compute(double p) const
68 {
69 
70  double value(1.0);
71  double z = p*_radial;
72  double zSq = z*z;
73 
74  if (_LL == 0) {
75  value = 1.0;
76  } else if (_LL == 1) {
77  value = sqrt(1.0/(1.0 + zSq));
78  } else if (_LL == 2) {
79  value = sqrt(1.0/(zSq*(zSq + 3.0) + 9.0));
80  }
81 
82  return value;
