230 EvtComplex EvtDalitzReso::evaluate(const EvtDalitzPoint& x)
231 {
232  if(!(x.isValid())){return EvtComplex(0.,0.);}
233 
234  double m = sqrt(x.q(_pairRes));
235 
236  if (_typeN==NON_RES)
237  return EvtComplex(1.0,0.0);
238 
239  // do use always hash table (speed up fitting)
240  if (_typeN==K_MATRIX || _typeN==K_MATRIX_I || _typeN==K_MATRIX_II)
241  return Fvector( m*m, _kmatrix_index );
242 
243  if (_typeN==LASS)
244  return lass(m*m);
245 
246  if (_typeN==FLATTE)
247  return flatte(m);
248 
249  EvtComplex amp(1.0,0.0);
250 
251  if (fabs(_dp.bigM() - x.bigM()) > 0.000001) {
252  _vb = EvtTwoBodyVertex(_m0,_dp.m(EvtCyclic3::other(_pairRes)),x.bigM(),_spin);
253  _vb.set_f(_f_b);
254  }
255  EvtTwoBodyKine vb(m,x.m(EvtCyclic3::other(_pairRes)),x.bigM());
256  EvtTwoBodyKine vd(_massFirst,_massSecond,m);
257 
258  EvtComplex prop(0,0);
259  if (_typeN==NBW) {
260  prop = propBreitWigner(_m0,_g0,m);
261  } else if (_typeN==GAUSS_CLEO || _typeN==GAUSS_CLEO_ZEMACH) {
262  prop = propGauss(_m0,_g0,m);
263  } else {
264  if (_coupling2==Undefined) {
265  // single BW
266  double g = (_g0<=0. || _vd.pD()<=0.)? -_g0 : _g0*_vd.widthFactor(vd); // running width
267  if (_typeN==GS_CLEO || _typeN==GS_CLEO_ZEMACH) {
268  // Gounaris-Sakurai (GS)
269  assert(_massFirst==_massSecond);
270  prop = propGounarisSakurai(_m0,fabs(_g0),_vd.pD(),m,g,vd.p());
271 // std::cout<<_vd.pD()<<std::endl;
272  } else {
273 
274  // standard relativistic BW
275  prop = propBreitWignerRel(_m0,g,m);
276  }
277  } else {
278  // coupled width BW
279 
280  EvtComplex G1,G2;
281  switch (_coupling2) {
282  case PicPic: {
283  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
284  double mPic = 1.3957018e-01;
285  G2 = _g2*_g2*psFactor(mPic,mPic,m);
286  break;
287  }
288  case PizPiz: {
289  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
290  double mPiz = 1.3497660e-01;
291  G2 = _g2*_g2*psFactor(mPiz,mPiz,m);
292  break;
293  }
294  case PiPi: {
295  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
296  double mPic = 1.3957018e-01;
297  double mPiz = 1.3497660e-01;
298  G2 = _g2*_g2*psFactor(mPic,mPic,mPiz,mPiz,m);
299  break;
300  }
301  case KcKc: {
302  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
303  double mKc = 4.9367700e-01;
304  G2 = _g2*_g2*psFactor(mKc,mKc,m);
305  break;
306  }
307  case KzKz: {
308  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
309  double mKz = 4.9761400e-01;
310  G2 = _g2*_g2*psFactor(mKz,mKz,m);
311  break;
312  }
313  case KK: {
314  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
315  double mKc = 4.9367700e-01;
316  double mKz = 4.9761400e-01;
317  G2 = _g2*_g2*psFactor(mKc,mKc,mKz,mKz,m);
318  break;
319  }
320  case EtaPic: {
321  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
322  double mEta = 5.4785300e-01;
323  double mPic = 1.3957018e-01;
324  G2 = _g2*_g2*psFactor(mEta,mPic,m);
325  break;
326  }
327  case EtaPiz: {
328  G1 = _g1*_g1*psFactor(_massFirst,_massSecond,m);
329  double mEta = 5.4785300e-01;
330  double mPiz = 1.3497660e-01;
331  G2 = _g2*_g2*psFactor(mEta,mPiz,m);
332  break;
333  }
334  case PicPicKK: {
335  double mPic = 1.3957018e-01;
336  //G1 = _g1*_g1*psFactor(mPic,mPic,m);
337  G1 = _g1*psFactor(mPic,mPic,m);
338  double mKc = 4.9367700e-01;
339  double mKz = 4.9761400e-01;
340  //G2 = _g2*_g2*psFactor(mKc,mKc,mKz,mKz,m);
341  G2 = _g2*psFactor(mKc,mKc,mKz,mKz,m);
342  break;
343  }
344  default:
345  std::cout << "EvtDalitzReso:evaluate(): PANIC, wrong coupling2 state." << std::endl;
346  assert(0);
347  break;
348  }
349  // calculate standard couple BW propagator
350  if (_coupling2 != WA76)
351  prop = _g1*propBreitWignerRelCoupled(_m0,G1,G2,m);
352  }
353  }
354  amp *= prop;
355 
356  // Compute form-factors (Blatt-Weisskopf penetration factor)
357  amp *= _vb.formFactor(vb);
358  amp *= _vd.formFactor(vd);
359 
360  // Compute numerator (angular distribution)
361  amp *= numerator(x,vb,vd);
362 
363  // Compute electromagnetic mass mixing factor
364  if (_m0_mix>0.) {
365  EvtComplex prop_mix;
366  if (_typeN==NBW) {
367  prop_mix = propBreitWigner(_m0_mix,_g0_mix,m);
368  } else {
369  assert(_g1<0.); // running width only
370  double g_mix = _g0_mix*_vd.widthFactor(vd);
371  prop_mix = propBreitWignerRel(_m0_mix,g_mix,m);
372  }
373  amp *= mixFactor(prop,prop_mix);
374  }
375 
376  return amp;
377 }
