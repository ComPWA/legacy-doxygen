17 std::shared_ptr<FunctionTree> Flatte::createFunctionTree(
18  InputInfo Params, const ComPWA::FunctionTree::ParameterList &DataSample,
19  unsigned int pos, std::string suffix) {
20 
21  if (Params.Couplings.size() == 2)
22  Params.Couplings.push_back(Coupling(0.0, 0.0, 0.0));
23 
24  if (Params.Couplings.size() != 3)
25  throw std::runtime_error(
26  "AmpFlatteRes::SetCouplings() | Vector with "
27  "couplings has a wrong size. We expect either 2 or 3 couplings.");
28 
29  // Check if one of the coupling match the final state (_daughterMasses)
30  if (!(Params.DaughterMasses.first && Params.DaughterMasses.second))
31  LOG(INFO)
32  << "AmpFlatteRes::SetCouplings() | Masses of decay products not set. "
33  " Can not determine if correct couplings were set.";
34 
35  bool ok = false;
36  for (auto i : Params.Couplings) {
37  if (i.GetMassA() == Params.DaughterMasses.first->value() &&
38  i.GetMassB() == Params.DaughterMasses.second->value())
39  ok = true;
40  if (i.GetMassB() == Params.DaughterMasses.first->value() &&
41  i.GetMassA() == Params.DaughterMasses.second->value())
42  ok = true;
43  }
44  if (!ok)
45  throw std::runtime_error("AmpFlatteRes::SetCouplings() | No couplings "
46  "for the current decay particles set!");
47 
48  size_t sampleSize = DataSample.mDoubleValue(pos)->values().size();
49 
50  std::string NodeName = "Flatte" + suffix;
51  auto tr = std::make_shared<FunctionTree>(
52  NodeName, ComPWA::FunctionTree::MComplex("", sampleSize),
53  std::make_shared<FlatteStrategy>(""));
54 
55  tr->createLeaf("Mass", Params.Mass, NodeName);
56  for (unsigned int i = 0; i < Params.Couplings.size(); ++i) {
57  tr->createLeaf("g_" + std::to_string(i) + "_massA",
58  Params.Couplings.at(i).GetMassA(), NodeName);
59  tr->createLeaf("g_" + std::to_string(i) + "_massB",
60  Params.Couplings.at(i).GetMassB(), NodeName);
61  tr->createLeaf("g_" + std::to_string(i),
62  Params.Couplings.at(i).GetValueParameter(), NodeName);
63  }
64  tr->createLeaf("OrbitalAngularMomentum", Params.L, NodeName);
65  tr->createLeaf("MesonRadius", Params.MesonRadius, NodeName);
66  tr->createLeaf("FormFactorType", Params.FFType, NodeName);
67  tr->createLeaf(DataSample.mDoubleValue(pos)->name(),
68  DataSample.mDoubleValue(pos), NodeName);
69 
70  return tr;
71 }

73 void FlatteStrategy::execute(ParameterList &paras,
74  std::shared_ptr<Parameter> &out) {
75  if (out && checkType != out->type())
76  throw BadParameter("FlatteStrategy::execute() | Parameter type mismatch!");
77 
78 #ifndef NDEBUG
79  // Check parameter type
80  if (checkType != out->type())
81  throw(
82  WrongParType("FlatteStrategy::execute() | "
83  "Output parameter is of type " +
84  std::string(ComPWA::FunctionTree::ParNames[out->type()]) +
85  " and conflicts with expected type " +
86  std::string(ComPWA::FunctionTree::ParNames[checkType])));
87 
88  // How many parameters do we expect?
89  size_t check_nInt = 0;
90  size_t nInt = paras.intValues().size();
91  size_t check_nDouble = 13;
92  size_t nDouble = paras.doubleValues().size();
93  nDouble += paras.doubleParameters().size();
94  size_t check_nComplex = 0;
95  size_t nComplex = paras.complexValues().size();
96  size_t check_nMInteger = 0;
97  size_t nMInteger = paras.mIntValues().size();
98  size_t check_nMDouble = 1;
99  size_t nMDouble = paras.mDoubleValues().size();
100  size_t check_nMComplex = 0;
101  size_t nMComplex = paras.mComplexValues().size();
102 
103  // Check size of parameter list
104  if (nInt != check_nInt)
105  throw(BadParameter("FlatteStrategy::execute() | "
106  "Number of IntParameters does not match: " +
107  std::to_string(nInt) + " given but " +
108  std::to_string(check_nInt) + " expected."));
109  if (nDouble != check_nDouble)
110  throw(BadParameter("FlatteStrategy::execute() | "
111  "Number of FitParameters does not match: " +
112  std::to_string(nDouble) + " given but " +
113  std::to_string(check_nDouble) + " expected."));
114  if (nComplex != check_nComplex)
115  throw(BadParameter("FlatteStrategy::execute() | "
116  "Number of ComplexParameters does not match: " +
117  std::to_string(nComplex) + " given but " +
118  std::to_string(check_nComplex) + " expected."));
119  if (nMInteger != check_nMInteger)
120  throw(BadParameter("FlatteStrategy::execute() | "
121  "Number of MultiInt does not match: " +
122  std::to_string(nMInteger) + " given but " +
123  std::to_string(check_nMInteger) + " expected."));
124  if (nMDouble != check_nMDouble)
125  throw(BadParameter("FlatteStrategy::execute() | "
126  "Number of MultiDoubles does not match: " +
127  std::to_string(nMDouble) + " given but " +
128  std::to_string(check_nMDouble) + " expected."));
129  if (nMComplex != check_nMComplex)
130  throw(BadParameter("FlatteStrategy::execute() | "
131  "Number of MultiComplexes does not match: " +
132  std::to_string(nMComplex) + " given but " +
133  std::to_string(check_nMComplex) + " expected."));
134 #endif
135 
136  size_t n = paras.mDoubleValue(0)->values().size();
137  if (!out)
138  out = ComPWA::FunctionTree::MComplex("", n);
139 
140  auto par =
141  std::static_pointer_cast<Value<std::vector<std::complex<double>>>>(out);
142  auto &results = par->values(); // reference
143  if (results.size() != n) {
144  results.resize(n);
145  }
146  // calc function for each point
147  for (size_t ele = 0; ele < n; ele++) {
148  try {
149  // Generally we need to add a factor q^{2J+1} to each channel term.
150  // But since Flatte resonances are usually J=0 we neglect it here.
151  results.at(ele) = Flatte::dynamicalFunction(
152  paras.mDoubleValue(0)->values().at(ele),
153  paras.doubleParameter(0)->value(), // mass
154  paras.doubleValue(0)->value(), // g1_massA
155  paras.doubleValue(1)->value(), // g1_massB
156  paras.doubleParameter(1)->value(), // g1
157  paras.doubleValue(2)->value(), // g2_massA
158  paras.doubleValue(3)->value(), // g2_massB
159  paras.doubleParameter(2)->value(), // g2
160  paras.doubleValue(4)->value(), // g3_massA
161  paras.doubleValue(5)->value(), // g3_massB
162  paras.doubleParameter(3)->value(), // g3
163  paras.doubleValue(6)->value(), // OrbitalAngularMomentum
164  paras.doubleParameter(4)->value(), // mesonRadius
165  FormFactorType(paras.doubleValue(7)->value()) // ffType
166  );
167  } catch (std::exception &ex) {
168  LOG(ERROR) << "FlatteStrategy::execute() | " << ex.what();
169  throw(std::runtime_error("FlatteStrategy::execute() | "
170  "Evaluation of dynamic function failed!"));
171  }
172  }
