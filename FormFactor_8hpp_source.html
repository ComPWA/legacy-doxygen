24 inline double qSqValue(double sqrtS, double ma, double mb) {
25  double mapb = ma + mb;
26  double mamb = ma - mb;
27  double xSq = sqrtS * sqrtS;
28  double t1 = xSq - mapb * mapb;
29  double t2 = xSq - mamb * mamb;
30  return (t1 * t2 / (4 * xSq));
31 }

38 inline std::complex<double> phspFactor(double sqrtS, double ma, double mb) {
39  double s = sqrtS * sqrtS;
40  std::complex<double> i(0, 1);
41 
42  // == Two types of analytic continuation
43  // 1) Complex sqrt
44  // std::complex<double> rhoOld;
45  // rhoOld = sqrt(std::complex<double>(qSqValue(sqrtS,ma,mb))) /
46  //(8*M_PI*sqrtS); //PDG definition
47  // rhoOld = sqrt(std::complex<double>(qSqValue(sqrtS,ma,mb))) /
48  //(0.5*sqrtS); //BaBar definition
49  // return rhoOld; //complex sqrt
50 
51  // 2) Correct analytic continuation
52  // proper analytic continuation (PDG 2014 - Resonances (47.2.2))
53  // I'm not sure of this is correct for the case of two different masses ma
54  // and mb. Furthermore we divide by the factor 16*Pi*Sqrt[s]). This is
55  // more or less arbitrary and not mentioned in the reference, but it leads
56  // to a good agreement between both approaches.
57  std::complex<double> rho;
58  double q = std::sqrt(std::fabs(qSqValue(sqrtS, ma, mb) * 4 / s));
59  if ((ma + mb) * (ma + mb) < s) { // above threshold
60  rho = (-q / M_PI * log(std::fabs((1 + q) / (1 - q))) + i * q) /
61  (i * 16.0 * M_PI * sqrtS);
62  } else if (0 < s && s <= (ma + mb) * (ma + mb)) { // below threshold
63  rho = (-2.0 * q / M_PI * atan(1 / q)) / (i * 16.0 * M_PI * sqrtS);
64  } else if (s <= 0) { // below 0
65  rho = -q / M_PI * std::log(std::fabs((1 + q) / (1 - q)));
66  } else
67  throw std::runtime_error("phspFactor() | PhspFactor not "
68  "defined at sqrtS = " +
69  std::to_string((long double)sqrtS));
70 
71 #ifndef NDEBUG
72  if (rho.real() != rho.real() || rho.imag() != rho.imag()) {
73  std::stringstream ss;
74  ss << "phspFactor() | Result invalid (NaN)! sqrtS=" << sqrtS
75  << ", ma=" << ma << ", mb=" << mb;
76  throw std::runtime_error(ss.str());
77  }
78 #endif
79 
80  return rho; // correct analytical continuation
81 }

99 inline double FormFactor(std::complex<double> qValue, unsigned int orbitL,
100  double mesonRadius, FormFactorType type) {
101  if (mesonRadius == 0)
102  return 1.0; // disable form factors
103  if (type == FormFactorType::noFormFactor)
104  return 1.0; // disable form factors
105  if (type == FormFactorType::BlattWeisskopf && orbitL == 0) {
106  return 1.0;
107  }
108 
109  double ff = 0.0;
110 
111  if (type == FormFactorType::CrystalBarrel) {
112  // Form factor for a0(980) used by Crystal Barrel (Phys.Rev.D78-074023)
113  if (orbitL == 0) {
114  double qSq = std::norm(qValue);
115  double alpha = mesonRadius * mesonRadius / 6;
116  ff = std::exp(-alpha * qSq);
117  } else
118  throw std::runtime_error("FormFactor() | Form factors of type " +
119  std::string(formFactorTypeString[type]) +
120  " are implemented for spin 0 only!");
121  } else if (type == FormFactorType::BlattWeisskopf) {
122  // Blatt-Weisskopf form factors with normalization F(x=mR) = 1.
123  // Reference: S.U.Chung Annalen der Physik 4(1995) 404-430
124  // z = q / (interaction range). For the interaction range we assume
125  // 1/mesonRadius
126  if (orbitL == 0)
127  return 1.0;
128 
129  double qSq = std::norm(qValue);
130  double z = qSq * mesonRadius * mesonRadius;
131  // Events below threshold. What should we do if event is below threshold?
132  // Shouldn't really influence the result because resonances at threshold
133  // don't have spin(?).
134  // Ref. for Blatt-Weisskopf: Phys. Rev. D 48, 1225
135  z = std::fabs(z);
136 
137  if (orbitL == 1) {
138  ff = std::sqrt(2 * z / (z + 1));
139  } else if (orbitL == 2) {
140  ff = std::sqrt(13 * z * z / ((z - 3) * (z - 3) + 9 * z));
141  } else if (orbitL == 3) {
142  ff = std::sqrt(277 * z * z * z /
143  (z * (z - 15) * (z - 15) + 9 * (2 * z - 5) * (2 * z - 5)));
144  } else if (orbitL == 4) {
145  ff = std::sqrt(12746 * z * z * z * z /
146  ((z * z - 45 * z + 105) * (z * z - 45 * z + 105) +
147  25 * z * (2 * z - 21) * (2 * z - 21)));
148  } else
149  throw std::runtime_error("FormFactor() | Form factors of type " +
150  std::string(formFactorTypeString[type]) +
151  " are implemented for spins up to 4!");
152  } else {
153  throw std::runtime_error("FormFactor() | Form factor type " +
154  std::to_string((long long int)type) +
155  " not specified!");
156  }
157 
158  return ff;
159 }
