75 void FunctionTree::insertNode(std::shared_ptr<TreeNode> node,
76  std::string parent) {
77 
78  auto parentNode = Head->findNode(parent);
79  if (!parentNode)
80  throw TreeBuildError("FunctionTree::insertNode | Cannot insert node " +
81  node->name() + ", already exists!");
82 
83  // Reuse existing node
84  auto oldNode = Head->findNode(node->name());
85  if (oldNode) {
86  oldNode->Parents.push_back(parentNode);
87  parentNode->update();
88  return;

90 
91  node->Parents.push_back(parentNode);
