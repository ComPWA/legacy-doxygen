53 bool HelicityKinematics::isWithinPhaseSpace(const DataPoint &point) const {
54  unsigned int pos = 0;
55  for (auto bounds : InvMassBounds) {
56  if (point.KinematicVariableList[pos] < bounds.first ||
57  point.KinematicVariableList[pos] > bounds.second) {
58  return false;
59  }
60  if (point.KinematicVariableList[pos + 1] < 0 ||
61  point.KinematicVariableList[pos + 1] > M_PI)
62  return false;
63  if (point.KinematicVariableList[pos + 2] < -M_PI ||
64  point.KinematicVariableList[pos + 2] > M_PI)
65  return false;
66 
67  pos += 3;
68  }
69 
70  return true;
71 }

97 void HelicityKinematics::createAllSubsystems() {
98  std::vector<IndexListTuple> CurrentSubsystems;
99  LOG(INFO) << "creating all Subsystems!";
100 
101  std::vector<IndexListTuple> AllSubsystems;
102  // add current subsystems
103  for (auto const &SubSys : subSystems()) {
104  AllSubsystems.push_back(
105  std::make_tuple(KinematicsInfo.convertFinalStateIDToPositionIndex(
106  SubSys.getFinalStates()[0]),
107  KinematicsInfo.convertFinalStateIDToPositionIndex(
108  SubSys.getFinalStates()[1]),
109  KinematicsInfo.convertFinalStateIDToPositionIndex(
110  SubSys.getRecoilState()),
111  KinematicsInfo.convertFinalStateIDToPositionIndex(
112  SubSys.getParentRecoilState())));
113  }
114 
115  IndexList FinalStateIDs(KinematicsInfo.getFinalStateParticleCount());
116  unsigned int i = 0;
117  std::generate(FinalStateIDs.begin(), FinalStateIDs.end(),
118  [&i]() -> unsigned int { return i++; });
119  for (auto const &x : redistributeIndexLists(FinalStateIDs, IndexList())) {
120  CurrentSubsystems.push_back(
121  std::make_tuple(x.first, x.second, IndexList(), IndexList()));
122  }
123 
124  while (CurrentSubsystems.size() > 0) {
125  auto TempSubsystems(CurrentSubsystems);
126  CurrentSubsystems.clear();
127  for (auto const &x : TempSubsystems) {
128  IndexList CurrentA(std::get<0>(x));
129  IndexList CurrentB(std::get<1>(x));
130  IndexList CurrentRecoil(std::get<2>(x));
131  IndexList CurrentParentRecoil(std::get<3>(x));
132  auto SortedEntry = sortSubsystem(x);
133  auto result = std::find_if(
134  AllSubsystems.begin(), AllSubsystems.end(),
135  [&SortedEntry, this](const IndexListTuple &element) -> bool {
136  return sortSubsystem(element) == SortedEntry;
137  });
138  if (result == AllSubsystems.end()) {
139  AllSubsystems.push_back(SortedEntry);
140  }
141 
142  // create more combinations on this level
143  for (auto const &ABnew : redistributeIndexLists(CurrentA, CurrentB)) {
144  CurrentSubsystems.push_back(std::make_tuple(
145  ABnew.first, ABnew.second, CurrentRecoil, CurrentParentRecoil));
146  }
147  // Try to go a level deeper for A
148  if (CurrentA.size() > 1) {
149  for (auto const &ABnew :
150  redistributeIndexLists(CurrentA, IndexList())) {
151  // use current B as recoil
152  // and shift recoil to parent recoil
153  CurrentSubsystems.push_back(std::make_tuple(ABnew.first, ABnew.second,
154  CurrentB, CurrentRecoil));
155  }
156  }
157  // same for B
158  if (CurrentB.size() > 1) {
159  for (auto const &ABnew :
160  redistributeIndexLists(CurrentB, IndexList())) {
161  CurrentSubsystems.push_back(std::make_tuple(ABnew.first, ABnew.second,
162  CurrentA, CurrentRecoil));
163  }
164  }
165  }
166  }
167  for (auto const &x : AllSubsystems) {
168  addSubSystem(std::get<0>(x), std::get<1>(x), std::get<2>(x),
169  std::get<3>(x));
170  }
171 }

173 HelicityKinematics::IndexListTuple
174 HelicityKinematics::sortSubsystem(const IndexListTuple &SubSys) const {
175  IndexList FinalStateA(std::get<0>(SubSys));
176  IndexList FinalStateB(std::get<1>(SubSys));
177  IndexList RecoilState(std::get<2>(SubSys));
178  IndexList ParentRecoilState(std::get<3>(SubSys));
179 
180  // create sorted entry
181  std::sort(FinalStateA.begin(), FinalStateA.end());
182  std::sort(FinalStateB.begin(), FinalStateB.end());
183  std::sort(RecoilState.begin(), RecoilState.end());
184  std::sort(ParentRecoilState.begin(), ParentRecoilState.end());
185 
186  IndexListTuple SortedTuple;
187  if (FinalStateA > FinalStateB)
188  SortedTuple = std::make_tuple(FinalStateB, FinalStateA, RecoilState,
189  ParentRecoilState);
190  else
191  SortedTuple = std::make_tuple(FinalStateA, FinalStateB, RecoilState,
192  ParentRecoilState);
193  return SortedTuple;
194 }

196 std::vector<std::pair<IndexList, IndexList>>
197 HelicityKinematics::redistributeIndexLists(const IndexList &A,
198  const IndexList &B) const {
199  std::vector<std::pair<IndexList, IndexList>> NewIndexLists;
200  if (A.size() < B.size())
201  throw std::runtime_error("HelicityKinematics::redistributeIndexLists(): A "
202  "cannot have less content than B!");
203  if (A.size() == 2 && B.size() == 0) {
204  NewIndexLists.push_back(std::make_pair(IndexList{A[0]}, IndexList{A[1]}));
205  } else if (A.size() - B.size() > 1) {
206  for (unsigned int i = 0; i < A.size(); ++i) {
207  IndexList TempB(B);
208  TempB.push_back(A[i]);
209  IndexList TempA(A);
210  TempA.erase(TempA.begin() + i);
211  NewIndexLists.push_back(std::make_pair(TempA, TempB));
212  }
213  }
214  return NewIndexLists;
215 }

217 unsigned int HelicityKinematics::addSubSystem(const SubSystem &subSys) {
218  // We calculate the variables currently for two-body decays
219  if (subSys.getFinalStates().size() != 2) {
220  std::stringstream ss;
221  ss << "HelicityKinematics::addSubSystem(const SubSystem "
222  "&subSys): Number of final state particles = "
223  << subSys.getFinalStates().size() << ", which is != 2";
224  throw std::runtime_error(ss.str());
225  }
226  unsigned int pos(Subsystems.size());
227  auto const result = std::find(Subsystems.begin(), Subsystems.end(), subSys);
228  if (result == Subsystems.end()) {
229  Subsystems.push_back(subSys);
230  InvMassBounds.push_back(calculateInvMassBounds(subSys));
231  std::stringstream ss;
232  ss << subSys;
233  VariableNames.push_back("mSq" + ss.str());
234  VariableNames.push_back("theta" + ss.str());
235  VariableNames.push_back("phi" + ss.str());
236  } else {
237  pos = result - Subsystems.begin();
238  }
239  return pos;
240 }

263 double HelicityKinematics::helicityAngle(double M, double m, double m2,
264  double mSpec, double invMassSqA,
265  double invMassSqB) const {
266  // Calculate energy and momentum of m1/m2 in the invMassSqA rest frame
267  double eCms = (invMassSqA + m * m - m2 * m2) / (2.0 * std::sqrt(invMassSqA));
268  double pCms = eCms * eCms - m * m;
269  // Calculate energy and momentum of mSpec in the invMassSqA rest frame
270  double eSpecCms =
271  (M * M - mSpec * mSpec - invMassSqA) / (2.0 * std::sqrt(invMassSqA));
272  double pSpecCms = eSpecCms * eSpecCms - mSpec * mSpec;
273  double cosAngle =
274  -(invMassSqB - m * m - mSpec * mSpec - 2.0 * eCms * eSpecCms) /
275  (2.0 * std::sqrt(pCms * pSpecCms));
276 
277  return cosAngle;
278 }

280 void HelicityKinematics::convert(
281  const Event &event, DataPoint &point, const SubSystem &sys,
282  const std::pair<double, double> &limits) const {
283  FourMomentum FinalA, FinalB;
284  for (auto s : sys.getFinalStates().at(0)) {
285  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
286  FinalA += event.ParticleList[index].fourMomentum();
287  }
288 
289  for (auto s : sys.getFinalStates().at(1)) {
290  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
291  FinalB += event.ParticleList[index].fourMomentum();
292  }
293 
294  // Four momentum of the decaying resonance
295  FourMomentum State = FinalA + FinalB;
296  double mSq = State.invMassSq();
297 
298  QFT::Vector4<double> DecayingState(State);
299  QFT::Vector4<double> Daughter(FinalA);
300 
301  // calculate the recoil and parent recoil
302  auto const &RecoilState = sys.getRecoilState();
303 
304  // the first step is boosting everything into the rest system of the
305  // decaying state
306  Daughter.Boost(DecayingState);
307 
308  if (RecoilState.size() > 0) {
309  FourMomentum TempRecoil;
310  for (auto s : RecoilState) {
311  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
312  TempRecoil += event.ParticleList[index].fourMomentum();
313  }
314  QFT::Vector4<double> Recoil(TempRecoil);
315 
316  Recoil.Boost(DecayingState);
317 
318  // rotate recoil so that recoil points in the z direction
319  Daughter.RotateZ(-Recoil.Phi());
320  Daughter.RotateY(M_PI - Recoil.Theta());
321 
322  auto const &ParentRecoilState = sys.getParentRecoilState();
323  // in case there is no parent recoil, it is artificially along z
324  QFT::Vector4<double> ParentRecoil(0.0, 0.0, 0.0, 1.0);
325  if (ParentRecoilState.size() > 0) {
326  FourMomentum TempParentRecoil;
327  for (auto s : ParentRecoilState) {
328  unsigned int index =
329  KinematicsInfo.convertFinalStateIDToPositionIndex(s);
330  TempParentRecoil += event.ParticleList[index].fourMomentum();
331  }
332  ParentRecoil = TempParentRecoil;
333  }
334 
335  ParentRecoil.Boost(DecayingState);
336  ParentRecoil.RotateZ(-Recoil.Phi());
337  ParentRecoil.RotateY(M_PI - Recoil.Theta());
338 
339  // rotate around the z-axis so that the parent recoil lies in the x-z
340  // plane
341  Daughter.RotateZ(M_PI - ParentRecoil.Phi());
342  }
343 
344  double cosTheta = Daughter.CosTheta();
345  double phi = Daughter.Phi();
346 
347  point.Weight = event.Weight;
348  point.KinematicVariableList.push_back(mSq);
349  point.KinematicVariableList.push_back(std::acos(cosTheta));
350  point.KinematicVariableList.push_back(phi);
351 }

363 std::pair<double, double>
364 HelicityKinematics::calculateInvMassBounds(const SubSystem &sys) const {
367  std::pair<double, double> lim(0.0, 0.0);
368  // Sum up masses of all final state particles
369  for (auto j : sys.getFinalStates())
370  lim.first += KinematicsInfo.calculateFinalStateIDMassSum(j);
371 
372  double S = KinematicsInfo.getInitialStateInvariantMassSquared();
373  double RemainderMass(0.0);
374  for (auto x : KinematicsInfo.getFinalStateMasses())
375  RemainderMass += x;
376  RemainderMass -= lim.first;
377 
378  lim.first *= lim.first;
379  // to improve precision: (M - m)^2 -> S - 2 sqrt(S) m + m^2 with S=M^2
380  lim.second =
381  S - 2 * std::sqrt(S) * RemainderMass + RemainderMass * RemainderMass;
382 
383  // extend the invariant mass interval by the numeric double precision
384  // otherwise quite a few events at the phase space boundary can be lost
385  // due to numerical imprecision of event generators
386  // (especially drastic effect, if gammas are in the final state)
387  lim.first -= std::numeric_limits<double>::epsilon() * lim.first;
388  lim.second += std::numeric_limits<double>::epsilon() * lim.second;
389  return lim;
390 }
