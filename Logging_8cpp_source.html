43 void Logging::setLogLevel(std::string Level) {
44  // Capitalize string
45  std::transform(Level.begin(), Level.end(), Level.begin(), ::toupper);
46 
47  // Normally use the hierarchy mode of easyloggingcpp, e.g.
48  // el::Loggers::addFlag(el::LoggingFlag::HierarchicalLogging);
49  // el::Loggers::setLoggingLevel(el::Level::Fatal);
50  // However, the hierarchy of the easyloggingcpp log levels is currently not
51  // convenient and has to be manually reordered to:
52  // TRACE, DEBUG, INFO, WARNING, ERROR, FATAL
53 
54  std::vector<el::Level> OffLevels;
55  if (Level == "TRACE") {
56  // trace is the highest level, outputs everything
57  } else if (Level == "DEBUG")
58  OffLevels = {el::Level::Trace};
59  else if (Level == "INFO")
60  OffLevels = {el::Level::Trace, el::Level::Debug};
61  else if (Level == "WARNING")
62  OffLevels = {el::Level::Trace, el::Level::Debug, el::Level::Info};
63  else if (Level == "ERROR")
64  OffLevels = {el::Level::Trace, el::Level::Debug, el::Level::Info,
65  el::Level::Warning};
66  else if (Level == "FATAL")
67  OffLevels = {el::Level::Trace, el::Level::Debug, el::Level::Info,
68  el::Level::Warning, el::Level::Error};
69  else {
70  OffLevels = {el::Level::Trace, el::Level::Debug};
71  LOG(WARNING) << "Logging::setLogLevel() | Log level " + Level +
72  " unknown. Setting log level to [Info] instead!";
73  }
74 
75  el::Logger *logger =
76  ELPP->registeredLoggers()->get(el::base::consts::kDefaultLoggerId);
77 
78  for (auto x : OffLevels) {
79  logger->configurations()->set(x, el::ConfigurationType::Enabled, "0");
80  }
81  logger->reconfigure();
82 };
