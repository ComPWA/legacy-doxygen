68 inline void* mmapTHP(size_t bytes) {
69  // Initializes in zero-initialized data section
70  static void* hint;
71 
72  // Optimistically try to use a last huge page aligned region end
73  // as a hint for mmap.
74  hint = hint ? (void*)((uintptr_t)hint - bytes) : hint;
75  void* result = mmap_impl(bytes, hint);
76 
77  // Something went wrong
78  if (result == MAP_FAILED) {
79  hint = NULL;
80  return MAP_FAILED;
81  }
82 
83  // Otherwise, fall back to the slow path - map oversized region
84  // and trim excess parts.
85  if (!isAligned(result, HUGE_PAGE_SIZE)) {
86  // Undo previous try
87  munmap(result, bytes);
88 
89  // Map oversized on huge page size region
90  result = mmap_impl(bytes + HUGE_PAGE_SIZE);
91 
92  // Something went wrong
93  if (result == MAP_FAILED) {
94  hint = NULL;
95  return MAP_FAILED;
96  }
97 
98  // Misalignment offset
99  uintptr_t offset = 0;
100 
101  if (!isAligned(result, HUGE_PAGE_SIZE)) {
102  // Trim excess head of a region if it is no aligned
103  offset = HUGE_PAGE_SIZE - ((uintptr_t)result & (HUGE_PAGE_SIZE - 1));
104  munmap(result, offset);
105 
106  // New region begining
107  result = (void*)((uintptr_t)result + offset);
108  }
109 
110  // Trim excess tail of a region
111  munmap((void*)((uintptr_t)result + bytes), HUGE_PAGE_SIZE - offset);
112  }
113 
114  // Assume, that mmap virtual addresses grow down by default
115  // So, set a hint as a result of a last successfull allocation
116  // and then use it minus requested size as a new mapping point.
117  // TODO: Atomic store is meant here, fence not needed, but
118  // currently we don't have such function.
119  hint = result;
120 
121  MALLOC_ASSERT(isAligned(result, HUGE_PAGE_SIZE), "Mapped address is not aligned on huge page size.");
122 
123  return result;
124 }

127 void* MapMemory (size_t bytes, PageType pageType)
128 {
129  void* result = 0;
130  int prevErrno = errno;
131 
132  switch (pageType) {
133  case REGULAR:
134  {
135  result = mmap_impl(bytes);
136  break;
137  }
138  case PREALLOCATED_HUGE_PAGE:
139  {
140  MALLOC_ASSERT((bytes % HUGE_PAGE_SIZE) == 0, "Mapping size should be divisible by huge page size");
141  result = mmap_impl(bytes, NULL, __TBB_MAP_HUGETLB);
142  break;
143  }
144  case TRANSPARENT_HUGE_PAGE:
145  {
146  MALLOC_ASSERT((bytes % HUGE_PAGE_SIZE) == 0, "Mapping size should be divisible by huge page size");
147  result = mmapTHP(bytes);
148  break;
149  }
150  default:
151  {
152  MALLOC_ASSERT(false, "Unknown page type");
153  }
154  }
155 
156  if (result == MAP_FAILED) {
157  errno = prevErrno;
158  return 0;
159  }
160 
161  return result;
162 }
