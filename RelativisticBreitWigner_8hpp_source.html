67 inline std::complex<double>
68 dynamicalFunction(double mSq, double mR, double ma, double mb, double width,
69  unsigned int L, double mesonRadius, FormFactorType ffType) {
70 
71  std::complex<double> i(0, 1);
72  double sqrtS = std::sqrt(mSq);
73 
74  // Phase space factors at sqrt(s) and at the resonance position
75  auto phspFactorSqrtS = phspFactor(sqrtS, ma, mb);
76  auto phspFactormR = phspFactor(mR, ma, mb);
77 
78  // Check if we have an event which is exactly at the phase space boundary
79  if (phspFactorSqrtS == std::complex<double>(0, 0))
80  return std::complex<double>(0, 0);
81 
82  // The each FormFactor includes a term q^L. Therefore only q instead of
83  // q^(2L+1) has to be calculated. Also because phspFactor ~ q/sqrt(s) (see
84  // PDG2018, equation 48.2) the factor \frac{M_R}{\sqrt{s}} can also be omitted
85  std::complex<double> qRatio = (phspFactorSqrtS / phspFactormR);
86 
87  double ffR = FormFactor(mR, ma, mb, L, mesonRadius, ffType);
88  double ff = FormFactor(sqrtS, ma, mb, L, mesonRadius, ffType);
89  std::complex<double> barrierTermSq = qRatio * (ff * ff) / (ffR * ffR);
90 
91  // Calculate normalized vertex function gammaA(s_R) at the resonance position
92  // (see PDG2018, Chapter 48.2)
93  std::complex<double> gammaA(1, 0); // spin==0
94  if (L > 0) {
95  std::complex<double> qR = std::pow(qValue(mR, ma, mb), L);
96  gammaA = ffR * qR;
97  }
98 
99  // Coupling to the final state (ma, mb)
100  std::complex<double> g_final =
101  widthToCoupling(mR, width, gammaA, phspFactorSqrtS);
102 
103  std::complex<double> denom(mR * mR - mSq, 0);
104  denom += (-1.0) * i * sqrtS * (width * barrierTermSq);
105 
106  std::complex<double> result = g_final / denom;
107 
108  assert(
109  (!std::isnan(result.real()) || !std::isinf(result.real())) &&
110  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
111  assert(
112  (!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
113  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
114 
115  return result;
116 }
