71 inline std::complex<double>
72 dynamicalFunction(double mSq, double mR, double ma, double mb, double width,
73  unsigned int L, double mesonRadius, FormFactorType ffType) {
74 
75  std::complex<double> i(0, 1);
76  double sqrtS = std::sqrt(mSq);
77 
78  // Phase space factors at sqrt(s) and at the resonance position
79  auto phspFactorSqrtS = phspFactor(sqrtS, ma, mb);
80  auto phspFactormR = phspFactor(mR, ma, mb);
81 
82  // Check if we have an event which is exactly at the phase space boundary
83  if (phspFactorSqrtS == std::complex<double>(0, 0))
84  return std::complex<double>(0, 0);
85 
86  // The each FormFactor includes a term q^L. Therefore only q instead of
87  // q^(2L+1) has to be calculated. Also because phspFactor ~ q/sqrt(s) (see
88  // PDG2018, equation 48.2) the factor \frac{M_R}{\sqrt{s}} can also be omitted
89  std::complex<double> qRatio = (phspFactorSqrtS / phspFactormR);
90 
91  double ffR = FormFactor(mR, ma, mb, L, mesonRadius, ffType);
92  double ff = FormFactor(sqrtS, ma, mb, L, mesonRadius, ffType);
93  std::complex<double> barrierTermSq = qRatio * (ff * ff) / (ffR * ffR);
94 
95  // Calculate normalized vertex function gammaA(s_R) at the resonance position
96  // (see PDG2018, Chapter 48.2)
97  std::complex<double> gammaA(1, 0); // spin==0
98  if (L > 0) {
99  std::complex<double> qR = std::pow(qValue(mR, ma, mb), L);
100  gammaA = ffR * qR;
101  }
102 
103  // Coupling to the final state (ma, mb)
104  std::complex<double> g_final =
105  widthToCoupling(mR, width, gammaA, phspFactorSqrtS);
106 
107  std::complex<double> denom(mR * mR - mSq, 0);
108  denom += (-1.0) * i * sqrtS * (width * barrierTermSq);
109 
110  std::complex<double> result = g_final / denom;
111 
112  assert(
113  (!std::isnan(result.real()) || !std::isinf(result.real())) &&
114  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
115  assert(
116  (!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
117  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
118 
119  return result;
120 }
