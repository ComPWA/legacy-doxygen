44  TClonesArray *pParticles(&Particles);

47  fTree->GetBranch("Particles")->SetAutoDelete(false);

49  fTree->SetBranchAddress("weight", &feventWeight);

51  unsigned int NumberEventsToRead(NumberEventsToProcess);
52  if (NumberEventsToProcess <= 0 || NumberEventsToProcess > fTree->GetEntries())
53  NumberEventsToRead = fTree->GetEntries();

55  std::vector<ComPWA::Event> Events;
56  Events.reserve(NumberEventsToRead);
57 
58  for (unsigned int i = 0; i < NumberEventsToRead; ++i) {
59  Event evt;
60  Particles.Clear();
61  fTree->GetEntry(i);
62 
63  // Get number of particle in TClonesrray
64  unsigned int nParts = Particles.GetEntriesFast();
65 
66  TParticle *partN;
67  TLorentzVector inN;
68  for (unsigned int part = 0; part < nParts; part++) {
69  partN = 0;

71  if (!partN)
72  continue;
73  partN->Momentum(inN);
74  evt.ParticleList.push_back(

76  } // particle loop

78 
79  Events.push_back(evt);
80  } // end event loop
