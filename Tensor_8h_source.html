168  void Boost(double __bx, double __by, double __bz) {
169 
170  // check to see if bx,by,bz are all less than 1
171  if (std::abs(__bx) >= 1 || std::abs(__by) >= 1 || std::abs(__bz) >= 1)
172  std::cout << "Error! Attempt to boost using invalid boost vector."
173  << std::endl;
174  assert((std::abs(__bx) < 1) && (std::abs(__by) < 1) &&
175  (std::abs(__bz) < 1));
176 
177  Tensor<double> lt(2); // Lorentz transformation tensor
178  double gamma = 1.0 / sqrt(1.0 - __bx * __bx - __by * __by - __bz * __bz);
179  double gamFact = (gamma * gamma) / (gamma + 1.0);
180 
181  // set up the Lorentz transformation tensor
182  lt.Zero();
183 
184  lt(0, 0) = gamma;
185  lt(0, 1) = gamma * __bx;
186  lt(0, 2) = gamma * __by;
187  lt(0, 3) = gamma * __bz;
188 
189  lt(1, 1) = (__bx * __bx * gamFact) + 1;
190  lt(1, 2) = __bx * __by * gamFact;
191  lt(1, 3) = __bx * __bz * gamFact;
192 
193  lt(2, 2) = (__by * __by * gamFact) + 1;
194  lt(2, 3) = __by * __bz * gamFact;
195 
196  lt(3, 3) = (__bz * __bz * gamFact) + 1;
197 
198  lt(1, 0) = lt(0, 1);
199  lt(2, 0) = lt(0, 2);
200  lt(2, 1) = lt(1, 2);
201  lt(3, 0) = lt(0, 3);
202  lt(3, 1) = lt(1, 3);
203  lt(3, 2) = lt(2, 3);
204 
205  this->Transform(lt);
206  }

219  void Rotate(double __alpha, double __beta, double __gamma) {
220 
221  double ca = cos(__alpha);
222  double sa = sin(__alpha);
223  double cb = cos(__beta);
224  double sb = sin(__beta);
225  double cg = cos(__gamma);
226  double sg = sin(__gamma);
227 
228  Tensor<double> lt(2); // Lorentz transformation tensor
229 
230  lt.Zero();
231 
232  lt(0, 0) = 1.0;
233 
234  lt(1, 1) = ca * cb * cg - sa * sg;
235  lt(1, 2) = sa * cb * cg + ca * sg;
236  lt(1, 3) = -sb * cg;
237 
238  lt(2, 1) = -ca * cb * sg - sa * cg;
239  lt(2, 2) = -sa * cb * sg + ca * cg;
240  lt(2, 3) = sb * sg;
241 
242  lt(3, 1) = ca * sb;
243  lt(3, 2) = sa * sb;
244  lt(3, 3) = cb;
245 
246  this->Transform(lt);

509  template <typename T>
510  Tensor<typename MultType<_Tp, T>::Type>
511  operator%(const Tensor<T> &__tensor) const {
512 
513  int rank = this->_rank + __tensor._rank;
514  Tensor<typename MultType<_Tp, T>::Type> ret(rank);
515 
516  // if either tensor is rank 0, just return this*__tensor
517  if ((_rank == 0) || (__tensor._rank == 0))
518  return (*this) * __tensor;
519  else { // we actually have to do some work
520  TensorIndex index(rank);
521  TensorIndex ind1(this->_rank);
522  TensorIndex ind2(__tensor._rank);
523 
524  int size1 = ind1.Size();
525  // int size2 = ind2.Size();
526 
527  while (index.IsValid()) { // loop over ret's elements
528 
529  // set up this' indicies
530  for (int i = 0; i < size1; i++)
531  ind1.SetIndex(i, index[i]);
532  // set up _tensor's indicies
533  for (int i = size1; i < index.Size(); i++)
534  ind2.SetIndex(i - size1, index[i]);
535

537  ret(index) = (this->Element(ind1)) * (__tensor(ind2));
538  index++;
539  }
540  }

723  template <typename T>
724  Tensor<typename MultType<_Tp, T>::Type>
725  Contract(const Tensor<T> &__tensor, int __num_indicies = -1) const {
726 
727  int ind1max, ind2st, sumsize, nterm, rank;
728  Tensor<typename MultType<_Tp, T>::Type> ret;
729  // MetricTensor g;
730  Tensor<double> g(2); // Metric tensor
731  g.Element(0, 0) = 1.0;
732  g.Element(1, 1) = -1.0;
733  g.Element(2, 2) = -1.0;
734  g.Element(3, 3) = -1.0;
735 
736  double gFactors;
737  typename MultType<_Tp, T>::Type element;
738  ind1max = 0;
739  ind2st = 0;
740  sumsize = 0;
741 
742  if (_rank ==
743  0) { // if this is rank 0, just multiply __tensor by this' value
744  ret.SetRank(__tensor._rank);
745  ret = _data[0] * __tensor;
746  return ret;
747  } else if (__tensor._rank == 0) { //__tensor is rank 0...
748  ret.SetRank(_rank);
749  ret = (*this) * __tensor._data[0];
750  return ret;
751  }
752 
753  if (__num_indicies > _rank || __num_indicies > __tensor._rank) {
754  std::cout << "<Tensor::Contract> Error! Can't contract " << __num_indicies
755  << " between a " << _rank << " rank and a " << __tensor._rank
756  << " rank tensor." << std::endl;
757  abort();
758  }
759 
760  if (__num_indicies > 0)
761  rank = _rank + __tensor._rank - 2 * __num_indicies;
762  else
763  rank = abs(_rank - __tensor._rank);
764 
765  // set ret's rank and create the summed TensorIndex
766  ret.SetRank(rank);
767  TensorIndex indSummed;
768 
769  if (__num_indicies < 0) {
770  // check to see which tensor has the smaller rank (calculate how many
771  // summed indicies are needed)
772  if (_rank <= __tensor._rank)
773  sumsize = _rank;
774  else
775  sumsize = __tensor._rank;
776  } else
777  sumsize = __num_indicies;
778  indSummed.Resize(2 * sumsize);
779 
780  TensorIndex ind1(this->_rank);
781  TensorIndex ind2(__tensor._rank);
782 
783  int size1 = ind1.Size();
784  int size2 = ind2.Size();
785 
786  if (rank > 0) { // rank > 0, we need to do all the loops
787  TensorIndex index(rank);
788 
789  while (index.IsValid()) { // loop over ret's elements
790 
791  // check to see if this will have any free indicies
792  if ((size1 - sumsize) > 0)
793  ind1max = size1 - sumsize;
794  else
795  ind1max = 0;
796 
797  // set this index (except last ??(number of summed indicies) indicies)
798  for (int i = 0; i < ind1max; i++)
799  ind1.SetIndex(i, index[i]);
800 
801  // set __tensor index (except 1st ??(summed indicies) indicies)
802  ind2st = sumsize;
803  for (int i = ind2st; i < size2; i++)
804  ind2.SetIndex(i, index[ind1max + (i - ind2st)]);
805 
806  nterm = 0;
807  while (indSummed.IsValid()) { // loop over summed indicies
808 
809  gFactors = g(indSummed[0], indSummed[0 + indSummed.Size() / 2]);
810  // get the needed amount of metric tensor factors
811  for (int i = 1; i < indSummed.Size() / 2; i++) {
812  gFactors *= g(indSummed[i], indSummed[i + indSummed.Size() / 2]);
813  }
814  if (gFactors != 0.0) {
815  nterm++;
816  // set up last ?? this and 1st ?? __tensor indicies
817  for (int i = ind1max; i < size1; i++)
818  ind1.SetIndex(i, indSummed[i - ind1max]);
819  for (int i = size1 - ind1max; i < indSummed.Size(); i++)
820  ind2.SetIndex(i - (size1 - ind1max), indSummed[i]);
821 
822  // multiply the metric tensor factor by this and __tensor elements
823  element = (this->Element(ind1)) * (__tensor(ind2)) * gFactors;
824 
825  // add to each element this*__tensor*g*g...*g with correct # of g's
826  if (nterm == 1)
827  ret(index) = element;
828  else
829  ret(index) += element;
830  }
831  ++indSummed;
832  }
833  // reset the summed indicies, step up index to next ret element
834  indSummed.Reset();
835  ++index;
836  }
837  } else { // both are same rank tensors (R is rank 0)
838  nterm = 0;
839 
840  // loop over summed indicies (only loop needed in this case)
841  while (indSummed.IsValid()) {
842  gFactors = g(indSummed[0], indSummed[0 + indSummed.Size() / 2]);
843 
844  // get the needed amount of metric tensor factors
845  for (int i = 1; i < indSummed.Size() / 2; i++)
846  gFactors *= g(indSummed[i], indSummed[i + indSummed.Size() / 2]);
847 
848  if (gFactors != 0.0) {
849  nterm++;
850  for (int i = 0; i < indSummed.Size() / 2; i++) {
851  ind1.SetIndex(i, indSummed[i]);
852  ind2.SetIndex(i, indSummed[i + indSummed.Size() / 2]);
853  }
854  element = (this->Element(ind1)) * (__tensor(ind2)) * gFactors;

939  void Transform(const Tensor<double> &__lt) {
940 
941  if (__lt.Rank() != 2)
942  std::cout << "Error! Lorentz transformation tensor NOT rank 2."
943  << std::endl;
944  assert(__lt.Rank() == 2);
945  int rank = this->Rank();
946  if (rank > 0) { // if rank 0 no transformation needed
947  TensorIndex index(rank);
948  TensorIndex indSummed(rank);
949  int nterm;
950  double lamFact;
951  // make a copy
952  Tensor<_Tp> copy(*this);
953 
954  while (index.IsValid()) { // loop over elements of this tensor
955  nterm = 0;
956  while (indSummed.IsValid()) {
957  // get the appropriate number of Lambda_mu_nu factors
958  lamFact = __lt(index[0], indSummed[0]);
959  for (int i = 1; i < rank; i++)
960  lamFact *= __lt(index[i], indSummed[i]);
961  if (lamFact != 0.0) {
962  nterm++;
963  // add to each element this*Lambda*Lambda*...*Lambda
964  if (nterm == 1)
965  (*this)(index) = lamFact * (copy(indSummed));
966  else
967  (*this)(index) += lamFact * (copy(indSummed));
968  }
969  ++indSummed;
970  }
971  // reset summed indicies, step up index to next element
972  indSummed.Reset();
973  ++index;
974  }
975  }
976  }
