70 std::shared_ptr<Parameter> TreeNode::recalculate() const {
71  // has been changed or is lead node -> return Parameter
72  if (OutputParameter && (!HasChanged || !ChildNodes.size()))
73  return OutputParameter;
74 
75  std::shared_ptr<Parameter> result;
76  if (OutputParameter)
77  result = OutputParameter;
78 
79  ParameterList newVals;
80  for (auto ch : ChildNodes) {
81  auto p = ch->parameter();
82  if (p->isParameter())
83  newVals.addParameter(p);
84  else
85  newVals.addValue(p);
86  }
87  try {
88  Strat->execute(newVals, result);
89  } catch (std::exception &ex) {
90  LOG(INFO) << "TreeNode::Recalculate() | Strategy " << Strat
91  << " failed on node " << name() << ": " << ex.what();
92  throw;
93  }
94 
95  return result;
96 }

117 std::string TreeNode::print(int level, std::string prefix) const {
118  std::stringstream oss;
119  oss << prefix << Name;
120 
121  auto p = recalculate();
122  if (!ChildNodes.size()) { // Print leaf nodes
123  if (p->name() != "")
124  oss << " [" << p->name() << "]";
125  oss << " = " << p->val_to_str() << std::endl;
126  } else { // Print non-leaf nodes
127  oss << " [";
128  if (!OutputParameter)
129  oss << "-, ";
130  oss << ChildNodes.size() << "]";
131  if (OutputParameter && HasChanged)
132  oss << " = ?";
133  else
134  oss << " = " << p->val_to_str() << std::endl;
135  }
136 
137  // Abort recursion
138  if (level == 0)
139  return oss.str();
140 
141  for (auto ch : ChildNodes) {
142  oss << ch->print(level - 1, prefix + ". ");
143  }
144  return oss.str();

153 void TreeNode::deleteParentLinks(std::shared_ptr<TreeNode> parent) {
154  // Remove parent node from this node's Parents
155  auto r = std::find(Parents.begin(), Parents.end(), parent);
156  if (r != Parents.end())
157  Parents.erase(r);
158 
159  // If this node does not have any remaining parents we need to delete links
160  // down to tree to ensure that shared_ptr's reference count goes to zero.
161  // We do this recursively until we arrive at a node which still has other
162  // parents.
163  if (!Parents.size()) {
164  for (auto ch : childNodes()) {
165  ch->deleteParentLinks(shared_from_this());
166  }
167  if (OutputParameter)
168  OutputParameter->Detach(shared_from_this());
169  }
170 }
