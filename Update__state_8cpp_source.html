222 char GetAdjacentCellState(
223  char* source, // pointer to source data block
224  int x, // logical width of field
225  int y, // logical height of field
226  int cellNumber, // number of cell position to examine
227  int cp // which adjacent position
228  )
229 {
230 /*
231 cp
232 *-- cp=1 ... --- cp=8 (summary: -1-2-3-
233 -x- -x- -4-x-5-
234 --- --* -6-7-8- )
235 */
236  char cellState = 0; // return value
237 
238  // set up boundary flags to trigger field-wrap logic
239  bool onTopRow = false;
240  bool onBottomRow = false;
241  bool onLeftColumn = false;
242  bool onRightColumn = false;
243 
244  // check to see if cell is on top row
245  if (cellNumber < x)
246  {
247  onTopRow = true;
248  }
249  // check to see if cell is on bottom row
250  if ((x*y)-cellNumber <= x)
251  {
252  onBottomRow = true;
253  }
254  // check to see if cell is on left column
255  if (cellNumber%x == 0)
256  {
257  onLeftColumn = true;
258  }
259  // check to see if cell is on right column
260  if ((cellNumber+1)%x == 0)
261  {
262  onRightColumn = true;
263  }
264 
265  switch (cp)
266  {
267  case 1:
268  if (onTopRow && onLeftColumn)
269  {
270  return *(source+((x*y)-1));
271  }
272  if (onTopRow && !onLeftColumn)
273  {
274  return *(source+(((x*y)-x)+(cellNumber-1)));
275  }
276  if (onLeftColumn && !onTopRow)
277  {
278  return *(source+(cellNumber-1));
279  }
280  return *((source+cellNumber)-(x+1));
281 
282  case 2:
283  if (onTopRow)
284  {
285  return *(source+(((x*y)-x)+cellNumber));
286  }
287  return *((source+cellNumber)-x);
288 
289  case 3:
290  if (onTopRow && onRightColumn)
291  {
292  return *(source+((x*y)-x));
293  }
294  if (onTopRow && !onRightColumn)
295  {
296  return *(source+(((x*y)-x)+(cellNumber+1)));
297  }
298  if (onRightColumn && !onTopRow)
299  {
300  return *(source+((cellNumber-(x*2))+1));
301  }
302  return *(source+(cellNumber-(x-1)));
303 
304  case 4:
305  if (onRightColumn)
306  {
307  return *(source+(cellNumber-(x-1)));
308  }
309  return *(source+(cellNumber+1));
310 
311  case 5:
312  if (onBottomRow && onRightColumn)
313  {
314  return *source;
315  }
316  if (onBottomRow && !onRightColumn)
317  {
318  return *(source+((cellNumber-((x*y)-x))+1));
319  }
320  if (onRightColumn && !onBottomRow)
321  {
322  return *(source+(cellNumber+1));
323  }
324  return *(source+(((cellNumber+x))+1));
325 
326  case 6:
327  if (onBottomRow)
328  {
329  return *(source+(cellNumber-((x*y)-x)));
330  }
331  return *(source+(cellNumber+x));
332 
333  case 7:
334  if (onBottomRow && onLeftColumn)
335  {
336  return *(source+(x-1));
337  }
338  if (onBottomRow && !onLeftColumn)
339  {
340  return *(source+(cellNumber-((x*y)-x)-1));
341  }
342  if (onLeftColumn && !onBottomRow)
343  {
344  return *(source+(cellNumber+((x*2)-1)));
345  }
346  return *(source+(cellNumber+(x-1)));
347 
348  case 8:
349  if (onLeftColumn)
350  {
351  return *(source+(cellNumber+(x-1)));
352  }
353  return *(source+(cellNumber-1));
354  }

358 char CheckCell(Matrix * m_matrix, int cellNumber)
359 {
360  char total = 0;
361  char* source = m_matrix->data;
362  //look around to find cell's with status "alive"
363  for(int i=1; i<9; i++)
364  {
365  total += GetAdjacentCellState(source, m_matrix->width, m_matrix->height, cellNumber, i);
366  }
367  // if the number of adjacent live cells is < 2 or > 3, the result is a dead
368  // cell regardless of its current state. (A live cell dies of loneliness if it
369  // has less than 2 neighbors, and of overcrowding if it has more than 3; a new
370  // cell is born in an empty spot only if it has exactly 3 neighbors.
371  if (total < 2 || total > 3)
372  {
373  return 0;
374  }
375 
376  // if we get here and the cell position holds a living cell, it stays alive
377  if (*(source+cellNumber))
378  {
379  return 1;
380  }
381 
382  // we have an empty position. If there are only 2 neighbors, the position stays
383  // empty.
384  if (total == 2)
385  {
386  return 0;
387  }
388 
389  // we have an empty position and exactly 3 neighbors. A cell is born.
390  return 1;
391 }
