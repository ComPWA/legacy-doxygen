38 inline int factorial(int __i) {
39  int f = 1;
40  if ((__i == 0) || (__i == 1))
41  f = 1;
42  else {
43  while (__i > 0) {
44  f = f * __i;
45  __i--;
46  }
47  }
48  return f;
49 }

63 inline double Clebsch(const double _j1, const double _m1, const double _j2,
64  const double _m2, const double _J, const double _M) {
65  // convert to pure integers (each 2*spin)
66  int j1 = (int)(2. * _j1);
67  int m1 = (int)(2. * _m1);
68  int j2 = (int)(2. * _j2);
69  int m2 = (int)(2. * _m2);
70  int J = (int)(2. * _J);
71  int M = (int)(2. * _M);
72 
73  if ((m1 + m2) != M)
74  return 0.;
75  if (m1 > j1)
76  return 0;
77  if (m2 > j2)
78  return 0;
79 
80  double n0, n1, n2, n3, n4, n5, d0, d1, d2, d3, d4, A, exp;
81  int nu = 0;
82 
83  double sum = 0;
84  while (((d3 = (j1 - j2 - M) / 2 + nu) < 0) ||
85  ((n2 = (j1 - m1) / 2 + nu) < 0)) {
86  nu++;
87  }
88  while (((d1 = (J - j1 + j2) / 2 - nu) >= 0) &&
89  ((d2 = (J + M) / 2 - nu) >= 0) &&
90  ((n1 = (j2 + J + m1) / 2 - nu) >= 0)) {
91  d3 = ((j1 - j2 - M) / 2 + nu);
92  n2 = ((j1 - m1) / 2 + nu);
93  d0 = dfact((double)nu);
94  exp = nu + (j2 + m2) / 2;
95  n0 = (double)pow(-1., exp);
96  sum += ((n0 * dfact(n1) * dfact(n2)) /
97  (d0 * dfact(d1) * dfact(d2) * dfact(d3)));
98  nu++;
99  }
100 
101  if (sum == 0)
102  return 0;
103 
104  n0 = J + 1;
105  n1 = dfact((double)(J + j1 - j2) / 2);
106  n2 = dfact((double)(J - j1 + j2) / 2);
107  n3 = dfact((double)(j1 + j2 - J) / 2);
108  n4 = dfact((double)(J + M) / 2);
109  n5 = dfact((J - M) / 2);
110 
111  d0 = dfact((double)(j1 + j2 + J) / 2 + 1);
112  d1 = dfact((double)(j1 - m1) / 2);
113  d2 = dfact((double)(j1 + m1) / 2);
114  d3 = dfact((double)(j2 - m2) / 2);
115  d4 = dfact((double)(j2 + m2) / 2);
116 
117  A = ((double)(n0 * n1 * n2 * n3 * n4 * n5)) /
118  ((double)(d0 * d1 * d2 * d3 * d4));
119 
120  return sqrt(A) * sum;
121 }

130 inline double Wigner_d(const double _J, const double _M, const double _N,
131  double _beta) {
132 
133  int J = 2 * _J;
134  int M = 2 * _M;
135  int N = 2 * _N;
136  double beta = _beta;
137 
138  int temp_M, k, k_low, k_hi;
139  double const_term = 0.0, sum_term = 0.0, d = 1.0;
140  int m_p_n, j_p_m, j_p_n, j_m_m, j_m_n;
141  int kmn1, kmn2, jmnk, jmk, jnk;
142  double kk;
143 
144  if (J < 0 || abs(M) > J || abs(N) > J) {
145  std::cerr << std::endl;
146  std::cerr << "d: you have entered an illegal number for J, M, N." << std::endl;
147  std::cerr << "Must follow these rules: J >= 0, abs(M) <= J, and abs(N) <= J."
148  << std::endl;
149  std::cerr << "J = " << J << " M = " << M << " N = " << N << std::endl;
150  return 0.;
151  }
152 
153  if (beta < 0) {
154  beta = fabs(beta);
155  temp_M = M;
156  M = N;
157  N = temp_M;
158  }
159 
160  m_p_n = (M + N) / 2;
161  j_p_m = (J + M) / 2;
162  j_m_m = (J - M) / 2;
163  j_p_n = (J + N) / 2;
164  j_m_n = (J - N) / 2;
165 
166  kk = (double)factorial(j_p_m) * (double)factorial(j_m_m) *
167  (double)factorial(j_p_n) * (double)factorial(j_m_n);
168  const_term = pow((-1.0), (j_p_m)) * sqrt(kk);
169 
170  k_low = MAX(0, m_p_n);
171  k_hi = MIN(j_p_m, j_p_n);
172 
173  for (k = k_low; k <= k_hi; k++) {
174 
175  kmn1 = 2 * k - (M + N) / 2;
176  jmnk = J + (M + N) / 2 - 2 * k;
177  jmk = (J + M) / 2 - k;
178  jnk = (J + N) / 2 - k;
179  kmn2 = k - (M + N) / 2;
180 
181  sum_term +=
182  pow((-1.0), (k)) *
183  ((pow(cos(beta / 2.0), kmn1)) * (pow(sin(beta / 2.0), jmnk))) /
184  (factorial(k) * factorial(jmk) * factorial(jnk) * factorial(kmn2));
185  }
186 
187  d = const_term * sum_term;
188  return d;
189 }
