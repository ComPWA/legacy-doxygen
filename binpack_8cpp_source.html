39 typedef size_t size_type; // to represent non-zero indices, capacities, etc.
40 typedef size_t value_type; // the type of items we are attempting to pack into bins
41 typedef std::vector<value_type> bin; // we use a simple vector to represent a bin
42 // Our bin packers will be function nodes in the graph that take value_type items and
43 // return a dummy value. They will also implicitly send packed bins to the bin_buffer
44 // node, and unused items back to the value_pool node:
45 typedef multifunction_node<value_type, tuple<value_type, bin>, rejecting> bin_packer;
46 // Items are placed into a pool that all bin packers grab from, represent by a queue_node:
47 typedef queue_node<value_type> value_pool;
48 // Packed bins are placed in this buffer waiting to be serially printed and/or accounted for:
49 typedef buffer_node<bin> bin_buffer;
50 // Packed bins are taken from the_bin_buffer and processed by the_writer:
51 typedef function_node<bin, continue_msg, rejecting> bin_writer;
52 // Items are injected into the graph when this node sends them to the_value_pool:
53 typedef source_node<value_type> value_source;

75 class bin_filler {
76  typedef bin_packer::output_ports_type ports_type;
77  bin my_bin; // the current bin that this bin_filler is packing
78  size_type my_used; // capacity of bin used by current contents (not to be confused with my_bin.size())
79  size_type relax, relax_val; // relaxation counter for determining when to settle for a non-full bin
80  bin_packer* my_bin_packer; // ptr to the bin packer that this body object is associated with
81  size_type bin_index; // index of the encapsulating bin packer in the global bins array
82  value_type looking_for; // the minimum size of item this bin_packer will accept
83  value_pool* the_value_pool; // the queue of incoming values
84  bool done; // flag to indicate that this binpacker has been deactivated
85  public:
86  bin_filler(size_t bidx, value_pool* _q) :
87  my_used(0), relax(0), relax_val(0), my_bin_packer(NULL), bin_index(bidx), looking_for(V), the_value_pool(_q), done(false) {}
88  void operator()(const value_type& item, ports_type& p) {
89  if (!my_bin_packer) my_bin_packer = bins[bin_index];
90  if (done) get<0>(p).try_put(item); // this bin_packer is done packing items; put item back to pool
91  else if (item > V) { // signal that packed_sum has reached item_sum at some point
92  size_type remaining = active_bins--;
93  if (remaining == 1 && packed_sum == item_sum) { // this is the last bin and it has seen everything
94  // this bin_packer may not have seen everything, so stay active
95  if (my_used>0) get<1>(p).try_put(my_bin);
96  my_bin.clear();
97  my_used = 0;
98  looking_for = V;
99  ++active_bins;
100  }
101  else if (remaining == 1) { // this is the last bin, but there are remaining items
102  get<0>(p).try_put(V+1); // send out signal
103  ++active_bins;
104  }
105  else if (remaining > 1) { // this is not the last bin; deactivate
106  if (my_used < V/(1+optimality*.1)) { // this bin is ill-utilized; throw back items and deactivate
107  packed_sum -= my_used;
108  packed_items -= my_bin.size();
109  for (size_type i=0; i<my_bin.size(); ++i)
110  get<0>(p).try_put(my_bin[i]);
111  the_value_pool->remove_successor(*my_bin_packer); // deactivate
112  done = true;
113  get<0>(p).try_put(V+1); // send out signal
114  }
115  else { // this bin is well-utilized; send out bin and deactivate
116  the_value_pool->remove_successor(*my_bin_packer); // build no more bins
117  done = true;
118  if (my_used>0) get<1>(p).try_put(my_bin);
119  get<0>(p).try_put(V+1); // send out signal
120  }
121  }
122  }
123  else if (item <= V-my_used && item >= looking_for) { // this item can be packed
124  my_bin.push_back(item);
125  my_used += item;
126  packed_sum += item;
127  ++packed_items;
128  looking_for = V-my_used;
129  relax = 0;
130  if (packed_sum == item_sum) {
131  get<0>(p).try_put(V+1); // send out signal
132  }
133  if (my_used == V) {
134  get<1>(p).try_put(my_bin);
135  my_bin.clear();
136  my_used = 0;
137  looking_for = V;
138  }
139  }
140  else { // this item can't be packed; relax constraints
141  ++relax;
142  if (relax >= (N-packed_items)/optimality) { // this bin_packer has looked through enough items
143  relax = 0;
144  --looking_for; // accept a wider range of items
145  if (looking_for == 0 && my_used < V/(1+optimality*.1) && my_used > 0 && active_bins > 1) {
146  // this bin_packer is ill-utilized and can't find items; deactivate and throw back items
147  size_type remaining = active_bins--;
148  if (remaining > 1) { // not the last bin_packer
149  the_value_pool->remove_successor(*my_bin_packer); // deactivate
150  done = true;
151  }
152  else active_bins++; // can't deactivate last bin_packer
153  packed_sum -= my_used;
154  packed_items -= my_bin.size();
155  for (size_type i=0; i<my_bin.size(); ++i)
156  get<0>(p).try_put(my_bin[i]);
157  my_bin.clear();
158  my_used = 0;
159  }
160  else if (looking_for == 0 && (my_used >= V/(1+optimality*.1) || active_bins == 1)) {
161  // this bin_packer can't find items but is well-utilized, so send it out and reset
162  get<1>(p).try_put(my_bin);
163  my_bin.clear();
164  my_used = 0;
165  looking_for = V;
166  }
167  }
168  get<0>(p).try_put(item); // put unused item back to pool
169  }
170  }

188 // the terminal function_node uses this to gather stats and print bin information
189 class bin_printer {
190  value_type running_count;
191  size_type item_count;
192  value_type my_min, my_max;
193  double avg;
194 public:
195  bin_printer() : running_count(0), item_count(0), my_min(V), my_max(0), avg(0) {}
196  continue_msg operator()(bin b) {
197  value_type sum=0;
198  ++B;
199  if (verbose)
200  std::cout << "[ ";
201  for (size_type i=0; i<b.size(); ++i) {
202  if (verbose)
203  std::cout << b[i] << " ";
204  sum+=b[i];
205  ++item_count;
206  }
207  if (sum < my_min) my_min = sum;
208  if (sum > my_max) my_max = sum;
209  avg += sum;
210  running_count += sum;
211  if (verbose)
212  std::cout << "]=" << sum << "; Done/Packed/Total cap: " << running_count << "/" << packed_sum << "/" << item_sum
213  << " items:" << item_count << "/" << packed_items << "/" << N << " B=" << B << std::endl;
214  if (item_count == N) { // should be the last; print stats
215  avg = avg/(double)B;
216  if (!silent)
217  std::cout << "SUMMARY: #Bins used: " << B << "; Avg size: " << avg << "; Max size: " << my_max
218  << "; Min size: " << my_min << "\n Lower bound on optimal #bins: " << min_B
219  << "; Start #bins: " << num_bin_packers << std::endl;
220  }
221  return continue_msg(); // need to return something
222  }

262  task_scheduler_init init(p);
