91 void box_intersect(box * bx, ray * ry) {
92  flt a, tx1, tx2, ty1, ty2, tz1, tz2;
93  flt tnear, tfar;
94 
95  tnear= -FHUGE;
96  tfar= FHUGE;
97 
98  if (ry->d.x == 0.0) {
99  if ((ry->o.x < bx->min.x) || (ry->o.x > bx->max.x)) return;
100  }
101  else {
102  tx1 = (bx->min.x - ry->o.x) / ry->d.x;
103  tx2 = (bx->max.x - ry->o.x) / ry->d.x;
104  if (tx1 > tx2) { a=tx1; tx1=tx2; tx2=a; }
105  if (tx1 > tnear) tnear=tx1;
106  if (tx2 < tfar) tfar=tx2;
107  }
108  if (tnear > tfar) return;
109  if (tfar < 0.0) return;
110 
111  if (ry->d.y == 0.0) {
112  if ((ry->o.y < bx->min.y) || (ry->o.y > bx->max.y)) return;
113  }
114  else {
115  ty1 = (bx->min.y - ry->o.y) / ry->d.y;
116  ty2 = (bx->max.y - ry->o.y) / ry->d.y;
117  if (ty1 > ty2) { a=ty1; ty1=ty2; ty2=a; }
118  if (ty1 > tnear) tnear=ty1;
119  if (ty2 < tfar) tfar=ty2;
120  }
121  if (tnear > tfar) return;
122  if (tfar < 0.0) return;
123 
124  if (ry->d.z == 0.0) {
125  if ((ry->o.z < bx->min.z) || (ry->o.z > bx->max.z)) return;
126  }
127  else {
128  tz1 = (bx->min.z - ry->o.z) / ry->d.z;
129  tz2 = (bx->max.z - ry->o.z) / ry->d.z;
130  if (tz1 > tz2) { a=tz1; tz1=tz2; tz2=a; }
131  if (tz1 > tnear) tnear=tz1;
132  if (tz2 < tfar) tfar=tz2;
133  }
134  if (tnear > tfar) return;
135  if (tfar < 0.0) return;
136 
137  add_intersection(tnear, (object *) bx, ry);
138  add_intersection(tfar, (object *) bx, ry);
139 }

141 void box_normal(box * bx, vector * pnt, ray * incident, vector * N) {
142  vector a, b, c;
143  flt t;
144 
145  c.x=(bx->max.x + bx->min.x) / 2.0;
146  c.y=(bx->max.y + bx->min.y) / 2.0;
147  c.z=(bx->max.z + bx->min.z) / 2.0;
148 
149  VSub((vector *) pnt, &c, N);
150  b=(*N);
151 
152  a.x=fabs(N->x);
153  a.y=fabs(N->y);
154  a.z=fabs(N->z);
155 
156  N->x=0.0; N->y=0.0; N->z=0.0;
157 
158  t=MYMAX(a.x, MYMAX(a.y, a.z));
159 
160  if (t==a.x) N->x=b.x;
161 
162  if (t==a.y) N->y=b.y;
163 
164  if (t==a.z) N->z=b.z;
165 
166  VNorm(N);
167 }
