61 ray camray(scenedef *scene, int x, int y) {
62  ray ray1, newray;
63  vector projcent;
64  vector projpixel;
65  flt px, py, sx, sy;
66 
67  sx = (flt) scene->hres;
68  sy = (flt) scene->vres;
69 
70  /* calculate the width and height of the image plane given the */
71  /* aspect ratio, image resolution, and zoom factor */
72 
73  px=((sx / sy) / scene->aspectratio) / scene->camzoom;
74  py=1.0 / scene->camzoom;
75 
76  /* assuming viewvec is a unit vector, then the center of the */
77  /* image plane is the camera center + vievec */
78  projcent.x = scene->camcent.x + scene->camviewvec.x;
79  projcent.y = scene->camcent.y + scene->camviewvec.y;
80  projcent.z = scene->camcent.z + scene->camviewvec.z;
81 
82  /* starting from the center of the image plane, we move the */
83  /* center of the pel we're calculating, to */
84  /* projcent + (rightvec * x distance) */
85  ray1.o=projcent;
86  ray1.d=scene->camrightvec;
87  projpixel=Raypnt(&ray1, ((x*px/sx) - (px / 2.0)));
88 
89  /* starting from the horizontally translated pel, we move the */
90  /* center of the pel we're calculating, to */
91  /* projcent + (upvec * y distance) */
92  ray1.o=projpixel;
93  ray1.d=scene->camupvec;
94  projpixel=Raypnt(&ray1, ((y*py/sy) - (py / 2.0)));
95 
96  /* now that we have the exact pel center in the image plane */
97  /* we create the real primary ray that will be used by the */
98  /* rest of the system. */
99  /* The ray is expected to be re-normalized elsewhere, we're */
100  /* only really concerned about getting its direction right. */
101  newray.o=scene->camcent;
102  VSub(&projpixel, &scene->camcent, &newray.d);
103  newray.depth = scene->raydepth;
104  newray.flags = RT_RAY_REGULAR; /* camera only generates primary rays */
105 
106  return newray;
107 }
