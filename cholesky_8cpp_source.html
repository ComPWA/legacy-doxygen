71 // Creates tiled array
72 static double ***create_tile_array( double *A, int n, int b ) {
73  const int p = n/b;
74  double ***tile = (double ***)calloc( sizeof( double ** ), p );
75 
76  for ( int j = 0; j < p; ++j ) {
77  tile[j] = (double **)calloc( sizeof( double * ), p );
78  }
79 
80  for ( int j = 0; j < p; ++j ) {
81  for ( int i = 0; i < p; ++i ) {
82  double *temp_block = (double *)calloc( sizeof( double ), b*b );
83 
84  for ( int A_j = j*b, T_j = 0; T_j < b; ++A_j, ++T_j ) {
85  for ( int A_i = i*b, T_i = 0; T_i < b; ++A_i, ++T_i ) {
86  temp_block[T_j*b+T_i] = A[A_j*n+A_i];
87  }
88  }
89 
90  tile[j][i] = temp_block;
91  }
92  }
93  return tile;
94 }

96 static void collapse_tile_array( double ***tile, double *A, int n, int b ) {
97  const int p = n/b;
98 
99  for ( int j = 0; j < p; ++j ) {
100  for ( int i = 0; i < p; ++i ) {
101  double *temp_block = tile[j][i];
102 
103  for ( int A_j = j*b, T_j = 0; T_j < b; ++A_j, ++T_j ) {
104  for ( int A_i = i*b, T_i = 0; T_i < b; ++A_i, ++T_i ) {
105  A[A_j*n+A_i] = temp_block[T_j*b+T_i];
106  }
107  }
108 
109  free( temp_block );
110  tile[j][i] = NULL;
111  }
112 
113  free( tile[j] );
114  }
115 
116  free( tile );
117 }

127  bool check_if_valid( double *A0, double *C, double *A, int n ) {
128  char transa = 'n', transb = 't';
129  double alpha = 1;
130  double beta = 0;
131 
132  for ( int i = 0; i < n; ++i ) {
133  for ( int j = i+1; j < n; ++j ) {
134  A0[j*n+i] = 0.;
135  }
136  }
137 
138  dgemm ( &transa, &transb, &n, &n, &n, &alpha, A0, &n, A0, &n, &beta, C, &n );
139 
140  for ( int j = 0; j < n; ++j ) {
141  for ( int i = 0; i < n; ++i ) {
142  const double epsilon = std::abs( A[j*n+i]*0.1 );
143 
144  if ( std::abs( C[j*n+i] - A[j*n+i] ) > epsilon ) {
145  printf( "ERROR: %s did not validate at C(%d,%d) = %lf != A(%d,%d) = %lf\n",
146  name.c_str(), i, j, C[j*n+i], i, j, A[j*n+i] );
147  printf( "ERROR: %g; %g < %g < %g\n", epsilon, A[j*n+i] - epsilon, C[j*n+i], A[j*n+i] + epsilon );
148  return false;
149  }
150  }
151  }
152  return true;
153  }

222  double *A_block = tile[k][j];
223  double *L_block = tile[k][k];
224  char uplo = 'l', side = 'r', transa = 't', diag = 'n';
225  double alpha = 1;

230 static void call_dsyr2k( double ***tile, int b, int k, int j, int i ) {
231  double *A_block = tile[i][j];
232  char transa = 'n', transb = 't';
233  char uplo = 'l';
234  double alpha = -1;
235  double beta = 1;
236 
237  if ( i == j ) { // Diagonal block
238  double *L_block = tile[k][i];
239  dsyrk( &uplo, &transa, &b, &b, &alpha, L_block, &b, &beta, A_block, &b );
240  } else { // Non-diagonal block
241  double *L2_block = tile[k][i];
242  double *L1_block = tile[k][j];
243  dgemm( &transa, &transb, &b, &b, &b, &alpha, L1_block, &b, L2_block, &b, &beta, A_block, &b );
244  }
245  return;
246 }

337  int p;

339 public:
340  dtrsm_body( int p_, int b_ ) : p(p_), b(b_) {}
341 
342  void operator()( const t2_t &in, dtrsm_node_t::output_ports_type &ports ) {
343  using tbb::flow::get;
344 
345  tagged_tile_t in0 = get<0>( in );
346  tagged_tile_t in1 = get<1>( in );
347  int k = in0.first.a[0];
348  int j = in0.first.a[1];
349  tile_t L_block = in0.second;
350  tile_t A_block = in1.second;
351  tag_t t;
352  t.tag = 0;
353  t.a[0] = k;
354  char uplo = 'l', side = 'r', transa = 't', diag = 'n';
355  double alpha = 1;
356  dtrsm( &side, &uplo, &transa, &diag, &b, &b, &alpha, L_block, &b, A_block, &b);
357 
358  // Send to rest of my row
359  t.a[1] = j;
360  for ( int i = k+1; i <= j; ++i ) {
361  t.a[2] = i;
362  get<0>( ports ).try_put( std::make_pair( t, A_block ) );
363  }
364 
365  // Send to transposed row
366  t.a[2] = j;
367  for ( int i = j; i < p; ++i ) {
368  t.a[1] = i;
369  get<1>( ports ).try_put( std::make_pair( t, A_block ) );

382 
383  tag_t t;
384  t.tag = 0;
385  char transa = 'n', transb = 't';
386  char uplo = 'l';
387  double alpha = -1;
388  double beta = 1;
389 
390  tagged_tile_t in0 = get<0>( in );
391  tagged_tile_t in1 = get<1>( in );
392  tagged_tile_t in2 = get<2>( in );
393  int k = in2.first.a[0];
394  int j = in2.first.a[1];
395  int i = in2.first.a[2];
396 
397  tile_t A_block = in2.second;
398  if ( i == j ) { // Diagonal block
399  tile_t L_block = in0.second;
400  dsyrk( &uplo, &transa, &b, &b, &alpha, L_block, &b, &beta, A_block, &b );
401  } else { // Non-diagonal block
402  tile_t L1_block = in0.second;
403  tile_t L2_block = in1.second;
404  dgemm( &transa, &transb, &b, &b, &b, &alpha, L1_block, &b, L2_block, &b, &beta, A_block, &b );
405  }
406 
407  // All outputs flow to next step
408  t.a[0] = k+1;
409  t.a[1] = j;
410  t.a[2] = i;
411  if ( k != p-1 && j == k+1 && i == k+1 ) {
412  get<0>( ports ).try_put( std::make_pair( t, A_block ) );
413  }
414 
415  if ( k < p-2 ) {
416  if ( i == k+1 && j > i ) {
417  t.a[0] = k+1;
418  t.a[1] = j;
419  get<1>( ports ).try_put( std::make_pair( t, A_block ) );
420  }
421 
422  if ( j != k+1 && i != k+1 ) {
423  t.a[0] = k+1;
424  t.a[1] = j;
425  t.a[2] = i;
426  get<2>( ports ).try_put( std::make_pair( t, A_block ) );
427  }
428  }
429  }

466  make_edge( output_port<1>( dtrsm_node ), input_port<1>( dsyr2k_join ) );

468  make_edge( dsyr2k_join, dsyr2k_node );
469 
470  // Now we need to send out the tiles to their first nodes
471  tag_t t;
472  t.tag = 0;
473  t.a[0] = 0;
474  t.a[1] = 0;
475  t.a[2] = 0;
476

478  // k == 0, j == 0, i == 0
479  dpotf2_node.try_put( std::make_pair( t, tile[0][0] ) );
480 
481  // Send to feedback input (port 1) of each dtrsm
482  // k == 0, j == 1..p-1
483  for ( int j = 1; j < p; ++j ) {
484  t.a[1] = j;
485  input_port<1>( dtrsm_join ).try_put( std::make_pair( t, tile[0][j] ) );
486  }
487 
488  // Send to feedback input (port 2) of each dsyr2k
489  // k == 0
490  for ( int i = 1; i < p; ++i ) {
491  t.a[2] = i;
492 
493  for ( int j = i; j < p; ++j ) {
494  t.a[1] = j;
495  input_port<2>( dsyr2k_join ).try_put( std::make_pair( t, tile[i][j] ) );
496  }
497  }
