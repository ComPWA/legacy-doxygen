144 void micro_queue::push( const void* item, ticket k, concurrent_queue_base& base ) {
145  k &= -concurrent_queue_rep::n_queue;
146  page* p = NULL;
147  size_t index = modulo_power_of_two( k/concurrent_queue_rep::n_queue, base.items_per_page );
148  if( !index ) {
149  size_t n = sizeof(page) + base.items_per_page*base.item_size;
150  p = static_cast<page*>(operator new( n ));
151  p->mask = 0;
152  p->next = NULL;
153  }
154  {
155  push_finalizer finalizer( *this, k+concurrent_queue_rep::n_queue );
156  spin_wait_until_eq( tail_counter, k );
157  if( p ) {
158  spin_mutex::scoped_lock lock( page_mutex );
159  if( page* q = tail_page )
160  q->next = p;
161  else
162  head_page = p;
163  tail_page = p;
164  } else {
165  p = tail_page;
166  }
167  base.copy_item( *p, index, item );
168  // If no exception was thrown, mark item as present.
169  p->mask |= uintptr_t(1)<<index;
170  }
171 }

173 bool micro_queue::pop( void* dst, ticket k, concurrent_queue_base& base ) {
174  k &= -concurrent_queue_rep::n_queue;
175  spin_wait_until_eq( head_counter, k );
176  spin_wait_while_eq( tail_counter, k );
177  page *p = head_page;
178  __TBB_ASSERT( p, NULL );
179  size_t index = modulo_power_of_two( k/concurrent_queue_rep::n_queue, base.items_per_page );
180  bool success = false;
181  {
182  pop_finalizer finalizer( *this, k+concurrent_queue_rep::n_queue, index==base.items_per_page-1 ? p : NULL );
183  if( p->mask & uintptr_t(1)<<index ) {
184  success = true;
185  base.assign_and_destroy_item( dst, *p, index );
186  }
187  }
188  return success;
189 }

226 void concurrent_queue_base::internal_push( const void* src ) {
227  concurrent_queue_rep& r = *my_rep;
228  concurrent_queue_rep::ticket k = r.tail_counter++;
229  if( my_capacity<concurrent_queue_rep::infinite_capacity ) {
230  // Capacity is limited, wait to not exceed it
231  atomic_backoff backoff;
232  while( (ptrdiff_t)(k-r.head_counter)>=const_cast<volatile ptrdiff_t&>(my_capacity) )
233  backoff.pause();
234  }
235  r.choose(k).push(src,k,*this);

246 bool concurrent_queue_base::internal_pop_if_present( void* dst ) {
247  concurrent_queue_rep& r = *my_rep;
248  concurrent_queue_rep::ticket k;
249  do {
250  for( atomic_backoff b;;b.pause() ) {
251  k = r.head_counter;
252  if( r.tail_counter<=k ) {
253  // Queue is empty
254  return false;
255  }
256  // Queue had item with ticket k when we looked. Attempt to get that item.
257  if( r.head_counter.compare_and_swap(k+1,k)==k ) {
258  break;
259  }
260  // Another thread snatched the item, so pause and retry.
261  }
262  } while( !r.choose(k).pop(dst,k,*this) );
263  return true;
264 }

266 bool concurrent_queue_base::internal_push_if_not_full( const void* src ) {
267  concurrent_queue_rep& r = *my_rep;
268  concurrent_queue_rep::ticket k;
269  for( atomic_backoff b;;b.pause() ) {
270  k = r.tail_counter;
271  if( (ptrdiff_t)(k-r.head_counter)>=my_capacity ) {
272  // Queue is full
273  return false;
274  }
275  // Queue had empty slot with ticket k when we looked. Attempt to claim that slot.
276  if( r.tail_counter.compare_and_swap(k+1,k)==k )
277  break;
278  // Another thread claimed the slot, so pause and retry.
279  }
280  r.choose(k).push(src,k,*this);
281  return true;
282 }
