53  static segment_index_t find_segment_end( const concurrent_vector_base& v ) {
54  const size_t pointers_per_long_segment = sizeof(void*)==4 ? 32 : 64;
55  const size_t pointers_per_short_segment = 2;
56  //unsigned u = v.my_segment==v.my_storage ? pointers_per_short_segment : pointers_per_long_segment;
57  segment_index_t u = v.my_segment==(&(v.my_storage[0])) ? pointers_per_short_segment : pointers_per_long_segment;
58  segment_index_t k = 0;
59  while( k<u && v.my_segment[k].array )
60  ++k;
61  return k;
62  }
63  static void extend_segment_if_necessary( concurrent_vector_base& v, size_t k ) {
64  const size_t pointers_per_short_segment = 2;
65  if( k>=pointers_per_short_segment && v.my_segment==v.my_storage ) {
66  extend_segment(v);

71 void concurrent_vector_base::helper::extend_segment( concurrent_vector_base& v ) {
72  const size_t pointers_per_long_segment = sizeof(void*)==4 ? 32 : 64;
73  segment_t* s = (segment_t*)NFS_Allocate( pointers_per_long_segment, sizeof(segment_t), NULL );
74  std::memset( static_cast<void*>(s), 0, pointers_per_long_segment*sizeof(segment_t) );
75  // If other threads are trying to set pointers in the short segment, wait for them to finish their
76  // assignments before we copy the short segment to the long segment.
77  atomic_backoff backoff;
78  while( !v.my_storage[0].array || !v.my_storage[1].array ) backoff.pause();
79  s[0] = v.my_storage[0];
80  s[1] = v.my_storage[1];
81  if( v.my_segment.compare_and_swap( s, v.my_storage )!=v.my_storage )
82  NFS_Free(s);
83 }

89 void concurrent_vector_base::internal_reserve( size_type n, size_type element_size, size_type max_size ) {
90  if( n>max_size ) {
91  __TBB_THROW( std::length_error("argument to concurrent_vector::reserve exceeds concurrent_vector::max_size()") );
92  }
93  for( segment_index_t k = helper::find_segment_end(*this); segment_base(k)<n; ++k ) {
94  helper::extend_segment_if_necessary(*this,k);
95  size_t m = segment_size(k);
96  __TBB_ASSERT( !my_segment[k].array, "concurrent operation during reserve(...)?" );
97  my_segment[k].array = NFS_Allocate( m, element_size, NULL );
98  }

118 void concurrent_vector_base::internal_assign( const concurrent_vector_base& src, size_type element_size, internal_array_op1 destroy, internal_array_op2 assign, internal_array_op2 copy ) {
119  size_type n = src.my_early_size;
120  while( my_early_size>n ) {
121  segment_index_t k = segment_index_of( my_early_size-1 );
122  size_type b=segment_base(k);
123  size_type new_end = b>=n ? b : n;
124  __TBB_ASSERT( my_early_size>new_end, NULL );
125  destroy( (char*)my_segment[k].array+element_size*(new_end-b), my_early_size-new_end );
126  my_early_size = new_end;
127  }
128  size_type dst_initialized_size = my_early_size;
129  my_early_size = n;
130  size_type b;
131  for( segment_index_t k=0; (b=segment_base(k))<n; ++k ) {
132  helper::extend_segment_if_necessary(*this,k);
133  size_t m = segment_size(k);
134  if( !my_segment[k].array )
135  my_segment[k].array = NFS_Allocate( m, element_size, NULL );
136  if( m>n-b ) m = n-b;
137  size_type a = 0;
138  if( dst_initialized_size>b ) {
139  a = dst_initialized_size-b;
140  if( a>m ) a = m;
141  assign( my_segment[k].array, src.my_segment[k].array, a );
142  m -= a;
143  a *= element_size;
144  }
145  if( m>0 )
146  copy( (char*)my_segment[k].array+a, (char*)src.my_segment[k].array+a, m );
147  }
148  __TBB_ASSERT( src.my_early_size==n, "detected use of concurrent_vector::operator= with right side that was concurrently modified" );
149 }

186 void concurrent_vector_base::internal_grow( const size_type start, size_type finish, size_type element_size, internal_array_op1 init ) {
187  __TBB_ASSERT( start<finish, "start must be less than finish" );
188  size_t tmp = start;
189  do {
190  segment_index_t k_old = segment_index_of( tmp );
191  size_type base = segment_base(k_old);
192  size_t n = segment_size(k_old);
193  helper::extend_segment_if_necessary(*this,k_old);
194  segment_t& s = my_segment[k_old];
195  void* array = s.array;
196  if( !array ) {
197  if( base==tmp ) {
198  __TBB_ASSERT( !s.array, NULL );
199  array = NFS_Allocate( n, element_size, NULL );
200  ITT_NOTIFY( sync_releasing, &s.array );
201  s.array = array;
202  } else {
203  ITT_NOTIFY(sync_prepare, &s.array);
204  spin_wait_while_eq( s.array, (void*)0 );
205  ITT_NOTIFY(sync_acquired, &s.array);
206  array = s.array;
207  }
208  }
209  size_type j_begin = tmp-base;
210  size_type j_end = n > finish-base ? finish-base : n;
211  (*init)( (void*)((char*)array+element_size*j_begin), j_end-j_begin );
212  tmp = base+j_end;
213  } while( tmp<finish );
214 }

216 void concurrent_vector_base::internal_clear( internal_array_op1 destroy, bool reclaim_storage ) {
217  // Set "my_early_size" early, so that subscripting errors can be caught.
218  // FIXME - doing so may be hurting exception safety
219  __TBB_ASSERT( my_segment, NULL );
220  size_type finish = my_early_size;
221  my_early_size = 0;
222  while( finish>0 ) {
223  segment_index_t k_old = segment_index_of(finish-1);
224  segment_t& s = my_segment[k_old];
225  __TBB_ASSERT( s.array, NULL );
226  size_type base = segment_base(k_old);
227  size_type j_end = finish-base;
228  __TBB_ASSERT( j_end, NULL );
229  (*destroy)( s.array, j_end );
230  finish = base;
231  }
232 
233  // Free the arrays
234  if( reclaim_storage ) {
235  size_t k = helper::find_segment_end(*this);
236  while( k>0 ) {
237  --k;
238  segment_t& s = my_segment[k];
239  void* array = s.array;
240  s.array = NULL;
241  NFS_Free( array );
242  }
243  // Clear short segment.
244  my_storage[0].array = NULL;
245  my_storage[1].array = NULL;
246  segment_t* s = my_segment;
247  if( s!=my_storage ) {
248  my_segment = my_storage;
249  NFS_Free( s );
250  }
251  }
