16 /*
17  This file contains the Parallel STL-based implementation of quickhull algorithm.
18  Quickhull algorithm description:
19  1. Initial phase
20  1) Find two points that guaranteed to belong to the convex hull. Min and max points in X can be used for it.
21  2) Divide the initial set of points in two subsets by the line formed by two points from previous step.
22  This subset will be processed recursively.
23 
24  2. Iteration Phase
25  1) Divide current subset by dividing line [p1,p2] into right and left subsets.
26  2) New point (p) of the convex hull is found as farthest point of right subset from the dividing line.
27  3) If the right subset has more than 1 point, repeat the iteration phase with the right subset and dividing lines [p1,p] and [p,p2].
28 
29  The implementation based on std::copy_if, std::max_element and std::minmax_element algorithms of Parallel STL.
30  Each of the algorithms use par_unseq policy. In order to get effect of the policy usage problem size should be big enough.
31  By default problem size was set as 5M points. With point set with less than 500K points par_unseq policy could be inefficient.
32  Correctness of the convex hull is checked by std::any_of algorithm with counting iterators.
33 */

54 void divide_and_conquer(Policy exec, Iterator first, Iterator last, pointVec_t &H, point_t p1, point_t fp, point_t p2) {
55  pointVec_t H1;
56  //decomposes the work and combines the results
57  if (first - last > grain_size){ //for small set size parallel policy could be inefficient
58  find_hull_points(exec, first, last, H, p1, fp);
59  find_hull_points(exec, first, last, H1, fp, p2);
60  }
61  else {
62  find_hull_points(pstl::execution::unseq, first, last, H, p1, fp);
63  find_hull_points(pstl::execution::unseq, first, last, H1, fp, p2);
64  }
65  H.insert(H.end(), H1.cbegin(), H1.cend());
66 }

68 //Find points of the convex hull on right sides of the segments [p1, p2]
69 template<typename Policy, typename Iterator>
70 void find_hull_points(Policy exec, Iterator first, Iterator last, pointVec_t &H, point_t p1, point_t p2) {
71 
72  pointVec_t P_reduced(last - first);
73 
74  //Find points from the range [first, last-1] that are on the right side of the segment [p1,p2]
75  Iterator end = std::copy_if(exec, first, last, P_reduced.begin(),
76  [&p1, &p2](const point_t& pnt) {
77  return cross_product(p1, p2, pnt) > 0;
78  });
79 
80  if ((end - P_reduced.cbegin()) < 2) {
81  //Add points into the hull
82  H.push_back(p1);
83  H.insert(H.end(), P_reduced.cbegin(), end);
84  }
85  else {
86  //Find the farthest point from the segment [p1,p1], it will be in the convex hull
87  auto far_point = *std::max_element(exec, P_reduced.cbegin(), end,
88  [&p1, &p2](const point_t & pnt1, const point_t & pnt2) {
89  double how_far1 = cross_product(p1, p2, pnt1);
90  double how_far2 = cross_product(p1, p2, pnt2);
91  return how_far1 == how_far2 ? pnt1 < pnt2 : how_far1 < how_far2;
92  });
93 
94  //Repeat for segments [p1, far_point] and [far_point, p2] with points from [P_reduced.cbegin(), end-1]
95  divide_and_conquer(exec, P_reduced.cbegin(), end, H, p1, far_point, p2);
96  }
97 }

99 //Quickhull algorithm
100 //The algorihm based on the divide and conquer technique
101 void quickhull(const pointVec_t &points, pointVec_t &hull) {
102  if (points.size() < 2) {
103  hull.insert(hull.end(), points.cbegin(), points.cend());
104  return;
105  }
106  //Find left and right most points, they will be in the convex hull
107 #if __INTEL_COMPILER == 1900 && PSTL_VERSION >= 200 && PSTL_VERSION <= 204
108  // A workaround for incorrectly working minmax_element
109  point_t p1 = *std::min_element(pstl::execution::par_unseq, points.cbegin(), points.cend());
110  point_t p2 = *std::max_element(pstl::execution::par_unseq, points.cbegin(), points.cend());
111 #else
112  auto minmaxx = std::minmax_element(pstl::execution::par_unseq, points.cbegin(), points.cend());
113  point_t p1 = *minmaxx.first;
114  point_t p2 = *minmaxx.second;
115 #endif
116 
117  //Divide the set of points into two subsets, which will be processed recursively
118  divide_and_conquer(pstl::execution::par_unseq, points.cbegin(), points.cend(), hull, p1, p2, p1);
119 }

121 // Check if a polygon is convex
122 bool is_convex(const pointVec_t & points) {
123  return std::all_of(pstl::execution::par_unseq,
124  pstl::counting_iterator<size_t>(size_t(0)),
125  pstl::counting_iterator<size_t>(points.size()),
126  [&points](size_t i) {
127  point_t p0(points[i]);
128  point_t p1(points[(i + 1) % points.size()]);
129  point_t p2(points[(i + 2) % points.size()]);
130  return (cross_product(p0, p1, p2) < 0);
131  });
132 }
