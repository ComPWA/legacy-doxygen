99  template < typename T ,typename rng_functor_type>
100  point<T> GenerateRNDPoint(size_t& count, rng_functor_type random, size_t rand_max) {
101  /* generates random points on 2D plane so that the cluster
102  is somewhat circle shaped */
103  const size_t maxsize=500;
104  T x = random()*2.0/(double)rand_max - 1;
105  T y = random()*2.0/(double)rand_max - 1;
106  T r = (x*x + y*y);
107  if(r>1) {
108  count++;
109  if(count>10) {
110  if (random()/(double)rand_max > 0.5)
111  x /= r;
112  if (random()/(double)rand_max > 0.5)
113  y /= r;
114  count = 0;
115  }
116  else {
117  x /= r;
118  y /= r;
119  }
120  }
121 
122  x = (x+1)*0.5*maxsize;
123  y = (y+1)*0.5*maxsize;
124 
125  return point<T>(x,y);
126  }
