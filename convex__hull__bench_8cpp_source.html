127 void divide_and_conquer(const pointVec_t &P, pointVec_t &H, point_t p1, point_t p2) {
128  assert(P.size() >= 2);
129  pointVec_t P_reduced;
130  pointVec_t H1, H2;
131  point_t p_far = divide(P, P_reduced, p1, p2);
132  if (P_reduced.size()<2) {
133  H.push_back(p1);
134  H.insert(H.end(), P_reduced.begin(), P_reduced.end());
135  }
136  else {
137  divide_and_conquer(P_reduced, H1, p1, p_far);
138  divide_and_conquer(P_reduced, H2, p_far, p2);
139 
140  H.insert(H.end(), H1.begin(), H1.end());
141  H.insert(H.end(), H2.begin(), H2.end());
142  }

291 template<typename BodyType>
292 void initialize(pointVec_t &points) {
293  //This function generate the same series of point on every call.
294  //Reproducibility is needed for benchmarking to produce reliable results.
295  //It is achieved through the following points:
296  // - FillRNDPointsVector_buf instance has its own local instance
297  // of random number generator, which in turn does not use any global data
298  // - tbb::simple_partitioner produce the same set of ranges on every call to
299  // tbb::parallel_for
300  // - local RNG instances are seeded by the starting indexes of corresponding ranges
301  // - grow_to_at_least() enables putting points into the resulting vector in deterministic order
302  // (unlike concurrent push_back or grow_by).
303 
304  // In the buffered version, a temporary storage for as much as grainSize elements
305  // is allocated inside the body. Since auto_partitioner may increase effective
306  // range size which would cause a crash, simple partitioner has to be used.
307 
308  tbb::parallel_for(range_t(0, cfg::numberOfPoints, BodyType::grainSize),
309  BodyType(points), tbb::simple_partitioner());
310 }
