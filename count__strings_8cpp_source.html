153 const char *GetLetters(int type, const int part)
154 {
155  Sound *sounds; int rate, i = 0;
156  if(type & 1)
157  sounds = Vowels, rate = rand() % VowelsRatesSum[part];
158  else
159  sounds = Consonants, rate = rand() % ConsonantsRatesSum[part];
160  do {
161  rate -= sounds[i++].rates[part];
162  } while(rate > 0);
163  return sounds[--i].chars;
164 }
