140  void operator()(const int threadID) const {

142  sleep(throughput_window);

144  }
145  else {
146  event e, tmp;
147  unsigned int num_operations = 0;
148  for (;;) {
149  // pop an event
150  if (do_pop(e, nThread, implementation)) {
151  num_operations++;
152  // do the event
153  busy_wait(e.elapse*contention_unit);
154  while (e.spawn > 0) {
155  tmp.spawn = ((e.spawn+1-min_spawn) % ((max_spawn-min_spawn)+1))+min_spawn;
156  tmp.timestamp = e.timestamp + e.elapse;
157  e.timestamp = tmp.timestamp;
158  e.elapse = ((e.elapse+1-min_elapse) % ((max_elapse-min_elapse)+1))+min_elapse;
159  tmp.elapse = e.elapse;
160  do_push(tmp, nThread, implementation);
161  num_operations++;
162  e.spawn--;
163  busy_wait(e.elapse*contention_unit);
164  if (done) break;
165  }
166  }
167  if (done) break;
168  }
169  operation_count += num_operations;
170  }
171  }
