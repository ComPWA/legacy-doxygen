131 static void rt_sheightfield(void * tex, vector ctr, int m, int n,
132  apiflt * field, apiflt wx, apiflt wy) {
133  vector * vertices;
134  vector * normals;
135  vector offset;
136  apiflt xinc, yinc;
137  int x, y, addr;
138 
139  vertices = (vector *) malloc(m*n*sizeof(vector));
140  normals = (vector *) malloc(m*n*sizeof(vector));
141 
142  offset.x = ctr.x - (wx / 2.0);
143  offset.y = ctr.z - (wy / 2.0);
144  offset.z = ctr.y;
145 
146  xinc = wx / ((apiflt) m);
147  yinc = wy / ((apiflt) n);
148 
149  /* build vertex list */
150  for (y=0; y<n; y++) {
151  for (x=0; x<m; x++) {
152  addr = y*m + x;
153  vertices[addr] = rt_vector(
154  x * xinc + offset.x,
155  field[addr] + offset.z,
156  y * yinc + offset.y);
157  }
158  }
159 
160  /* build normals from vertex list */
161  for (x=1; x<m; x++) {
162  normals[x] = normals[(n - 1)*m + x] = rt_vector(0.0, 1.0, 0.0);
163  }
164  for (y=1; y<n; y++) {
165  normals[y*m] = normals[y*m + (m-1)] = rt_vector(0.0, 1.0, 0.0);
166  }
167  for (y=1; y<(n-1); y++) {
168  for (x=1; x<(m-1); x++) {
169  addr = y*m + x;
170 
171  normals[addr] = rt_vector(
172  -(field[addr + 1] - field[addr - 1]) / (2.0 * xinc),
173  1.0,
174  -(field[addr + m] - field[addr - m]) / (2.0 * yinc));
175 
176  MyVNorm(&normals[addr]);
177  }
178  }
179 
180  /* generate actual triangles */
181  for (y=0; y<(n-1); y++) {
182  for (x=0; x<(m-1); x++) {
183  addr = y*m + x;
184 
185  rt_stri(tex, vertices[addr], vertices[addr + 1 + m], vertices[addr + 1],
186  normals[addr], normals[addr + 1 + m], normals[addr + 1]);
187  rt_stri(tex, vertices[addr], vertices[addr + m], vertices[addr + 1 + m],
188  normals[addr], normals[addr + m], normals[addr + 1 + m]);
189  }
190  }
191 
192  free(normals);
193  free(vertices);
194 } /* end of smoothed heightfield */
