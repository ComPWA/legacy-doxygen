42 MinuitResult MinuitIF::optimize(ComPWA::Estimator::Estimator<double> &Estimator,
43  ComPWA::FitParameterList InitialParameters) {
44  LOG(DEBUG) << "MinuitIF::optimize() | Start";
45  std::chrono::steady_clock::time_point StartTime =
46  std::chrono::steady_clock::now();
47 
48  double InitialEstimatorValue(Estimator.evaluate());
49 
50  MnUserParameters upar;
51  size_t FreeParameters(0);
52  for (auto Param : InitialParameters) {
53  if (Param.Name == "")
54  throw BadParameter("MinuitIF::optimize() | FitParameter without name in "
55  "list. Since FitParameter names are unique we stop "
56  "here.");
57  // If no error is set or error set to 0 we use a default error,
58  // otherwise minuit treats this parameter as fixed
59  double Error(0.5 * (Param.Error.first + Param.Error.second));
60  if (Error == 0.0) {
61  if (Param.Value != 0.0)
62  Error = Param.Value * 0.1;
63  else
64  Error = 0.01;
65  }
66 
67  if (Param.HasBounds) {
68  bool rt = upar.Add(Param.Name, Param.Value, Error, Param.Bounds.first,
69  Param.Bounds.second);
70  if (!rt)
71  throw BadParameter(
72  "MinuitIF::optimize() | FitParameter " + Param.Name +
73  " can not be added to Minuit2 (internal) parameters.");
74  } else {
75  bool rt = upar.Add(Param.Name, Param.Value, Error);
76  if (!rt)
77  throw BadParameter(
78  "MinuitIF::optimize() | FitParameter " + Param.Name +
79  " can not be added to Minuit2 (internal) parameters.");
80  }
81  if (!Param.IsFixed)
82  FreeParameters++;
83  if (Param.IsFixed)
84  upar.Fix(Param.Name);
85  }
86 
87  ROOT::Minuit2::MinuitFcn Function(Estimator);
88 
89  LOG(INFO) << "MinuitIF::optimize() | Number of parameters (free): "
90  << InitialParameters.size() << " (" << FreeParameters << ")";
91 
92  // use MnStrategy class to set all options for the fit
93  MnStrategy strat(1); // using default strategy = 1 (medium)
94 
95  try { // try to read xml file for minuit setting
96  std::ifstream ifs("MinuitStrategy.xml");
97  boost::archive::xml_iarchive ia(ifs, boost::archive::no_header);
98  ia >> BOOST_SERIALIZATION_NVP(strat);
99  ifs.close();
100  LOG(DEBUG) << "Minuit strategy parameters: from MinuitStrategy.xml";
101  } catch (std::exception &ex) {
102  }
103 
104  LOG(DEBUG) << "Gradient number of steps: " << strat.GradientNCycles();
105  LOG(DEBUG) << "Gradient step tolerance: " << strat.GradientStepTolerance();
106  LOG(DEBUG) << "Gradient tolerance: " << strat.GradientTolerance();
107  LOG(DEBUG) << "Hesse number of steps: " << strat.HessianNCycles();
108  LOG(DEBUG) << "Hesse gradient number of steps: "
109  << strat.HessianGradientNCycles();
110  LOG(DEBUG) << "Hesse step tolerance: " << strat.HessianStepTolerance();
111  LOG(DEBUG) << "Hesse G2 tolerance: " << strat.HessianG2Tolerance();
112 
113  // MIGRAD
114  MnMigrad migrad(Function, upar, strat);
115  double maxfcn = 0.0;
116  double tolerance = 0.1;
117 
118  // From the MINUIT2 Documentation:
119  // Minimize the function MnMigrad()(maxfcn, tolerance)
120  // \param maxfcn : max number of function calls (if = 0) default is
121  // used which is set to 200 + 100 * npar + 5 * npar**2
122  // \param tolerance : value used for terminating iteration procedure.
123  // For example, MIGRAD will stop iterating when edm (expected
124  // distance from minimum) will be: edm < tolerance * 10**-3
125  // Default value of tolerance used is 0.1
126  LOG(INFO) << "MinuitIF::optimize() | Starting migrad: "
127  "maxCalls="
128  << maxfcn << " tolerance=" << tolerance;
129 
130  FunctionMinimum minMin = migrad(maxfcn, tolerance); //(maxfcn,tolerance)
131 
132  LOG(INFO) << "MinuitIF::optimize() | Migrad finished! "
133  "Minimum is valid = "
134  << minMin.IsValid();
135 
136  // HESSE
137  MnHesse hesse(strat);
138  if (minMin.IsValid() && UseHesse) {
139  LOG(INFO) << "MinuitIF::optimize() | Starting hesse";
140  hesse(Function, minMin); // function minimum minMin is updated by hesse
141  LOG(INFO) << "MinuitIF::optimize() | Hesse finished";
142  } else
143  LOG(INFO) << "MinuitIF::optimize() | Migrad failed to "
144  "find minimum! Skip hesse and minos!";
145 
146  LOG(INFO) << "MinuitIF::optimize() | Minimization finished! "
147  "LH = "
148  << std::setprecision(10) << minMin.Fval();
149 
150  std::chrono::steady_clock::time_point EndTime =
151  std::chrono::steady_clock::now();
152 
153  // Create fit result, before changes to parameters in
154  // FunctionMinimum might occur
155  auto MinState = minMin.UserState();
156  FitResult BaseResult{
157  InitialParameters,
158  getFinalParameters(MinState, InitialParameters),
159  InitialEstimatorValue,
160  MinState.Fval(),
161  std::chrono::duration_cast<std::chrono::seconds>(EndTime - StartTime),
162  getCovarianceMatrix(MinState)};
163  MinuitResult Result(BaseResult, minMin);
164 
165  // In case Minos should be used, recalculate the errors
166  if (minMin.IsValid() && UseMinos) {
167  // MINOS
168  MnMinos minos(Function, minMin, strat);
169  size_t id = 0;
170  for (auto &FinalPar : Result.FinalParameters) {
171  if (FinalPar.IsFixed)
172  continue;
173 
174  // asymmetric errors -> run minos
175  LOG(INFO) << "MinuitIF::optimize() | Run minos "
176  "for parameter ["
177  << id << "] " << FinalPar.Name << "...";
178  MinosError err = minos.Minos(id);
179  FinalPar.Error = err();
180  ++id;
181  }
182  }
183 
184  return Result;
185 }

187 FitParameterList MinuitIF::getFinalParameters(
188  const ROOT::Minuit2::MnUserParameterState &minState,
189  FitParameterList InitialParameters) {
190  FitParameterList FinalParameters(InitialParameters);
191 
192  for (auto &FinalPar : FinalParameters) {
193  if (FinalPar.IsFixed)
194  continue;
195 
196  // central value
197  double val = minState.Value(FinalPar.Name);
198  double err = minState.Error(FinalPar.Name);
199 
200  // shift to [-pi;pi] if parameter is a phase
201  if (FinalPar.Name.find("phase") != FinalPar.Name.npos)
202  val = ComPWA::Utils::shiftAngle(val);
203  FinalPar.Value = val;
204  FinalPar.Error = std::make_pair(err, err);
205  }
206  return FinalParameters;
207 }

209 std::vector<std::vector<double>> MinuitIF::getCovarianceMatrix(
210  const ROOT::Minuit2::MnUserParameterState &minState) {
211  std::vector<std::vector<double>> CovarianceMatrix;
212 
213  if (minState.HasCovariance()) {
214  auto NumFreeParameter = minState.Parameters().Trafo().VariableParameters();
215  ROOT::Minuit2::MnUserCovariance minuitCovMatrix = minState.Covariance();
216  // Size of Minuit covariance vector is given by dim*(dim+1)/2.
217  // dim is the dimension of the covariance matrix.
218  // The dimension can therefore be calculated as
219  // dim = -0.5+-0.5 sqrt(8*size+1)
220  assert(minuitCovMatrix.Nrow() == NumFreeParameter);
221  CovarianceMatrix = std::vector<std::vector<double>>(
222  NumFreeParameter, std::vector<double>(NumFreeParameter));
223  for (unsigned i = 0; i < NumFreeParameter; ++i)
224  for (unsigned j = i; j < NumFreeParameter; ++j) {
225  CovarianceMatrix.at(i).at(j) = minuitCovMatrix(j, i);
226  CovarianceMatrix.at(j).at(i) = minuitCovMatrix(j, i); // fill lower half
227  }
228 
229  } else {
230  LOG(ERROR)
231  << "MinuitIF::createResult(): no valid covariance matrix available!";
232  }
233  return CovarianceMatrix;
234 }
