187 static void scalableMallocCheckSize(void *object, size_t size)
188 {
189 #if __clang__
190 // This prevents Clang from throwing out the calls to new & delete in CheckNewDeleteOverload().
191  static void *v = object;
192  Harness::suppress_unused_warning(v);
193 #endif
194  ASSERT(object, NULL);
195  if (size >= minLargeObjectSize) {
196  LargeMemoryBlock *lmb = ((LargeObjectHdr*)object-1)->memoryBlock;
197  ASSERT(uintptr_t(lmb)<uintptr_t(((LargeObjectHdr*)object-1))
198  && lmb->objectSize >= size, NULL);
199  }
200 #if MALLOC_UNIXLIKE_OVERLOAD_ENABLED || MALLOC_ZONE_OVERLOAD_ENABLED
201  ASSERT(malloc_usable_size(object) >= size, NULL);
202 #elif MALLOC_WINDOWS_OVERLOAD_ENABLED
203  // Check that _msize works correctly
204  ASSERT(_msize(object) >= size, NULL);
205  ASSERT(size<8 || _aligned_msize(object,8,0) >= size, NULL);
206 #endif
207 }

242 void CheckPvalloc(void *(*pvalloc_p)(size_t), void (*free_p)(void*))
243 {
244  const long memoryPageSize = sysconf(_SC_PAGESIZE);
245  // request large object with not power-of-2 size
246  const size_t largeSz = alignUp(minLargeObjectSize, 16*1024) + 1;
247 
248  for (size_t sz = 0; sz<=largeSz; sz+=largeSz) {
249  void *ptr = pvalloc_p(sz);
250  scalableMallocCheckSize(ptr, sz? alignUp(sz, memoryPageSize) : memoryPageSize);
251  ASSERT(is_aligned(ptr, memoryPageSize), NULL);
252  free_p(ptr);
253  }
