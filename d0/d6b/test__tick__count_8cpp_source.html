68 void TestSimpleDelay( int ntrial, double duration, double tolerance ) {
69  double total_worktime = 0;
70  // Iteration -1 warms up the code cache.
71  for( int trial=-1; trial<ntrial; ++trial ) {
72  tbb::tick_count t0 = tbb::tick_count::now();
73  if( duration ) WaitForDuration(duration);
74  tbb::tick_count t1 = tbb::tick_count::now();
75  if( trial>=0 ) {

96 static tbb::atomic<int> Counter1, Counter2;
97 static tbb::atomic<bool> Flag1, Flag2;
98 static tbb::tick_count *tick_count_array;
99 static double barrier_time;
100 
101 struct TickCountDifferenceBody {
102  TickCountDifferenceBody( int num_threads ) {
103  Counter1 = Counter2 = num_threads;
104  Flag1 = Flag2 = false;
105  }
106  void operator()( int id ) const {
107  bool last = false;
108  // The first barrier.
109  if ( --Counter1 == 0 ) last = true;
110  while ( !last && !Flag1.load<tbb::acquire>() ) __TBB_Pause( 1 );
111  // Save a time stamp of the first barrier releasing.
112  tick_count_array[id] = tbb::tick_count::now();
113 
114  // The second barrier.
115  if ( --Counter2 == 0 ) Flag2.store<tbb::release>(true);
116  // The last thread should release threads from the first barrier after it reaches the second
117  // barrier to avoid a deadlock.
118  if ( last ) Flag1.store<tbb::release>(true);
119  // After the last thread releases threads from the first barrier it waits for a signal from
120  // the second barrier.
121  while ( !Flag2.load<tbb::acquire>() ) __TBB_Pause( 1 );
122 
123  if ( last )
124  // We suppose that the barrier time is a time interval between the moment when the last
125  // thread reaches the first barrier and the moment when the same thread is released from
126  // the second barrier. This time is not accurate time of two barriers but it is
127  // guaranteed that it does not exceed it.
128  barrier_time = (tbb::tick_count::now() - tick_count_array[id]).seconds() / 2;
129  }
130  ~TickCountDifferenceBody() {
131  ASSERT( Counter1 == 0 && Counter2 == 0, NULL );
132  }
133 };

136 void TestTickCountDifference( int n ) {
137  const double tolerance = 3E-4;
138  tick_count_array = new tbb::tick_count[n];
139 
140  int num_trials = 0;
141  tbb::tick_count start_time = tbb::tick_count::now();
142  do {
143  NativeParallelFor( n, TickCountDifferenceBody( n ) );
144  if ( barrier_time > tolerance )
145  // The machine seems to be oversubscibed so skip the test.
146  continue;
147  for ( int i = 0; i < n; ++i ) {
148  for ( int j = 0; j < i; ++j ) {
149  double diff = (tick_count_array[i] - tick_count_array[j]).seconds();
150  if ( diff < 0 ) diff = -diff;
151  if ( diff > tolerance )
152  REPORT( "Warning: cross-thread tick_count difference = %g > %g = tolerance\n", diff, tolerance );
153  ASSERT( diff < 3 * tolerance, "Too big difference." );
154  }
155  }
156  // During 5 seconds we are trying to get 10 successful trials.
157  } while ( ++num_trials < 10 && (tbb::tick_count::now() - start_time).seconds() < 5 );
158  REMARK( "Difference test time: %g sec\n", (tbb::tick_count::now() - start_time).seconds() );
159  ASSERT( num_trials == 10, "The machine seems to be heavily oversubscibed, difference test was skipped." );
160  delete[] tick_count_array;
161 }

163 void TestResolution() {
164  static double target_value = 0.314159265358979323846264338327950288419;
165  static double step_value = 0.00027182818284590452353602874713526624977572;
166  static int range_value = 100;
167  double avg_diff = 0.0;
168  double max_diff = 0.0;
169  for( int i = -range_value; i <= range_value; ++i ) {
170  double my_time = target_value + step_value * i;
171  tbb::tick_count::interval_t t0(my_time);
172  double interval_time = t0.seconds();
173  avg_diff += (my_time - interval_time);
174  if ( max_diff < my_time-interval_time) max_diff = my_time-interval_time;
175  // time always truncates
176  ASSERT(interval_time >= 0 && my_time - interval_time < tbb::tick_count::resolution(), "tick_count resolution out of range");
177  }
178  avg_diff = (avg_diff/(2*range_value+1))/tbb::tick_count::resolution();
179  max_diff /= tbb::tick_count::resolution();
180  REMARK("avg_diff = %g ticks, max_diff = %g ticks\n", avg_diff, max_diff);
181 }
