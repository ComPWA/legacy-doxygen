191 template<typename Flavor, size_t Pad>
192 void Flog( int nthread ) {
193  tbb::tick_count T0 = tbb::tick_count::now();
194  for( int i=0; i<N; ++i ) {
195  for ( int mode = 0; mode < 4; ++mode) {
196  FooRange<Pad> r( 0, i );
197  const FooRange<Pad> rc = r;
198  FooBody<Pad> f( Array );
199  const FooBody<Pad> fc = f;
200  memset( Array, 0, sizeof(Array) );
201  FooBodyCount = 1;
202  switch (mode) {
203  case 0: {
204  empty_partitioner_tag p;
205  Invoker< Flavor, empty_partitioner_tag, FooRange<Pad>, FooBody<Pad> > invoke_for;
206  invoke_for( rc, fc, p );
207  }
208  break;
209  case 1: {
210  Invoker< Flavor, const tbb::simple_partitioner, FooRange<Pad>, FooBody<Pad> > invoke_for;
211  invoke_for( rc, fc, tbb::simple_partitioner() );
212  }
213  break;
214  case 2: {
215  Invoker< Flavor, const tbb::auto_partitioner, FooRange<Pad>, FooBody<Pad> > invoke_for;
216  invoke_for( rc, fc, tbb::auto_partitioner() );
217  }
218  break;
219  case 3: {
220  static tbb::affinity_partitioner affinity;
221  Invoker< Flavor, tbb::affinity_partitioner, FooRange<Pad>, FooBody<Pad> > invoke_for;
222  invoke_for( rc, fc, affinity );
223  }
224  break;
225  }
226  for( int j=0; j<i; ++j )
227  ASSERT( Array[j]==1, NULL );
228  for( int j=i; j<N; ++j )
229  ASSERT( Array[j]==0, NULL );
230  ASSERT( FooBodyCount==1, NULL );
231  }
232  }
233  tbb::tick_count T1 = tbb::tick_count::now();
234  REMARK("time=%g\tnthread=%d\tpad=%d\n",(T1-T0).seconds(),nthread,int(Pad));
235 }

255 {

257  const T pfor_buffer_actual_size = static_cast<T>(PFOR_BUFFER_ACTUAL_SIZE);
258  // Testing parallel_for with different step values
259  InvokerStep< Flavor, Partitioner, T, TestFunctor<T> > invoke_for;
260  for (T begin = 0; begin < pfor_buffer_test_size - 1; begin += pfor_buffer_test_size / 10 + 1) {

262  for (step = 1; step < pfor_buffer_test_size; step++) {
263  memset(pfor_buffer, 0, pfor_buffer_actual_size * sizeof(size_t));
264  if (step == 1){
265  invoke_for(begin, pfor_buffer_test_size, TestFunctor<T>(), p);
266  } else {
267  invoke_for(begin, pfor_buffer_test_size, step, TestFunctor<T>(), p);
268  }
269  // Verifying that parallel_for processed all items it should
270  for (T i = begin; i < pfor_buffer_test_size; i = i + step) {

272  pfor_buffer[i] = 0;
273  }
274  // Verifying that no extra items were processed and right bound of array wasn't crossed
275  for (T i = 0; i < pfor_buffer_actual_size; i++) {

610  void update_ets(BlockedRangeWhitebox& range) {
611  std::pair<MapType::iterator, MapType::iterator> equal_range = ets.local().equal_range(&range);
612  for (MapType::iterator it = equal_range.first; it != equal_range.second;++it) {
613  if (it->second.first <= range.begin() && range.end() <= it->second.second) {

638 struct CombineBody {
639  MapType operator()(MapType x, const MapType& y) const {
640  x.insert(y.begin(), y.end());
641  for (MapType::iterator it = x.begin(); it != x.end();++it)
642  for (MapType::iterator internal_it = x.begin(); internal_it != x.end(); ++internal_it) {
643  if (it != internal_it && internal_it->second.first <= it->second.first && it->second.second <= internal_it->second.second) {

646  }

652 range_borders combine_range(const MapType& map) {
653  range_borders result_range = map.begin()->second;
654  for (MapType::const_iterator it = map.begin(); it != map.end(); it++)
655  result_range = range_borders((std::min)(result_range.first, it->second.first), (std::max)(result_range.second, it->second.second));
656  return result_range;

672  for (MapType::iterator it = combined_map.begin(); it != combined_map.end(); ++it) {
673  size_t size = it->second.second - it->second.first;
674  ASSERT((size == expected_size || size == expected_size + 1), "Incorrect post-proportional range size");
