77 void CModel::init_and_terminate( int maxthread ) {
78  for( int i=0; i<200; ++i ) {
79  switch( i&3 ) {
80  default: {
81  tbb::task_scheduler_init init( rand() % maxthread + 1 );
82  break;
83  }
84  case 0: {
85  tbb::task_scheduler_init init;

87  }
88  case 1: {
89  tbb::task_scheduler_init init( tbb::task_scheduler_init::automatic );

91  }
92  case 2: {
93  tbb::task_scheduler_init init( tbb::task_scheduler_init::deferred );
94  init.initialize( rand() % maxthread + 1 );
95  init.terminate();
96  break;
97  }
