283  async_activity(int expected_items, int sleep_time = 50) : my_expected_items(expected_items), my_sleep_time(sleep_time) {
284  my_quit = false;
285  tbb::tbb_thread( ServiceThreadBody( this ) ).swap( my_service_thread );
286  }
287 
288 private:
289 
290  async_activity( const async_activity& ) : my_expected_items(UNKNOWN_NUMBER_OF_ITEMS), my_sleep_time(0) { }
291 
292 public:
293  ~async_activity() {
294  stop();
295  my_service_thread.join();
296  }
297 
298  void submit( const input_type &input, gateway_type& gateway ) {
299  work_type work = { input, &gateway};
300  my_work_queue.push( work );
301  }
302 
303  void process() {
304  do {
305  work_type work;
306  if( my_work_queue.try_pop( work ) ) {
307  Harness::Sleep(my_sleep_time);
308  ++async_activity_processed_msg_count;
309  output_type output;
310  wrapper_helper<output_type, output_type>::copy_value(work.input, output);
311  wrapper_helper<output_type, output_type>::check(work.input, output);
312  work.gateway->try_put(output);
313  if ( my_expected_items == UNKNOWN_NUMBER_OF_ITEMS || int(async_activity_processed_msg_count) == my_expected_items ) {
314  work.gateway->release_wait();
315  }
316  }
317  } while( my_quit == false || !my_work_queue.empty());
318  }
319 
320  void stop() {
321  my_quit = true;
322  }
323 
324  bool should_reserve_each_time() {
325  if ( my_expected_items == UNKNOWN_NUMBER_OF_ITEMS )
326  return true;
327  else
328  return false;
329  }
330 
331 private:
332 
333  const int my_expected_items;
334  const int my_sleep_time;
335 
336  tbb::concurrent_queue< work_type > my_work_queue;
337 
338  tbb::atomic< bool > my_quit;
339 
340  tbb::tbb_thread my_service_thread;
341 };
