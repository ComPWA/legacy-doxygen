67  // Try sequences of various lengths.
68  for (size_t n = 0; n <= 100000; n = n <= 16 ? n + 1 : size_t(3.1415 * n))
69  {
70  Sequence<In> in(n, [=](size_t k) -> In {
71  // Sprinkle "42" and "50" early, so that short sequences have non-zero count.
72  return convert((n - k - 1) % 3 == 0 ? 42 : (n - k - 2) % 5 == 0 ? 50 : 3 * (int(k) % 1000 - 500));
73  });
