116 class my_handler {
117  pq_t *pq;
118 public:
119  my_handler() {}
120  my_handler(pq_t *pq_) : pq(pq_) {}
121  void operator()(tbb::aggregator_operation* op_list) const {
122  while (op_list) {
123  op_data& request = static_cast<op_data&>(*op_list);
124  op_list = op_list->next();
125  request.start();
126  if (request.tid >= 0) pq->push(request.tid);
127  else {
128  ASSERT(!pq->empty(), "queue should not be empty!");
129  int elem = pq->top();
130  pq->pop();
131  shared_data[elem]++;
132  }
133  request.finish();
134  }
135  }
