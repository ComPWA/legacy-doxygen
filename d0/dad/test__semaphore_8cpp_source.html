54 // set semaphore to initial value
55 // see that semaphore only allows that number of threads to be active
56 class Body: NoAssign {
57  const int nThreads;
58  const int nIters;
59  tbb::internal::semaphore &mySem;
60  vector<int> &ourCounts;
61  vector<double> &tottime;
62  static const int tickCounts = 1; // millisecond
63  static const int innerWait = 5; // millisecond
64 public:
65  Body(int nThread_, int nIter_, semaphore &mySem_,
66  vector<int>& ourCounts_,
67  vector<double>& tottime_
68  ) : nThreads(nThread_), nIters(nIter_), mySem(mySem_), ourCounts(ourCounts_), tottime(tottime_) { sBarrier.initialize(nThread_); pCount = 0; }
69 void operator()(const int tid) const {
70  sBarrier.wait();
71  for(int i=0; i < nIters; ++i) {
72  Harness::Sleep( tid * tickCounts );
73  tbb::tick_count t0 = tbb::tick_count::now();
74  mySem.P();
75  tbb::tick_count t1 = tbb::tick_count::now();
76  tottime[tid] += (t1-t0).seconds();
77  int curval = ++pCount;
78  if(curval > ourCounts[tid]) ourCounts[tid] = curval;
79  Harness::Sleep( innerWait );
80  --pCount;
81  ASSERT((int)pCount >= 0, NULL);
82  mySem.V();
83  }
84 }
85 };

88 void testSemaphore( int semInitCnt, int extraThreads ) {
89  semaphore my_sem(semInitCnt);
90  // tbb::task_scheduler_init init(tbb::task_scheduler_init::deferred);
91  int nThreads = semInitCnt + extraThreads;
92  vector<int> maxVals(nThreads);
93  vector<double> totTimes(nThreads);
94  int nIters = 10;
95  Body myBody(nThreads, nIters, my_sem, maxVals, totTimes);
96 
97  REMARK( " sem(%d) with %d extra threads\n", semInitCnt, extraThreads);
98  pCount = 0;
99  NativeParallelFor(nThreads, myBody);
100  if(extraThreads == 0) {
101  double allPWaits = 0;
102  for(vector<double>::const_iterator j = totTimes.begin(); j != totTimes.end(); ++j) {
103  allPWaits += *j;
104  }
105  allPWaits /= static_cast<double>(nThreads * nIters);
106  REMARK("Average wait for P() in uncontested case for nThreads = %d is %g\n", nThreads, allPWaits);
107  }
108  ASSERT(!pCount, "not all threads decremented pCount");
109  int maxCount = -1;
110  for(vector<int>::const_iterator i=maxVals.begin(); i!= maxVals.end();++i) {
111  maxCount = max(maxCount,*i);
112  }
113  ASSERT(maxCount <= semInitCnt,"too many threads in semaphore-protected increment");
114  if(maxCount < semInitCnt) {
115  REMARK("Not enough threads in semaphore-protected region (%d < %d)\n", static_cast<int>(maxCount), semInitCnt);
116  }
117 }

225 // send a bunch of non-Null "tokens" to consumer, then a NULL.
226 void FilterBase::Produce(const int /*tid*/) {
227  nextBuffer[0] = 0; // just in case we provide no tokens
228  sBarrier.wait();
229  while(totTokens) {
230  while(!myTokens)
231  mySem.P();
232  // we have a slot available.
233  --myTokens; // moving this down reduces spurious wakeups
234  --totTokens;
235  if(totTokens)
236  nextBuffer[curToken&(MAX_TOKENS-1)] = curToken*3+1;
237  else
238  nextBuffer[curToken&(MAX_TOKENS-1)] = 0;
239  ++curToken;
240  Harness::Sleep(myWait);
241  unsigned temp = ++otherTokens;
242  if(temp == 1)
243  nextSem.V();
244  }
245  nextSem.V(); // final wakeup
246 }

248 void FilterBase::Consume(const int /*tid*/) {
249  unsigned myToken;
250  sBarrier.wait();
251  do {
252  while(!myTokens)
253  mySem.P();
254  // we have a slot available.
255  --myTokens; // moving this down reduces spurious wakeups
256  myToken = myBuffer[curToken&(MAX_TOKENS-1)];
257  if(myToken) {
258  ASSERT(myToken == curToken*3+1, "Error in received token");
259  ++curToken;
260  Harness::Sleep(myWait);
261  unsigned temp = ++otherTokens;
262  if(temp == 1)
263  nextSem.V();
264  }
265  } while(myToken);
266  // end of processing
267  ASSERT(curToken + 1 == totTokens, "Didn't receive enough tokens");
268 }
