193 void check(boost::property_tree::ptree &Tree, const std::string &KeyType,
194  const std::string &KeyValue, int &Count, bool CheckValue,
195  double Value, bool CheckBounds, double Min, double Max,
196  bool CheckFix, bool Fix) {
197 
198  for (const auto &Node : Tree) {
199  // if this node is not parameter, recursively check this node
200  if (Node.first != "Parameter") {
201  auto Child = Node.second;
202  check(Child, KeyType, KeyValue, Count, CheckValue, Value, CheckBounds,
203  Min, Max, CheckFix, Fix);
204  continue;
205  }
206 
207  // if this is a parameter, but not the target one, continue
208  if (KeyValue != Node.second.get<std::string>("<xmlattr>." + KeyType)) {
209  continue;
210  }
211 
212  bool ValueOK(true), BoundsOK(true), FixOK(true);
213  // check the target parameter's properties and Count the right one
214  if (CheckValue && abs(Value - Node.second.get<double>("Value")) > 1e-6) {
215  ValueOK = false;
216  }
217  if (CheckBounds && (abs(Min - Node.second.get<double>("Min")) > 1e-6 ||
218  abs(Max - Node.second.get<double>("Max")) > 1e-6)) {
219  BoundsOK = false;
220  }
221  if (CheckFix && Node.second.get<bool>("Fix") != Fix) {
222  FixOK = false;
223  }
224 
225  if (ValueOK && BoundsOK && FixOK)
226  Count++;
227  }
228 }
