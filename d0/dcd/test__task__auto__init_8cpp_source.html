135 class FireAndForgetTask : public tbb::task {
136  tbb::task* execute () __TBB_override {
137  // Let another master thread proceed requesting new TLS keys
138  FafStarted = true;
139  UseAFewNewTlsKeys();
140  // Wait while another master thread dirtied its new TLS slots
141  spin_wait_until_eq( FafCanFinish, true );
142  FafCompleted = true;
143  return NULL;
144  }
145 public: // to make gcc 3.2.3 happy
146  ~FireAndForgetTask() {
147  ASSERT(FafCompleted, "FireAndForgetTask got erroneously cancelled?");
148  }

154 struct DriverThreadBody : NoAssign, Harness::NoAfterlife {
155  void operator() ( int id ) const {
156  ASSERT( id < 2, "Only two test driver threads are expected" );
157  // a barrier is required to ensure both threads started; otherwise the test may deadlock:
158  // the first thread would execute FireAndForgetTask at shutdown and wait for FafCanFinish,
159  // while the second thread wouldn't even start waiting for the loader lock hold by the first one.
160  if ( id == 0 ) {
161  driver_barrier.wait();
162  // Prepare global data
163  g_Root1 = new( tbb::task::allocate_root() ) tbb::empty_task;
164  g_Root2 = new( tbb::task::allocate_root() ) tbb::empty_task;
165  g_Root3 = new( tbb::task::allocate_root() ) tbb::empty_task;
166  g_Task = new( g_Root3->allocate_child() ) tbb::empty_task;
167  g_Root3->set_ref_count(2);
168  // Run tests
169  NativeParallelFor( NumTestFuncs, TestThreadBody() );
170  ASSERT( g_NumTestsExecuted == NumTestFuncs, "Test driver: Wrong number of tests executed" );
171 
172  // This test checks the validity of temporarily restoring the value of
173  // the last TLS slot for a given key during the termination of an
174  // auto-initialized master thread (in governor::auto_terminate).
175  // If anything goes wrong, generic_scheduler::cleanup_master() will assert.
176  // The context for this task must be valid till the task completion.
177 #if __TBB_TASK_GROUP_CONTEXT
178  tbb::task &r = *new( tbb::task::allocate_root(*g_Ctx) ) FireAndForgetTask;
179 #else
180  tbb::task &r = *new( tbb::task::allocate_root() ) FireAndForgetTask;
181 #endif /* __TBB_TASK_GROUP_CONTEXT */
182  tbb::task::spawn(r);
183  }
184  else {
185 #if __TBB_TASK_GROUP_CONTEXT
186  tbb::task_group_context ctx;
187  g_Ctx = &ctx;
188 #endif /* __TBB_TASK_GROUP_CONTEXT */
189  driver_barrier.wait();
190  spin_wait_until_eq( FafStarted, true );
191  UseAFewNewTlsKeys();
192  FafCanFinish = true;
193  spin_wait_until_eq( FafCompleted, true );
194  }
195  }
196 };
