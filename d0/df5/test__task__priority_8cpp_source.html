207 template<class NodeType>
208 class MasterBodyBase : NoAssign, Harness::NoAfterlife {
209 protected:
210  uintptr_t m_opts;
211 
212 public:
213  void RunTaskForest ( int id ) const {
214  ASSERT( id < g_NumMasters, NULL );
215  g_LeavesExecuted[id] = 0;
216  int d = BaseDepth + id;
217  tbb::task_scheduler_init init(P-1);
218  tbb::task_group_context ctx (tbb::task_group_context::isolated);
219  tbb::empty_task &r = *new( tbb::task::allocate_root(ctx) ) tbb::empty_task;
220  const int R = 4;
221  r.set_ref_count( R * P + 1 );
222  // Only PreemptionActivator thread changes its task tree priority in preemption test mode
223  const uintptr_t opts = (id == PreemptionActivatorId) ? m_opts : (m_opts & ~(uintptr_t)TestPreemption);
224  for ( int i = 0; i < R; ++i ) {
225  for ( int j = 1; j < P; ++j )
226  r.spawn( *new(r.allocate_child()) NodeType(id, MinBaseDepth + id, opts, &r) );
227  r.spawn( *new(r.allocate_child()) NodeType(id, d, opts, &r) );
228  }
229  int count = 1;
230  intptr_t lastExecuted = 0;
231  while ( r.ref_count() > 1 ) {
232  // Give workers time to make some progress.
233  for ( int i = 0; i < 10 * count; ++i )
234  __TBB_Yield();
235 #if __TBB_TASK_PRIORITY
236  if ( lastExecuted == g_LeavesExecuted[id] ) {
237  // No progress. Likely all workers left to higher priority arena,
238  // and then returned to RML. Request workers back from RML.
239  tbb::task::enqueue( *new(tbb::task::allocate_root() ) tbb::empty_task, id == 0 ? Low : High );
240  Harness::Sleep(count);
241 #if __TBB_ipf
242  // Increased sleep periods are required on systems with unfair hyperthreading (Itanium(R) 2 processor)
243  count += 10;
244 #endif
245  }
246  else {
247  count = 1;
248  lastExecuted = g_LeavesExecuted[id];
249  }
250 #else /* !__TBB_TASK_PRIORITY */
251  (void)lastExecuted;
252  tbb::task::enqueue( *new(tbb::task::allocate_root() ) tbb::empty_task );
253 #endif /* !__TBB_TASK_PRIORITY */
254  }
255  ASSERT( g_LeavesExecuted[id] == R * ((1 << d) + ((P - 1) * (1 << (MinBaseDepth + id)))), NULL );
256  g_LeavesExecuted[id] = -1;
257  tbb::task::destroy(r);
258  }
259 
260  MasterBodyBase ( uintptr_t opts ) : m_opts(opts) {}
261 };

528 void TestSetPriority() {
