24 // ForwardIterator is like type Iterator, but restricted to be a forward iterator.
25 // Only the forward iterator signatures that are necessary for tests are present.
26 // Post-increment in particular is deliberatly omitted since our templates should avoid using it
27 // because of efficiency considerations.

30 {
31  public:
32  typedef IteratorTag iterator_category;
33  typedef typename std::iterator_traits<Iterator>::value_type value_type;
34  typedef typename std::iterator_traits<Iterator>::difference_type difference_type;
35  typedef typename std::iterator_traits<Iterator>::pointer pointer;
36  typedef typename std::iterator_traits<Iterator>::reference reference;
37 
38  protected:
39  Iterator my_iterator;
40  typedef value_type element_type;
41 
42  public:
43  ForwardIterator() = default;
44  explicit ForwardIterator(Iterator i) : my_iterator(i) {}
45  reference operator*() const { return *my_iterator; }
46  Iterator operator->() const { return my_iterator; }
47  ForwardIterator
48  operator++()
49  {
50  ++my_iterator;
51  return *this;
52  }
53  ForwardIterator operator++(int32_t)
54  {
55  auto retval = *this;
56  my_iterator++;
57  return retval;
58  }
59  friend bool
60  operator==(const ForwardIterator& i, const ForwardIterator& j)
61  {
62  return i.my_iterator == j.my_iterator;

66  {

114 // Adapters for creating different types of iterators.
115 //
116 // In this block we implemented some adapters for creating differnet types of iterators.
117 // It's needed for extending the unit testing of Parallel STL algorithms.
118 // We have adapters for iterators with different tags (forward_iterator_tag, bidirectional_iterator_tag), reverse iterators.
119 // The input iterator should be const or non-const, non-reverse random access iterator.
120 // Iterator creates in "MakeIterator":
121 // firstly, iterator is "packed" by "IteratorTypeAdapter" (creating forward or bidirectional iterator)
122 // then iterator is "packed" by "ReverseAdapter" (if it's possible)
123 // So, from input iterator we may create, for example, reverse bidirectional iterator.
124 // "Main" functor for testing iterators is named "invoke_on_all_iterator_types".
