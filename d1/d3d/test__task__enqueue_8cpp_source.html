33 class EnqueuedTask : public tbb::task {
34  task* my_successor;
35  int my_enqueue_order;
36  int* my_track;
37  tbb::task* execute() __TBB_override {
38  // Capture execution order in the very beginning
39  int execution_order = 2 - my_successor->decrement_ref_count();
40  // Create some local work.
41  TaskGenerator& p = *new( allocate_root() ) TaskGenerator(2,2);
42  spawn_root_and_wait(p);
43  if( execution_order==2 ) { // the "slower" of two peer tasks
44  ++nCompletedPairs;
45  // Of course execution order can differ from dequeue order.
46  // But there is no better approximation at hand; and a single worker
47  // will execute in dequeue order, which is enough for our check.
48  if (my_enqueue_order==execution_order)
49  ++nOrderedPairs;
50  FireTwoTasks(my_track);
51  destroy(*my_successor);
52  }
53  return NULL;
54  }
55 public:
56  EnqueuedTask( task* successor, int enq_order, int* track )
57  : my_successor(successor), my_enqueue_order(enq_order), my_track(track) {}
58 
59  // Create and enqueue two tasks
60  static void FireTwoTasks( int* track ) {
61  int progress = ++*track;
62  if( progress < PairsPerTrack ) {
63  task* successor = new (allocate_root()) tbb::empty_task;
64  successor->set_ref_count(2);
65  enqueue( *new (allocate_root()) EnqueuedTask(successor, 1, track) );
66  enqueue( *new (allocate_root()) EnqueuedTask(successor, 2, track) );
67  }
68  }

71  static tbb::atomic<int> nOrderedPairs;

87 class ProgressMonitor {
88 public:
89  void operator() ( ) {
90  int track_snapshot[nTracks];
91  int stall_count = 0, uneven_progress_count = 0, last_progress_mask = 0;
92  for(int i=0; i<nTracks; ++i)
93  track_snapshot[i]=0;
94  bool completed;
95  do {
96  // Yield repeatedly for at least 1 usec
97  TimedYield( 1E-6 );
98  int overall_progress = 0, progress_mask = 0;
99  const int all_progressed = (1<<nTracks) - 1;
100  completed = true;
101  for(int i=0; i<nTracks; ++i) {
102  int ti = TaskTracks[i];
103  int pi = ti-track_snapshot[i];
104  if( pi ) progress_mask |= 1<<i;
105  overall_progress += pi;
106  completed = completed && ti==PairsPerTrack;
107  track_snapshot[i]=ti;
108  }
109  // The constants in the next asserts are subjective and may need correction.
110  if( overall_progress )
111  stall_count=0;
112  else {
113  ++stall_count;
114  // no progress; consider it dead.
115  ASSERT(stall_count < stall_threshold, "no progress on enqueued tasks; deadlock, or the machine is heavily oversubscribed?");
116  }
117  if( progress_mask==all_progressed || progress_mask^last_progress_mask ) {
118  uneven_progress_count = 0;
119  last_progress_mask = progress_mask;
120  }
121  else if ( overall_progress > 2 ) {
122  ++uneven_progress_count;
123  // The threshold of 32 is 4x bigger than what was observed on a 8-core machine with oversubscription.
124  ASSERT_WARNING(uneven_progress_count < 32,
125  "some enqueued tasks seem stalling; no simultaneous progress, or the machine is oversubscribed? Investigate if repeated");
126  }
127  } while( !completed );
128  }
129 };

132  REMARK("Testing task::enqueue for %d threads\n", p);

137  TaskTracks[i] = -1; // to accommodate for the starting call
138  EnqueuedTask::FireTwoTasks(TaskTracks+i);
139  }
140  ProgressMonitor pm;
141  tbb::tbb_thread thr( pm );
142  if(mode==1) {
143  // do some parallel work in the meantime
144  for(int i=0; i<10; i++) {
145  TaskGenerator& g = *new( tbb::task::allocate_root() ) TaskGenerator(2,5);
146  tbb::task::spawn_root_and_wait(g);
147  TimedYield( 1E-6 );
148  }
149  }
150  if( mode==2 ) {
151  // Additionally enqueue a bunch of empty tasks. The goal is to test that tasks
152  // allocated and enqueued by a thread are safe to use after the thread leaves TBB.
153  tbb::task* root = new (tbb::task::allocate_root()) tbb::empty_task;
154  root->set_ref_count(100);
155  for( int i=0; i<100; ++i )
156  tbb::task::enqueue( *new (root->allocate_child()) tbb::empty_task );
157  init.terminate(); // master thread deregistered
158  }
159  thr.join();
160  ASSERT(EnqueuedTask::nCompletedPairs==nTracks*PairsPerTrack, NULL);
161  ASSERT(EnqueuedTask::nOrderedPairs<EnqueuedTask::nCompletedPairs,
162  "all task pairs executed in enqueue order; de facto guarantee is too strong?");
163  }
164 }
