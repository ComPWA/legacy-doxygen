479 //
480 // Check connection: function_node<continue_msg, SENDING_TYPE> <-> function_node<RECEIVING_TYPE>
481 // R E C E I V I N G T Y P E
482 // S 'bAllowed' | int | float | A | async_msg | async_msg | async_msg | UserAsync | UserAsync | UserAsync |
483 // E value | | | | <int> | <float> | <A> | _int | _float | _A |
484 // N -------------------------------------------------------------------------------------------------------------
485 // D int | Y | | | Y | | | Y | | |
486 // I float | | Y | | | Y | | | Y | |
487 // N A | | | Y | | | Y | | | Y |
488 // G async_msg<int> | Y | | | Y | | | | | |
489 // async_msg<float> | | Y | | | Y | | | | |
490 // T async_msg<A> | | | Y | | | Y | | | |
491 // Y UserAsync_int | Y | | | | | | Y | | |
492 // P UserAsync_float | | Y | | | | | | Y | |
493 // E UserAsync_A | | | Y | | | | | | Y |
494 //
495  // Test make_edge() & remove_edge() availability
496  static const bool bAllowed = is_same_type<T1, T2>::value
497  || is_same_type<typename async_helpers<T1>::filtered_type, T2>::value
498  || is_same_type<T1, typename async_helpers<T2>::filtered_type>::value;
499 
500  static const bool bConvertable = bAllowed
501  || std::is_base_of<T1, T2>::value
502  || (is_same_type<typename async_helpers<T1>::filtered_type, int>::value && is_same_type<T2, float>::value)
503  || (is_same_type<typename async_helpers<T1>::filtered_type, float>::value && is_same_type<T2, int>::value);
504 
505  __TBB_STATIC_ASSERT( (bAllowed == CheckerMakeEdge<function_node<continue_msg, T1>, function_node<T2> >::value), "invalid connection Fn<T1> -> Fn<T2>" );
506  __TBB_STATIC_ASSERT( (bAllowed == CheckerMakeEdge<queue_node<T1>, function_node<T2> >::value), "invalid connection Queue<T1> -> Fn<T2>" );
507 
508  // Test make_edge() & remove_edge() availability with output_port<N>(node&)
509  __TBB_STATIC_ASSERT( (bAllowed == CheckerMakeEdge<typename strip< decltype(
510  output_port<0>( *static_cast<multifunction_node< continue_msg, tuple<T1, int> >*>(0) ) ) >::type,
511  function_node<T2> >::value), "invalid connection MultuFn<0><T1,int> -> Fn<T2>" );
512 
513  __TBB_STATIC_ASSERT( (bAllowed == CheckerMakeEdge<typename strip< decltype(
514  output_port<1>( *static_cast<multifunction_node< continue_msg, tuple<int, T1> >*>(0) ) ) >::type,
515  function_node<T2> >::value), "invalid connection MultuFn<1><int, T1> -> Fn<T2>" );
516 
517  // Test untyped_sender connections
