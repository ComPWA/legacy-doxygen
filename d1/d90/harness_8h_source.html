150 void print_call_stack() {
151  Harness::internal::print_stack_mutex.lock();
152  fflush(stdout); fflush(stderr);
153  #if BACKTRACE_FUNCTION_AVAILABLE
154  const int sz = 100; // max number of frames to capture
155  void *buff[sz];
156  int n = backtrace(buff, sz);
157  REPORT("Call stack info (%d):\n", n);
158  backtrace_symbols_fd(buff, n, fileno(stdout));
159  #elif __SUNPRO_CC
160  REPORT("Call stack info:\n");
161  printstack(fileno(stdout));
162  #elif _WIN32_WINNT > 0x0501 && _MSC_VER>=1500 && !__TBB_WIN8UI_SUPPORT
163  const int sz = 62; // XP limitation for number of frames
164  void *buff[sz];
165  int n = CaptureStackBackTrace(0, sz, buff, NULL);
166  REPORT("Call stack info (%d):\n", n);
167  static LONG once = 0;
168  if( !InterlockedExchange(&once, 1) )
169  SymInitialize(GetCurrentProcess(), NULL, TRUE);
170  const int len = 255; // just some reasonable string buffer size
171  union { SYMBOL_INFO sym; char pad[sizeof(SYMBOL_INFO)+len]; };
172  sym.MaxNameLen = len;
173  sym.SizeOfStruct = sizeof( SYMBOL_INFO );
174  DWORD64 offset;
175  for(int i = 1; i < n; i++) { // skip current frame
176  if(!SymFromAddr( GetCurrentProcess(), DWORD64(buff[i]), &offset, &sym )) {
177  sym.Address = ULONG64(buff[i]); offset = 0; sym.Name[0] = 0;
178  }
179  REPORT("[%d] %016I64X+%04I64X: %s\n", i, sym.Address, offset, sym.Name); //TODO: print module name
180  }
181  #endif /*BACKTRACE_FUNCTION_AVAILABLE*/
182  Harness::internal::print_stack_mutex.unlock();
183 }

313 inline void ParseCommandLine( int argc, char* argv[] ) {
314  if( !argc ) REPORT("Command line with 0 arguments\n");
315  int i = 1;
316  if( i<argc ) {
317  if( strncmp( argv[i], "-v", 2 )==0 ) {
318  Verbose = true;
319  ++i;
320  }
321  }
322  if( i<argc ) {
323  char* endptr;
324  MinThread = strtol( argv[i], &endptr, 0 );
325  if( *endptr==':' )
326  MaxThread = strtol( endptr+1, &endptr, 0 );
327  else if( *endptr=='\0' )
328  MaxThread = MinThread;
329  if( *endptr!='\0' ) {
330  REPORT_FATAL_ERROR("garbled nthread range\n");
331  exit(1);
332  }
333  if( MinThread<0 ) {
334  REPORT_FATAL_ERROR("nthread must be nonnegative\n");
335  exit(1);
336  }
337  if( MaxThread<MinThread ) {
338  REPORT_FATAL_ERROR("nthread range is backwards\n");
339  exit(1);
340  }
341  ++i;
342  }
343 #if __TBB_STDARGS_BROKEN
344  if ( !argc )
345  argc = 1;
346  else {
347  while ( i < argc && argv[i][0] == 0 )
348  ++i;
349  }
350 #endif /* __TBB_STDARGS_BROKEN */
351  if( i!=argc ) {
352  REPORT_FATAL_ERROR("Usage: %s [-v] [nthread|minthread:maxthread]\n", argv[0] );
353  exit(1);
354  }
355 }

518  void start() {
519 #if _WIN32||_WIN64
520  unsigned thread_id;
521 #if __TBB_WIN8UI_SUPPORT
522  std::thread* thread_tmp=new std::thread(thread_function, this);
523  thread_handle = thread_tmp->native_handle();
524  thread_id = 0;
525 #else
526  unsigned stack_size = 0;
527 #if HARNESS_THREAD_STACK_SIZE
528  stack_size = HARNESS_THREAD_STACK_SIZE;
529 #endif
530  thread_handle = (HANDLE)_beginthreadex( NULL, stack_size, thread_function, this, 0, &thread_id );
531 #endif
532  ASSERT( thread_handle!=0, "NativeParallelFor: _beginthreadex failed" );
533 #else
534 #if __ICC==1100
535  #pragma warning (push)
536  #pragma warning (disable: 2193)
537 #endif /* __ICC==1100 */
538  // Some machines may have very large hard stack limit. When the test is
539  // launched by make, the default stack size is set to the hard limit, and
540  // calls to pthread_create fail with out-of-memory error.
541  // Therefore we set the stack size explicitly (as for TBB worker threads).
542 #if !defined(HARNESS_THREAD_STACK_SIZE)
543 #if __i386__||__i386||__arm__
544  const size_t stack_size = 1*MByte;
545 #elif __x86_64__
546  const size_t stack_size = 2*MByte;
547 #else
548  const size_t stack_size = 4*MByte;
549 #endif
550 #else
551  const size_t stack_size = HARNESS_THREAD_STACK_SIZE;
552 #endif /* HARNESS_THREAD_STACK_SIZE */
553  pthread_attr_t attr_stack;
554  int status = pthread_attr_init(&attr_stack);
555  ASSERT(0==status, "NativeParallelFor: pthread_attr_init failed");
556  status = pthread_attr_setstacksize( &attr_stack, stack_size );
557  ASSERT(0==status, "NativeParallelFor: pthread_attr_setstacksize failed");
558  status = pthread_create(&thread_id, &attr_stack, thread_function, this);
559  ASSERT(0==status, "NativeParallelFor: pthread_create failed");
560  pthread_attr_destroy(&attr_stack);
561 #if __ICC==1100
562  #pragma warning (pop)
563 #endif
564 #endif /* _WIN32||_WIN64 */
565  }

615 template<typename Index, typename Body>
616 void NativeParallelFor( Index n, const Body& body ) {
617  typedef NativeParallelForTask<Index,Body> task;
618 
619  if( n>0 ) {
620  // Allocate array to hold the tasks
621  task* array = static_cast<task*>(operator new( n*sizeof(task) ));
622 
623  // Construct the tasks
624  for( Index i=0; i!=n; ++i )
625  new( &array[i] ) task(i,body);
626 
627  // Start the tasks
628  for( Index i=0; i!=n; ++i )
629  array[i].start();
630 
631  // Wait for the tasks to finish and destroy each one.
632  for( Index i=n; i; --i ) {
633  array[i-1].wait_to_finish();
634  array[i-1].~task();
635  }
636 
637  // Deallocate the task array
638  operator delete(array);
639  }
640 }
