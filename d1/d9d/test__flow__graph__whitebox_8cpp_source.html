80 template< typename B >
81 void TestBufferingNode(const char * name) {
82  tbb::flow::graph g;
83  B bnode(g);
84  tbb::flow::function_node<int,int,tbb::flow::rejecting> fnode(g, tbb::flow::serial, serial_fn_body<int>(serial_fn_state0));
85  REMARK("Testing %s:", name);
86  for(int icnt = 0; icnt < 2; icnt++) {
87  bool reverse_edge = (icnt & 0x2) != 0;
88  serial_fn_state0 = 0; // reset to waiting state.
89  REMARK(" make_edge");
90  tbb::flow::make_edge(bnode, fnode);
91  ASSERT(!bnode.my_successors.empty(), "buffering node has no successor after make_edge");
92  REMARK(" try_put");
93  bnode.try_put(1); // will forward to the fnode
94  BACKOFF_WAIT(serial_fn_state0 == 0, "Timed out waiting for first put");
95  if(reverse_edge) {
96  REMARK(" try_put2");
97  bnode.try_put(2); // will reverse the edge
98  // cannot do a wait_for_all here; the function_node is still executing
99  BACKOFF_WAIT(!bnode.my_successors.empty(), "Timed out waiting after 2nd put");
100  // at this point the only task running is the one for the function_node.
101  ASSERT(bnode.my_successors.empty(), "successor not removed");
102  }
103  else {
104  ASSERT(!bnode.my_successors.empty(), "buffering node has no successor after forwarding message");
105  }
106  serial_fn_state0 = 0; // release the function_node.
107  if(reverse_edge) {
108  // have to do a second release because the function_node will get the 2nd item
109  BACKOFF_WAIT( serial_fn_state0 == 0, "Timed out waiting after 2nd put");
110  serial_fn_state0 = 0; // release the function_node.
111  }
112  g.wait_for_all();
113  REMARK(" remove_edge");
114  tbb::flow::remove_edge(bnode, fnode);
115  ASSERT(bnode.my_successors.empty(), "buffering node has a successor after remove_edge");
116  }
117  tbb::flow::join_node<tbb::flow::tuple<int,int>,tbb::flow::reserving> jnode(g);
118  tbb::flow::make_edge(bnode, tbb::flow::input_port<0>(jnode)); // will spawn a task
119  g.wait_for_all();
120  ASSERT(!bnode.my_successors.empty(), "buffering node has no successor after attaching to join");
121  REMARK(" reverse");
122  bnode.try_put(1); // the edge should reverse
123  g.wait_for_all();
124  ASSERT(bnode.my_successors.empty(), "buffering node has a successor after reserving");
125  REMARK(" reset()");
126  g.wait_for_all();
127  g.reset(); // should be in forward direction again
128  ASSERT(!bnode.my_successors.empty(), "buffering node has no successor after reset()");
129  REMARK(" remove_edge");
130  g.reset(tbb::flow::rf_clear_edges);
131  ASSERT(bnode.my_successors.empty(), "buffering node has a successor after reset(rf_clear_edges)");
132  tbb::flow::make_edge(bnode, tbb::flow::input_port<0>(jnode)); // add edge again
133  // reverse edge by adding to buffer.
134  bnode.try_put(1); // the edge should reverse
135  g.wait_for_all();
136  ASSERT(bnode.my_successors.empty(), "buffering node has a successor after reserving");
137  REMARK(" remove_edge(reversed)");
138  g.reset(tbb::flow::rf_clear_edges);
139  ASSERT(bnode.my_successors.empty(), "buffering node has no successor after reset()");
140  ASSERT(tbb::flow::input_port<0>(jnode).my_predecessors.empty(), "predecessor not reset");
141  REMARK(" done\n");
142  g.wait_for_all();
143 }

148 void TestContinueNode() {
149  tbb::flow::graph g;
150  tbb::flow::function_node<int> fnode0(g, tbb::flow::serial, serial_fn_body<int>(serial_fn_state0));
151  tbb::flow::continue_node<int> cnode(g, 1, serial_continue_body<int>(serial_continue_state0));
152  tbb::flow::function_node<int> fnode1(g, tbb::flow::serial, serial_fn_body<int>(serial_fn_state1));
153  tbb::flow::make_edge(fnode0, cnode);
154  tbb::flow::make_edge(cnode, fnode1);
155  REMARK("Testing continue_node:");
156  for( int icnt = 0; icnt < 2; ++icnt ) {
157  REMARK( " initial%d", icnt);
158  ASSERT(cnode.my_predecessor_count == 2, "predecessor addition didn't increment count");
159  ASSERT(!cnode.successors().empty(), "successors empty though we added one");
160  ASSERT(cnode.my_current_count == 0, "state of continue_receiver incorrect");
161  serial_continue_state0 = 0;
162  serial_fn_state0 = 0;
163  serial_fn_state1 = 0;
164 
165  fnode0.try_put(1); // start the first function node.
166  BACKOFF_WAIT(!serial_fn_state0, "Timed out waiting for function_node to start");
167  // Now the body of function_node 0 is executing.
168  serial_fn_state0 = 0; // release the node
169  // wait for node to count the message (or for the node body to execute, which would be wrong)
170  BACKOFF_WAIT(serial_continue_state0 == 0 && cnode.my_current_count == 0, "Timed out waiting for continue_state0 to change");
171  ASSERT(serial_continue_state0 == 0, "Improperly released continue_node");
172  ASSERT(cnode.my_current_count == 1, "state of continue_receiver incorrect");
173  if(icnt == 0) { // first time through, let the continue_node fire
174  REMARK(" firing");
175  fnode0.try_put(1); // second message
176  BACKOFF_WAIT(serial_fn_state0 == 0, "timeout waiting for continue_body to execute");
177  // Now the body of function_node 0 is executing.
178  serial_fn_state0 = 0; // release the node
179 
180  BACKOFF_WAIT(!serial_continue_state0,"continue_node didn't start"); // now we wait for the continue_node.
181  ASSERT(cnode.my_current_count == 0, " my_current_count not reset before body of continue_node started");
182  serial_continue_state0 = 0; // release the continue_node
183  BACKOFF_WAIT(!serial_fn_state1,"successor function_node didn't start"); // wait for the successor function_node to enter body
184  serial_fn_state1 = 0; // release successor function_node.
185  g.wait_for_all();
186 
187  // try a try_get()
188  {
189  int i;
190  ASSERT(!cnode.try_get(i), "try_get not rejected");
191  }
192 
193  REMARK(" reset");
194  ASSERT(!cnode.my_successors.empty(), "Empty successors in built graph (before reset)");
195  ASSERT(cnode.my_predecessor_count == 2, "predecessor_count reset (before reset)");
196  g.reset(); // should still be the same
197  ASSERT(!cnode.my_successors.empty(), "Empty successors in built graph (after reset)" );
198  ASSERT(cnode.my_predecessor_count == 2, "predecessor_count reset (after reset)");
199  }
200  else { // we're going to see if the rf_clear_edges resets things.
201  g.wait_for_all();
202  REMARK(" reset(rf_clear_edges)");
203  ASSERT(!cnode.my_successors.empty(), "Empty successors in built graph (before reset)");
204  ASSERT(cnode.my_predecessor_count == 2, "predecessor_count reset (before reset)");
205  g.reset(tbb::flow::rf_clear_edges); // should be in forward direction again
206  ASSERT(cnode.my_current_count == 0, "state of continue_receiver incorrect after reset(rf_clear_edges)");
207  ASSERT(cnode.my_successors.empty(), "buffering node has a successor after reset(rf_clear_edges)");
208  ASSERT(cnode.my_predecessor_count == cnode.my_initial_predecessor_count, "predecessor count not reset");
209  }
210  }
211 
212  REMARK(" done\n");
213

228 
229  tbb::flow::make_edge(fnode0, qnode1);
230  tbb::flow::make_edge(qnode0, fnode0);
231 
232  serial_fn_state0 = 2; // just let it go
233  // see if the darned thing will work....
234  qnode0.try_put(1);

236  int ii;
237  ASSERT(qnode1.try_get(ii) && ii == 1, "output not passed");

240 
241  tbb::flow::make_edge(fnode1, qnode1);
242  tbb::flow::make_edge(qnode0, fnode1);
243 
244  serial_fn_state0 = 2; // just let it go
245  // see if the darned thing will work....
246  qnode0.try_put(1);
247  g.wait_for_all();
248  ASSERT(qnode1.try_get(ii) && ii == 1, "output not passed");
249  tbb::flow::remove_edge(qnode0, fnode1);

251 
252  // rejecting
253  serial_fn_state0 = 0;
254  tbb::flow::make_edge(fnode0, qnode1);
255  tbb::flow::make_edge(qnode0, fnode0);
256  REMARK("Testing rejecting function_node:");

267  ASSERT(!qnode0.my_successors.empty(), "empty successors after reset()");

271  REMARK("\n");

284  ASSERT(!qnode0.my_successors.empty(), "empty successors after reset()");
285  ASSERT(fnode1.my_predecessors.empty(), "predecessor not reversed");
286  tbb::flow::remove_edge(qnode0, fnode1);
287  tbb::flow::remove_edge(fnode1, qnode1);
288  REMARK("\n");
289

291  tbb::flow::make_edge(qnode0, fnode0);
292  REMARK(" start_func");
293  qnode0.try_put(1);
294  BACKOFF_WAIT(serial_fn_state0 == 0, "Timed out waiting after 1st put");
295  // now if we put an item to the queues the edges to the function_node will reverse.
296  REMARK(" put_node(2)");
297  qnode0.try_put(2); // start queue node.
298  // wait for the edges to reverse
299  BACKOFF_WAIT(fnode0.my_predecessors.empty(), "Timed out waiting");
300  ASSERT(!fnode0.my_predecessors.empty(), "function_node edge not reversed");

306  g.reset(tbb::flow::rf_clear_edges);
307  ASSERT(fnode0.my_predecessors.empty() && qnode0.my_successors.empty(), "function_node edge not removed");
308  ASSERT(fnode0.my_successors.empty(), "successor to fnode not removed");
309  REMARK(" done\n");
310 }
