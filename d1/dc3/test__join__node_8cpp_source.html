58  tbb::flow::graph g;

61  // source_node for continue messages

64  input_join_type * my_input_join = makeJoin<2, input_join_type, tbb::flow::reserving>::create(g);

66  tbb::flow::make_edge(snode, tbb::flow::get<1>(my_input_join->input_ports()));
67  // queue to hold the tags
68  tbb::flow::queue_node<int> tag_queue(g);
69  tbb::flow::make_edge(tag_queue, tbb::flow::input_port<0>(*my_input_join));
70  // add all the function_nodes that are inputs to the tag-matching join
71  source_node_helper<N, JType>::add_recirc_func_nodes(*my_join, *my_input_join, g);
72  // add the function_node that accepts the output of the join and emits the int tag it was based on

75  // now the recirculating part (output back to the queue)
76  tbb::flow::make_edge(recreate_tag, tag_queue);
77 
78  // put the tags into the queue
79  for(int t = 1; t<=maxTag; ++t) tag_queue.try_put(t);
80 
81  input_count = Recirc_count;

83

85  snode.activate();
86 
87  // wait for everything to stop

89

91

94  std::vector<bool> out_tally(maxTag, false);
95  for(int i = 0; i < maxTag; ++i) {
96  ASSERT(tag_queue.try_get(j), "not enough tags in queue");
