318  static void uniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
319  const RangeStatisticData& stat)
320  {
321  // Checking that all threads were given a task
322  if (rangeSize >= settings->thread_num) {
323  uint64_t disparity =
324  max(stat.m_rangeNum, settings->thread_num) - min(stat.m_rangeNum, settings->thread_num);
325  if (disparity > settings->above_threads_size_tolerance) {
326  REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges(%llu)-#threads(%llu)|=%llu > %llu=tolerance\n",
327  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size), stat.m_rangeNum,
328  settings->thread_num, disparity, uint64_t(settings->above_threads_size_tolerance));
329  ASSERT(disparity <= settings->above_threads_size_tolerance, "Incorrect number of range "
330  "objects was created before work balancing phase started");
331  }
332  } else if (settings->ensure_non_empty_size && rangeSize != 0) {
333  uint64_t disparity = max(stat.m_rangeNum, rangeSize) - min(stat.m_rangeNum, rangeSize);
334  if (disparity > settings->below_threads_size_tolerance ) {
335  REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges-range size|=%llu > %llu=tolerance\n",
336  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
337  disparity, uint64_t(settings->below_threads_size_tolerance));
338  ASSERT(disparity <= settings->below_threads_size_tolerance, "Incorrect number of range objects"
339  " was created before work balancing phase started");
340  }
341  }
342  // Checking difference between min and max number of range iterations
343  size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
344  if (diff > settings->between_min_max_ranges_tolerance) {
345  REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=tolerance\n",
346  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
347  uint64_t(diff), uint64_t(settings->between_min_max_ranges_tolerance));
348  ASSERT(diff <= settings->between_min_max_ranges_tolerance, "Uniform iteration distribution error");
349  }
350  }
351  // Checker for test cases where ranges don't provide feedback during proportional split to
352  // partitioner and differ from tbb::blocked_range implementation in their splitting algorithm
353  static void nonuniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
354  const RangeStatisticData& stat)
355  {
356  if (stat.m_rangeNum > settings->thread_num) {
357  REPORT("ERROR: '%s (f=%d|e=%d)': %llu=#ranges > #threads=%llu\n",
358  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
359  uint64_t(stat.m_rangeNum), uint64_t(settings->thread_num));
360  ASSERT(stat.m_rangeNum <= settings->thread_num,
361  "Incorrect number of range objects was created before work balancing phase started");
362  }
363  // Checking difference between min and max number of range iterations
364  size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
365  if (diff > rangeSize) {
366  REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=initial range size\n",
367  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
368  uint64_t(diff), uint64_t(rangeSize));
369  ASSERT(diff <= rangeSize, "Iteration distribution error");
370  }
371  }

376  template <typename Range, typename Partitioner, typename T>
377  void test(use_case_settings_t& settings, T factors[], size_t (*rsgFunc)(T*, unsigned, size_t)
378  = &default_range_size_generator<T>) const
379  {
380  for (unsigned i = 0; i < settings.factors_array_len; ++i) {
381  size_t range_end = rsgFunc(factors, i, settings.thread_num);
382  RangeStatisticData stat = { /*range num=*/ 0, /*minimal size of range=*/ 0,
383  /*maximal size of range=*/ 0, /*minimal size of range was not rewritten yet=*/ false };
384  Range range = Range(settings.range_begin, range_end, &stat, settings.provide_feedback,
385  settings.ensure_non_empty_size);
386  Partitioner my_partitioner;
387  test_case(range, SimpleBody(), my_partitioner, NULL);
388  size_t range_size = range_end - settings.range_begin;
389  const char* rangeName = typeid(range).name();
390  settings.checker(rangeName, range_size, &settings, stat);
391  }
