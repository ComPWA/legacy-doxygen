108 template <typename In, typename Out, typename UnaryOp, typename BinaryOp>
109 void
110 test(UnaryOp unary_op, Out init, BinaryOp binary_op, Out trash)
111 {
112  for (size_t n = 0; n <= 100000; n = n <= 16 ? n + 1 : size_t(3.1415 * n))
113  {
114  Sequence<In> in(n, [](size_t k) { return In(k ^ encryption_mask); });
115 
116  Out tmp = init;
117  Sequence<Out> expected(n, [&](size_t k) -> Out {
118  if (inclusive)
119  {
120  tmp = binary_op(tmp, unary_op(in[k]));
121  return tmp;
122  }
123  else
124  {
125  Out val = tmp;
126  tmp = binary_op(tmp, unary_op(in[k]));
127  return val;
128  }
129  });
130 
131  Sequence<Out> out(n, [&](size_t) { return trash; });
132 
133  auto result =
134  inclusive
135  ? transform_inclusive_scan_serial(in.cbegin(), in.cend(), out.fbegin(), unary_op, init, binary_op)
136  : transform_exclusive_scan_serial(in.cbegin(), in.cend(), out.fbegin(), unary_op, init, binary_op);
137  check_and_reset(expected.begin(), out.begin(), out.size(), trash);
138 
139  invoke_on_all_policies(test_transform_scan(), in.begin(), in.end(), out.begin(), out.end(), expected.begin(),
140  expected.end(), in.size(), unary_op, init, binary_op, trash);
141  invoke_on_all_policies(test_transform_scan(), in.cbegin(), in.cend(), out.begin(), out.end(), expected.begin(),
142  expected.end(), in.size(), unary_op, init, binary_op, trash);
143  }
