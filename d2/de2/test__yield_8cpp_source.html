35 struct RoundRobin: NoAssign {
36  const int number_of_threads;
37  RoundRobin( long p ) : number_of_threads(p) {}
38  void operator()( long k ) const {
39  tbb::tick_count t0 = tbb::tick_count::now();
40  for( long i=0; i<10000; ++i ) {
41  // Wait for previous thread to notify us
42  for( int j=0; CyclicCounter!=k && !Quit; ++j ) {
43  __TBB_Yield();
44  if( j%100==0 ) {
45  tbb::tick_count t1 = tbb::tick_count::now();
46  if( (t1-t0).seconds()>=1.0*number_of_threads ) {
47  REPORT("Warning: __TBB_Yield failing to yield with %d threads (or system is heavily loaded)\n",number_of_threads);
48  Quit = true;
49  return;
50  }
51  }
52  }
53  // Notify next thread that it can run
54  CyclicCounter = (k+1)%number_of_threads;
55  }
56  }
57 };
