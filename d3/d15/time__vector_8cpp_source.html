51 class Timer {

53 public:
54  Timer() { tick = tbb::tick_count::now(); }
55  double get_time() { return (tbb::tick_count::now() - tick).seconds(); }
56  double diff_time(const Timer &newer) { return (newer.tick - tick).seconds(); }
57  double mark_time() { tick_count t1(tbb::tick_count::now()), t2(tick); tick = t1; return (t1 - t2).seconds(); }
58  double mark_time(const Timer &newer) { tick_count t(tick); tick = newer.tick; return (tick - t).seconds(); }
