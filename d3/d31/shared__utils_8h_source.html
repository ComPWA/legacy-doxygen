94 template <int BUF_LINE_SIZE, int N>
95 void parseFile(const char* file, const parseFileItem (&items)[N]) {
96  // Tries to find all items in each line
97  int found[N] = { 0 };
98  // If all items found, stop forward file reading
99  int numFound = 0;
100  // Line storage
101  char buf[BUF_LINE_SIZE];
102 
103  if (FILE *f = fopen(file, "r")) {
104  while (numFound < N && fgets(buf, BUF_LINE_SIZE, f)) {
105  for (int i = 0; i < N; ++i) {
106  if (!found[i] && 1 == sscanf(buf, items[i].format, &items[i].value)) {
107  ++numFound;
108  found[i] = 1;
109  }
110  }
111  }
112  fclose(f);
113  }
114 }
