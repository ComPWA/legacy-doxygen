29 /* Each slab block and each large memory object header contains BackRefIdx
30  * that points out in some BackRefBlock which points back to this block or header.
31  */
32 struct BackRefBlock : public BlockI {
33  BackRefBlock *nextForUse; // the next in the chain of blocks with free items
34  FreeObject *bumpPtr; // bump pointer moves from the end to the beginning of the block
35  FreeObject *freeList;
36  // list of all blocks that were allocated from raw mem (i.e., not from backend)
37  BackRefBlock *nextRawMemBlock;
38  int allocatedCount; // the number of objects allocated
39  BackRefIdx::master_t myNum; // the index in the master
40  MallocMutex blockMutex;
41  // true if this block has been added to the listForUse chain,
42  // modifications protected by masterMutex
43  bool addedToForUse;
44 
45  BackRefBlock(const BackRefBlock *blockToUse, intptr_t num) :
46  nextForUse(NULL), bumpPtr((FreeObject*)((uintptr_t)blockToUse + slabSize - sizeof(void*))),
47  freeList(NULL), nextRawMemBlock(NULL), allocatedCount(0), myNum(num),
48  addedToForUse(false) {
49  memset(&blockMutex, 0, sizeof(MallocMutex));
50 
51  MALLOC_ASSERT(!(num >> CHAR_BIT*sizeof(BackRefIdx::master_t)),
52  "index in BackRefMaster must fit to BackRefIdx::master");
53  }
54  // clean all but header
55  void zeroSet() { memset(this+1, 0, BackRefBlock::bytes-sizeof(BackRefBlock)); }
56  static const int bytes = slabSize;
57 };

62 struct BackRefMaster {
63 /* On 64-bit systems a slab block can hold up to ~2K back pointers to slab blocks
64  * or large objects, so it can address at least 32MB. The master array of 256KB
65  * holds 32K pointers to such blocks, addressing ~1 TB.
66  * On 32-bit systems there is ~4K back pointers in a slab block, so ~64MB can be addressed.
67  * The master array of 8KB holds 2K pointers to leaves, so ~128 GB can addressed.
68  */
69  static const size_t bytes = sizeof(uintptr_t)>4? 256*1024 : 8*1024;
70  static const int dataSz;
71 /* space is reserved for master table and 4 leaves
72  taking into account VirtualAlloc allocation granularity */
73  static const int leaves = 4;
74  static const size_t masterSize = BackRefMaster::bytes+leaves*BackRefBlock::bytes;
75  // The size of memory request for a few more leaf blocks;
76  // selected to match VirtualAlloc granularity
77  static const size_t blockSpaceSize = 64*1024;
78 
79  Backend *backend;
80  BackRefBlock *active; // if defined, use it for allocations
81  BackRefBlock *listForUse; // the chain of data blocks with free items
82  BackRefBlock *allRawMemBlocks;
83  intptr_t lastUsed; // index of the last used block
84  bool rawMemUsed;
85  MallocMutex requestNewSpaceMutex;
86  BackRefBlock *backRefBl[1]; // the real size of the array is dataSz
87 
88  BackRefBlock *findFreeBlock();
89  void addToForUseList(BackRefBlock *bl);
90  void initEmptyBackRefBlock(BackRefBlock *newBl);
91  bool requestNewSpace();
92 };

100 bool initBackRefMaster(Backend *backend)
101 {
102  bool rawMemUsed;
103  BackRefMaster *master =
104  (BackRefMaster*)backend->getBackRefSpace(BackRefMaster::masterSize,
105  &rawMemUsed);
106  if (! master)
107  return false;
108  master->backend = backend;
109  master->listForUse = master->allRawMemBlocks = NULL;
110  master->rawMemUsed = rawMemUsed;
111  master->lastUsed = -1;
112  memset(&master->requestNewSpaceMutex, 0, sizeof(MallocMutex));
113  for (int i=0; i<BackRefMaster::leaves; i++) {
114  BackRefBlock *bl = (BackRefBlock*)((uintptr_t)master + BackRefMaster::bytes + i*BackRefBlock::bytes);
115  bl->zeroSet();
116  master->initEmptyBackRefBlock(bl);
117  if (i)
118  master->addToForUseList(bl);
119  else // active leaf is not needed in listForUse
120  master->active = bl;
121  }
122  // backRefMaster is read in getBackRef, so publish it in consistent state
123  FencedStore((intptr_t&)backRefMaster, (intptr_t)master);
124  return true;
125 }

127 void destroyBackRefMaster(Backend *backend)
128 {
129  if (backRefMaster) { // Is initBackRefMaster() called?
130  for (BackRefBlock *curr=backRefMaster->allRawMemBlocks; curr; ) {
131  BackRefBlock *next = curr->nextRawMemBlock;
132  // allRawMemBlocks list is only for raw mem blocks
133  backend->putBackRefSpace(curr, BackRefMaster::blockSpaceSize,
134  /*rawMemUsed=*/true);
135  curr = next;
136  }
137  backend->putBackRefSpace(backRefMaster, BackRefMaster::masterSize,
138  backRefMaster->rawMemUsed);
139  }

149 void BackRefMaster::initEmptyBackRefBlock(BackRefBlock *newBl)
150 {
151  intptr_t nextLU = lastUsed+1;
152  new (newBl) BackRefBlock(newBl, nextLU);
153  MALLOC_ASSERT(nextLU < dataSz, NULL);
154  backRefBl[nextLU] = newBl;
155  // lastUsed is read in getBackRef, and access to backRefBl[lastUsed]
156  // is possible only after checking backref against current lastUsed
157  FencedStore(lastUsed, nextLU);

160 bool BackRefMaster::requestNewSpace()
161 {
162  bool isRawMemUsed;
163  MALLOC_STATIC_ASSERT(!(blockSpaceSize % BackRefBlock::bytes),
164  "Must request space for whole number of blocks.");
165 
166  if (backRefMaster->dataSz <= lastUsed + 1) // no space in master
167  return false;
168 
169  // only one thread at a time may add blocks
170  MallocMutex::scoped_lock newSpaceLock(requestNewSpaceMutex);
171 
172  if (listForUse) // double check that only one block is available
173  return true;
174  BackRefBlock *newBl =
175  (BackRefBlock*)backend->getBackRefSpace(blockSpaceSize, &isRawMemUsed);
176  if (!newBl) return false;
177 
178  // touch a page for the 1st time without taking masterMutex ...
179  for (BackRefBlock *bl = newBl; (uintptr_t)bl < (uintptr_t)newBl + blockSpaceSize;
180  bl = (BackRefBlock*)((uintptr_t)bl + BackRefBlock::bytes))
181  bl->zeroSet();
182 
183  MallocMutex::scoped_lock lock(masterMutex); // ... and share under lock
184 
185  const size_t numOfUnusedIdxs = backRefMaster->dataSz - lastUsed - 1;
186  if (numOfUnusedIdxs <= 0) { // no space in master under lock, roll back
187  backend->putBackRefSpace(newBl, blockSpaceSize, isRawMemUsed);
188  return false;
189  }
190  // It's possible that only part of newBl is used, due to lack of indices in master.
191  // This is OK as such underutilization is possible only once for backreferneces table.
192  int blocksToUse = min(numOfUnusedIdxs, blockSpaceSize / BackRefBlock::bytes);
193 
194  // use the first block in the batch to maintain the list of "raw" memory
195  // to be released at shutdown
196  if (isRawMemUsed) {
197  newBl->nextRawMemBlock = backRefMaster->allRawMemBlocks;
198  backRefMaster->allRawMemBlocks = newBl;
199  }
200  for (BackRefBlock *bl = newBl; blocksToUse>0;
201  bl = (BackRefBlock*)((uintptr_t)bl + BackRefBlock::bytes), blocksToUse--) {
202  initEmptyBackRefBlock(bl);
203  if (active->allocatedCount == BR_MAX_CNT)
204  active = bl; // active leaf is not needed in listForUse
205  else
206  addToForUseList(bl);
207  }
208  return true;
209 }

211 BackRefBlock *BackRefMaster::findFreeBlock()
212 {
213  if (active->allocatedCount < BR_MAX_CNT)
214  return active;
215 
216  if (listForUse) { // use released list
217  MallocMutex::scoped_lock lock(masterMutex);
218 
219  if (active->allocatedCount == BR_MAX_CNT && listForUse) {
220  active = listForUse;
221  listForUse = listForUse->nextForUse;
222  MALLOC_ASSERT(active->addedToForUse, ASSERT_TEXT);
223  active->addedToForUse = false;
224  }
225  } else // allocate new data node
226  if (!requestNewSpace())
227  return NULL;
228  return active;
229 }

231 void *getBackRef(BackRefIdx backRefIdx)
232 {
233  // !backRefMaster means no initialization done, so it can't be valid memory
234  // see addEmptyBackRefBlock for fences around lastUsed
235  if (!FencedLoad((intptr_t&)backRefMaster)
236  || backRefIdx.getMaster() > FencedLoad(backRefMaster->lastUsed)
237  || backRefIdx.getOffset() >= BR_MAX_CNT)
238  return NULL;
239  return *(void**)((uintptr_t)backRefMaster->backRefBl[backRefIdx.getMaster()]
240  + sizeof(BackRefBlock)+backRefIdx.getOffset()*sizeof(void*));
241 }

251 BackRefIdx BackRefIdx::newBackRef(bool largeObj)
252 {
253  BackRefBlock *blockToUse;
254  void **toUse;
255  BackRefIdx res;
256  bool lastBlockFirstUsed = false;
257 
258  do {
259  MALLOC_ASSERT(backRefMaster, ASSERT_TEXT);
260  blockToUse = backRefMaster->findFreeBlock();
261  if (!blockToUse)
262  return BackRefIdx();
263  toUse = NULL;
264  { // the block is locked to find a reference
265  MallocMutex::scoped_lock lock(blockToUse->blockMutex);
266 
267  if (blockToUse->freeList) {
268  toUse = (void**)blockToUse->freeList;
269  blockToUse->freeList = blockToUse->freeList->next;
270  MALLOC_ASSERT(!blockToUse->freeList ||
271  ((uintptr_t)blockToUse->freeList>=(uintptr_t)blockToUse
272  && (uintptr_t)blockToUse->freeList <
273  (uintptr_t)blockToUse + slabSize), ASSERT_TEXT);
274  } else if (blockToUse->allocatedCount < BR_MAX_CNT) {
275  toUse = (void**)blockToUse->bumpPtr;
276  blockToUse->bumpPtr =
277  (FreeObject*)((uintptr_t)blockToUse->bumpPtr - sizeof(void*));
278  if (blockToUse->allocatedCount == BR_MAX_CNT-1) {
279  MALLOC_ASSERT((uintptr_t)blockToUse->bumpPtr
280  < (uintptr_t)blockToUse+sizeof(BackRefBlock),
281  ASSERT_TEXT);
282  blockToUse->bumpPtr = NULL;
283  }
284  }
285  if (toUse) {
286  if (!blockToUse->allocatedCount && !backRefMaster->listForUse)
287  lastBlockFirstUsed = true;
288  blockToUse->allocatedCount++;
289  }
290  } // end of lock scope
291  } while (!toUse);
292  // The first thread that uses the last block requests new space in advance;
293  // possible failures are ignored.
294  if (lastBlockFirstUsed)
295  backRefMaster->requestNewSpace();
296 
297  res.master = blockToUse->myNum;
298  uintptr_t offset =
299  ((uintptr_t)toUse - ((uintptr_t)blockToUse + sizeof(BackRefBlock)))/sizeof(void*);
300  // Is offset too big?
301  MALLOC_ASSERT(!(offset >> 15), ASSERT_TEXT);
302  res.offset = offset;
303  if (largeObj) res.largeObj = largeObj;
304 
305  return res;
306 }

308 void removeBackRef(BackRefIdx backRefIdx)
309 {
310  MALLOC_ASSERT(!backRefIdx.isInvalid(), ASSERT_TEXT);
311  MALLOC_ASSERT(backRefIdx.getMaster()<=backRefMaster->lastUsed
312  && backRefIdx.getOffset()<BR_MAX_CNT, ASSERT_TEXT);
313  BackRefBlock *currBlock = backRefMaster->backRefBl[backRefIdx.getMaster()];
314  FreeObject *freeObj = (FreeObject*)((uintptr_t)currBlock + sizeof(BackRefBlock)
315  + backRefIdx.getOffset()*sizeof(void*));
316  MALLOC_ASSERT(((uintptr_t)freeObj>(uintptr_t)currBlock &&
317  (uintptr_t)freeObj<(uintptr_t)currBlock + slabSize), ASSERT_TEXT);
318  {
319  MallocMutex::scoped_lock lock(currBlock->blockMutex);
320 
321  freeObj->next = currBlock->freeList;
322  MALLOC_ASSERT(!freeObj->next ||
323  ((uintptr_t)freeObj->next > (uintptr_t)currBlock
324  && (uintptr_t)freeObj->next <
325  (uintptr_t)currBlock + slabSize), ASSERT_TEXT);
326  currBlock->freeList = freeObj;
327  currBlock->allocatedCount--;
328  }
329  // TODO: do we need double-check here?
330  if (!currBlock->addedToForUse && currBlock!=backRefMaster->active) {
331  MallocMutex::scoped_lock lock(masterMutex);
332 
333  if (!currBlock->addedToForUse && currBlock!=backRefMaster->active)
334  backRefMaster->addToForUseList(currBlock);
335  }
336 }
