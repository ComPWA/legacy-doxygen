21 /* The test uses "single produces multiple consumers" (SPMC )pattern to check
22  if the memory of the tasks stolen by consumer threads is returned to the
23  producer thread and is reused.
24 
25  The test consists of a series of iterations, which execute a task tree.
26  the test fails is the memory consumption is not stabilized during some
27  number of iterations.
28 
29  After the memory consumption stabilized the memory state is perturbed by
30  switching producer thread, and the check is repeated.
31 */

79 class TaskGenerator: public tbb::task {
80  const int my_child_count;
81  int my_depth;
82 public:
83  TaskGenerator(int child_count, int d) : my_child_count(child_count), my_depth(d) {
84  ASSERT(my_child_count>1, "The TaskGenerator should produce at least two children");
85  }
86  tbb::task* execute() __TBB_override {
87  if( my_depth>0 ) {
88  int child_count = my_child_count;
89  scheduler* my_sched = internal::governor::local_scheduler();
90  tbb::task& c = *new( allocate_continuation() ) tbb::empty_task;
91  c.set_ref_count( child_count );
92  recycle_as_child_of(c);
93  --child_count;
94  if( Producer==my_sched ) {
95  // produce a task and put it into Exchanger
96  tbb::task* t = new( c.allocate_child() ) tbb::empty_task;
97  --child_count;
98  t = Exchanger.fetch_and_store(t);
99  if( t ) spawn(*t);
100  } else {
101  tbb::task* t = Exchanger.fetch_and_store(NULL);
102  if( t ) spawn(*t);
103  }
104  while( child_count ) {
105  tbb::task* t = new( c.allocate_child() ) TaskGenerator(my_child_count, my_depth-1);
106  if( my_depth >4 ) enqueue(*t);
107  else spawn(*t);
108  --child_count;
109  }
110  --my_depth;
111  return this;
112  } else {
113  tbb::task* t = Exchanger.fetch_and_store(NULL);
114  if( t ) spawn(*t);
115  return NULL;
116  }
117  }
118 };

131 void RunTaskGenerators( bool switchProducer = false, bool checkProducer = false ) {
132  if( switchProducer )
133  Producer = NULL;
134  tbb::task* dummy_root = new( tbb::task::allocate_root() ) tbb::empty_task;
135  dummy_root->set_ref_count( 2 );
136  // If no producer, start elections; some worker will take the role
137  if( Producer )
138  tbb::task::spawn( *new( dummy_root->allocate_child() ) tbb::empty_task );
139  else
140  tbb::task::spawn( *new( dummy_root->allocate_child() ) ChangeProducer );
141  if( checkProducer && !Producer )
142  REPORT("Warning: producer has not changed after 10 attempts; running on a single core?\n");
143  for( int j=0; j<TaskGeneratorsIterations; ++j ) {
144  if( j&1 ) {
145  tbb::task& t = *new( tbb::task::allocate_root() ) TaskGenerator(/*child_count=*/4, /*depth=*/6);
146  tbb::task::spawn_root_and_wait(t);
147  } else {
148  tbb::task& t = *new (tbb::task::allocate_additional_child_of(*dummy_root))
149  TaskGenerator(/*child_count=*/4, /*depth=*/6);
150  tbb::task::enqueue(t);
151  }
152  }
153  dummy_root->wait_for_all();
154  tbb::task::destroy( *dummy_root );
155 }

193 void TestTaskReclamation() {
194  REMARK("testing task reclamation\n");
195 
196  size_t initial_amount_of_memory = 0;
197  double task_count_sum = 0;
198  double task_count_sum_square = 0;
199  double average, sigma;
200 
201  tbb::task_scheduler_init init (MinThread);
202  REMARK("Starting with %d threads\n", MinThread);
203  // For now, the master will produce "additional" tasks; later a worker will replace it;
204  Producer = internal::governor::local_scheduler();
205  int N = InitialStatsIterations;
206  // First N iterations fill internal buffers and collect initial statistics
207  for( int i=0; i<N; ++i ) {
208  // First N iterations fill internal buffers and collect initial statistics
209  RunTaskGenerators();
210  RunTaskListGenerator();
211 
212  size_t m = GetMemoryUsage();
213  if( m-initial_amount_of_memory > 0)
214  initial_amount_of_memory = m;
215 
216  intptr_t n = internal::governor::local_scheduler()->get_task_node_count( /*count_arena_workers=*/true );
217  task_count_sum += n;
218  task_count_sum_square += n*n;
219 
220  REMARK( "Consumed %ld bytes and %ld objects (iteration=%d)\n", long(m), long(n), i );
221  }
222  // Calculate statistical values
223  average = task_count_sum / N;
224  sigma = sqrt( (task_count_sum_square - task_count_sum*task_count_sum/N)/N );
225  REMARK("Average task count: %g, sigma: %g, sum: %g, square sum:%g \n", average, sigma, task_count_sum, task_count_sum_square);
226 
227  int last_error_iteration = 0,
228  producer_switch_iteration = 0,
229  producer_switches = 0;
230  bool switchProducer = false,
231  checkProducer = false;
232  for( int i=0; i < MaxIterations; ++i ) {
233  // These iterations check for excessive memory use and unreasonable task count
234  RunTaskGenerators( switchProducer, checkProducer );
235  RunTaskListGenerator();
236 
237  intptr_t n = internal::governor::local_scheduler()->get_task_node_count( /*count_arena_workers=*/true );
238  size_t m = GetMemoryUsage();
239 
240  if( (m-initial_amount_of_memory > 0) && (n > average+4*sigma) ) {
241  // Use 4*sigma interval (for normal distribution, 3*sigma contains ~99% of values).
242  REMARK( "Warning: possible leak of up to %ld bytes; currently %ld cached task objects (iteration=%d)\n",
243  static_cast<unsigned long>(m-initial_amount_of_memory), long(n), i );
244  last_error_iteration = i;
245  initial_amount_of_memory = m;
246  } else {
247  REMARK( "Consumed %ld bytes and %ld objects (iteration=%d)\n", long(m), long(n), i );
248  }
249  if ( i == last_error_iteration + AsymptoticRange ) {
250  if ( producer_switches++ == NumProducerSwitches )
251  break;
252  else {
253  last_error_iteration = producer_switch_iteration = i;
254  switchProducer = true;
255  }
256  }
257  else {
258  switchProducer = false;
259  checkProducer = producer_switch_iteration && (i == producer_switch_iteration + ProducerCheckTimeout);
260  }
261  }
262  ASSERT( last_error_iteration < MaxIterations - AsymptoticRange, "The amount of allocated tasks keeps growing. Leak is possible." );
263 }
