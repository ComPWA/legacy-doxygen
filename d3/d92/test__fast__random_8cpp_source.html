55 inline void CheckProbability ( double probability, double expectedProbability, int index, int numIndices, void* seed ) {
56  double lowerBound = expectedProbability / AcceptableDeviation,
57  upperBound = expectedProbability * AcceptableDeviation;
58  if ( probability < lowerBound ) {
59  if ( !NumLowOutliers )
60  REMARK( "Warning: Probability %.3f of hitting index %d among %d elements is out of acceptable range (%.3f - %.3f) for seed %p\n",
61  probability, index, numIndices, lowerBound, upperBound, seed );
62  ++NumLowOutliers;
63  }

72 struct CheckDistributionBody {
73  void operator() ( int id ) const {
74  uintptr_t randomRange = id + MinThread;
75  uintptr_t *curHits = new uintptr_t[randomRange]
76 #if TEST_TOTAL_SEQUENCE
77  , *totalHits = new uintptr_t[randomRange]
78 #endif
79  ;
80  double expectedProbability = 1./randomRange;
81  // Loop through different seeds
82  for ( uintptr_t i = 0; i < NumSeeds; ++i ) {
83  // Seed value mimics the one used by the TBB task scheduler
84  void* seed = (char*)&curHits + i * 16;
85  tbb::internal::FastRandom random( seed );
86  // According to Section 3.2.1.2 of Volume 2 of Knuth's Art of Computer Programming
87  // the following conditions must be hold for m=2^32:
88  ASSERT((random.c&1)!=0, "c is relatively prime to m");
89  ASSERT((random.a-1)%4==0, "a-1 is a multiple of p, for every prime p dividing m."
90  " And a-1 is a multiple of 4, if m is a multiple of 4");
91 
92  memset( curHits, 0, randomRange * sizeof(uintptr_t) );
93 #if TEST_TOTAL_SEQUENCE
94  memset( totalHits, 0, randomRange * sizeof(uintptr_t) );
95 #endif
96  const uintptr_t seriesLen = randomRange * SeriesBaseLen,
97  experimentLen = NumSeries * seriesLen;
98  uintptr_t *curSeries = new uintptr_t[seriesLen], // circular buffer
99  randsGenerated = 0;
100  // Initialize statistics
101  while ( randsGenerated < seriesLen ) {
102  uintptr_t idx = random.get() % randomRange;
103  ++curHits[idx];
104 #if TEST_TOTAL_SEQUENCE
105  ++totalHits[idx];
106 #endif
107  curSeries[randsGenerated++] = idx;
108  }
109  while ( randsGenerated < experimentLen ) {
110  for ( uintptr_t j = 0; j < randomRange; ++j ) {
111  CheckProbability( double(curHits[j])/seriesLen, expectedProbability, j, randomRange, seed );
112 #if TEST_TOTAL_SEQUENCE
113  CheckProbability( double(totalHits[j])/randsGenerated, expectedProbability, j, randomRange, seed );
114 #endif
115  }
116  --curHits[curSeries[randsGenerated % seriesLen]];
117  int idx = random.get() % randomRange;
118  ++curHits[idx];
119 #if TEST_TOTAL_SEQUENCE
120  ++totalHits[idx];
121 #endif
122  curSeries[randsGenerated++ % seriesLen] = idx;
123  }
124  delete [] curSeries;
125  }
126  delete [] curHits;
127 #if TEST_TOTAL_SEQUENCE
128  delete [] totalHits;
129 #endif
130  }

169 int TestMain () {
170  ASSERT( AcceptableDeviation < 100, NULL );
171  MinThread = max(MinThread, 2);
172  MaxThread = max(MinThread, MaxThread);
173  double NumChecks = double(NumSeeds) * (MaxThread - MinThread + 1) * (MaxThread + MinThread) / 2.0 * (SeriesBaseLen * NumSeries - SeriesBaseLen);
174  REMARK( "Number of distribution quality checks %g\n", NumChecks );
175  NumLowOutliers = NumHighOutliers = 0;
176  // Parallelism is used in this test only to speed up the long serial checks
177  // Essentially it is a loop over random number ranges
178  // Ideally tbb::parallel_for could be used to parallelize the outermost loop
179  // in CheckDistributionBody, but it is not used to avoid unit test contamination.
180  int P = tbb::tbb_thread::hardware_concurrency();
181  enum {reproducibilitySeedsToTest=1000};
182  enum {reproducibilitySeriesLen=100};
183  CheckReproducibilityBody<reproducibilitySeriesLen,reproducibilitySeedsToTest> CheckReproducibility(reproducibilitySeedsToTest/MaxThread);
184  while ( MinThread <= MaxThread ) {
185  int ThreadsToRun = min(P, MaxThread - MinThread + 1);
186  REMARK("Checking random range [%d;%d)\n", MinThread, MinThread+ThreadsToRun);
187  NativeParallelFor( ThreadsToRun, CheckDistributionBody() );
188  NativeParallelFor( ThreadsToRun, CheckReproducibility );
189  MinThread += P;
190  }
191  double observedProbabilityOfOutliers = (NumLowOutliers + NumHighOutliers) / NumChecks;
192  if ( observedProbabilityOfOutliers > AcceptableProbabilityOfOutliers ) {
193  if ( NumLowOutliers )
194  REPORT( "Warning: %d cases of too low probability of a given number detected\n", (int)NumLowOutliers );
195  if ( NumHighOutliers )
196  REPORT( "Warning: %d cases of too high probability of a given number detected\n", (int)NumHighOutliers );
197  ASSERT( observedProbabilityOfOutliers <= AcceptableProbabilityOfOutliers, NULL );
198  }
199  return Harness::Done;
