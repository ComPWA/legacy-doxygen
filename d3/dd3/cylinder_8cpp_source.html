97 static void cylinder_intersect(cylinder * cyl, ray * ry) {
98  vector rc, n, D, O;
99  flt t, s, tin, tout, ln, d;
100 
101  rc.x = ry->o.x - cyl->ctr.x;
102  rc.y = ry->o.y - cyl->ctr.y;
103  rc.z = ry->o.z - cyl->ctr.z;
104 
105  VCross(&ry->d, &cyl->axis, &n);
106 
107  VDOT(ln, n, n);
108  ln=sqrt(ln); /* finish length calculation */
109 
110  if (ln == 0.0) { /* ray is parallel to the cylinder.. */
111  VDOT(d, rc, cyl->axis);
112  D.x = rc.x - d * cyl->axis.x;
113  D.y = rc.y - d * cyl->axis.y;
114  D.z = rc.z - d * cyl->axis.z;
115  VDOT(d, D, D);
116  d = sqrt(d);
117  tin = -FHUGE;
118  tout = FHUGE;
119  /* if (d <= cyl->rad) then ray is inside cylinder.. else outside */
120  }
121 
122  VNorm(&n);
123  VDOT(d, rc, n);
124  d = fabs(d);
125 
126  if (d <= cyl->rad) { /* ray intersects cylinder.. */
127  VCross(&rc, &cyl->axis, &O);
128  VDOT(t, O, n);
129  t = - t / ln;
130  VCross(&n, &cyl->axis, &O);
131  VNorm(&O);
132  VDOT(s, ry->d, O);
133  s = fabs(sqrt(cyl->rad*cyl->rad - d*d) / s);
134  tin = t - s;
135  add_intersection(tin, (object *) cyl, ry);
136  tout = t + s;
137  add_intersection(tout, (object *) cyl, ry);
138  }
139 }

142  vector a,b,c;

211 static void fcylinder_intersect(cylinder * cyl, ray * ry) {
212  vector rc, n, O, hit, tmp2, ctmp4;
213  flt t, s, tin, tout, ln, d, tmp, tmp3;
214 
215  rc.x = ry->o.x - cyl->ctr.x;
216  rc.y = ry->o.y - cyl->ctr.y;
217  rc.z = ry->o.z - cyl->ctr.z;
218 
219  VCross(&ry->d, &cyl->axis, &n);
220 
221  VDOT(ln, n, n);
222  ln=sqrt(ln); /* finish length calculation */
223 
224  if (ln == 0.0) { /* ray is parallel to the cylinder.. */
225  return; /* in this case, we want to miss or go through the "hole" */
226  }
227 
228  VNorm(&n);
229  VDOT(d, rc, n);
230  d = fabs(d);
231 
232  if (d <= cyl->rad) { /* ray intersects cylinder.. */
233  VCross(&rc, &cyl->axis, &O);
234  VDOT(t, O, n);
235  t = - t / ln;
236  VCross(&n, &cyl->axis, &O);
237  VNorm(&O);
238  VDOT(s, ry->d, O);
239  s = fabs(sqrt(cyl->rad*cyl->rad - d*d) / s);
240  tin = t - s;
241 
242  RAYPNT(hit, (*ry), tin);
243 
244  ctmp4=cyl->axis;
245  VNorm(&ctmp4);
246 
247  tmp2.x = hit.x - cyl->ctr.x;
248  tmp2.y = hit.y - cyl->ctr.y;
249  tmp2.z = hit.z - cyl->ctr.z;
250 
251  VDOT(tmp, tmp2, ctmp4);
252  VDOT(tmp3, cyl->axis, cyl->axis);
253 
254  if ((tmp > 0.0) && (tmp < sqrt(tmp3)))
255  add_intersection(tin, (object *) cyl, ry);
256  tout = t + s;
257 
258  RAYPNT(hit, (*ry), tout);
259 
260  tmp2.x = hit.x - cyl->ctr.x;
261  tmp2.y = hit.y - cyl->ctr.y;
262  tmp2.z = hit.z - cyl->ctr.z;
263 
264  VDOT(tmp, tmp2, ctmp4);
265  VDOT(tmp3, cyl->axis, cyl->axis);
266 
267  if ((tmp > 0.0) && (tmp < sqrt(tmp3)))
268  add_intersection(tout, (object *) cyl, ry);
269  }
