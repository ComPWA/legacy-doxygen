61 size_t GetMemoryUsage(MemoryStatType stat = currentUsage) {
62  ASSERT(stat==currentUsage || stat==peakUsage, NULL);
63 #if __TBB_WIN8UI_SUPPORT
64  return 0;
65 #elif _WIN32
66  PROCESS_MEMORY_COUNTERS mem;
67  bool status = GetProcessMemoryInfo(GetCurrentProcess(), &mem, sizeof(mem))!=0;
68  ASSERT(status, NULL);
69  return stat==currentUsage? mem.PagefileUsage : mem.PeakPagefileUsage;
70 #elif __linux__
71  long unsigned size = 0;
72  FILE *fst = fopen("/proc/self/status", "r");
73  ASSERT(fst, NULL);
74  const int BUF_SZ = 200;
75  char buf_stat[BUF_SZ];
76  const char *pattern = stat==peakUsage ? "VmPeak: %lu" : "VmSize: %lu";
77  while (NULL != fgets(buf_stat, BUF_SZ, fst)) {
78  if (1==sscanf(buf_stat, pattern, &size)) {
79  ASSERT(size, "Invalid value of memory consumption.");
80  break;
81  }
82  }
83  // VmPeak is available in kernels staring 2.6.15
84  if (stat!=peakUsage || LinuxKernelVersion() >= 2006015)
85  ASSERT(size, "Invalid /proc/self/status format, pattern not found.");
86  fclose(fst);
87  return size*1024;
88 #elif __APPLE__ && !__ARM_ARCH
89  // TODO: find how detect peak virtual memory size under macOS
90  if (stat == peakUsage)
91  return 0;
92  kern_return_t status;
93  task_basic_info info;
94  mach_msg_type_number_t msg_type = TASK_BASIC_INFO_COUNT;
95  status = task_info(mach_task_self(), TASK_BASIC_INFO, reinterpret_cast<task_info_t>(&info), &msg_type);
96  ASSERT(status==KERN_SUCCESS, NULL);
97  return info.virtual_size - shared_size;
98 #else
99  return 0;
100 #endif
101 }
