111 /* test that pools in small space are either usable or not created
112  (i.e., exception raised) */
113 void TestSmallFixedSizePool()
114 {
115  char *buf;
116  bool allocated = false;
117 
118  for (size_t sz = 0; sz < 64*1024; sz = sz? 3*sz : 3) {
119  buf = (char*)malloc(sz);
120 #if TBB_USE_EXCEPTIONS
121  try {
122  tbb::fixed_pool pool(buf, sz);
123 /* Check that pool is usable, i.e. such an allocation exists,
124  that can be fulfilled from the pool. 16B allocation fits in 16KB slabs,
125  so it requires at least 16KB. Requirement of 9KB allocation is more modest.
126 */
127  allocated = pool.malloc( 16 ) || pool.malloc( 9*1024 );
128  } catch (std::invalid_argument&) {
129  ASSERT(!sz, "expect std::invalid_argument for zero-sized pool only");
130  } catch (...) {
131  ASSERT(0, "wrong exception type;");
132  }
133 #else
134 /* Do not test high-level pool interface because pool ctor emit exception
135  on creation failure. Instead test same functionality via low-level interface.
136  TODO: add support for configuration with disabled exceptions to pools.
137 */
138  rml::MemPoolPolicy pol(fixedBufGetMem, NULL, 0, /*fixedSizePool=*/true,
139  /*keepMemTillDestroy=*/false);
140  rml::MemoryPool *pool;
141  FixedPool fixedPool(buf, sz);
142 
143  rml::MemPoolError ret = pool_create_v1((intptr_t)&fixedPool, &pol, &pool);
144 
145  if (ret == rml::POOL_OK) {
146  allocated = pool_malloc(pool, 16) || pool_malloc(pool, 9*1024);
147  pool_destroy(pool);
148  } else
149  ASSERT(ret == rml::NO_MEMORY, "Expected that pool either valid "
150  "or have no memory to be created");
151 #endif
152  free(buf);
153  }
154  ASSERT(allocated, "Maximal buf size should be enough to create working fixed_pool");
155 #if TBB_USE_EXCEPTIONS
156  try {
157  tbb::fixed_pool pool(NULL, 10*1024*1024);
158  ASSERT(0, "Useless allocator with no memory must not be created");
159  } catch (std::invalid_argument&) {
160  } catch (...) {
161  ASSERT(0, "wrong exception type; expected invalid_argument");
162  }
163 #endif
164 }
