72 template<typename T>
73 class BaseFilter: public T {
74  bool* const my_done;
75  const bool my_is_last;
76  bool my_is_running;
77 public:
78  tbb::atomic<tbb::internal::Token> current_token;
79  BaseFilter( tbb::filter::mode type, bool done[], bool is_last ) :
80  T(type),
81  my_done(done),
82  my_is_last(is_last),
83  my_is_running(false),
84  current_token()
85  {}
86  virtual Buffer* get_buffer( void* item ) {
87  current_token++;
88  return static_cast<Buffer*>(item);
89  }
90  void* operator()( void* item ) __TBB_override {
91  // Check if work is done only on one thread when ntokens==1 or
92  // when pipeline has only one filter that is serial and non-thread-bound
93  if( is_serial_execution && !this->is_bound() ) {
94  // Get id of current thread
95  tbb::tbb_thread::id id = tbb::this_tbb_thread::get_id();
96  // At first execution, set thread_id to current thread id.
97  // Serialized execution is expected, so there should be no race.
98  if( thread_id == id0 )
99  thread_id = id;
100  // Check if work is done on one thread
101  ASSERT( thread_id == id, "non-thread-bound stages executed on different threads when must be executed on a single one");
102  }
103  Harness::ConcurrencyTracker ct;
104  if( this->is_serial() )
105  ASSERT( !my_is_running, "premature entry to serial stage" );
106  my_is_running = true;
107  Buffer* b = get_buffer(item);
108  if( b ) {
109  if(!this->is_bound() && sleeptime > 0) {
110  if(this->is_serial()) {
111  Harness::Sleep((int)sleeptime);
112  }
113  else {
114  // early parallel tokens sleep longer...
115  int i = (int)((5 - b->sequence_number) * sleeptime);
116  if(i < (int)sleeptime) i = (int)sleeptime;
117  Harness::Sleep(i);
118  }
119  }
120  if( this->is_ordered() ) {
121  if( b->sequence_number == Buffer::unused )
122  b->sequence_number = current_token-1;
123  else
124  ASSERT( b->sequence_number==current_token-1, "item arrived out of order" );
125  } else if( this->is_serial() ) {
126  if( b->sequence_number != current_token-1 && b->sequence_number != Buffer::unused )
127  out_of_order_count++;
128  }
129  ASSERT( b->id < StreamSize, NULL );
130  ASSERT( !my_done[b->id], "duplicate processing of token?" );
131  ASSERT( b->is_busy, NULL );
132  my_done[b->id] = true;
133  if( my_is_last ) {
134  b->id = Buffer::unused;
135  b->sequence_number = Buffer::unused;
136  __TBB_store_with_release(b->is_busy, false);
137  }
138  }
139  my_is_running = false;
140  return b;
141  }

144 template<typename T>
145 class InputFilter: public BaseFilter<T> {
146  tbb::spin_mutex input_lock;
147  Buffer buffer[MaxBuffer];
148  const tbb::internal::Token my_number_of_tokens;
149 public:
150  InputFilter( tbb::filter::mode type, tbb::internal::Token ntokens, bool done[], bool is_last ) :
151  BaseFilter<T>(type, done, is_last),
152  my_number_of_tokens(ntokens)
153  {}
154  Buffer* get_buffer( void* ) __TBB_override {
155  unsigned long next_input;
156  unsigned free_buffer = 0;
157  { // lock protected scope
158  tbb::spin_mutex::scoped_lock lock(input_lock);
159  if( this->current_token>=StreamSize )
160  return NULL;
161  next_input = this->current_token++;
162  // once in a while, emulate waiting for input; this only makes sense for serial input
163  if( this->is_serial() && WaitTest.required() )
164  WaitTest.probe( );
165  while( free_buffer<MaxBuffer )
166  if( __TBB_load_with_acquire(buffer[free_buffer].is_busy) )
167  ++free_buffer;
168  else {
169  buffer[free_buffer].is_busy = true;
170  break;
171  }
172  }
173  ASSERT( free_buffer<my_number_of_tokens, "premature reuse of buffer" );
174  Buffer* b = &buffer[free_buffer];
175  ASSERT( &buffer[0] <= b, NULL );
176  ASSERT( b <= &buffer[MaxBuffer-1], NULL );
177  ASSERT( b->id == Buffer::unused, NULL);
178  b->id = next_input;
179  ASSERT( b->sequence_number == Buffer::unused, NULL);
180  return b;
181  }

293 double PipelineTest::TestOneConfiguration(unsigned numeral, unsigned nthread, unsigned number_of_filters, tbb::internal::Token ntokens)
294 {
295  // Build pipeline
296  tbb::pipeline pipeline;
297  tbb::filter* filter[MaxFilters];
298  unsigned temp = numeral;
299  // parallelism_limit is the upper bound on the possible parallelism
300  unsigned parallelism_limit = 0;
301  // number of thread-bound-filters in the current sequence
302  unsigned number_of_tb_filters = 0;
303  // ordinal numbers of thread-bound-filters in the current sequence
304  unsigned array_of_tb_filter_numbers[MaxFilters];
305  if(!ContainsBoundFilter(numeral)) return 0.0;
306  for( unsigned i=0; i<number_of_filters; ++i, temp/=number_of_filter_types ) {
307  bool is_bound = temp%number_of_filter_types&0x1;
308  tbb::filter::mode filter_type;
309  if( is_bound ) {
310  filter_type = tb_filters_table[temp%number_of_filter_types/number_of_non_tb_filter_types];
311  } else
312  filter_type = non_tb_filters_table[temp%number_of_filter_types/number_of_tb_filter_types];
313  const bool is_last = i==number_of_filters-1;
314  if( is_bound ) {
315  if( i == 0 )
316  filter[i] = new InputFilter<tbb::thread_bound_filter>(filter_type,ntokens,Done[i],is_last);
317  else
318  filter[i] = new BaseFilter<tbb::thread_bound_filter>(filter_type,Done[i],is_last);
319  array_of_tb_filter_numbers[number_of_tb_filters] = i;
320  number_of_tb_filters++;
321  } else {
322  if( i == 0 )
323  filter[i] = new InputFilter<tbb::filter>(filter_type,ntokens,Done[i],is_last);
324  else
325  filter[i] = new BaseFilter<tbb::filter>(filter_type,Done[i],is_last);
326  }
327  pipeline.add_filter(*filter[i]);
328  if ( filter[i]->is_serial() ) {
329  parallelism_limit += 1;
330  } else {
331  parallelism_limit = nthread;
332  }
333  }
334  ASSERT(number_of_tb_filters,NULL);
335  clear_global_state();
336  // Account for clipping of parallelism.
337  if( parallelism_limit>nthread )
338  parallelism_limit = nthread;
339  if( parallelism_limit>ntokens )
340  parallelism_limit = (unsigned)ntokens;
341  StreamSize = nthread; // min( MaxStreamSize, nthread * MaxStreamItemsPerThread );
342 
343  for( unsigned i=0; i<number_of_filters; ++i ) {
344  static_cast<BaseFilter<tbb::filter>*>(filter[i])->current_token=0;
345  }
346  tbb::tbb_thread* t[MaxFilters];
347  for( unsigned j = 0; j<number_of_tb_filters; j++)
348  t[j] = new tbb::tbb_thread(process_loop(), static_cast<tbb::thread_bound_filter*>(filter[array_of_tb_filter_numbers[j]]));
349  if( ntokens == 1 || ( number_of_filters == 1 && number_of_tb_filters == 0 && filter[0]->is_serial() ))
350  is_serial_execution = true;
351  double strttime = GetCPUUserTime();
352  pipeline.run( ntokens );
353  double endtime = GetCPUUserTime();
354  for( unsigned j = 0; j<number_of_tb_filters; j++)
355  t[j]->join();
356  ASSERT( !Harness::ConcurrencyTracker::InstantParallelism(), "filter still running?" );
357  for( unsigned i=0; i<number_of_filters; ++i )
358  ASSERT( static_cast<BaseFilter<tbb::filter>*>(filter[i])->current_token==StreamSize, NULL );
359  for( unsigned i=0; i<MaxFilters; ++i )
360  for( unsigned j=0; j<StreamSize; ++j ) {
361  ASSERT( Done[i][j]==(i<number_of_filters), NULL );
362  }
363  if( Harness::ConcurrencyTracker::PeakParallelism() < parallelism_limit )
364  REMARK( "nthread=%lu ntokens=%lu MaxParallelism=%lu parallelism_limit=%lu\n",
365  nthread, ntokens, Harness::ConcurrencyTracker::PeakParallelism(), parallelism_limit );
366  for( unsigned i=0; i < number_of_filters; ++i ) {
367  delete filter[i];
368  filter[i] = NULL;
369  }
370  for( unsigned j = 0; j<number_of_tb_filters; j++)
371  delete t[j];
372  pipeline.clear();
373  return endtime - strttime;
374 } // TestOneConfiguration

408 void PipelineTest::TestIdleSpinning( unsigned nthread) {
409  unsigned sample_setups[] = {
410  // in the comments below, s == serial, o == serial out-of-order,
411  // B == thread bound, Bo == thread bound out-of-order, p == parallel
412  1, // B s s s
413  5, // s B s s
414  25, // s s B s
415  125, // s s s B
416  6, // B B s s
417  26, // B s B s
418  126, // B s s B
419  30, // s B B s
420  130, // s B s B
421  150, // s s B B
422  31, // B B B s
423  131, // B B s B
424  155, // s B B B
425  495, // s p p Bo
426  71, // B p o s
427  355, // s B p o
428  95, // s p Bo s
429  475, // s s p Bo
430  };
431  const int nsetups = sizeof(sample_setups) / sizeof(unsigned);
432  const int ntests = 4;
433  const double bignum = 1000000000.0;
434  const double allowable_slowdown = 3.5;
435  unsigned zero_count = 0;
436 
437  REMARK( "testing idle spinning with %lu threads\n", nthread );
438  tbb::internal::Token max_tokens = nthread < MaxBuffer ? nthread : MaxBuffer;
439  for( int i=0; i<nsetups; ++i ) {
440  unsigned numeral = sample_setups[i];
441  unsigned temp = numeral;
442  unsigned nbound = 0;
443  while(temp) {
444  if((temp%number_of_filter_types)&0x01) nbound++;
445  temp /= number_of_filter_types;
446  }
447  sleeptime = 20.0;
448  double s0 = bignum;
449  double s1 = bignum;
450  int v0cnt = 0;
451  int v1cnt = 0;
452  double s0sum = 0.0;
453  double s1sum = 0.0;
454  REMARK(" TestOneConfiguration, pipeline == ");
455  PrintConfiguration(numeral, MaxFilters);
456  REMARK(", max_tokens== %d\n", (int)max_tokens);
457  for(int j = 0; j < ntests; ++j) {
458  double s1a = TestOneConfiguration(numeral, nthread, MaxFilters, max_tokens);
459  double s0a = TestOneConfiguration((unsigned)0, nthread, MaxFilters, max_tokens);
460  s1sum += s1a;
461  s0sum += s0a;
462  if(s0a > 0.0) {
463  ++v0cnt;
464  s0 = (s0a < s0) ? s0a : s0;
465  }
466  else {
467  ++zero_count;
468  }
469  if(s1a > 0.0) {
470  ++v1cnt;
471  s1 = (s1a < s1) ? s1a : s1;
472  }
473  else {
474  ++zero_count;
475  }
476  }
477  if(s0 == bignum || s1 == bignum) continue;
478  s0sum /= (double)v0cnt;
479  s1sum /= (double)v1cnt;
480  double slowdown = (s1sum-s0sum)/s0sum;
481  if(slowdown > allowable_slowdown)
482  REMARK( "with %lu threads configuration %lu has slowdown > %g (%g)\n", nthread, numeral, allowable_slowdown, slowdown );
483  }
484  REMARK("Total of %lu zero times\n", zero_count);
485 }
