77 int Fib ( int n ) {
78  if( n < 2 ) {
79  if ( g_sim ) {
80  tbb::task_scheduler_init tsi(P_nested);
81  }
82  return n;
83  } else {
84  tbb::task_scheduler_init *tsi = NULL;
85  tbb::task *cur = NULL;
86  if ( g_sim ) {
87  if ( n % 2 == 0 ) {
88  if ( g_sim == tbbsched_auto_only || (g_sim == tbbsched_mixed && n % 4 == 0) ) {
89  // Trigger TBB scheduler auto-initialization
90  cur = &tbb::task::self();
91  }
92  else {
93  ASSERT ( g_sim == tbbsched_explicit_only || (g_sim == tbbsched_mixed && n % 4 != 0), NULL );
94  // Initialize TBB scheduler explicitly
95  tsi = new tbb::task_scheduler_init(P_nested);
96  }
97  }
98  }
99  int x, y;
100  x = cilk_spawn Fib(n-2);
101  y = cilk_spawn Fib(n-1);
102  cilk_sync;
103  if ( tsi )
104  delete tsi;
105  return x+y;
106  }
