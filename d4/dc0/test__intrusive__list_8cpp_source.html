77 template<class List, class Iterator>
78 void CheckListNodes ( List& il, int valueStep ) {
79  ASSERT( il.size()==unsigned(NumElements/valueStep), "Wrong size of the list" );
80  ASSERT( !il.empty(), "Incorrect result of empty() or the list is corrupted" );
81  int i;
82  Iterator it = il.begin();
83  for ( i = valueStep - 1; it != il.end(); ++it, i += valueStep ) {
84  ASSERT( it->Data() == i, "Unexpected node value while iterating forward" );
85  ASSERT( (*it).m_Canary == NoliMeTangere, "Memory corruption" );
86  }
87  ASSERT( i == NumElements + valueStep - 1, "Wrong number of list elements while iterating forward" );
88  it = il.end();
89  for ( i = NumElements - 1, it--; it != il.end(); --it, i -= valueStep ) {
90  ASSERT( (*it).Data() == i, "Unexpected node value while iterating backward" );
91  ASSERT( it->m_Canary == NoliMeTangere, "Memory corruption" );
92  }
93  ASSERT( i == -1, "Wrong number of list elements while iterating backward" );
94 }

96 template<class List, class Item>
97 void TestListOperations () {
98  typedef typename List::iterator iterator;
99  List il;
100  for ( int i = NumElements - 1; i >= 0; --i )
101  il.push_front( *new Item(i) );
102  CheckListNodes<const List, typename List::const_iterator>( il, 1 );
103  iterator it = il.begin();
104  for ( ; it != il.end(); ++it ) {
105  Item &item = *it;
106  it = il.erase( it ); // also advances the iterator
107  delete &item;
108  }
109  CheckListNodes<List, iterator>( il, 2 );
110  for ( it = il.begin(); it != il.end(); ++it ) {
111  Item &item = *it;
112  il.remove( *it++ ); // extra advance here as well
113  delete &item;
114  }
115  CheckListNodes<List, iterator>( il, 4 );
116  for ( it = il.begin(); it != il.end(); ) {
117  Item &item = *it++; // the iterator advances only here
118  il.remove( item );
119  delete &item;
120  }
121  ASSERT( il.size()==0, "The list has wrong size or not all items were removed" );
122  ASSERT( il.empty(), "Incorrect result of empty() or not all items were removed" );
123 }
