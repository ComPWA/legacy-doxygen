163 void ThreadLevelRecorder::change_level( int delta ) {
164  int x = level+=delta;
165  tbb::tick_count t = tbb::tick_count::now();
166  unsigned k = next++;
167  if( k<max_record_count ) {
168  record& r = array[k];
169  r.time = t;
170  r.nthread = x;
171  }
172 }

174 void ThreadLevelRecorder::dump() {
175  FILE* f = fopen("time.txt","w");
176  if( !f ) {
177  perror("fopen(time.txt)\n");
178  exit(1);
179  }
180  unsigned limit = next;
181  if( limit>max_record_count ) {
182  // Clip
183  limit = next;
184  }
185  for( unsigned i=0; i<limit; ++i ) {
186  fprintf(f,"%f\t%d\n",(array[i].time-array[0].time).seconds(),array[i].nthread);
187  }
188  fclose(f);
189 }

249 void TBBWork() {
250  if( AvailWork>=0 ) {
251  int k = --AvailWork;
252  if( k==-1 ) {
253  TBB_RunTime.server->adjust_job_count_estimate(-(TBB_MaxThread-1));
254  ++CompletionCount;
255  } else if( k>=0 ) {
256  for( int j=0; j<4; ++j ) {
257  OMP_Team team( *OMP_RunTime.server );
258  int n = OMP_RunTime.server->try_increase_load( OMP_ParallelRegionSize-1, /*strict=*/false );
259  team.barrier = 0;
260  ::rml::job* array[OMP_ParallelRegionSize-1];
261  if( n>0)
262  OMP_RunTime.server->get_threads( n, &team, array );
263  // Master does work inside parallel region too.
264  OMPWork();
265  // Master waits for workers to finish
266  if( n>0 )
267  while( team.barrier!=unsigned(n) ) {
268  __TBB_Yield();
269  }
270  }
271  ++CompletionCount;
272  }
273  }
274 }
