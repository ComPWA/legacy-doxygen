333  bool is_power_of_2( T val ) {
334  size_t intval = size_t(val);
335  return (intval&(intval-1)) == size_t(0);
336  }
337  int step_function_plus(int previous, double step){
338  return static_cast<int>(previous+step);
339  }
340  int step_function_multiply(int previous, double multiply){
341  return static_cast<int>(previous*multiply);
342  }
343  // "Power-of-2 ladder": nsteps is the desired number of steps between any subsequent powers of 2.
344  // The actual step is the quotient of the nearest smaller power of 2 divided by that number (but at least 1).
345  // E.g., '1:32:#4' means 1,2,3,4,5,6,7,8,10,12,14,16,20,24,28,32
346  int step_function_power2_ladder(int previous, double nsteps){
347  int steps = int(nsteps);
348  assert( is_power_of_2(steps) ); // must be a power of 2
349  // The actual step is 1 until the value is twice as big as nsteps
350  if( previous < 2*steps )
351  return previous+1;
352  // calculate the previous power of 2
353  int prev_power2 = previous/2; // start with half the given value
354  int rshift = 1; // and with the shift of 1;
355  while( int shifted = prev_power2>>rshift ) { // shift the value right; while the result is non-zero,
356  prev_power2 |= shifted; // add the bits set in 'shifted';
357  rshift <<= 1; // double the shift, as twice as many top bits are set;
358  } // repeat.
359  ++prev_power2; // all low bits set; now it's just one less than the desired power of 2
360  assert( is_power_of_2(prev_power2) );
361  assert( (prev_power2<=previous)&&(2*prev_power2>previous) );
362  // The actual step value is the previous power of 2 divided by steps
363  return previous + (prev_power2/steps);
364  }
