212 SceneHandle rt_newscene(void) {
213  scenedef * scene;
214  SceneHandle voidscene;
215 
216  scene = (scenedef *) malloc(sizeof(scenedef));
217  memset(scene, 0, sizeof(scenedef)); /* clear all valuas to 0 */
218 
219  voidscene = (SceneHandle) scene;
220 
221  rt_outputfile(voidscene, "/dev/null"); /* default output file (.tga) */
222  rt_resolution(voidscene, 512, 512); /* 512x512 resolution */
223  rt_verbose(voidscene, 0); /* verbose messages off */
224  rt_rawimage(voidscene, NULL); /* raw image output off */
225  rt_boundmode(voidscene, RT_BOUNDING_ENABLED); /* spatial subdivision on */
226  rt_boundthresh(voidscene, MAXOCTNODES); /* default threshold */
227  rt_displaymode(voidscene, RT_DISPLAY_ENABLED); /* video output on */
228  rt_camerasetup(voidscene, 1.0, 1.0, 0, 6,
229  rt_vector(0.0, 0.0, 0.0),
230  rt_vector(0.0, 0.0, 1.0),
231  rt_vector(0.0, 1.0, 0.0));
232 
233  return scene;
234 }

241 void apitextotex(apitexture * apitex, texture * tex) {
242  switch(apitex->texturefunc) {
243  case 0:
244  tex->texfunc=(color(*)(void *, void *, void *))(standard_texture);
245  break;
246 
247  case 1:
248  tex->texfunc=(color(*)(void *, void *, void *))(checker_texture);
249  break;
250 
251  case 2:
252  tex->texfunc=(color(*)(void *, void *, void *))(grit_texture);
253  break;
254 
255  case 3:
256  tex->texfunc=(color(*)(void *, void *, void *))(marble_texture);
257  break;
258 
259  case 4:
260  tex->texfunc=(color(*)(void *, void *, void *))(wood_texture);
261  break;
262 
263  case 5:
264  tex->texfunc=(color(*)(void *, void *, void *))(gnoise_texture);
265  break;
266 
267  case 6:
268  tex->texfunc=(color(*)(void *, void *, void *))(cyl_checker_texture);
269  break;
270 
271  case 7:
272  tex->texfunc=(color(*)(void *, void *, void *))(image_sphere_texture);
273  tex->img=AllocateImage((char *)apitex->imap);
274  break;
275 
276  case 8:
277  tex->texfunc=(color(*)(void *, void *, void *))(image_cyl_texture);
278  tex->img=AllocateImage((char *)apitex->imap);
279  break;
280 
281  case 9:
282  tex->texfunc=(color(*)(void *, void *, void *))(image_plane_texture);
283  tex->img=AllocateImage((char *)apitex->imap);
284  break;
285 
286  default:
287  tex->texfunc=(color(*)(void *, void *, void *))(standard_texture);
288  break;
289  }
290 
291  tex->ctr = apitex->ctr;
292  tex->rot = apitex->rot;
293  tex->scale = apitex->scale;
294  tex->uaxs = apitex->uaxs;
295  tex->vaxs = apitex->vaxs;
296  tex->ambient = apitex->ambient;
297  tex->diffuse = apitex->diffuse;
298  tex->specular = apitex->specular;
299  tex->opacity = apitex->opacity;
300  tex->col = apitex->col;
301 
302  tex->islight = 0;
303  tex->shadowcast = 1;
304  tex->phong = 0.0;
305  tex->phongexp = 0.0;
306  tex->phongtype = 0;
307 }
