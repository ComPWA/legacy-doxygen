71 // T must be arithmetic, and shouldn't wrap around for reasonable sizes of Count (which is now 150, and maxPorts is 10,
72 // so the max number generated right now is 1500 or so.) Source will generate a series of TT with value
73 // (init_val + (i-1)*addend) * my_mult, where i is the i-th invocation of the body. We are attaching addend
74 // source nodes to a join_port, and each will generate part of the numerical series the port is expecting
75 // to receive. If there is only one source node, the series order will be maintained; if more than one,

95 // if we start N source_bodys they will all have the addend N, and my_count should be initialized to 0 .. N-1.
96 // the output tuples should have all the sequence, but the order will in general vary.
97 template<typename TupleType>
98 class source_body {
99  typedef TupleType TT;
100  static const int N = tbb::flow::tuple_size<TT>::value;
101  int my_count;
102  int addend;
103 public:
104  source_body(int init_val, int addto) : my_count(init_val), addend(addto) { }
105  void operator=( const source_body& other) { my_count = other.my_count; addend = other.addend; }
106  bool operator()( TT &v) {
107  if(my_count >= Count) return false;
108  tuple_helper<N>::set_element(v, my_count);
109  my_count += addend;
110  return true;
111  }
112 };

209 // parallel_test: create source_nodes that feed tuples into the split node
210 // and queue_nodes that receive the output.
211 template<typename SType>
212 class parallel_test {
213 public:
214  typedef typename SType::input_type TType;
215  typedef tbb::flow::source_node<TType> source_type;
216  static const int N = tbb::flow::tuple_size<TType>::value;
217  static void test() {
218  source_type* all_source_nodes[MaxNSources];
219  sink_node_helper<N,SType>::print_parallel_remark();
220  REMARK(" >\n");
221  for(int i=0; i < MaxPorts; ++i) {
222  all_sink_nodes[i] = NULL;
223  }
224  // try test for # sources 1 .. MaxNSources
225  for(int nInputs = 1; nInputs <= MaxNSources; ++nInputs) {
226  tbb::flow::graph g;
227  SType* my_split = makeSplit<N,SType>::create(g);
228 
229  // add sinks first so when sources start spitting out values they are there to catch them
230  sink_node_helper<N, SType>::add_sink_nodes((*my_split), g);
231 
232  // now create nInputs source_nodes, each spitting out i, i+nInputs, i+2*nInputs ...
233  // each element of the tuple is i*(n+1), where n is the tuple element index (1-N)
234  for(int i = 0; i < nInputs; ++i) {
235  // create source node
236  source_type *s = new source_type(g, source_body<TType>(i, nInputs) );
237  tbb::flow::make_edge(*s, *my_split);
238  all_source_nodes[i] = s;
239  }
240 
241  g.wait_for_all();
242 
243  // check that we got Count values in each output queue, and all the index values
244  // are there.
245  sink_node_helper<N, SType>::check_sink_values();
246 
247  sink_node_helper<N, SType>::remove_sink_nodes(*my_split);
248  for(int i = 0; i < nInputs; ++i) {
249  delete all_source_nodes[i];
250  }
251  makeSplit<N,SType>::destroy(my_split);
252  }
253  }
254 };

259 template<typename SType>
260 void test_one_serial( SType &my_split, tbb::flow::graph &g) {
261  typedef typename SType::input_type TType;
262  static const int TUPLE_SIZE = tbb::flow::tuple_size<TType>::value;
263  sink_node_helper<TUPLE_SIZE, SType>::add_sink_nodes(my_split,g);
264  typedef TType q3_input_type;
265  tbb::flow::queue_node< q3_input_type > q3(g);
266 
267  tbb::flow::make_edge( q3, my_split );
268 
269  // fill the queue with its value one-at-a-time
270  flags.clear();
271  for (int i = 0; i < Count; ++i ) {
272  TType v;
273  tuple_helper<TUPLE_SIZE>::set_element(v, i);
274  ASSERT(my_split.try_put(v), NULL);
275  flags.push_back(false);
276  }
277 
278  g.wait_for_all();
279 
280  sink_node_helper<TUPLE_SIZE,SType>::check_sink_values();
281 
282  sink_node_helper<TUPLE_SIZE, SType>::remove_sink_nodes(my_split);
283
