62 template< typename InputType, typename OutputType, typename Body >
63 void buffered_levels( size_t concurrency, Body body ) {
64 
65  // Do for lc = 1 to concurrency level
66  for ( size_t lc = 1; lc <= concurrency; ++lc ) {
67  tbb::flow::graph g;
68 
69  // Set the execute_counter back to zero in the harness
70  harness_graph_executor<InputType, OutputType>::execute_count = 0;
71  // Set the number of current executors to zero.
72  harness_graph_executor<InputType, OutputType>::current_executors = 0;
73  // Set the max allowed executors to lc. There is a check in the functor to make sure this is never exceeded.
74  harness_graph_executor<InputType, OutputType>::max_executors = lc;
75 
76  // Create the function_node with the appropriate concurrency level, and use default buffering
77  tbb::flow::function_node< InputType, OutputType > exe_node( g, lc, body );
78  tbb::flow::function_node<InputType, InputType> pass_thru( g, tbb::flow::unlimited, pass_through<InputType>());
79 
80  // Create a vector of identical exe_nodes and pass_thrus
81  std::vector< tbb::flow::function_node< InputType, OutputType > > exe_vec(2, exe_node);
82  std::vector< tbb::flow::function_node< InputType, InputType > > pass_thru_vec(2, pass_thru);
83  // Attach each pass_thru to its corresponding exe_node
84  for (size_t node_idx=0; node_idx<exe_vec.size(); ++node_idx) {
85  tbb::flow::make_edge(pass_thru_vec[node_idx], exe_vec[node_idx]);
86  }
87 
88  // TODO: why the test is executed serially for the node pairs, not concurrently?
89  for (size_t node_idx=0; node_idx<exe_vec.size(); ++node_idx) {
90  // For num_receivers = 1 to MAX_NODES
91  for (size_t num_receivers = 1; num_receivers <= MAX_NODES; ++num_receivers ) {
92  // Create num_receivers counting receivers and connect the exe_vec[node_idx] to them.
93  std::vector< harness_mapped_receiver<OutputType>* > receivers(num_receivers);
94  for (size_t i = 0; i < num_receivers; i++) {
95  receivers[i] = new harness_mapped_receiver<OutputType>(g);
96  }
97 
98  for (size_t r = 0; r < num_receivers; ++r ) {
99  tbb::flow::make_edge( exe_vec[node_idx], *receivers[r] );
100  }
101 
102  // Do the test with varying numbers of senders
103  harness_counting_sender<InputType> *senders = NULL;
104  for (size_t num_senders = 1; num_senders <= MAX_NODES; ++num_senders ) {
105  // Create num_senders senders, set there message limit each to N, and connect them to pass_thru_vec[node_idx]
106  senders = new harness_counting_sender<InputType>[num_senders];
107  for (size_t s = 0; s < num_senders; ++s ) {
108  senders[s].my_limit = N;
109  senders[s].register_successor(pass_thru_vec[node_idx] );
110  }
111 
112  // Initialize the receivers so they know how many senders and messages to check for
113  for (size_t r = 0; r < num_receivers; ++r ) {
114  receivers[r]->initialize_map( N, num_senders );
115  }
116 
117  // Do the test
118  NativeParallelFor( (int)num_senders, parallel_put_until_limit<InputType>(senders) );
119  g.wait_for_all();
120 
121  // confirm that each sender was requested from N times
122  for (size_t s = 0; s < num_senders; ++s ) {
123  size_t n = senders[s].my_received;
124  ASSERT( n == N, NULL );
125  ASSERT( senders[s].my_receiver == &pass_thru_vec[node_idx], NULL );
126  }
127  // validate the receivers
128  for (size_t r = 0; r < num_receivers; ++r ) {
129  receivers[r]->validate();
130  }
131  delete [] senders;
132  }
133  for (size_t r = 0; r < num_receivers; ++r ) {
134  tbb::flow::remove_edge( exe_vec[node_idx], *receivers[r] );
135  }
136  ASSERT( exe_vec[node_idx].try_put( InputType() ) == true, NULL );
137  g.wait_for_all();
138  for (size_t r = 0; r < num_receivers; ++r ) {
139  // since it's detached, nothing should have changed
140  receivers[r]->validate();
141  }
142 
143  for (size_t i = 0; i < num_receivers; i++) {
144  delete receivers[i];
145  }
146 
147  } // for num_receivers
148  } // for node_idx
149  } // for concurrency level lc
150 }

266 template< typename InputType, typename OutputType, typename Body >
267 void concurrency_levels( size_t concurrency, Body body ) {
268 
269  for ( size_t lc = 1; lc <= concurrency; ++lc ) {
270  tbb::flow::graph g;
271 
272  // Set the execute_counter back to zero in the harness
273  harness_graph_executor<InputType, OutputType>::execute_count = 0;
274  // Set the number of current executors to zero.
275  harness_graph_executor<InputType, OutputType>::current_executors = 0;
276  // Set the max allowed executors to lc. There is a check in the functor to make sure this is never exceeded.
277  harness_graph_executor<InputType, OutputType>::max_executors = lc;
278 
279  typedef tbb::flow::function_node< InputType, OutputType, tbb::flow::rejecting > fnode_type;
280  fnode_type exe_node( g, lc, body );
281 
282  for (size_t num_receivers = 1; num_receivers <= MAX_NODES; ++num_receivers ) {
283 
284  std::vector< harness_counting_receiver<OutputType> > receivers(num_receivers, harness_counting_receiver<OutputType>(g));
285 
286 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
287  ASSERT(exe_node.successor_count() == 0, NULL);
288  ASSERT(exe_node.predecessor_count() == 0, NULL);
289 #endif
290 
291  for (size_t r = 0; r < num_receivers; ++r ) {
292  tbb::flow::make_edge( exe_node, receivers[r] );
293  }
294 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
295  ASSERT(exe_node.successor_count() == num_receivers, NULL);
296  typename fnode_type::successor_list_type my_succs;
297  exe_node.copy_successors(my_succs);
298  ASSERT(my_succs.size() == num_receivers, NULL);
299  typename fnode_type::predecessor_list_type my_preds;
300  exe_node.copy_predecessors(my_preds);
301  ASSERT(my_preds.size() == 0, NULL);
302 #endif
303 
304  harness_counting_sender<InputType> *senders = NULL;
305 
306  for (size_t num_senders = 1; num_senders <= MAX_NODES; ++num_senders ) {
307  senders = new harness_counting_sender<InputType>[num_senders];
308  {
309  // Exclusively lock m to prevent exe_node from finishing
310  tbb::spin_rw_mutex::scoped_lock l( harness_graph_executor<InputType, OutputType>::template mutex_holder<tbb::spin_rw_mutex>::mutex );
311 
312  // put to lc level, it will accept and then block at m
313  for ( size_t c = 0 ; c < lc ; ++c ) {
314  ASSERT( exe_node.try_put( InputType() ) == true, NULL );
315  }
316  // it only accepts to lc level
317  ASSERT( exe_node.try_put( InputType() ) == false, NULL );
318 
319  for (size_t s = 0; s < num_senders; ++s ) {
320  // register a sender
321  senders[s].my_limit = N;
322  exe_node.register_predecessor( senders[s] );
323  }
324 
325  } // release lock at end of scope, setting the exe node free to continue
326  // wait for graph to settle down
327  g.wait_for_all();
328 
329  // confirm that each sender was requested from N times
330  for (size_t s = 0; s < num_senders; ++s ) {
331  size_t n = senders[s].my_received;
332  ASSERT( n == N, NULL );
333  ASSERT( senders[s].my_receiver == &exe_node, NULL );
334  }
335  // confirm that each receivers got N * num_senders + the initial lc puts
336  for (size_t r = 0; r < num_receivers; ++r ) {
337  size_t n = receivers[r].my_count;
338  ASSERT( n == num_senders*N+lc, NULL );
339  receivers[r].my_count = 0;
340  }
341  delete [] senders;
342  }
343  for (size_t r = 0; r < num_receivers; ++r ) {
344  tbb::flow::remove_edge( exe_node, receivers[r] );
345  }
346  ASSERT( exe_node.try_put( InputType() ) == true, NULL );
347  g.wait_for_all();
348  for (size_t r = 0; r < num_receivers; ++r ) {
349  ASSERT( int(receivers[r].my_count) == 0, NULL );
350  }
351  }
352  }
353 }

402

404

409  tbb::flow::make_edge( exe_node, receivers[r] );

411 
412  NativeParallelFor( p, parallel_puts<InputType>(exe_node) );
413  g.wait_for_all();
414

416  size_t ec = harness_graph_executor<InputType, OutputType>::execute_count;

420  // 3) the nodes will send to multiple successors.
421  ASSERT( (int)c == p*N, NULL );
422  }
423  for (size_t r = 0; r < num_receivers; ++r ) {
424  tbb::flow::remove_edge( exe_node, receivers[r] );
425  }
426  }
427  }
