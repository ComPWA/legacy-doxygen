47 void ThreadState::loop() {
48  for(;;) {
49  ++clock;
50  if( ack==request ) {
51  thread_monitor::cookie c;
52  monitor.prepare_wait(c);
53  if( ack==request ) {
54  REMARK("%p: request=%d ack=%d\n", this, request, ack );
55  monitor.commit_wait(c);
56  } else
57  monitor.cancel_wait();
58  } else {
59  // Throw in delay occasionally
60  switch( request%8 ) {
61  case 0:
62  case 1:
63  case 5:
64  rml::internal::thread_monitor::yield();
65  }
66  int r = request;
67  ack = request;
68  if( !r ) return;
69  }
70  }
71 }
