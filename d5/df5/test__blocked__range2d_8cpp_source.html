72  for( int row_y=row_x; row_y<10; ++row_y ) {
73  row_type row_i = MakeAbstractValueType<RowTag>(row_x);
74  row_type row_j = MakeAbstractValueType<RowTag>(row_y);
75  for( int row_grain=1; row_grain<10; ++row_grain ) {
76  for( int col_x=-10; col_x<10; ++col_x ) {
77  for( int col_y=col_x; col_y<10; ++col_y ) {
78  col_type col_i = MakeAbstractValueType<ColTag>(col_x);
79  col_type col_j = MakeAbstractValueType<ColTag>(col_y);
80  for( int col_grain=1; col_grain<10; ++col_grain ) {
81  range_type r( row_i, row_j, row_grain, col_i, col_j, col_grain );
82  AssertSameType( r.is_divisible(), true );

84  AssertSameType( static_cast<range_type::row_range_type::const_iterator*>(0), static_cast<row_type*>(0) );
85  AssertSameType( static_cast<range_type::col_range_type::const_iterator*>(0), static_cast<col_type*>(0) );
86  AssertSameType( r.rows(), tbb::blocked_range<row_type>( row_i, row_j, 1 ));
87  AssertSameType( r.cols(), tbb::blocked_range<col_type>( col_i, col_j, 1 ));
88  ASSERT( r.empty()==(row_x==row_y||col_x==col_y), NULL );
