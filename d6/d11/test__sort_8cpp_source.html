41 class ParanoidKey
42 {
44  int32_t value;
46  int32_t index;
48  static const int32_t Empty = -1;
50  static const int32_t Dead = -2;
51  // True if key object has comparable value
52  bool
53  isLive() const
54  {
55  return (uint32_t)(index) < LastIndex;
56  }
57  // True if key object has been constructed.
58  bool
59  isConstructed() const
60  {
61  return isLive() || index == Empty;
62  }
63 
64  public:
65  ParanoidKey()
66  {
67  ++KeyCount;
68  index = Empty;
69  value = Empty;
70  }
71  ParanoidKey(const ParanoidKey& k) : value(k.value), index(k.index)
72  {
73  EXPECT_TRUE(k.isLive(), "source for copy-constructor is dead");
74  ++KeyCount;
75  }
76  ~ParanoidKey()
77  {
78  EXPECT_TRUE(isConstructed(), "double destruction");
79  index = Dead;
80  --KeyCount;
81  }
82  ParanoidKey&
83  operator=(const ParanoidKey& k)
84  {
85  EXPECT_TRUE(k.isLive(), "source for copy-assignment is dead");
86  EXPECT_TRUE(isConstructed(), "destination for copy-assignment is dead");
87  value = k.value;
88  index = k.index;
89  return *this;
90  }
91  ParanoidKey(int32_t index, int32_t value, OddTag) : index(index), value(value) {}
92  ParanoidKey(ParanoidKey&& k) : value(k.value), index(k.index)
93  {
94  EXPECT_TRUE(k.isConstructed(), "source for move-construction is dead");
95 // std::stable_sort() fails in move semantics on paranoid test before VS2015
96 #if !defined(_MSC_VER) || _MSC_VER >= 1900
97  k.index = Empty;
98 #endif
99  ++KeyCount;
100  }
101  ParanoidKey&
102  operator=(ParanoidKey&& k)
103  {
104  EXPECT_TRUE(k.isConstructed(), "source for move-assignment is dead");
105  EXPECT_TRUE(isConstructed(), "destination for move-assignment is dead");

201 template <typename T, typename Compare, typename Convert>
202 void
203 test_sort(Compare compare, Convert convert)
204 {
205  for (size_t n = 0; n < 100000; n = n <= 16 ? n + 1 : size_t(3.1415 * n))
206  {
207  LastIndex = n + 2;
208  // The rand()%(2*n+1) encourages generation of some duplicates.
209  // Sequence is padded with an extra element at front and back, to detect overwrite bugs.
210  Sequence<T> in(n + 2, [=](size_t k) { return convert(k, rand() % (2 * n + 1)); });
211  Sequence<T> expected(in);
212  Sequence<T> tmp(in);
213  invoke_on_all_policies(test_sort_with_compare(), tmp.begin(), tmp.end(), expected.begin(), expected.end(),
214  in.begin(), in.end(), in.size(), compare);
215  }
216 }
