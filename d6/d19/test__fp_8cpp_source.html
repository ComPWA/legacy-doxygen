77 void TestArenaFpuEnvPropagation( int id ) {
78  // TBB scheduler instance in a master thread captures the FPU control state
79  // at the moment of its initialization and passes it to the workers toiling
80  // on its behalf.
81  for( int k = 0; k < NumSseModes; ++k ) {
82  int sse_mode = SseModes[(k + id) % NumSseModes];
83  SetSseMode( sse_mode );
84  for( int i = 0; i < NumRoundingModes; ++i ) {
85  int mode = RoundingModes[(i + id) % NumRoundingModes];
86  SetRoundingMode( mode );
87  // New mode must be set before TBB scheduler is initialized
88  tbb::task_scheduler_init init;
89  tbb::parallel_for( 0, N, 1, RoundingModeCheckBody(mode, sse_mode) );
90  ASSERT( GetRoundingMode() == mode, NULL );
91  }
92  }

317 void TestContextFpuEnv() {
318  // Prepare contexts' fp modes.
319  for ( int i = 0, modeNum = 0; i < NumRoundingModes; ++i ) {
320  const int roundingMode = RoundingModes[i];
321  SetRoundingMode( roundingMode );
322  for( int j = 0; j < NumSseModes; ++j, ++modeNum ) {
323  const int sseMode = SseModes[j];
324  SetSseMode( sseMode );
325 
326  contexts[modeNum] = new tbb::task_group_context( tbb::task_group_context::isolated,
327  tbb::task_group_context::default_traits | tbb::task_group_context::fp_settings );
328  roundingModes[modeNum] = roundingMode;
329  sseModes[modeNum] = sseMode;
330  }
331  }
332  // Prepare arenas' fp modes.
333  for ( int arenaNum = 0; arenaNum < numArenas; ++arenaNum ) {
334  roundingModes[numModes+arenaNum] = roundingModes[arenaNum%numModes];
335  sseModes[numModes+arenaNum] = sseModes[arenaNum%numModes];
336  }
337  NativeParallelFor( numArenas, TestContextFpuEnvNativeLoopBody() );
338 #if TBB_USE_EXCEPTIONS
339  NativeParallelFor( numArenas, TestContextFpuEnvEhNativeLoopBody() );
340 #endif
341  for ( int modeNum = 0; modeNum < numModes; ++modeNum )
342  delete contexts[modeNum];
343 }
