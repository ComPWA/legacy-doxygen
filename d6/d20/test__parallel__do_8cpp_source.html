104 void do_work ( const value_t& depth, tbb::parallel_do_feeder<value_t>& feeder ) {
105  ++g_tasks_observed;
106  value_t new_value(depth.value()-1);
107  for( size_t i = 0; i < depth.value(); ++i) {
108  if (i%2) feeder.add( new_value ); // pass lvalue
109  else feeder.add( value_t(depth.value()-1) ); // pass rvalue
110  }
111 }

262  g_tasks_expected = 0;
263  for ( size_t i=0; i < depth; ++i )
