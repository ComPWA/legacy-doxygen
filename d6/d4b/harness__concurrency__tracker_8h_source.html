45 class ConcurrencyTracker {
46  bool m_Outer;
47 
48  static void Started () {
49  unsigned p = ++ctInstantParallelism;
50  unsigned q = ctPeakParallelism;
51  while( q<p ) {
52  q = ctPeakParallelism.compare_and_swap(p,q);
53  }
54  }
55 
56  static void Stopped () {
57  ASSERT ( ctInstantParallelism > 0, "Mismatched call to ConcurrencyTracker::Stopped()" );
58  --ctInstantParallelism;
59  }
60 public:
61  ConcurrencyTracker() : m_Outer(false) {
62  uintptr_t nested = ctNested;
63  ASSERT (nested == 0 || nested == 1, NULL);
64  if ( !ctNested ) {
65  Started();
66  m_Outer = true;
67  ctNested = 1;
68  }
69  }
70  ~ConcurrencyTracker() {
71  if ( m_Outer ) {
72  Stopped();
73  ctNested = 0;
74  }
75  }
76 
77  static unsigned PeakParallelism() { return ctPeakParallelism; }
78  static unsigned InstantParallelism() { return ctInstantParallelism; }
79 
80  static void Reset() {
81  ASSERT (ctInstantParallelism == 0, "Reset cannot be called when concurrency tracking is underway");
82  ctInstantParallelism = ctPeakParallelism = 0;
83  }
84 }; // ConcurrencyTracker

87 struct ExactConcurrencyLevel : NoCopy {
88  typedef tbb::combinable<size_t> Combinable;
89 private:
90  Harness::SpinBarrier *myBarrier;
91  // count unique worker threads
92  Combinable *myUniqueThreads;
93  mutable tbb::atomic<size_t> myActiveBodyCnt;
94  // output parameter for parallel_for body to report that max is reached
95  mutable bool myReachedMax;
96  // zero timeout means no barrier is used during concurrency level detection
97  const double myTimeout;
98  const size_t myConcLevel;
99  const bool myCrashOnFail;
100 
101  static tbb::mutex global_mutex;
102 
103  ExactConcurrencyLevel(double timeout, size_t concLevel, Combinable *uniq, bool crashOnFail) :
104  myBarrier(NULL), myUniqueThreads(uniq), myReachedMax(false),
105  myTimeout(timeout), myConcLevel(concLevel), myCrashOnFail(crashOnFail) {
106  myActiveBodyCnt = 0;
107  }
108  bool run() {
109  const int LOOP_ITERS = 100;
110  tbb::combinable<size_t> uniq;
111  Harness::SpinBarrier barrier((unsigned)myConcLevel, /*throwaway=*/true);
112  if (myTimeout != 0.)
113  myBarrier = &barrier;
114  if (!myUniqueThreads)
115  myUniqueThreads = &uniq;
116  tbb::parallel_for((size_t)0, myConcLevel*LOOP_ITERS, *this, tbb::simple_partitioner());
117  return myReachedMax;
118  }
119 public:
120  void operator()(size_t) const {
121  size_t v = ++myActiveBodyCnt;
122  ASSERT(v <= myConcLevel, "Number of active bodies is too high.");
123  if (v == myConcLevel) // record that the max expected concurrency was observed
124  myReachedMax = true;
125  // try to get barrier when 1st time in the thread
126  if (myBarrier && !myBarrier->timed_wait_noerror(myTimeout))
127  ASSERT(!myCrashOnFail, "Timeout was detected.");
128 
129  myUniqueThreads->local() = 1;
130  for (int i=0; i<100; i++)
131  __TBB_Pause(1);
132  --myActiveBodyCnt;
133  }
134 
135  enum Mode {
136  None,
137  // When multiple blocking checks are performed, there might be not enough
138  // concurrency for all of them. Serialize check() calls.
139  Serialize
140  };
141 
142  // check that we have never got more than concLevel threads,
143  // and that in some moment we saw exactly concLevel threads
144  static void check(size_t concLevel, Mode m = None) {
145  ExactConcurrencyLevel o(30., concLevel, NULL, /*crashOnFail=*/true);
146 
147  tbb::mutex::scoped_lock lock;
148  if (m == Serialize)
149  lock.acquire(global_mutex);
150  bool ok = o.run();
151  ASSERT(ok, NULL);
152  }
153 
154  static bool isEqual(size_t concLevel) {

156  return o.run();
157  }

159  static void checkLessOrEqual(size_t concLevel, tbb::combinable<size_t> *unique) {
160  ExactConcurrencyLevel o(0., concLevel, unique, /*crashOnFail=*/true);

165  }
