171 static void tri_intersect(tri * trn, ray * ry) {
172  vector tvec, pvec, qvec;
173  flt det, inv_det, t, u, v;
174 
175  /* begin calculating determinant - also used to calculate U parameter */
176  CROSS(pvec, ry->d, trn->edge2);
177 
178  /* if determinant is near zero, ray lies in plane of triangle */
179  det = DOT(trn->edge1, pvec);
180 
181  if (det > -EPSILON && det < EPSILON)
182  return;
183 
184  inv_det = 1.0 / det;
185 
186  /* calculate distance from vert0 to ray origin */
187  SUB(tvec, ry->o, trn->v0);
188 
189  /* calculate U parameter and test bounds */
190  u = DOT(tvec, pvec) * inv_det;
191  if (u < 0.0 || u > 1.0)
192  return;
193 
194  /* prepare to test V parameter */
195  CROSS(qvec, tvec, trn->edge1);
196 
197  /* calculate V parameter and test bounds */
198  v = DOT(ry->d, qvec) * inv_det;
199  if (v < 0.0 || u + v > 1.0)
200  return;
201 
202  /* calculate t, ray intersects triangle */
203  t = DOT(trn->edge2, qvec) * inv_det;
204 
205  add_intersection(t,(object *) trn, ry);
206 }

222 static void stri_normal(stri * trn, vector * pnt, ray * incident, vector * N) {
223  flt U, V, W, lensqr;
224  vector P, tmp, norm;
225 
226  CROSS(norm, trn->edge1, trn->edge2);
227  lensqr = DOT(norm, norm);
228 
229  VSUB((*pnt), trn->v0, P);
230 
231  CROSS(tmp, P, trn->edge2);
232  U = DOT(tmp, norm) / lensqr;
233 
234  CROSS(tmp, trn->edge1, P);
235  V = DOT(tmp, norm) / lensqr;
236 
237  W = 1.0 - (U + V);
238 
239  N->x = W*trn->n0.x + U*trn->n1.x + V*trn->n2.x;
240  N->y = W*trn->n0.y + U*trn->n1.y + V*trn->n2.y;
241  N->z = W*trn->n0.z + U*trn->n1.z + V*trn->n2.z;
242 
243  VNorm(N);
244 }
