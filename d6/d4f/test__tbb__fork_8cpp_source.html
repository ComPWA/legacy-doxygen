163 // check for memory leak during TBB task scheduler init/terminate life cycle
164 // TODO: move to test_task_scheduler_init after workers waiting productization
165 void TestSchedulerMemLeaks()
166 {
167  const int ITERS = 10;
168  int it;
169 
170  for (it=0; it<ITERS; it++) {
171  size_t memBefore = GetMemoryUsage();
172 #if _MSC_VER && _DEBUG
173  // _CrtMemCheckpoint() and _CrtMemDifference are non-empty only in _DEBUG
174  _CrtMemState stateBefore, stateAfter, diffState;
175  _CrtMemCheckpoint(&stateBefore);
176 #endif
177  for (int i=0; i<100; i++) {
178  tbb::task_scheduler_init sch(1);
179  for (int k=0; k<10; k++) {
180  tbb::empty_task *t = new( tbb::task::allocate_root() ) tbb::empty_task();
181  tbb::task::enqueue(*t);
182  }
183  bool ok = sch.blocking_terminate(std::nothrow);
184  ASSERT(ok, NULL);
185  }
186 #if _MSC_VER && _DEBUG
187  _CrtMemCheckpoint(&stateAfter);
188  int ret = _CrtMemDifference(&diffState, &stateBefore, &stateAfter);
189  ASSERT(!ret, "It must be no memory leaks at this point.");
190 #endif
191  if (GetMemoryUsage() <= memBefore)
192  break;
193  }
194  ASSERT(it < ITERS, "Memory consumption has not stabilized. Memory Leak?");
195 }
