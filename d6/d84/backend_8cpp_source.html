161 class GuardedSize : tbb::internal::no_copy {
162  uintptr_t value;
163 public:
164  enum State {
165  LOCKED,
166  COAL_BLOCK, // block is coalescing now
167  MAX_LOCKED_VAL = COAL_BLOCK,
168  LAST_REGION_BLOCK, // used to mark last block in region
169  // values after this are "normal" block sizes
170  MAX_SPEC_VAL = LAST_REGION_BLOCK
171  };
172 
173  void initLocked() { value = LOCKED; }
174  void makeCoalscing() {
175  MALLOC_ASSERT(value == LOCKED, ASSERT_TEXT);
176  value = COAL_BLOCK;
177  }
178  size_t tryLock(State state) {
179  size_t szVal, sz;
180  MALLOC_ASSERT(state <= MAX_LOCKED_VAL, ASSERT_TEXT);
181  for (;;) {
182  sz = FencedLoad((intptr_t&)value);
183  if (sz <= MAX_LOCKED_VAL)
184  break;
185  szVal = AtomicCompareExchange((intptr_t&)value, state, sz);
186 
187  if (szVal==sz)
188  break;
189  }
190  return sz;
191  }
192  void unlock(size_t size) {
193  MALLOC_ASSERT(value <= MAX_LOCKED_VAL, "The lock is not locked");
194  MALLOC_ASSERT(size > MAX_LOCKED_VAL, ASSERT_TEXT);
195  FencedStore((intptr_t&)value, size);
196  }
197  bool isLastRegionBlock() const { return value==LAST_REGION_BLOCK; }
198  friend void Backend::IndexedBins::verify();
199 };

379 // Try to get a block from a bin.
380 // If the remaining free space would stay in the same bin,
381 // split the block without removing it.
382 // If the free space should go to other bin(s), remove the block.
383 // alignedBin is true, if all blocks in the bin have slab-aligned right side.
384 FreeBlock *Backend::IndexedBins::getFromBin(int binIdx, BackendSync *sync,
385  size_t size, bool needAlignedRes, bool alignedBin, bool wait,
386  int *binLocked)
387 {
388  Bin *b = &freeBins[binIdx];
389 try_next:
390  FreeBlock *fBlock = NULL;
391  if (b->head) {
392  bool locked;
393  MallocMutex::scoped_lock scopedLock(b->tLock, wait, &locked);
394 
395  if (!locked) {
396  if (binLocked) (*binLocked)++;
397  return NULL;
398  }
399 
400  for (FreeBlock *curr = b->head; curr; curr = curr->next) {
401  size_t szBlock = curr->tryLockBlock();
402  if (!szBlock) {
403  // block is locked, re-do bin lock, as there is no place to spin
404  // while block coalescing
405  goto try_next;
406  }
407 
408  if (alignedBin || !needAlignedRes) {
409  size_t splitSz = szBlock - size;
410  // If we got a block as split result,
411  // it must have a room for control structures.
412  if (szBlock >= size && (splitSz >= FreeBlock::minBlockSize ||
413  !splitSz))
414  fBlock = curr;
415  } else {
416  void *newB = alignUp(curr, slabSize);
417  uintptr_t rightNew = (uintptr_t)newB + size;
418  uintptr_t rightCurr = (uintptr_t)curr + szBlock;
419  // appropriate size, and left and right split results
420  // are either big enough or non-existent
421  if (rightNew <= rightCurr
422  && (newB==curr ||
423  (uintptr_t)newB-(uintptr_t)curr >= FreeBlock::minBlockSize)
424  && (rightNew==rightCurr ||
425  rightCurr - rightNew >= FreeBlock::minBlockSize))
426  fBlock = curr;
427  }
428  if (fBlock) {
429  // consume must be called before result of removing from a bin
430  // is visible externally.
431  sync->blockConsumed();
432  if (alignedBin && needAlignedRes &&
433  Backend::sizeToBin(szBlock-size) == Backend::sizeToBin(szBlock)) {
434  // free remainder of fBlock stay in same bin,
435  // so no need to remove it from the bin
436  // TODO: add more "still here" cases
437  FreeBlock *newFBlock = fBlock;
438  // return block from right side of fBlock
439  fBlock = (FreeBlock*)((uintptr_t)newFBlock + szBlock - size);
440  MALLOC_ASSERT(isAligned(fBlock, slabSize), "Invalid free block");
441  fBlock->initHeader();
442  fBlock->setLeftFree(szBlock - size);
443  newFBlock->setMeFree(szBlock - size);
444 
445  fBlock->sizeTmp = size;
446  } else {
447  b->removeBlock(fBlock);
448  if (freeBins[binIdx].empty())
449  bitMask.set(binIdx, false);
450  fBlock->sizeTmp = szBlock;
451  }
452  break;
453  } else { // block size is not valid, search for next block in the bin
454  curr->setMeFree(szBlock);
455  curr->rightNeig(szBlock)->setLeftFree(szBlock);
456  }
457  }
458  }
459  return fBlock;
460 }

682 FreeBlock *Backend::askMemFromOS(size_t blockSize, intptr_t startModifiedCnt,
683  int *lockedBinsThreshold, int numOfLockedBins,
684  bool *splittableRet)
685 {
686  FreeBlock *block;
687  // The block sizes can be divided into 3 groups:
688  // 1. "quite small": popular object size, we are in bootstarp or something
689  // like; request several regions.
690  // 2. "quite large": we want to have several such blocks in the region
691  // but not want several pre-allocated regions.
692  // 3. "huge": exact fit, we allocate only one block and do not allow
693  // any other allocations to placed in a region.
694  // Dividing the block sizes in these groups we are trying to balance between
695  // too small regions (that leads to fragmentation) and too large ones (that
696  // leads to excessive address space consumption). If a region is "too
697  // large", allocate only one, to prevent fragmentation. It supposedly
698  // doesn't hurt performance, because the object requested by user is large.
699  // Bounds for the groups are:
700  const size_t maxBinned = getMaxBinnedSize();
701  const size_t quiteSmall = maxBinned / 8;
702  const size_t quiteLarge = maxBinned;
703 
704  if (blockSize >= quiteLarge) {
705  // Do not interact with other threads via semaphores, as for exact fit
706  // we can't share regions with them, memory requesting is individual.
707  block = addNewRegion(blockSize, MEMREG_ONE_BLOCK, /*addToBin=*/false);
708  if (!block)
709  return releaseMemInCaches(startModifiedCnt, lockedBinsThreshold, numOfLockedBins);
710  *splittableRet = false;
711  } else {
712  const size_t regSz_sizeBased = alignUp(4*maxRequestedSize, 1024*1024);
713  // Another thread is modifying backend while we can't get the block.
714  // Wait while it leaves and re-do the scan
715  // before trying other ways to extend the backend.
716  if (bkndSync.waitTillBlockReleased(startModifiedCnt)
717  // semaphore is protecting adding more more memory from OS
718  || memExtendingSema.wait())
719  return (FreeBlock*)VALID_BLOCK_IN_BIN;
720 
721  if (startModifiedCnt != bkndSync.getNumOfMods()) {
722  memExtendingSema.signal();
723  return (FreeBlock*)VALID_BLOCK_IN_BIN;
724  }
725 
726  if (blockSize < quiteSmall) {
727  // For this size of blocks, add NUM_OF_REG "advance" regions in bin,
728  // and return one as a result.
729  // TODO: add to bin first, because other threads can use them right away.
730  // This must be done carefully, because blocks in bins can be released
731  // in releaseCachesToLimit().
732  const unsigned NUM_OF_REG = 3;
733  block = addNewRegion(regSz_sizeBased, MEMREG_FLEXIBLE_SIZE, /*addToBin=*/false);
734  if (block)
735  for (unsigned idx=0; idx<NUM_OF_REG; idx++)
736  if (! addNewRegion(regSz_sizeBased, MEMREG_FLEXIBLE_SIZE, /*addToBin=*/true))
737  break;
738  } else {
739  block = addNewRegion(regSz_sizeBased, MEMREG_SEVERAL_BLOCKS, /*addToBin=*/false);
740  }
741  memExtendingSema.signal();
742 
743  // no regions found, try to clean cache
744  if (!block || block == (FreeBlock*)VALID_BLOCK_IN_BIN)
745  return releaseMemInCaches(startModifiedCnt, lockedBinsThreshold, numOfLockedBins);
746  // Since a region can hold more than one block it can be splitted.
747  *splittableRet = true;
748  }
749  // after asking memory from OS, release caches if we above the memory limits
750  releaseCachesToLimit();
751 
752  return block;
753 }

755 void Backend::releaseCachesToLimit()
756 {
757  if (!memSoftLimit || totalMemSize <= memSoftLimit)
758  return;
759  size_t locTotalMemSize, locMemSoftLimit;
760 
761  scanCoalescQ(/*forceCoalescQDrop=*/false);
762  if (extMemPool->softCachesCleanup() &&
763  (locTotalMemSize = FencedLoad((intptr_t&)totalMemSize)) <=
764  (locMemSoftLimit = FencedLoad((intptr_t&)memSoftLimit)))
765  return;
766  // clean global large-object cache, if this is not enough, clean local caches
767  // do this in several tries, because backend fragmentation can prevent
768  // region from releasing
769  for (int cleanLocal = 0; cleanLocal<2; cleanLocal++)
770  while (cleanLocal?
771  extMemPool->allLocalCaches.cleanup(extMemPool, /*cleanOnlyUnused=*/true)
772  : extMemPool->loc.decreasingCleanup())
773  if ((locTotalMemSize = FencedLoad((intptr_t&)totalMemSize)) <=
774  (locMemSoftLimit = FencedLoad((intptr_t&)memSoftLimit)))
775  return;
776  // last chance to match memSoftLimit
777  extMemPool->hardCachesCleanup();
778 }

807 // try to allocate size Byte block in available bins
808 // needAlignedRes is true if result must be slab-aligned
809 FreeBlock *Backend::genericGetBlock(int num, size_t size, bool needAlignedBlock)
810 {
811  FreeBlock *block = NULL;
812  const size_t totalReqSize = num*size;
813  // no splitting after requesting new region, asks exact size
814  const int nativeBin = sizeToBin(totalReqSize);
815 
816  requestBootstrapMem();
817  // If we found 2 or less locked bins, it's time to ask more memory from OS.
818  // But nothing can be asked from fixed pool. And we prefer wait, not ask
819  // for more memory, if block is quite large.
820  int lockedBinsThreshold = extMemPool->fixedPool || size>=maxBinned_SmallPage? 0 : 2;
821 
822  // Find maximal requested size limited by getMaxBinnedSize()
823  AtomicUpdate(maxRequestedSize, totalReqSize, MaxRequestComparator(this));
824  scanCoalescQ(/*forceCoalescQDrop=*/false);
825 
826  bool splittable = true;
827  for (;;) {
828  const intptr_t startModifiedCnt = bkndSync.getNumOfMods();
829  int numOfLockedBins;
830 
831  do {
832  numOfLockedBins = 0;
833 
834  // TODO: try different bin search order
835  if (needAlignedBlock) {
836  block = freeAlignedBins.findBlock(nativeBin, &bkndSync, num*size,
837  /*needAlignedBlock=*/true, /*alignedBin=*/true,
838  &numOfLockedBins);
839  if (!block)
840  block = freeLargeBins.findBlock(nativeBin, &bkndSync, num*size,
841  /*needAlignedBlock=*/true, /*alignedBin=*/false,
842  &numOfLockedBins);
843  } else {
844  block = freeLargeBins.findBlock(nativeBin, &bkndSync, num*size,
845  /*needAlignedBlock=*/false, /*alignedBin=*/false,
846  &numOfLockedBins);
847  if (!block)
848  block = freeAlignedBins.findBlock(nativeBin, &bkndSync, num*size,
849  /*needAlignedBlock=*/false, /*alignedBin=*/true,
850  &numOfLockedBins);
851  }
852  } while (!block && numOfLockedBins>lockedBinsThreshold);
853 
854  if (block)
855  break;
856 
857  if (!(scanCoalescQ(/*forceCoalescQDrop=*/true)
858  | extMemPool->softCachesCleanup())) {
859  // bins are not updated,
860  // only remaining possibility is to ask for more memory
861  block =
862  askMemFromOS(totalReqSize, startModifiedCnt, &lockedBinsThreshold,
863  numOfLockedBins, &splittable);
864  if (!block)
865  return NULL;
866  if (block != (FreeBlock*)VALID_BLOCK_IN_BIN) {
867  // size can be increased in askMemFromOS, that's why >=
868  MALLOC_ASSERT(block->sizeTmp >= size, ASSERT_TEXT);
869  break;
870  }
871  // valid block somewhere in bins, let's find it
872  block = NULL;
873  }
874  }
875  MALLOC_ASSERT(block, ASSERT_TEXT);
876  if (splittable)
877  block = toAlignedBin(block, block->sizeTmp)?
878  splitAlignedBlock(block, num, size, needAlignedBlock) :
879  splitUnalignedBlock(block, num, size, needAlignedBlock);
880  // matched blockConsumed() from startUseBlock()
881  bkndSync.blockReleased();
882 
883  return block;
884 }

1052 FreeBlock *Backend::doCoalesc(FreeBlock *fBlock, MemRegion **mRegion)
1053 {

1055  size_t resSize = fBlock->sizeTmp;
1056  MemRegion *memRegion = NULL;
1057 
1058  fBlock->markCoalescing(resSize);
1059  resBlock->blockInBin = false;
1060 
1061  // coalescing with left neighbor
1062  size_t leftSz = fBlock->trySetLeftUsed(GuardedSize::COAL_BLOCK);
1063  if (leftSz != GuardedSize::LOCKED) {
1064  if (leftSz == GuardedSize::COAL_BLOCK) {
1065  coalescQ.putBlock(fBlock);
1066  return NULL;
1067  } else {
1068  FreeBlock *left = fBlock->leftNeig(leftSz);
1069  size_t lSz = left->trySetMeUsed(GuardedSize::COAL_BLOCK);
1070  if (lSz <= GuardedSize::MAX_LOCKED_VAL) {
1071  fBlock->setLeftFree(leftSz); // rollback
1072  coalescQ.putBlock(fBlock);
1073  return NULL;
1074  } else {
1075  MALLOC_ASSERT(lSz == leftSz, "Invalid header");
1076  left->blockInBin = true;
1077  resBlock = left;
1078  resSize += leftSz;
1079  resBlock->sizeTmp = resSize;
1080  }
1081  }
1082  }
1083  // coalescing with right neighbor
1084  FreeBlock *right = fBlock->rightNeig(fBlock->sizeTmp);
1085  size_t rightSz = right->trySetMeUsed(GuardedSize::COAL_BLOCK);
1086  if (rightSz != GuardedSize::LOCKED) {
1087  // LastFreeBlock is on the right side
1088  if (GuardedSize::LAST_REGION_BLOCK == rightSz) {
1089  right->setMeFree(GuardedSize::LAST_REGION_BLOCK);
1090  memRegion = static_cast<LastFreeBlock*>(right)->memRegion;
1091  } else if (GuardedSize::COAL_BLOCK == rightSz) {
1092  if (resBlock->blockInBin) {
1093  resBlock->blockInBin = false;
1094  removeBlockFromBin(resBlock);
1095  }
1096  coalescQ.putBlock(resBlock);
1097  return NULL;
1098  } else {
1099  size_t rSz = right->rightNeig(rightSz)->
1100  trySetLeftUsed(GuardedSize::COAL_BLOCK);
1101  if (rSz <= GuardedSize::MAX_LOCKED_VAL) {
1102  right->setMeFree(rightSz); // rollback
1103  if (resBlock->blockInBin) {
1104  resBlock->blockInBin = false;
1105  removeBlockFromBin(resBlock);
1106  }
1107  coalescQ.putBlock(resBlock);
1108  return NULL;
1109  } else {
1110  MALLOC_ASSERT(rSz == rightSz, "Invalid header");
1111  removeBlockFromBin(right);
1112  resSize += rightSz;
1113 
1114  // Is LastFreeBlock on the right side of right?
1115  FreeBlock *nextRight = right->rightNeig(rightSz);
1116  size_t nextRightSz = nextRight->
1117  trySetMeUsed(GuardedSize::COAL_BLOCK);
1118  if (nextRightSz > GuardedSize::MAX_LOCKED_VAL) {
1119  if (nextRightSz == GuardedSize::LAST_REGION_BLOCK)
1120  memRegion = static_cast<LastFreeBlock*>(nextRight)->memRegion;
1121 
1122  nextRight->setMeFree(nextRightSz);
1123  }
1124  }
1125  }
1126  }
1127  if (memRegion) {
1128  MALLOC_ASSERT((uintptr_t)memRegion + memRegion->allocSz >=
1129  (uintptr_t)right + sizeof(LastFreeBlock), ASSERT_TEXT);
1130  MALLOC_ASSERT((uintptr_t)memRegion < (uintptr_t)resBlock, ASSERT_TEXT);
1131  *mRegion = memRegion;
1132  } else
1133  *mRegion = NULL;
1134  resBlock->sizeTmp = resSize;
1135  return resBlock;
1136 }

1138 bool Backend::coalescAndPutList(FreeBlock *list, bool forceCoalescQDrop,
1139  bool reportBlocksProcessed)
1140 {
1141  bool regionReleased = false;
1142 
1143  for (FreeBlock *helper; list;
1144  list = helper,
1145  // matches block enqueue in CoalRequestQ::putBlock()
1146  reportBlocksProcessed? coalescQ.blockWasProcessed() : (void)0) {
1147  MemRegion *memRegion;
1148  bool addToTail = false;
1149 
1150  helper = list->nextToFree;
1151  FreeBlock *toRet = doCoalesc(list, &memRegion);
1152  if (!toRet)
1153  continue;
1154 
1155  if (memRegion && memRegion->blockSz == toRet->sizeTmp
1156  && !extMemPool->fixedPool) {
1157  if (extMemPool->regionsAreReleaseable()) {
1158  // release the region, because there is no used blocks in it
1159  if (toRet->blockInBin)
1160  removeBlockFromBin(toRet);
1161  releaseRegion(memRegion);
1162  regionReleased = true;
1163  continue;
1164  } else // add block from empty region to end of bin,
1165  addToTail = true; // preserving for exact fit
1166  }
1167  size_t currSz = toRet->sizeTmp;
1168  int bin = sizeToBin(currSz);
1169  bool toAligned = toAlignedBin(toRet, currSz);
1170  bool needAddToBin = true;
1171 
1172  if (toRet->blockInBin) {
1173  // Does it stay in same bin?
1174  if (toRet->myBin == bin && toRet->aligned == toAligned)
1175  needAddToBin = false;
1176  else {
1177  toRet->blockInBin = false;
1178  removeBlockFromBin(toRet);
1179  }
1180  }
1181 
1182  // Does not stay in same bin, or bin-less; add it
1183  if (needAddToBin) {
1184  toRet->prev = toRet->next = toRet->nextToFree = NULL;
1185  toRet->myBin = NO_BIN;
1186 
1187  // If the block is too small to fit in any bin, keep it bin-less.
1188  // It's not a leak because the block later can be coalesced.
1189  if (currSz >= minBinnedSize) {
1190  toRet->sizeTmp = currSz;
1191  IndexedBins *target = toAligned? &freeAlignedBins : &freeLargeBins;
1192  if (forceCoalescQDrop) {
1193  target->addBlock(bin, toRet, toRet->sizeTmp, addToTail);
1194  } else if (!target->tryAddBlock(bin, toRet, addToTail)) {
1195  coalescQ.putBlock(toRet);
1196  continue;
1197  }
1198  }
1199  toRet->sizeTmp = 0;
1200  }
1201  // Free (possibly coalesced) free block.
1202  // Adding to bin must be done before this point,
1203  // because after a block is free it can be coalesced, and
1204  // using its pointer became unsafe.
1205  // Remember that coalescing is not done under any global lock.
1206  toRet->setMeFree(currSz);
1207  toRet->rightNeig(currSz)->setLeftFree(currSz);
1208  }
1209  return regionReleased;
1210 }
