502 void ipc_worker::start_shutdown(bool join) {
503  state_t s;
504 
505  do {
506  s = my_state;
507  __TBB_ASSERT( s!=st_quit, NULL );
508  } while( my_state.compare_and_swap( st_quit, s )!=s );
509  if( s==st_normal || s==st_starting ) {
510  // May have invalidated invariant for sleeping, so wake up the thread.
511  // Note that the notify() here occurs without maintaining invariants for my_slack.
512  // It does not matter, because my_state==st_quit overrides checking of my_slack.
513  my_thread_monitor.notify();
514  // Do not need release handle in st_init state,
515  // because in this case the thread wasn't started yet.
516  // For st_starting release is done at launch site.
517  if( s==st_normal )
518  release_handle( my_handle, join );
519  }
520 }

541 void ipc_worker::run() {
542  my_server.propagate_chain_reaction();
543 
544  // Transiting to st_normal here would require setting my_handle,
545  // which would create race with the launching thread and
546  // complications in handle management on Windows.
547 
548  ::rml::job& j = *my_client.create_one_job();
549  state_t state = my_state;
550  while( state!=st_quit && state!=st_stop ) {
551  if( my_server.my_slack>=0 ) {
552  my_client.process(j);
553  } else {
554  ipc_thread_monitor::cookie c;
555  // Prepare to wait
556  my_thread_monitor.prepare_wait(c);
557  // Check/set the invariant for sleeping
558  state = my_state;
559  if( state!=st_quit && state!=st_stop && my_server.try_insert_in_asleep_list(*this) ) {
560  if( my_server.my_n_thread > 1 ) my_server.release_active_thread();
561  my_thread_monitor.commit_wait(c);
562  my_server.propagate_chain_reaction();
563  } else {
564  // Invariant broken
565  my_thread_monitor.cancel_wait();
566  }
567  }
568  state = my_state;
569  }
570  my_client.cleanup(j);
571 
572  my_server.remove_server_ref();
573 }

575 inline bool ipc_worker::wake_or_launch() {
576  if( ( my_state==st_init && my_state.compare_and_swap( st_starting, st_init )==st_init ) ||
577  ( my_state==st_stop && my_state.compare_and_swap( st_starting, st_stop )==st_stop ) ) {
578  // after this point, remove_server_ref() must be done by created thread
579 #if USE_WINTHREAD
580  my_handle = ipc_thread_monitor::launch( thread_routine, this, my_server.my_stack_size, &this->my_index );
581 #elif USE_PTHREAD
582  {
583  affinity_helper fpa;
584  fpa.protect_affinity_mask( /*restore_process_mask=*/true );
585  my_handle = ipc_thread_monitor::launch( thread_routine, this, my_server.my_stack_size );
586  if( my_handle == 0 ) {
587  // Unable to create new thread for process
588  // However, this is expected situation for the use cases of this coordination server
589  state_t s = my_state.compare_and_swap( st_init, st_starting );
590  if (st_starting != s) {
591  // Do shutdown during startup. my_handle can't be released
592  // by start_shutdown, because my_handle value might be not set yet
593  // at time of transition from st_starting to st_quit.
594  __TBB_ASSERT( s==st_quit, NULL );
595  release_handle( my_handle, my_server.my_join_workers );
596  }
597  return false;
598  } else {
599  my_server.my_ref_count++;
600  }
601  // Implicit destruction of fpa resets original affinity mask.
602  }
603 #endif /* USE_PTHREAD */
604  state_t s = my_state.compare_and_swap( st_normal, st_starting );
605  if( st_starting!=s ) {
606  // Do shutdown during startup. my_handle can't be released
607  // by start_shutdown, because my_handle value might be not set yet
608  // at time of transition from st_starting to st_quit.
609  __TBB_ASSERT( s==st_quit, NULL );
610  release_handle( my_handle, my_server.my_join_workers );
611  }
612  }
613  else {
614  my_thread_monitor.notify();
615  }
616 
617  return true;
618 }

939 void ipc_server::wake_some( int additional_slack, int active_threads ) {
940  __TBB_ASSERT( additional_slack>=0, NULL );
941  ipc_worker* wakee[2];
942  ipc_worker **w = wakee;
943  {
944  asleep_list_mutex_type::scoped_lock lock(my_asleep_list_mutex);
945  while( active_threads>0 && my_asleep_list_root && w<wakee+2 ) {
946  if( additional_slack>0 ) {
947  if( additional_slack+my_slack<=0 ) // additional demand does not exceed surplus supply
948  break;
949  --additional_slack;
950  } else {
951  // Chain reaction; Try to claim unit of slack
952  int old;
953  do {
954  old = my_slack;
955  if( old<=0 ) goto done;
956  } while( my_slack.compare_and_swap( old-1, old )!=old );
957  }
958  // Pop sleeping worker to combine with claimed unit of slack
959  my_asleep_list_root = (*w++ = my_asleep_list_root)->my_next;
960  --active_threads;
961  }
962  if( additional_slack ) {
963  // Contribute our unused slack to my_slack.
964  my_slack += additional_slack;
965  }
966  }
967 done:
968  while( w>wakee ) {
969  if( !(*--w)->wake_or_launch() ) {
970  add_stop_thread();
971  do {
972  } while( !try_insert_in_asleep_list_forced(**w) );
973  release_active_thread();
974  }
975  }
976  while( active_threads ) {
977  release_active_thread();
978  --active_threads;
979  }
980 }
