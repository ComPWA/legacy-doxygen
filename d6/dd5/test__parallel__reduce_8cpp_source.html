72  ++FooBodyCount;

80  void join( FooBody& s ) {
81  ASSERT( s.forked==1, NULL );
82  ASSERT( this!=&s, NULL );
83  ASSERT( this==s.parent, NULL );
84  ASSERT( end==s.begin, NULL );
85  end = s.end;
86  sum += s.sum;
87  join_count += s.join_count + 1;
88  s.forked = 2;
89  }

91  for( size_t k=r.begin; k<r.end; ++k )
92  ++sum;
93  if( is_new ) {

95  begin = r.begin;
96  } else
97  ASSERT( end==r.begin, NULL );
98  end = r.end;
99  }

106 void Flog( int nthread, bool interference=false ) {
107  for (int mode = 0; mode < 4; mode++) {
108  tbb::tick_count T0 = tbb::tick_count::now();
109  long join_count = 0;
110  tbb::affinity_partitioner ap;
111  for( size_t i=0; i<=1000; ++i ) {
112  FooBody f;
113  f.sum = 0;
114  f.parent = NULL;
115  f.join_count = 0;
116  f.is_new = true;
117  f.forked = 0;
118  f.begin = ~size_t(0);
119  f.end = ~size_t(0);
120  ASSERT( FooBodyCount==1, NULL );
121  switch (mode) {
122  case 0:
123  tbb::parallel_reduce( MinimalRange(i), f );
124  break;
125  case 1:
126  tbb::parallel_reduce( MinimalRange(i), f, tbb::simple_partitioner() );
127  break;
128  case 2:
129  tbb::parallel_reduce( MinimalRange(i), f, tbb::auto_partitioner() );
130  break;
131  case 3:
132  tbb::parallel_reduce( MinimalRange(i), f, ap );
133  break;
134  }
135  join_count += f.join_count;
136  ASSERT( FooBodyCount==1, NULL );
137  ASSERT( f.sum==i, NULL );
138  ASSERT( f.begin==(i==0 ? ~size_t(0) : 0), NULL );
139  ASSERT( f.end==(i==0 ? ~size_t(0) : i), NULL );
140  }
141  tbb::tick_count T1 = tbb::tick_count::now();
142  REMARK("time=%g join_count=%ld ForkCount=%ld nthread=%d%s\n",

374 template <class Partitioner>
375 void TestDeterministicReductionFor() {
376  const int N = 1000;
377  const tbb::blocked_range<int> range(0, N);
378  typedef ReduceBody<RotOp> BodyType;
379  BodyType::result_type R1 =
380  parallel_deterministic_reduce_invoker<BodyType, Partitioner>::run(range);
381  for ( int i=0; i<100; ++i ) {
382  BodyType::result_type R2 =
383  parallel_deterministic_reduce_invoker<BodyType, Partitioner>::run(range);
384  ASSERT( R1 == R2, "parallel_deterministic_reduce behaves differently from run to run" );
385 #if __TBB_CPP11_LAMBDAS_PRESENT
386  typedef RotOp::Type Type;
387  Type R3 = parallel_deterministic_reduce_lambda_invoker<Type, Partitioner>::run(
388  range,
389  [](const tbb::blocked_range<int>& br, Type value) -> Type {
390  Harness::ConcurrencyTracker ct;
391  for ( int ii = br.begin(); ii != br.end(); ++ii ) {
392  RotOp op;
393  value = op(value, ii);
394  }
395  return value;
396  },
397  [](const Type& v1, const Type& v2) -> Type {
398  RotOp op;
399  return op.join(v1,v2);
400  }
401  );
402  ASSERT( R1 == R3, "lambda-based parallel_deterministic_reduce behaves differently from run to run" );
403 #endif /* LAMBDAS */
