49 class FillRNDPointsVector_buf {
50  pointVec_t &points;
51 public:
52  static const size_t grainSize = cfg::generateGrainSize;
53 
54  explicit FillRNDPointsVector_buf(pointVec_t& _points)
55  : points(_points) {}
56 
57  void operator()(const range_t& range) const {
58  util::rng the_rng(range.begin());
59  const size_t i_end = range.end();
60  size_t count = 0, j = 0;
61  point_t tmp_vec[grainSize];
62 
63  for(size_t i=range.begin(); i!=i_end; ++i) {
64  tmp_vec[j++] = util::GenerateRNDPoint<double>(count, the_rng, util::rng::max_rand);
65  }
66  //Here we have race condition. Elements being written to may be still under construction.
67  //For C++ 2003 it is workarounded by vector element type which default constructor does not touch memory,
68  //it being constructed on. See comments near default ctor of point class for more details.
69  //Strictly speaking it is UB.
70  //TODO: need to find more reliable/correct way
71  points.grow_to_at_least(range.end());
72  std::copy(tmp_vec, tmp_vec+j,points.begin()+range.begin());
73  }
74 };

76 void initialize(pointVec_t &points) {
77  //This function generate the same series of point on every call.
78  //Reproducibility is needed for benchmarking to produce reliable results.
79  //It is achieved through the following points:
80  // - FillRNDPointsVector_buf instance has its own local instance
81  // of random number generator, which in turn does not use any global data
82  // - tbb::simple_partitioner produce the same set of ranges on every call to
83  // tbb::parallel_for
84  // - local RNG instances are seeded by the starting indexes of corresponding ranges
85  // - grow_to_at_least() enables putting points into the resulting vector in deterministic order
86  // (unlike concurrent push_back or grow_by).
87 
88  // In the buffered version, a temporary storage for as much as grainSize elements
89  // is allocated inside the body. Since auto_partitioner may increase effective
90  // range size which would cause a crash, simple partitioner has to be used.
91  tbb::parallel_for(range_t(0, cfg::numberOfPoints, FillRNDPointsVector_buf::grainSize),
92  FillRNDPointsVector_buf(points), tbb::simple_partitioner());
93 }

95 class FindXExtremum {
96 public:
97  typedef enum {
98  minX, maxX
99  } extremumType;
100 
101  static const size_t grainSize = cfg::findExtremumGrainSize;
102 
103  FindXExtremum(const pointVec_t& points_, extremumType exType_)
104  : points(points_), exType(exType_), extrXPoint(points[0]) {}
105 
106  FindXExtremum(const FindXExtremum& fxex, tbb::split)
107  // Can run in parallel with fxex.operator()() or fxex.join().
108  // The data race reported by tools is harmless.
109  : points(fxex.points), exType(fxex.exType), extrXPoint(fxex.extrXPoint) {}
110 
111  void operator()(const range_t& range) {
112  const size_t i_end = range.end();
113  if(!range.empty()) {
114  for(size_t i = range.begin(); i != i_end; ++i) {
115  if(closerToExtremum(points[i])) {
116  extrXPoint = points[i];
117  }
118  }
119  }
120  }
121 
122  void join(const FindXExtremum &rhs) {
123  if(closerToExtremum(rhs.extrXPoint)) {
124  extrXPoint = rhs.extrXPoint;
125  }
126  }
127 
128  point_t extremeXPoint() {
129  return extrXPoint;
130  }
131 
132 private:
133  const pointVec_t &points;
134  const extremumType exType;
135  point_t extrXPoint;
136  bool closerToExtremum(const point_t &p) const {
137  switch(exType) {
138  case minX:
139  return p.x<extrXPoint.x; break;
140  case maxX:
141  return p.x>extrXPoint.x; break;
142  }
143  return false; // avoid warning
144  }
145 };

154 class SplitByCP_buf {
155  const pointVec_t &initialSet;
156  pointVec_t &reducedSet;
157  point_t p1, p2;
158  point_t farPoint;
159  double howFar;
160 public:
161  static const size_t grainSize = cfg::divideGrainSize;
162 
163  SplitByCP_buf( point_t _p1, point_t _p2,
164  const pointVec_t &_initialSet, pointVec_t &_reducedSet)
165  : p1(_p1), p2(_p2),
166  initialSet(_initialSet), reducedSet(_reducedSet),
167  howFar(0), farPoint(p1) {}
168 
169  SplitByCP_buf(SplitByCP_buf& sbcp, tbb::split)
170  : p1(sbcp.p1), p2(sbcp.p2),
171  initialSet(sbcp.initialSet), reducedSet(sbcp.reducedSet),
172  howFar(0), farPoint(p1) {}
173 
174  void operator()(const range_t& range) {
175  const size_t i_end = range.end();
176  size_t j = 0;
177  double cp;
178  point_t tmp_vec[grainSize];
179  for(size_t i = range.begin(); i != i_end; ++i) {
180  if( (initialSet[i] != p1) && (initialSet[i] != p2) ) {
181  cp = util::cross_product(p1, p2, initialSet[i]);
182  if(cp>0) {
183  tmp_vec[j++] = initialSet[i];
184  if(cp>howFar) {
185  farPoint = initialSet[i];
186  howFar = cp;
187  }
188  }
189  }
190  }
191 
192  appendVector(tmp_vec, j, reducedSet);
193  }
194 
195  void join(const SplitByCP_buf& rhs) {
196  if(rhs.howFar>howFar) {
197  howFar = rhs.howFar;
198  farPoint = rhs.farPoint;
199  }
200  }
201 
202  point_t farthestPoint() const {
203  return farPoint;
204  }
205 };

225 void divide_and_conquer(const pointVec_t &P, pointVec_t &H,
226  point_t p1, point_t p2) {
227  assert(P.size() >= 2);
228  pointVec_t P_reduced;
229  pointVec_t H1, H2;
230  point_t p_far = divide(P, P_reduced, p1, p2);
231  if (P_reduced.size()<2) {
232  H.push_back(p1);
233  appendVector(P_reduced, H);
234  }
235  else {
236  divide_and_conquer(P_reduced, H1, p1, p_far);
237  divide_and_conquer(P_reduced, H2, p_far, p2);
238 
239  appendVector(H1, H);
240  appendVector(H2, H);
241  }

244 void quickhull(const pointVec_t &points, pointVec_t &hull) {
245  if (points.size() < 2) {
246  appendVector(points, hull);
247  return;
248  }
249 
250  point_t p_maxx = extremum<FindXExtremum::maxX>(points);
251  point_t p_minx = extremum<FindXExtremum::minX>(points);
252 
253  pointVec_t H;
254 
255  divide_and_conquer(points, hull, p_maxx, p_minx);
256  divide_and_conquer(points, H, p_minx, p_maxx);
257 
258  appendVector(H, hull);
259 }
