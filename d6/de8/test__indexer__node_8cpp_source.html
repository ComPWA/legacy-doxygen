406 // TT must be arithmetic, and shouldn't wrap around for reasonable sizes of Count (which is now 150, and maxPorts is 10,
407 // so the max number generated right now is 1500 or so.) Source will generate a series of TT with value
408 // (init_val + (i-1)*addend) * my_mult, where i is the i-th invocation of the body. We are attaching addend
409 // source nodes to a indexer_port, and each will generate part of the numerical series the port is expecting
410 // to receive. If there is only one source node, the series order will be maintained; if more than one,
411 // this is not guaranteed.
412 // The manual specifies bodies can be assigned, so we can't hide the operator=.
413 template<typename TT>
414 class source_body {
415  TT my_mult;
416  int my_count;
417  int addend;
418 public:
419  source_body(TT multiplier, int init_val, int addto) : my_mult(multiplier), my_count(init_val), addend(addto) { }
420  bool operator()( TT &v) {
421  int lc = my_count;
422  v = my_mult * (TT)my_count;
423  my_count += addend;
424  return lc < Count;
425  }
426 };

456 template<int ELEM, typename INT>
457 class source_node_helper {
458 public:
459  typedef INT indexer_node_type;
460  typedef typename indexer_node_type::output_type TT;
461  typedef typename tbb::flow::tuple_element<ELEM-1,typename INT::tuple_types>::type IT;
462  typedef typename tbb::flow::source_node<IT> my_source_node_type;
463  static void print_remark() {
464  source_node_helper<ELEM-1,INT>::print_remark();
465  REMARK(", %s", name_of<IT>::name());
466  }
467  static void add_source_nodes(indexer_node_type &my_indexer, tbb::flow::graph &g, int nInputs) {
468  for(int i=0; i < nInputs; ++i) {
469  my_source_node_type *new_node = new my_source_node_type(g, source_body<IT>((IT)(ELEM+1), i, nInputs));
470  tbb::flow::make_edge(*new_node, tbb::flow::input_port<ELEM-1>(my_indexer));
471 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
472  ASSERT(new_node->successor_count() == 1, NULL);
473 #endif
474  all_source_nodes[ELEM-1][i] = (void *)new_node;
475  }
476 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
477  ASSERT(tbb::flow::input_port<ELEM-1>(my_indexer).predecessor_count() == (size_t)nInputs, NULL);
478 #endif
479  // add the next source_node
480  source_node_helper<ELEM-1, INT>::add_source_nodes(my_indexer, g, nInputs);
481  }
482  static void check_value(TT &v) {
483  if(v.tag() == ELEM-1) {
484  int ival = getval_helper<ELEM,INT>::get_integer_val(v);
485  ASSERT(!(ival%(ELEM+1)), NULL);
486  ival /= (ELEM+1);
487  ASSERT(!outputCheck[ELEM-1][ival], NULL);
488  outputCheck[ELEM-1][ival] = true;
489  }
490  else {
491  source_node_helper<ELEM-1,INT>::check_value(v);
492  }
493  }

502  }
