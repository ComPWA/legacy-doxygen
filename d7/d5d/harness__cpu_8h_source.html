33 /* Returns 0 if not implemented on platform. */
34 static double GetCPUUserTime() {
35 #if __TBB_WIN8UI_SUPPORT
36  return 0;
37 #elif _WIN32
38  FILETIME my_times[4];
39  bool status = GetProcessTimes(GetCurrentProcess(), my_times, my_times+1, my_times+2, my_times+3)!=0;
40  ASSERT( status, NULL );
41  LARGE_INTEGER usrtime;
42  usrtime.LowPart = my_times[3].dwLowDateTime;
43  usrtime.HighPart = my_times[3].dwHighDateTime;
44  return double(usrtime.QuadPart)*1E-7;
45 #else
46  // Generic UNIX, including __APPLE__
47 
48  // On Linux, there is no good way to get CPU usage info for the current process:
49  // getrusage(RUSAGE_SELF, ...) that is used now only returns info for the calling thread;
50  // getrusage(RUSAGE_CHILDREN, ...) only counts for finished children threads;
51  // tms_utime and tms_cutime got with times(struct tms*) are equivalent to the above items;
52  // finally, /proc/self/task/<task_id>/stat doesn't exist on older kernels
53  // and it isn't quite convenient to read it for every task_id.
54 
55  struct rusage resources;
56  bool status = getrusage(RUSAGE_SELF, &resources)==0;
57  ASSERT( status, NULL );
58  return (double(resources.ru_utime.tv_sec)*1E6 + double(resources.ru_utime.tv_usec))*1E-6;
59 #endif
60 }

65 // The resolution of GetCPUUserTime is 10-15 ms or so; waittime should be a few times bigger.
66 const double WAITTIME = 0.1; // in seconds, i.e. 100 ms
67 const double THRESHOLD = WAITTIME/100;
68 
69 static void TestCPUUserTime( int nthreads, int nactive = 1 ) {
70  // The test will always pass on Linux; read the comments in GetCPUUserTime for details
71  // Also it will not detect spinning issues on systems with only one processing core.
72 
73  int nworkers = nthreads-nactive;
74  if( !nworkers ) return;
75  double lastusrtime = GetCPUUserTime();
76  if( !lastusrtime ) return;
77 
78  static double minimal_waittime = WAITTIME,
79  maximal_waittime = WAITTIME * 10;
80  double usrtime_delta;
81  double waittime_delta;
82  tbb::tick_count stamp = tbb::tick_count::now();
83  volatile intptr_t k = (intptr_t)&usrtime_delta;
84  // wait for GetCPUUserTime update
85  while( (usrtime_delta=GetCPUUserTime()-lastusrtime) < THRESHOLD ) {
86  for ( int i = 0; i < 1000; ++i ) ++k; // do fake work without which user time can stall
87  if ( (waittime_delta = (tbb::tick_count::now()-stamp).seconds()) > maximal_waittime ) {
88  REPORT( "Warning: %.2f sec elapsed but user mode time is still below its threshold (%g < %g)\n",
89  waittime_delta, usrtime_delta, THRESHOLD );
90  break;
91  }
92  }
93  lastusrtime += usrtime_delta;
94 
95  // Wait for workers to go sleep
96  stamp = tbb::tick_count::now();
97  while( ((waittime_delta=(tbb::tick_count::now()-stamp).seconds()) < minimal_waittime)
98  || ((usrtime_delta=GetCPUUserTime()-lastusrtime) < THRESHOLD) )
99  {
100  for ( int i = 0; i < 1000; ++i ) ++k; // do fake work without which user time can stall
101  if ( waittime_delta > maximal_waittime ) {
102  REPORT( "Warning: %.2f sec elapsed but GetCPUUserTime reported only %g sec\n", waittime_delta, usrtime_delta );
103  break;
104  }
105  }
106 
107  // Test that all workers sleep when no work.
108  while( nactive>1 && usrtime_delta-nactive*waittime_delta<0 ) {
109  // probably the number of active threads was mispredicted
110  --nactive; ++nworkers;
111  }
112  double avg_worker_usrtime = (usrtime_delta-nactive*waittime_delta)/nworkers;
113 
114  if( avg_worker_usrtime > waittime_delta/2 )
115  REPORT( "ERROR: %d worker threads are spinning; waittime: %g; usrtime: %g; avg worker usrtime: %g\n",
116  nworkers, waittime_delta, usrtime_delta, avg_worker_usrtime);
117  else
118  REMARK("%d worker threads; waittime: %g; usrtime: %g; avg worker usrtime: %g\n",
119  nworkers, waittime_delta, usrtime_delta, avg_worker_usrtime);
120 }
