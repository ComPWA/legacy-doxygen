73 color shader(ray * incident) {
74  color col, diffuse, phongcol;
75  vector N, L, hit;
76  ray shadowray;
77  flt inten, t, Llen;
78  object * obj;
79  int numints, i;
80  point_light * li;
81 
82 
83  numints=closest_intersection(&t, &obj, incident->intstruct);
84  /* find the number of intersections */
85  /* and return the closest one. */
86 
87  if (numints < 1) {
88  /* if there weren't any object intersections then return the */
89  /* background color for the pixel color. */
90  return incident->scene->background;
91  }
92 
93  if (obj->tex->islight) { /* if the current object is a light, then we */
94  return obj->tex->col; /* will only use the objects ambient color */
95  }
96 
97  RAYPNT(hit, (*incident), t) /* find the point of intersection from t */
98  obj->methods->normal(obj, &hit, incident, &N); /* find the surface normal */
99 
100  /* execute the object's texture function */
101  col = obj->tex->texfunc(&hit, obj->tex, incident);
102 
103  diffuse.r = 0.0;
104  diffuse.g = 0.0;
105  diffuse.b = 0.0;
106  phongcol = diffuse;
107 
108  if ((obj->tex->diffuse > 0.0) || (obj->tex->phong > 0.0)) {
109  for (i=0; i<numlights; i++) { /* loop for light contributions */
110  li=lightlist[i]; /* set li=to the current light */
111  VSUB(li->ctr, hit, L) /* find the light vector */
112 
113  /* calculate the distance to the light from the hit point */
114  Llen = sqrt(L.x*L.x + L.y*L.y + L.z*L.z) + EPSILON;
115 
116  L.x /= Llen; /* normalize the light direction vector */
117  L.y /= Llen;
118  L.z /= Llen;
119 
120  VDOT(inten, N, L) /* light intensity */
121 
122  /* add in diffuse lighting for this light if we're facing it */
123  if (inten > 0.0) {
124  /* test for a shadow */
125  shadowray.intstruct = incident->intstruct;
126  shadowray.flags = RT_RAY_SHADOW | RT_RAY_BOUNDED;
127  incident->serial++;
128  shadowray.serial = incident->serial;
129  shadowray.mbox = incident->mbox;
130  shadowray.o = hit;
131  shadowray.d = L;
132  shadowray.maxdist = Llen;
133  shadowray.s = hit;
134  shadowray.e = li->ctr;
135  shadowray.scene = incident->scene;
136  reset_intersection(incident->intstruct);
137  intersect_objects(&shadowray);
138 
139  if (!shadow_intersection(incident->intstruct, Llen)) {
140  /* XXX now that opacity is in the code, have to be more careful */
141  ColorAddS(&diffuse, &li->tex->col, inten);
142 
143  /* phong type specular highlights */
144  if (obj->tex->phong > 0.0) {
145  flt phongval;
146  phongval = shade_phong(incident, &hit, &N, &L, obj->tex->phongexp);
147  if (obj->tex->phongtype)
148  ColorAddS(&phongcol, &col, phongval);
149  else
150  ColorAddS(&phongcol, &(li->tex->col), phongval);
151  }
152  }
153  }
154  }
155  }
156 
157  ColorScale(&diffuse, obj->tex->diffuse);
158 
159  col.r *= (diffuse.r + obj->tex->ambient); /* do a product of the */
160  col.g *= (diffuse.g + obj->tex->ambient); /* diffuse intensity with */
161  col.b *= (diffuse.b + obj->tex->ambient); /* object color + ambient */
162 
163  if (obj->tex->phong > 0.0) {
164  ColorAccum(&col, &phongcol);
165  }
166 
167  /* spawn reflection rays if necessary */
168  /* note: this will overwrite the old intersection list */
169  if (obj->tex->specular > 0.0) {
170  color specol;
171  specol = shade_reflection(incident, &hit, &N, obj->tex->specular);
172  ColorAccum(&col, &specol);
173  }
174 
175  /* spawn transmission rays / refraction */
176  /* note: this will overwrite the old intersection list */
177  if (obj->tex->opacity < 1.0) {
178  color transcol;
179  transcol = shade_transmission(incident, &hit, 1.0 - obj->tex->opacity);
180  ColorAccum(&col, &transcol);
181  }
182 
183  return col; /* return the color of the shaded pixel... */
184 }

187 color shade_reflection(ray * incident, vector * hit, vector * N, flt specular) {
188  ray specray;
189  color col;
190  vector R;
191 
192  VAddS(-2.0 * (incident->d.x * N->x +
193  incident->d.y * N->y +
194  incident->d.z * N->z), N, &incident->d, &R);
195 
196  specray.intstruct=incident->intstruct; /* what thread are we */
197  specray.depth=incident->depth - 1; /* go up a level in recursion depth */
198  specray.flags = RT_RAY_REGULAR; /* infinite ray, to start with */
199  specray.serial = incident->serial + 1; /* next serial number */
200  specray.mbox = incident->mbox;
201  specray.o=*hit;
202  specray.d=R; /* reflect incident ray about normal */
203  specray.o=Raypnt(&specray, EPSILON); /* avoid numerical precision bugs */
204  specray.maxdist = FHUGE; /* take any intersection */
205  specray.scene=incident->scene; /* global scenedef info */
206  col=trace(&specray); /* trace specular reflection ray */
207 
208  incident->serial = specray.serial; /* update the serial number */
209 
210  ColorScale(&col, specular);
211 
212  return col;
213 }

239 flt shade_phong(ray * incident, vector * hit,
240  vector * N, vector * L, flt specpower){
241  vector H, V;
242  flt inten;
243 
244  V = incident->d;
245  VScale(&V, -1.0);
246  VAdd(&V, L, &H);
247  VScale(&H, 0.5);
248  VNorm(&H);
249  inten = VDot(N, &H);
250  if (inten > 0.0)
251  inten = pow(inten, specpower);
252  else
253  inten = 0.0;
254 
255  return inten;
256 }
