86 void FireUpJobs( MyServer& server, MyClient& client, int n_thread, int n_extra, Checker* checker ) {
87  REMARK("client %d: calling adjust_job_count_estimate(%d)\n", client.client_id(),n_thread);
88  // Exercise independent_thread_number_changed, even for zero values.
89  server.independent_thread_number_changed( n_extra );
90 #if _WIN32||_WIN64
91  ::rml::server::execution_resource_t me;
92  server.register_master( me );
93 #endif /* _WIN32||_WIN64 */
94  // Experiments indicate that when oversubscribing, the main thread should wait a little
95  // while for the RML worker threads to do some work.
96  if( checker ) {
97  // Give RML time to respond to change in number of threads.
98  Harness::Sleep(1);
99  for( int k=0; k<n_thread; ++k )
100  client.job_array[k].processing_count = 0;
101  }
102  //close the gate to keep worker threads from returning to RML until a snapshot is taken
103  client.close_the_gate();
104  server.adjust_job_count_estimate( n_thread );
105  int n_used = 0;
106  if( checker ) {
107  Harness::Sleep(100);
108  for( int k=0; k<n_thread; ++k )
109  if( client.job_array[k].processing_count )
110  ++n_used;
111  }
112  // open the gate
113  client.open_the_gate();
114  // Logic further below presumes that jobs never starve, so undo previous call
115  // to independent_thread_number_changed before waiting on those jobs.
116  server.independent_thread_number_changed( -n_extra );
117  REMARK("client %d: wait for each job to be processed at least once\n",client.client_id());
118  // Calculate the number of jobs that are expected to get threads.
119  int expected = n_thread;
120  // Wait for expected number of jobs to be processed.
121 #if RML_USE_WCRM
122  int default_concurrency = server.default_concurrency();
123  if( N_TestConnections>0 ) {
124  if( default_concurrency+1>=8 && n_thread<=3 && N_TestConnections<=3 && (default_concurrency/int(N_TestConnections)-1)>=n_thread ) {
125 #endif /* RML_USE_WCRM */
126  for(;;) {
127  int n = 0;
128  for( int k=0; k<n_thread; ++k )
129  if( client.job_array[k].processing_count!=0 )
130  ++n;
131  if( n>=expected ) break;
132  server.yield();
133  }
134 #if RML_USE_WCRM
135  } else if( n_thread>0 ) {
136  for( int m=0; m<20; ++m ) {
137  int n = 0;
138  for( int k=0; k<n_thread; ++k )
139  if( client.job_array[k].processing_count!=0 )
140  ++n;
141  if( n>=expected ) break;
142  Harness::Sleep(1);
143  }
144  }
145  }
146 #endif /* RML_USE_WCRM */
147  server.adjust_job_count_estimate(-n_thread);
148 #if _WIN32||_WIN64
149  server.unregister_master( me );
150 #endif
151  // Give RML some time to respond
152  if( checker ) {
153  Harness::Sleep(1);
154  checker->check_number_of_threads_delivered( n_used, n_thread, n_extra );
155  }
156 }
