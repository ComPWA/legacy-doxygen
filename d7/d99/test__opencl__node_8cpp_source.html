91  cl_uint num_of_platforms = 0;
92  clGetPlatformIDs(0, 0, &num_of_platforms);
93  cl_platform_id* platforms = new cl_platform_id[num_of_platforms];

95  // Get IDs for all platforms:
96  clGetPlatformIDs(num_of_platforms, platforms, 0);
97

99  cl_uint selected_platform_index = 0;
100  cl_platform_id platform = platforms[selected_platform_index];
101 
102  // Count the number of plaform devices and compare with selector list
103  cl_uint device_count;
104  clGetDeviceIDs(platform, CL_DEVICE_TYPE_ALL, 0, NULL, &device_count);
105  // It should be the same
106  ASSERT(device_count == devices.size(), "Default device filter returned not all devices from the platform");
107 
108  // Retrieve device ids from the platform
109  cl_device_id* queuered_devices = (cl_device_id*) malloc(sizeof(cl_device_id) * device_count);
110  clGetDeviceIDs(platform, CL_DEVICE_TYPE_ALL, device_count, queuered_devices, NULL);
111 
112  // Compare retrieved device ids with defaults

114  cl_device_id searched_id = queuered_devices[i];

116  tbb::flow::opencl_device_list::const_iterator it = std::find_if(devices.cbegin(), devices.cend(),

607

609  REPORT_ONCE( "Known issue: the system does not have several devices in one platform\n" );

612  case ONE_DEVICE:
613  {

627 
628  return dl;
629  }
