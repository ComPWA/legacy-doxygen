90 void SerialConvolve( T c[], const T a[], int m, const T b[], int n ) {
91  for( int i=0; i<m+n-1; ++i ) {
92  int start = i<n ? 0 : i-n+1;

94  T sum = 0;
95  for( int j=start; j<finish; ++j )
96  sum += a[j]*b[i-j];
97  c[i] = sum;
98  }
