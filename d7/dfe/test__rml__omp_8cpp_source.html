106 void FireUpJobs( MyServer& server, MyClient& client, int max_thread, int n_extra, Checker* checker ) {
107  ASSERT( max_thread>=0, NULL );
108 #if _WIN32||_WIN64
109  ::rml::server::execution_resource_t me;
110  server.register_master( me );
111 #endif /* _WIN32||_WIN64 */
112  client.server = &server;
113  MyTeam team(server,size_t(max_thread));
114  MyServer::size_type n_thread = 0;
115  for( int iteration=0; iteration<4; ++iteration ) {
116  for( size_t i=0; i<team.max_thread; ++i )
117  team.info[i].ran = false;
118  switch( iteration ) {
119  default:
120  n_thread = int(max_thread);
121  break;
122  case 1:
123  // No change in number of threads
124  break;
125  case 2:
126  // Decrease number of threads.
127  n_thread = int(max_thread)/2;
128  break;
129  // Case 3 is same code as the default, but has effect of increasing the number of threads.
130  }
131  team.barrier = 0;
132  REMARK("client %d: server.run with n_thread=%d\n", client.client_id(), int(n_thread) );
133  server.independent_thread_number_changed( n_extra );
134  if( checker ) {
135  // Give RML time to respond to change in number of threads.
136  Harness::Sleep(1);
137  }
138  int n_delivered = server.try_increase_load( n_thread, StrictTeam );
139  ASSERT( !StrictTeam || n_delivered==int(n_thread), "server failed to satisfy strict request" );
140  if( n_delivered<0 ) {
141  REMARK( "client %d: oversubscription occurred (by %d)\n", client.client_id(), -n_delivered );
142  server.independent_thread_number_changed( -n_extra );
143  n_delivered = 0;
144  } else {
145  team.n_thread = n_delivered;
146  ::rml::job* job_array[JobArraySize];
147  job_array[n_delivered] = (::rml::job*)intptr_t(-1);
148  server.get_threads( n_delivered, &team, job_array );
149  __TBB_ASSERT( job_array[n_delivered]== (::rml::job*)intptr_t(-1), NULL );
150  for( int i=0; i<n_delivered; ++i ) {
151  MyJob* j = static_cast<MyJob*>(job_array[i]);
152  int s = j->state;
153  ASSERT( s==MyJob::idle||s==MyJob::busy, NULL );
154  }
155  server.independent_thread_number_changed( -n_extra );
156  REMARK("client %d: team size is %d\n", client.client_id(), n_delivered);
157  if( checker ) {
158  checker->check_number_of_threads_delivered( n_delivered, n_thread, n_extra );
159  }
160  // Protocol requires that master wait until workers have called "done_processing"
161  while( team.barrier!=n_delivered ) {
162  ASSERT( team.barrier>=0, NULL );
163  ASSERT( team.barrier<=n_delivered, NULL );
164  __TBB_Yield();
165  }
166  REMARK("client %d: team completed\n", client.client_id() );
167  for( int i=0; i<n_delivered; ++i ) {
168  ASSERT( team.info[i].ran, "thread on team allegedly delivered, but did not run?" );
169  }
170  }
171  for( MyServer::size_type i=n_delivered; i<MyServer::size_type(max_thread); ++i ) {
172  ASSERT( !team.info[i].ran, "thread on team ran with illegal index" );
173  }
174  }
175 #if _WIN32||_WIN64
176  server.unregister_master( me );
177 #endif
178 }
