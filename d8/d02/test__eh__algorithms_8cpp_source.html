163  Partitioner partitioner;
164  for( int i=0; i<2; ++i ) {
165  ResetGlobals();

170  SimpleParReduceBody<ParForBody> rb;
171  tbb::parallel_reduce( range_type(0, FLAT_RANGE, FLAT_GRAIN), rb, partitioner );
172  }
173  CATCH_AND_ASSERT();
174  // two cases: g_SolitaryException and !g_SolitaryException
175  // 1) g_SolitaryException: only one thread actually threw. There is only one context, so the exception
176  // (when caught) will cause that context to be cancelled. After this event, there may be one or
177  // more threads which are "in-flight", up to g_NumThreads, but no more will be started. The threads,
178  // when they start, if they see they are cancelled, TGCCancelled is incremented.
179  // 2) !g_SolitaryException: more than one thread can throw. The number of threads that actually
180  // threw is g_MasterExecutedThrow if only the master is allowed, else g_NonMasterExecutedThrow.
181  // Only one context, so TGCCancelled should be <= g_NumThreads.
182  //
183  // the reasoning is similar for nested algorithms in a single context (Test2).
184  //
185  // If a thread throws in a context, more than one subsequent task body may see the
186  // cancelled state (if they are scheduled before the state is propagated.) this is
187  // infrequent, but it occurs. So what was to be an assertion must be a remark.
188  ASSERT( g_TGCCancelled <= g_NumThreads, "Too many tasks ran after exception thrown");

191  ASSERT(g_CurExecuted <= g_ExecutedAtLastCatch + g_NumThreads, "Too many tasks survived exception");
192  if ( g_SolitaryException ) {
193  ASSERT(g_NumExceptionsCaught == 1, "No try_blocks in any body expected in this test");
194  ASSERT(g_NumExceptionsCaught == (g_ExceptionInMaster ? g_MasterExecutedThrow : g_NonMasterExecutedThrow),
195  "Not all throws were caught");
196  ASSERT(g_ExecutedAtFirstCatch == g_ExecutedAtLastCatch, "Too many exceptions occurred");
197  }
198  else {
199  ASSERT(g_NumExceptionsCaught >= 1, "No try blocks in any body expected in this test");
