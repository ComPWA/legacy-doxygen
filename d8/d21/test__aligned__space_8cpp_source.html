50 template<typename U, size_t N>
51 void TestAlignedSpaceN() {
52  typedef Minimal<U> T;
53  struct {
55  char pad;
56  tbb::aligned_space<T ,N> space;
57  } x;
58  AssertSameType( static_cast< T *>(0), x.space.begin() );
59  AssertSameType( static_cast< T *>(0), x.space.end() );
60  ASSERT( reinterpret_cast<void *>(x.space.begin())==reinterpret_cast< void *>(&x.space), NULL );
61  ASSERT( x.space.end()-x.space.begin()==N, NULL );
62  ASSERT( reinterpret_cast<void *>(x.space.begin())>=reinterpret_cast< void *>(&x.space), NULL );
63  ASSERT( x.space.end()<=reinterpret_cast< T *>(&x.space+1), NULL );
64  // Though not required, a good implementation of aligned_space<T,N> does not use any more space than a T[N].
65  SpaceWasted |= sizeof(x.space)!=sizeof(T)*N;
66  for( size_t k=1; k<N; ++k )
67  AssignToCheckAlignment( x.space.begin()[k-1], x.space.begin()[k] );
68 }
