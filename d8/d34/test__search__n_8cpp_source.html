60  const std::size_t max_n1 = 100000;
61  const T value = T(1);
62  for (std::size_t n1 = 0; n1 <= max_n1; n1 = n1 <= 16 ? n1 + 1 : size_t(3.1415 * n1))
63  {
64  std::size_t sub_n[] = {0, 1, 3, n1, (n1 * 10) / 8};
65  std::size_t res[] = {0, 1, n1 / 2, n1};
66  for (auto n2 : sub_n)
67  {
68  // Some of standard libraries return "first" in this case. We return "last" according to the standard
69  if (n2 == 0)
70  {
71  continue;
72  }
73  for (auto r : res)
74  {
75  Sequence<T> in(n1, [n1](std::size_t k) { return T(0); });
76  std::size_t i = r, isub = 0;
77  for (; i < n1 & isub < n2; ++i, ++isub)
78  in[i] = value;
79 
80  invoke_on_all_policies(test_one_policy(), in.begin(), in.begin() + n1, n2, value, std::equal_to<T>());
81  invoke_on_all_policies(test_one_policy(), in.cbegin(), in.cbegin() + n1, n2, value, std::equal_to<T>());
82  }
83  }
84  }
