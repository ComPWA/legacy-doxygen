86 LRESULT CALLBACK InternalWndProc(HWND hwnd, UINT iMsg, WPARAM wParam, LPARAM lParam)
87 {
88  switch (iMsg)
89  {
90  case WM_MOVE:
91  // Check to make sure our window exists before we tell it to repaint.
92  // This will fail the first time (while the window is being created).
93  if (hwnd) {
94  InvalidateRect(hwnd, NULL, FALSE);
95  UpdateWindow(hwnd);
96  }
97  return 0L;
98 
99  case WM_SIZE:
100  case WM_PAINT:
101  if( g_video->running && g_video->updating ) {
102  DrawBitmap();
103  Sleep(0);
104  }
105  break;
106  // Process all mouse and keyboard events
107  case WM_LBUTTONDOWN: mouse( 1, lParam ); break;
108  case WM_LBUTTONUP: mouse(-1, lParam ); break;
109  case WM_RBUTTONDOWN: mouse( 2, lParam ); break;
110  case WM_RBUTTONUP: mouse(-2, lParam ); break;
111  case WM_MBUTTONDOWN: mouse( 3, lParam ); break;
112  case WM_MBUTTONUP: mouse(-3, lParam ); break;
113  case WM_CHAR: g_video->on_key( (int)wParam); break;
114 
115  // some useless stuff
116  case WM_ERASEBKGND: return 1; // keeps erase-background events from happening, reduces chop
117  case WM_DISPLAYCHANGE: return 0;
118 
119  // Now, shut down the window...
120  case WM_DESTROY: PostQuitMessage(0); return 0;
121  }
122  // call user defined proc, if exists
123  return g_pUserProc? g_pUserProc(hwnd, iMsg, wParam, lParam) : DefWindowProc(hwnd, iMsg, wParam, lParam);
124 }

140  // Create a Direct2D render target.
141  if (SUCCEEDED(hr) && !m_pRenderTarget){
142  RECT rc;
143  GetClientRect(g_hAppWnd, &rc);
144 
145  bitmapSize = D2D1::SizeU(
146  rc.right - rc.left,
147  rc.bottom - rc.top
148  );
149 
150  hr = m_pD2DFactory->CreateHwndRenderTarget(
151  D2D1::RenderTargetProperties(),
152  D2D1::HwndRenderTargetProperties(g_hAppWnd, bitmapSize),
153  &m_pRenderTarget
154  );
155  if (SUCCEEDED(hr) && !m_pBitmap){
156  D2D1_PIXEL_FORMAT pixelFormat = D2D1::PixelFormat(
157  DXGI_FORMAT_B8G8R8A8_UNORM,
158  D2D1_ALPHA_MODE_IGNORE
159  );
160  D2D1_BITMAP_PROPERTIES bitmapProperties;
161  bitmapProperties.pixelFormat = pixelFormat;
162  m_pRenderTarget->GetDpi( &bitmapProperties.dpiX, &bitmapProperties.dpiY );
163  m_pRenderTarget->CreateBitmap(bitmapSize,bitmapProperties,&m_pBitmap);
164  m_pRenderTarget->DrawBitmap(m_pBitmap);
