211  void check_allocate( typename A::pointer array[], size_t i, size_t t ) const
212  {
213  ASSERT(array[i] == 0, NULL);
214  size_t size = i * (i&3);
215  array[i] = i&1 ? a.allocate(size, array[i>>3]) : a.allocate(size);
216  ASSERT(array[i] != 0, "allocator returned null");
217  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[i]));
218  for( size_t j=0; j<size*sizeof(typename A::value_type); ++j ) {
219  if(is_zero_filling<typename A::template rebind<void>::other>::value)
220  ASSERT( !s[j], NULL);
221  s[j] = PseudoRandomValue(i, t);
222  }
223  }

225  void check_deallocate( typename A::pointer array[], size_t i, size_t t ) const
226  {
227  ASSERT(array[i] != 0, NULL);
228  size_t size = i * (i&3);
229  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[i]));
230  for( size_t j=0; j<size*sizeof(typename A::value_type); ++j )
231  ASSERT( s[j] == PseudoRandomValue(i, t), "Thread safety test failed" );
232  a.deallocate(array[i], size);
233  array[i] = 0;
234  }

236  void operator()( size_t thread_id ) const {
237  typename A::pointer array[256];
238 
239  for( size_t k=0; k<256; ++k )
240  array[k] = 0;

242  size_t i = static_cast<unsigned char>(PseudoRandomValue(k,thread_id));
