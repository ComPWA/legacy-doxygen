47 class WorkEmulator: NoAssign {
48  M& m_mutex;
49  static volatile size_t s_anchor;
50 public:
51  void operator()( tbb::blocked_range<size_t>& range ) const {
52  for( size_t i=range.begin(); i!=range.end(); ++i ) {
53  typename M::scoped_lock lock(m_mutex);
54  for ( size_t j = 0; j!=range.end(); ++j )
55  s_anchor = (s_anchor - i) / 2 + (s_anchor + j) / 2;
56  }
57  }
58  WorkEmulator( M& mutex ) : m_mutex(mutex) {}
