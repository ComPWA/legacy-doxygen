163 std::pair<double, double>
164 pawianHelicityAngles(std::vector<Vector4<double>> levels) {
165 
166  if (levels.size() < 3) {
167  levels.insert(
168  levels.begin(),
169  Vector4<double>(std::sqrt(levels[0].Mass2() + 1.0), 0., 0., -1.0));
170  }
171  if (levels.size() < 4) {
172  levels.insert(levels.begin(), Vector4<double>(0., 0., 0., -2.0));
173  }
174 
175  Vector4<double> result = levels[3];
176  Vector4<double> refTrafo = levels[1];
177  Vector4<double> refRecoilTrafo = levels[0] - levels[1];
178  Vector4<double> mother = levels[2];
179 
180  // boost all vectors into the mother rest frame
181  result.Boost(mother);
182  refTrafo.Boost(mother);
183  refRecoilTrafo.Boost(mother);
184 
185  // rotate vectors so that refRecoilTrafo moves in the negative direction of
186  // the z-axis
187  result.RotateZ(-refTrafo.Phi());
188  result.RotateY(TMath::Pi() - refTrafo.Theta());
189 
190  refRecoilTrafo.RotateZ(-refTrafo.Phi());
191  refRecoilTrafo.RotateY(TMath::Pi() - refTrafo.Theta());
192 
193  // rotate around the z-axis so that refRecoil lies in the x-z plain
194  result.RotateZ(M_PI - refRecoilTrafo.Phi());
195  return std::make_pair(result.CosTheta(), result.Phi());
196 }

199 double evtGenDecayAngle(const Vector4<double> &p, const Vector4<double> &q,
200  const Vector4<double> &d) {
201  double pd = p * d;
202  double pq = p * q;
203  double qd = q * d;
204  double mp2 = p.Mass2();
205  double mq2 = q.Mass2();
206  double md2 = d.Mass2();
207 
208  double cost = (pd * mq2 - pq * qd) /
209  sqrt((pq * pq - mq2 * mp2) * (qd * qd - mq2 * md2));
210 
211  return cost;
212 }

214 double mag2r3(const Vector4<double> &p0, const Vector4<double> &p1) {
215  return std::pow(p0 * p1, 2) / p0.Mass2() - p1.Mass2();

218 // Calculate the 3-d dot product of 4-vectors p1 and p2 in the rest frame of

220 double dotr3(const Vector4<double> &p0, const Vector4<double> &p1,
221  const Vector4<double> &p2) {
222  return 1 / p0.Mass2() * (p0 * p1) * (p0 * p2) - p1 * p2;

225 Tensor<double> dual(const Tensor<double> &t2) {
226  Tensor<double> temp(2);
227  temp(0, 0) = 0.0;
228  temp(1, 1) = 0.0;
229  temp(2, 2) = 0.0;
230  temp(3, 3) = 0.0;
231 
232  temp(0, 1) = t2(3, 2) - t2(2, 3);
233  temp(0, 2) = -t2(3, 1) + t2(1, 3);
234  temp(0, 3) = t2(2, 1) - t2(1, 2);
235 
236  temp(1, 2) = -t2(3, 0) + t2(0, 3);
237  temp(1, 3) = t2(2, 0) - t2(0, 2);
238 
239  temp(2, 3) = -t2(1, 0) + t2(0, 1);
240 
241  temp(1, 0) = -temp(0, 1);
242  temp(2, 0) = -temp(0, 2);
243  temp(3, 0) = -temp(0, 3);
244 
245  temp(2, 1) = -temp(1, 2);
246  temp(3, 1) = -temp(1, 3);
247 
248  temp(3, 2) = -temp(2, 3);
249  return temp;
250 }

252 // Calculate ( \vec{p1} cross \vec{p2} ) \cdot \vec{p3} in rest frame of object
253 double scalartripler3(const Vector4<double> &p0, const Vector4<double> &p1,
254  const Vector4<double> &p2, const Vector4<double> &p3) {
255  Tensor<double> l = dual(p0 % p1) * p2;
256  return -1.0 / p0.Mass() * (l * p3);

259 // Calculate phi using the given 4 vectors (all in the same frame)
260 double evtGenDecayAnglePhi(const Vector4<double> &z, const Vector4<double> &p,
261  const Vector4<double> &q, const Vector4<double> &d) {
262  double eq = (p * q) / p.Mass();
263  double ed = (p * d) / p.Mass();
264  double mq = q.Mass();
265  double q2 = mag2r3(p, q);
266  double qd = dotr3(p, q, d);
267  double zq = dotr3(p, z, q);
268  double zd = dotr3(p, z, d);
269  double alpha = (eq - mq) / (q2 * mq) * qd - ed / mq;
270 
271  double y = scalartripler3(p, z, q, d) + alpha * scalartripler3(p, z, q, q);
272  double x = (zq * (qd + alpha * q2) - q2 * (zd + alpha * zq)) / std::sqrt(q2);
273 
274  double phi = std::atan2(y, x);
275 
276  return phi;
277 }
