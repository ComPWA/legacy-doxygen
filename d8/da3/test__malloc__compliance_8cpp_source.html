358 void InvariantDataRealloc(bool aligned, size_t maxAllocSize, bool checkData)
359 {
360  Harness::FastRandom fastRandom(1);
361  size_t size = 0, start = 0;
362  char *ptr = NULL,
363  // master to create copies and compare ralloc result against it
364  *master = (char*)Tmalloc(2*maxAllocSize);
365 
366  ASSERT(master, NULL);
367  ASSERT(!(2*maxAllocSize%sizeof(unsigned short)),
368  "The loop below expects that 2*maxAllocSize contains sizeof(unsigned short)");
369  for (size_t k = 0; k<2*maxAllocSize; k+=sizeof(unsigned short))
370  *(unsigned short*)(master+k) = fastRandom.get();
371 
372  for (int i=0; i<100; i++) {
373  // don't want sizeNew==0 here
374  const size_t sizeNew = fastRandom.get() % (maxAllocSize-1) + 1;
375  char *ptrNew = aligned?
376  (char*)Taligned_realloc(ptr, sizeNew, choose_random_alignment())
377  : (char*)Trealloc(ptr, sizeNew);
378  ASSERT(ptrNew, NULL);
379  // check that old data not changed
380  if (checkData)
381  ASSERT(!memcmp(ptrNew, master+start, min(size, sizeNew)), "broken data");
382 
383  // prepare fresh data, copying them from random position in master
384  size = sizeNew;
385  ptr = ptrNew;
386  if (checkData) {
387  start = fastRandom.get() % maxAllocSize;
388  memcpy(ptr, master+start, size);
389  }
390  }
391  if (aligned)
392  Taligned_realloc(ptr, 0, choose_random_alignment());
393  else
394  Trealloc(ptr, 0);
395  Tfree(master);
396 }
