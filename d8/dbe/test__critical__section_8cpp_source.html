129 RunOneCriticalSectionTest(int nThreads, int csWorkRatio, bool test_throw) {
130  tbb::task_scheduler_init init(tbb::task_scheduler_init::deferred);
131  tbb::enumerable_thread_specific<double> test_locals;
132  int myCount = 0;
133  BusyBody myBody(nThreads, csWorkRatio, test_locals, myCount, test_throw);
134  BusyBodyScoped myScopedBody(nThreads, csWorkRatio, test_locals, myCount, test_throw);
135  init.initialize(nThreads);
136  tbb::tick_count t0;
137  {
138  t0 = tbb::tick_count::now();
139  myCount = 0;
140  NativeParallelFor(nThreads, myBody);
141  ASSERT(myCount == (MAX_WORK - (MAX_WORK % nThreads)), NULL);
142  REMARK("%d threads, work ratio %d per cent, time %g", nThreads, csWorkRatio, (tbb::tick_count::now() - t0).seconds());
143  if (nThreads > 1) {
144  double etsSum = 0;
145  double etsMax = 0;
146  double etsMin = 0;
147  double etsSigmaSq = 0;
148  double etsSigma = 0;
149 
150  for(tbb::enumerable_thread_specific<double>::const_iterator ci = test_locals.begin(); ci != test_locals.end(); ci++) {
151  etsSum += *ci;
152  if(etsMax==0.0) {
153  etsMin = *ci;
154  }
155  else {
156  if(etsMin > *ci) etsMin = *ci;
157  }
158  if(etsMax < *ci) etsMax = *ci;
159  }
160  double etsAvg = etsSum / (double)nThreads;
161  for(tbb::enumerable_thread_specific<double>::const_iterator ci = test_locals.begin(); ci != test_locals.end(); ci++) {
162  etsSigma = etsAvg - *ci;
163  etsSigmaSq += etsSigma * etsSigma;
164  }
165  // an attempt to gauge the "fairness" of the scheduling of the threads. We figure
166  // the standard deviation, and compare it with the maximum deviation from the
167  // average time. If the difference is 0 that means all threads finished in the same
168  // amount of time. If non-zero, the difference is divided by the time, and the
169  // negative log is taken. If > 2, then the difference is on the order of 0.01*t
170  // where T is the average time. We aritrarily define this as "fair."
171  etsSigma = sqrt(etsSigmaSq/double(nThreads));
172  etsMax -= etsAvg; // max - a == delta1
173  etsMin = etsAvg - etsMin; // a - min == delta2
174  if(etsMax < etsMin) etsMax = etsMin;
175  etsMax -= etsSigma;
176  // ASSERT(etsMax >= 0, NULL); // shouldn't the maximum difference from the mean be > the stddev?
177  etsMax = (etsMax > 0.0) ? etsMax : 0.0; // possible rounding error
178  double fairness = etsMax / etsAvg;
179  if(fairness == 0.0) {
180  fairness = 100.0;
181  }
182  else fairness = - log10(fairness);
183  if(fairness > 2.0 ) {
184  REMARK(" Fair (%g)\n", fairness);
185  }
186  else {
187  REMARK(" Unfair (%g)\n", fairness);
188  }
189  }
190  myCount = 0;
191  NativeParallelFor(nThreads, myScopedBody);
192  ASSERT(myCount == (MAX_WORK - (MAX_WORK % nThreads)), NULL);
193 
194  }
195 
196  init.terminate();
197 }
