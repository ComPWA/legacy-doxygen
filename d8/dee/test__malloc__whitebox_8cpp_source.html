854  else tooLargeLOC = 0;

863  objects.pop_front();
864  }

870  CacheBinModel(CacheBinType &_cacheBin, size_t allocSize) : cacheBin(_cacheBin), size(allocSize) {
871  cacheBinModel.oldest = cacheBin.oldest;
872  cacheBinModel.lastCleanedAge = cacheBin.lastCleanedAge;
873  cacheBinModel.ageThreshold = cacheBin.ageThreshold;
874  cacheBinModel.usedSize = cacheBin.usedSize;
875  cacheBinModel.cachedSize = cacheBin.cachedSize;
876  cacheBinModel.meanHitRange = cacheBin.meanHitRange;
877  cacheBinModel.lastGet = cacheBin.lastGet;
878  }

880  uintptr_t currTime = ++cacheCurrTime;
881 
882  if ( objects.empty() ) {
883  const uintptr_t sinceLastGet = currTime - cacheBinModel.lastGet;
884  if ( ( cacheBinModel.ageThreshold && sinceLastGet > Props::LongWaitFactor*cacheBinModel.ageThreshold ) ||
885  ( cacheBinModel.lastCleanedAge && sinceLastGet > Props::LongWaitFactor*(cacheBinModel.lastCleanedAge - cacheBinModel.lastGet) ) )
886  cacheBinModel.lastCleanedAge = cacheBinModel.ageThreshold = 0;
887 
888  if (cacheBinModel.lastCleanedAge)
889  cacheBinModel.ageThreshold = Props::OnMissFactor*(currTime - cacheBinModel.lastCleanedAge);
890  } else {
891  uintptr_t obj_age = objects.back();
892  objects.pop_back();
893  if ( objects.empty() ) cacheBinModel.oldest = 0;
894 
895  intptr_t hitRange = currTime - obj_age;
896  cacheBinModel.meanHitRange = cacheBinModel.meanHitRange? (cacheBinModel.meanHitRange + hitRange)/2 : hitRange;
897 
898  cacheBinModel.cachedSize -= size;
899  }
900 
901  cacheBinModel.usedSize += size;
902  cacheBinModel.lastGet = currTime;
903 
904  if ( currTime % rml::internal::cacheCleanupFreq == 0 ) doCleanup();
905  }

915  cacheBinModel.lastCleanedAge = ++currTime;
916  cleanUpNeeded |= currTime % rml::internal::cacheCleanupFreq == 0;
917  num--;
918  }

921  currTime+=1;
922  cleanUpNeeded |= currTime % rml::internal::cacheCleanupFreq == 0;
923  if ( objects.empty() )
924  cacheBinModel.oldest = currTime;
925  objects.push_back(currTime);
926  }
927 
928  cacheBinModel.cachedSize += num*size;
929 
930  if ( cleanUpNeeded ) doCleanup();
931  }
