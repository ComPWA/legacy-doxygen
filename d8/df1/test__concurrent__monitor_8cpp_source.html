95 void QueuingMutex::ScopedLock::Acquire( QueuingMutex& m, size_t test_mode )
96 {
97  // Must set all fields before the fetch_and_store, because once the
98  // fetch_and_store executes, *this becomes accessible to other threads.
99  mutex = &m;
100  next = NULL;
101  going = 0;
102 
103  // The fetch_and_store must have release semantics, because we are
104  // "sending" the fields initialized above to other processors.
105  ScopedLock* pred = m.q_tail.fetch_and_store<tbb::release>(this);
106  if( pred ) {
107 #if TBB_USE_ASSERT
108  __TBB_control_consistency_helper(); // on "m.q_tail"
109  ASSERT( !pred->next, "the predecessor has another successor!");
110 #endif
111  pred->next = this;
112  for( int i=0; i<16; ++i ) {
113  if( going!=0ul ) break;
114  __TBB_Yield();
115  }
116  int x = int( test_mode%3 );
117  switch( x ) {
118  case 0:
119  mutex->waitq.wait( QueuingMutex_Until(going), QueuingMutex_Context(this) );
120  break;
121 #if __TBB_CPP11_LAMBDAS_PRESENT
122  case 1:
123  mutex->waitq.wait( [&](){ return going!=0ul; }, [=]() { return (uintptr_t)this; } );
124  break;
125 #endif
126  default:
127  SleepPerhaps();
128  break;
129  }
130  }
131 
132  // Acquire critical section indirectly from previous owner or directly from predecessor.
133  __TBB_control_consistency_helper(); // on either "m.q_tail" or "going"
134 }

137 void QueuingMutex::ScopedLock::Release( )
138 {
139  if( !next ) {
140  if( this == mutex->q_tail.compare_and_swap<tbb::release>(NULL, this) ) {
141  // this was the only item in the queue, and the queue is now empty.
142  goto done;
143  }
144  // Someone in the queue
145  spin_wait_while_eq( next, (ScopedLock*)0 );
146  }
147  __TBB_store_with_release(next->going, 1);
148  mutex->waitq.notify( PredicateEq(uintptr_t(next)) );
149 done:
150  Initialize();
151 }

155 {
