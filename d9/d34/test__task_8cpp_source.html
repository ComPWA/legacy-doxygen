31 class UnboundedlyRecursiveOnUnboundedStealingTask : public tbb::task {
32  typedef UnboundedlyRecursiveOnUnboundedStealingTask this_type;
33 
34  this_type *m_Parent;
35  const int m_Depth;
36  volatile bool m_GoAhead;
37 
38  // Well, virtually unboundedly, for any practical purpose
39  static const int max_depth = 1000000;
40 
41 public:
42  UnboundedlyRecursiveOnUnboundedStealingTask( this_type *parent_ = NULL, int depth_ = max_depth )
43  : m_Parent(parent_)
44  , m_Depth(depth_)
45  , m_GoAhead(true)
46  {}
47 
48  tbb::task* execute() __TBB_override {
49  // Using large padding array speeds up reaching stealing limit
50  const int paddingSize = 16 * 1024;
51  volatile char padding[paddingSize];
52  if( !m_Parent || (m_Depth > 0 && m_Parent->m_GoAhead) ) {
53  if ( m_Parent ) {
54  // We are stolen, let our parent start waiting for us
55  m_Parent->m_GoAhead = false;
56  }
57  tbb::task &t = *new( allocate_child() ) this_type(this, m_Depth - 1);
58  set_ref_count( 2 );
59  spawn( t );
60  // Give a willing thief a chance to steal
61  for( int i = 0; i < 1000000 && m_GoAhead; ++i ) {
62  ++padding[i % paddingSize];
63  __TBB_Yield();
64  }
65  // If our child has not been stolen yet, then prohibit it siring ones
66  // of its own (when this thread executes it inside the next wait_for_all)
67  m_GoAhead = false;
68  wait_for_all();
69  }
70  return NULL;
71  }
72 }; // UnboundedlyRecursiveOnUnboundedStealingTask

76 class RecursiveTask: public tbb::task {
77  const int m_ChildCount;
78  const int m_Depth;
80  void SpawnList( tbb::task_list& list, int bit_mask ) {
81  if( m_Depth&bit_mask ) {
82  // Take address to check that signature of spawn(task_list&) is static.
83  void (*s)(tbb::task_list&) = &tbb::task::spawn;
84  (*s)(list);
85  ASSERT( list.empty(), NULL );
86  wait_for_all();
87  } else {
88  spawn_and_wait_for_all(list);
89  ASSERT( list.empty(), NULL );
90  }
91  }
92 public:
93  RecursiveTask( int child_count, int depth_ ) : m_ChildCount(child_count), m_Depth(depth_) {}
94  tbb::task* execute() __TBB_override {
95  ++Count;
96  if( m_Depth>0 ) {
97  tbb::task_list list;
98  ASSERT( list.empty(), NULL );
99  for( int k=0; k<m_ChildCount; ++k ) {
100  list.push_back( *new( allocate_child() ) RecursiveTask(m_ChildCount/2,m_Depth-1 ) );
101  ASSERT( !list.empty(), NULL );
102  }
103  set_ref_count( m_ChildCount+1 );
104  SpawnList( list, 1 );
105  // Now try reusing this as the parent.
106  set_ref_count(2);
107  list.push_back( *new ( allocate_child() ) tbb::empty_task() );
108  SpawnList( list, 2 );
109  }
110  return NULL;
111  }

441 class DagTask: public tbb::task {
442  typedef unsigned long long number_t;
443  const int i, j;
444  number_t sum_from_left, sum_from_above;
445  void check_sum( number_t sum ) {
446  number_t expected_sum = 1;
447  for( int k=i+1; k<=i+j; ++k )
448  expected_sum *= k;
449  for( int k=1; k<=j; ++k )
450  expected_sum /= k;
451  ASSERT(sum==expected_sum, NULL);
452  }
453 public:
454  DagTask *successor_to_below, *successor_to_right;
455  DagTask( int i_, int j_ ) : i(i_), j(j_), sum_from_left(0), sum_from_above(0) {}
456  task* execute() __TBB_override {
457  ASSERT( ref_count()==0, NULL );
458  number_t sum = i==0 && j==0 ? 1 : sum_from_left+sum_from_above;
459  check_sum(sum);
460  ++execution_count;
461  if( DagTask* t = successor_to_right ) {
462  t->sum_from_left = sum;
463  if( t->decrement_ref_count()==0 )
464  // Test using spawn to evaluate DAG
465  spawn( *t );
466  }
467  if( DagTask* t = successor_to_below ) {
468  t->sum_from_above = sum;
469  if( t->add_ref_count(-1)==0 )
470  // Test using bypass to evaluate DAG
471  return t;
472  }
473  return NULL;
474  }
475  ~DagTask() {++destruction_count;}
476  static tbb::atomic<int> execution_count;
477  static tbb::atomic<int> destruction_count;
478 };
