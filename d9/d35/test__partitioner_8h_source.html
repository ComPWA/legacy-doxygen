52 // helper for calculating number of range objects created before balancing phase is started
53 // and for finding maximum and minimum number of iterations among all such ranges
54 // Note: class does not provide exclusive access to members
55 class RangeStatisticCollector {
56 public:
57  RangeStatisticCollector(RangeStatisticData *statisticData) :
58  m_statData(statisticData)
59  {
60  m_called = false;
61  if (m_statData)
62  m_statData->m_rangeNum = 1;
63  }
64 
65  // constructor is called from non-proportional split constructor of derived Range
66  RangeStatisticCollector(RangeStatisticCollector& sc, size_t rangeSize) {
67  if (!sc.m_called) {
68  // this is the first time non-proportional split constructor is called
69  // it means that work distribution phase has been completed and
70  // work balancing phase has been just started
71  sc.m_called = true;
72 
73  if (sc.m_statData) {
74  size_t *minRangeSize = &sc.m_statData->m_minRangeSize;
75  if (*minRangeSize > rangeSize || !sc.m_statData->m_wasMinRangeSizeWritten) { // if minimum is not an actual minimum
76  *minRangeSize = rangeSize;
77  sc.m_statData->m_wasMinRangeSizeWritten = true;
78  }
79  size_t *maxRangeSize = &sc.m_statData->m_maxRangeSize;
80  if (*maxRangeSize < rangeSize) { // if maximum is not an actual maximum
81  *maxRangeSize = rangeSize;
82  }
83  }
84  }
85  *this = sc;
86  // constructor is used on work balancing phase only, so no need to increment
87  // number of range objects created
88  }
89 
90  RangeStatisticCollector(RangeStatisticCollector& sc, proportional_split&) {
91  if (sc.m_statData)
92  sc.m_statData->m_rangeNum++;
93  *this = sc;
94  }
95 
96 private:
97  RangeStatisticData *m_statData;
98 
99  // turns to 'true' when non-proportional split constructor is called first time
100  bool m_called;
101 };

103 // Base class for fake ranges used in vaious tests for parallel
104 // algorithms as well as for partitioner
105 template <typename DerivedRange, typename T>
106 class RangeBase: public RangeStatisticCollector {
107 protected:
108  size_t my_begin, my_end;
109  bool m_provide_feedback;
110  bool m_ensure_non_empty_size;
111 public:
112  RangeBase(size_t _begin, size_t _end, RangeStatisticData *statData,
113  bool provide_feedback, bool ensure_non_empty_size)
114  : RangeStatisticCollector(statData)
115  , my_begin(_begin), my_end(_end)
116  , m_provide_feedback(provide_feedback)
117  , m_ensure_non_empty_size(ensure_non_empty_size)
118  { }
119  RangeBase(RangeBase& r, tbb::split) : RangeStatisticCollector(r, r.size()) {
120  *this = r;
121  size_t middle = r.my_begin + (r.my_end - r.my_begin) / 2u;
122  r.my_end = my_begin = middle;
123  }
124 
125  RangeBase(RangeBase& r, proportional_split& p) : RangeStatisticCollector(r, p) {
126  *this = r;
127  size_t original_size = r.size();
128  T right = self().compute_right_part(r, p);
129  size_t right_part = self().round(right);
130  if( m_ensure_non_empty_size ) {
131  right_part = (original_size == right_part) ? (original_size - 1) : right_part;
132  right_part = (right_part != 0) ? right_part : 1;
133  }
134  r.my_end = my_begin = r.my_end - right_part;
135 #if __TBB_ENABLE_RANGE_FEEDBACK
136  if( m_provide_feedback )
137  p.set_proportion(original_size - right_part, right_part);
138 #endif
139  if( m_ensure_non_empty_size )
140  ASSERT(r.my_end != r.my_begin && my_end != my_begin, "Incorrect range split");
141  }
142 
143  size_t begin() const { return my_begin; }
144  size_t end() const { return my_end; }
145  bool is_divisible() const { return (my_end - my_begin) > 1; }
146  bool empty() const { return my_end == my_begin; }
147  size_t size() const { return my_end - my_begin; }
148 
149  // helper methods (not part of the range concept)
150  DerivedRange& self() { return static_cast<DerivedRange&>(*this); }
151  size_t round(T part) { return size_t(part); }
152  T compute_right_part(RangeBase& r, proportional_split& p) {
153  return T(r.size() * T(p.right())) / T(p.left() + p.right());
154  }
155  bool is_ensure_non_emptiness() { return m_ensure_non_empty_size; }
156 };

281  : RangeBase<ExactSplitRange, size_t>(r, p) { }
282  size_t compute_right_part(RangeBase<ExactSplitRange, size_t>& r, proportional_split& p) {
283  size_t parts = size_t(p.left() + p.right());
284  size_t currSize = r.size();
285  size_t int_part = currSize / parts * p.right();
286  size_t remainder = currSize % parts * p.right();
287  int_part += remainder / parts;
288  remainder %= parts;
289  size_t right_part = int_part + (remainder > parts/2 ? 1 : 0);
290  return right_part;
291  }
292  static const bool is_splittable_in_proportion = true;
293 };

356  void push_subnode(TreeNode *&root_node, TreeNode *node) {
357  if (!root_node) {
358  root_node = node;
359  return;
360  } else if (are_nodes_equal(root_node, node)) {
361  // no need to push the same node
362  return;
363  }
364 
365  if (!has_children(root_node)) {
366  // if current root_node does not have children passed node
367  // should has one of the interval bounds to be equal to
368  // the same bound in the root_node
369  if (is_look_like_left_sibling(root_node, node))
370  push_subnode(root_node->m_left, node);
371  else
372  push_subnode(root_node->m_right, node);
373  return;
374  }
375 
376  if (has_left_child(root_node)) {
377  if (is_subnode(root_node->m_left, node)) {
378  push_subnode(root_node->m_left, node);
379  return;
380  }
381  push_subnode(root_node->m_right, node);
382  return;
383  }
384 
385  ASSERT(root_node->m_right != NULL, "Right child is NULL but must be present");
386  if (is_subnode(root_node->m_right, node)) {
387  push_subnode(root_node->m_right, node);
388  return;
389  }
390  push_subnode(root_node->m_left, node);
391  return;
392  }
393 
394  bool has_children(TreeNode *node) { return node->m_left || node->m_right; }
395 
396  bool is_look_like_left_sibling(TreeNode *root_node, TreeNode *node) {
397  if (root_node->m_range_begin == node->m_range_begin)
398  return true;
399  ASSERT(root_node->m_range_end == node->m_range_end, NULL);

401  }

403  bool has_left_child(TreeNode *node) { return node->m_left != NULL; }
404

406  return root_node->m_range_begin <= node->m_range_begin &&

408  }

516  * -------------------------------------------------------------------------------------------------------------
517  * 4 true not available proportional pMN, r(s), part(p) *
518  * -------------------------------------------------------------------------------------------------------------
519  * 5 false not available proportional p11, r(s), part(p)
520  * -------------------------------------------------------------------------------------------------------------
521  * 6 not defined not available proportional p11, r(s), part(p)
522  * -------------------------------------------------------------------------------------------------------------
523  * 1 true available simple s, r(s), part(s)
524  * -------------------------------------------------------------------------------------------------------------
525  * 2 false available simple s, r(s), part(s)
526  * -------------------------------------------------------------------------------------------------------------

529  * 4 true not available simple s, r(s), part(s)

532  * -------------------------------------------------------------------------------------------------------------
533  * 6 not defined not available simple s, r(s), part(s)
534  * -------------------------------------------------------------------------------------------------------------
535  *
536  * Legend:
537  * proportional - with proportional splits (e.g. affinity_partitioner)
538  * simple - without proportional splits (e.g. simple_partitioner, auto_partitioner)
