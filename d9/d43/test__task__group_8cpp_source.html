110  struct TaskFunctor {
111  SharedGroupBodyImpl *m_pOwner;
112  void operator () () const {
113  if ( m_pOwner->m_sharingMode & ParallelWait ) {
114  while ( Harness::ConcurrencyTracker::PeakParallelism() < m_pOwner->m_numThreads )
115  __TBB_Yield();
116  }
117  ++s_tasksExecuted;
118  }
119  };

121  TaskFunctor m_taskFunctor;
122 
123  void Spawn ( uint_t numTasks ) {
124  for ( uint_t i = 0; i < numTasks; ++i ) {
125  ++m_tasksSpawned;
126  Harness::ConcurrencyTracker ct;
127  m_taskGroup->run( m_taskFunctor );
128  }
129  ++m_threadsReady;
130  }
131 
132  void DeleteTaskGroup () {
133  delete m_taskGroup;
134  m_taskGroup = NULL;
135  }
136 
137  void Wait () {
138  while ( m_threadsReady != m_numThreads )
139  __TBB_Yield();
140  const uint_t numSpawned = c_numTasks0 + c_numTasks1 * (m_numThreads - 1);
141  ASSERT ( m_tasksSpawned == numSpawned, "Wrong number of spawned tasks. The test is broken" );
142  REMARK("Max spawning parallelism is %u out of %u\n", Harness::ConcurrencyTracker::PeakParallelism(), g_MaxConcurrency);
143  if ( m_sharingMode & ParallelWait ) {
144  m_barrier.wait( &Harness::ConcurrencyTracker::Reset );
145  {
146  Harness::ConcurrencyTracker ct;
147  m_taskGroup->wait();
148  }
149  if ( Harness::ConcurrencyTracker::PeakParallelism() == 1 )
150  REPORT ( "Warning: No parallel waiting detected in TestParallelWait\n" );
151  m_barrier.wait();
152  }
153  else
154  m_taskGroup->wait();
155  ASSERT ( m_tasksSpawned == numSpawned, "No tasks should be spawned after wait starts. The test is broken" );
156  ASSERT ( s_tasksExecuted == numSpawned, "Not all spawned tasks were executed" );
157  }
158 
159 public:
160  SharedGroupBodyImpl ( uint_t numThreads, uint_t sharingMode = 0 )
161  : m_numThreads(numThreads)
162  , m_sharingMode(sharingMode)
163  , m_taskGroup(NULL)
164  , m_barrier(numThreads)
165  {
166  ASSERT ( m_numThreads > 1, "SharedGroupBody tests require concurrency" );
167  ASSERT ( !(m_sharingMode & VagabondGroup) || m_numThreads == 2, "In vagabond mode SharedGroupBody must be used with 2 threads only" );
168  Harness::ConcurrencyTracker::Reset();
169  s_tasksExecuted = 0;
170  m_tasksSpawned = 0;
171  m_threadsReady = 0;
172  m_taskFunctor.m_pOwner = this;
173  }

175  void Run ( uint_t idx ) {
176 #if TBBTEST_USE_TBB
177  tbb::task_scheduler_init init(g_MaxConcurrency);
178 #endif
179  AssertLive();
180  if ( idx == 0 ) {
181  ASSERT ( !m_taskGroup && !m_tasksSpawned, "SharedGroupBody must be reset before reuse");
182  m_taskGroup = new Concurrency::task_group;
183  Spawn( c_numTasks0 );
184  Wait();
185  if ( m_sharingMode & VagabondGroup )
186  m_barrier.wait();
187  else
188  DeleteTaskGroup();
189  }
190  else {
191  while ( m_tasksSpawned == 0 )
192  __TBB_Yield();
193  ASSERT ( m_taskGroup, "Task group is not initialized");
194  Spawn (c_numTasks1);
195  if ( m_sharingMode & ParallelWait )
196  Wait();
197  if ( m_sharingMode & VagabondGroup ) {
198  ASSERT ( idx == 1, "In vagabond mode SharedGroupBody must be used with 2 threads only" );
199  m_barrier.wait();
200  DeleteTaskGroup();
201  }
202  }
203  AssertLive();
204  }
205 };
