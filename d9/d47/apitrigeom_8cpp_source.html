69 void rt_tri_fcylinder(void * tex, vector ctr, vector axis, apiflt rad) {
70  vector x, y, z, tmp;
71  double u, v, u2, v2;
72  int j;
73  vector p1, p2, p3, p4;
74  vector n1, n2;
75 
76  z = axis;
77  MyVNorm(&z);
78  tmp.x = z.y - 2.1111111;
79  tmp.y = -z.z + 3.14159267;
80  tmp.z = z.x - 3.915292342341;
81  MyVNorm(&z);
82  MyVNorm(&tmp);
83  MyVCross(&z, &tmp, &x);
84  MyVNorm(&x);
85  MyVCross(&x, &z, &y);
86  MyVNorm(&y);
87 
88  for (j=0; j<CYLFACETS; j++) {
89  u = rad * sin((6.28 * j) / (CYLFACETS - 1.0));
90  v = rad * cos((6.28 * j) / (CYLFACETS - 1.0));
91  u2 = rad * sin((6.28 * (j + 1.0)) / (CYLFACETS - 1.0));
92  v2 = rad * cos((6.28 * (j + 1.0)) / (CYLFACETS - 1.0));
93 
94  p1.x = p1.y = p1.z = 0.0;
95  p4 = p3 = p2 = p1;
96 
97  MyVAddS(u, &x, &p1, &p1);
98  MyVAddS(v, &y, &p1, &p1);
99  n1 = p1;
100  MyVNorm(&n1);
101  MyVAddS(1.0, &ctr, &p1, &p1);
102 
103 
104  MyVAddS(u2, &x, &p2, &p2);
105  MyVAddS(v2, &y, &p2, &p2);
106  n2 = p2;
107  MyVNorm(&n2);
108  MyVAddS(1.0, &ctr, &p2, &p2);
109 
110  MyVAddS(1.0, &axis, &p1, &p3);
111  MyVAddS(1.0, &axis, &p2, &p4);
112 
113  rt_stri(tex, p1, p2, p3, n1, n2, n1);
114  rt_stri(tex, p3, p2, p4, n1, n2, n2);
115  }
116 }
