85 void
86 check_multimap(MyMultiMap &m, int *targets, int tcount, int key) {
87  std::vector<bool> vfound(tcount,false);
88  std::pair<MyMultiMap::iterator, MyMultiMap::iterator> range = m.equal_range( key );
89  for(MyMultiMap::iterator it = range.first; it != range.second; ++it) {
90  bool found = false;
91  for( int i = 0; i < tcount; ++i) {
92  if((*it).second == targets[i]) {
93  if(!vfound[i]) { // we can insert duplicate values
94  vfound[i] = found = true;
95  break;
96  }
97  }
98  }
99  // just in case an extra value in equal_range...
100  ASSERT(found, "extra value from equal range");
101  }
102  for(int i = 0; i < tcount; ++i) ASSERT(vfound[i], "missing value");
103 }
