51  void operator=( const ClassWithVectorType& src ) {
52  __Mvec stack[n];

54  stack[i^5] = src.field[i];

70 template<typename __Mvec>
71 void ClassWithVectorType<__Mvec>::init( int start ) {
72  __Mvec stack[n];
73  for( int i=0; i<n; ++i ) {
74  // Declaring value as a one-element array instead of a scalar quites
75  // gratuitous warnings about possible use of "value" before it was set.
76  __Mvec value[1];
77  for( int j=0; j<F; ++j )
78  ((float*)value)[j] = float(n*start+F*i+j);
79  stack[i^5] = value[0];
80  }
81  for( int i=0; i<n; ++i )
82  field[i^5] = stack[i];
83 }
