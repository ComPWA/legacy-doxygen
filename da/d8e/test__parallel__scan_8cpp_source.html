117 template<typename T>
118 void Scan(const Range & r, bool is_final, Storage<T> & storage, std::vector<T> & sum, const std::vector<T> & addend) {
119  ASSERT(!is_final || (storage.my_range.begin() == 0 && storage.my_range.end() == r.begin()) || (storage.my_range.empty() && r.begin() == 0), NULL);
120  for (long i = r.begin(); i < r.end(); ++i) {
121  storage.my_total += addend[i];
122  if (is_final) {
123  ASSERT(AddendHistory[i] < USED_FINAL, "addend used 'finally' twice?");
124  AddendHistory[i] |= USED_FINAL;
125  sum[i] = storage.my_total;
126  VerifySum(42, i, int(sum[i]), __LINE__);
127  }
128  else {
129  ASSERT(AddendHistory[i] == UNUSED, "addend used too many times");
130  AddendHistory[i] |= USED_NONFINAL;
131  }
132  }
133  if (storage.my_range.empty())
134  storage.my_range = r;
135  else
136  storage.my_range = Range(storage.my_range.begin(), r.end(), 1);
137  Snooze(true);
138 }

148 template<typename T>
149 class Accumulator: BodyId {
150  const std::vector<T> &my_array;
151  std::vector<T> & my_sum;
152  Storage<T> storage;
153  enum state_type {
154  full, // Accumulator has sufficient information for final scan,
155  // i.e. has seen all iterations to its left.
156  // It's either the original Accumulator provided by the user
157  // or a Accumulator constructed by a splitting constructor *and* subsequently
158  // subjected to a reverse_join with a full accumulator.
159 
160  partial, // Accumulator has only enough information for pre_scan.
161  // i.e. has not seen all iterations to its left.
162  // It's an Accumulator created by a splitting constructor that
163  // has not yet been subjected to a reverse_join with a full accumulator.
164 
165  summary, // Accumulator has summary of iterations processed, but not necessarily
166  // the information required for a final_scan or pre_scan.
167  // It's the result of "assign".
168 
169  trash // Accumulator with possibly no useful information.
170  // It was the source for "assign".
171 
172  };
173  mutable state_type my_state;
175 
176  Accumulator* self;
177  Accumulator& operator= (const Accumulator& other);
178 public:
179  Accumulator( T init, const std::vector<T> & array, std::vector<T> & sum ) :
180  my_array(array), my_sum(sum), storage(init), my_state(full)
181  {
182  // Set self as last action of constructor, to indicate that object is fully constructed.
183  self = this;
184  }
185 #if PRINT_DEBUG
186  void print() const {
187  REPORT("%d [%ld..%ld)\n", id, storage.my_range.begin(), storage.my_range.end() );
188  }
189 #endif /* PRINT_DEBUG */
190  ~Accumulator() {
191 #if PRINT_DEBUG
192  REPORT("%d [%ld..%ld) destroyed\n",id, storage.my_range.begin(), storage.my_range.end() );
193 #endif /* PRINT_DEBUG */
194  // Clear self as first action of destructor, to indicate that object is not fully constructed.
195  self = 0;
196  }
197  Accumulator( Accumulator& a, tbb::split ) :
198  my_array(a.my_array), my_sum(a.my_sum), storage(0), my_state(partial)
199  {
200  ASSERT(a.my_state==full || a.my_state==partial, NULL);
201 #if PRINT_DEBUG
202  REPORT("%d forked from %d\n",id,a.id);
203 #endif /* PRINT_DEBUG */

351 void TestAccumulator( int mode, int nthread ) {
352  typedef int T;
353  std::vector<T> addend(MAXN);
354  std::vector<T> sum(MAXN);
355  for( long n=0; n<=MAXN; ++n ) {
356  for( long i=0; i<MAXN; ++i ) {

362  addend[i] = i;
363 
364  Accumulator<T> acc( 42, addend, sum );
365  tbb::tick_count t0 = tbb::tick_count::now();
366 #if PRINT_DEBUG
367  REPORT("--------- mode=%d range=[0..%ld)\n",mode,n);
368 #endif /* PRINT_DEBUG */
369  ScanIsRunning = true;
370 
371  switch (mode) {
372  case 0:
373  tbb::parallel_scan( Range( 0, n, 1 ), acc );
374  break;
375  case 1:
376  tbb::parallel_scan( Range( 0, n, 1 ), acc, tbb::simple_partitioner() );
377  break;
378  case 2:
379  tbb::parallel_scan( Range( 0, n, 1 ), acc, tbb::auto_partitioner() );
380  break;
381  }
382 
383  ScanIsRunning = false;
384 #if PRINT_DEBUG
385  REPORT("=========\n");
386 #endif /* PRINT_DEBUG */
387  Snooze(false);
388  tbb::tick_count t1 = tbb::tick_count::now();
389  long used_once_count = 0;
390  for( long i=0; i<n; ++i )
391  if( !(AddendHistory[i]&USED_FINAL) ) {

400  else
