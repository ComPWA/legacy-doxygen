41 bool spin_rw_mutex::internal_acquire_writer(spin_rw_mutex *mutex)
42 {
43  ITT_NOTIFY(sync_prepare, mutex);
44  for( atomic_backoff backoff;;backoff.pause() ) {
45  state_t s = mutex->state;
46  if( !(s & BUSY) ) { // no readers, no writers
47  if( CAS(mutex->state, WRITER, s) )
48  break; // successfully stored writer flag
49  backoff.reset(); // we could be very close to complete op.
50  } else if( !(s & WRITER_PENDING) ) { // no pending writers
51  __TBB_AtomicOR(&mutex->state, WRITER_PENDING);
52  }
53  }
54  ITT_NOTIFY(sync_acquired, mutex);
55  __TBB_ASSERT( (mutex->state & BUSY)==WRITER, "invalid state of a write lock" );
56  return false;
57 }

67 void spin_rw_mutex::internal_acquire_reader(spin_rw_mutex *mutex) {
68  ITT_NOTIFY(sync_prepare, mutex);
69  for( atomic_backoff backoff;;backoff.pause() ) {
70  state_t s = mutex->state;
71  if( !(s & (WRITER|WRITER_PENDING)) ) { // no writer or write requests
72  if( CAS(mutex->state, s+ONE_READER, s) )
73  break; // successfully stored increased number of readers
74  backoff.reset(); // we could be very close to complete op.
75  }
76  }
77  ITT_NOTIFY(sync_acquired, mutex);
78  __TBB_ASSERT( mutex->state & READERS, "invalid state of a read lock: no readers" );
79  __TBB_ASSERT( !(mutex->state & WRITER), "invalid state of a read lock: active writer" );
80 }

84 bool spin_rw_mutex::internal_upgrade(spin_rw_mutex *mutex) {
85  state_t s = mutex->state;
86  __TBB_ASSERT( s & READERS, "invalid state before upgrade: no readers " );
87  __TBB_ASSERT( !(s & WRITER), "invalid state before upgrade: active writer " );
88  // check and set writer-pending flag
89  // required conditions: either no pending writers, or we are the only reader
90  // (with multiple readers and pending writer, another upgrade could have been requested)
91  while( (s & READERS)==ONE_READER || !(s & WRITER_PENDING) ) {
92  if( CAS(mutex->state, s | WRITER_PENDING, s) )
93  {
94  ITT_NOTIFY(sync_prepare, mutex);
95  atomic_backoff backoff;
96  while( (mutex->state & READERS) != ONE_READER ) backoff.pause();
97  __TBB_ASSERT(mutex->state == (ONE_READER | WRITER_PENDING),"invalid state when upgrading to writer");
98  // both new readers and writers are blocked at this time
99  mutex->state = WRITER;
100  ITT_NOTIFY(sync_acquired, mutex);
101  __TBB_ASSERT( (mutex->state & BUSY) == WRITER, "invalid state after upgrade" );
102  return true; // successfully upgraded
103  } else {
104  s = mutex->state; // re-read
105  }
106  }
107  // slow reacquire
108  internal_release_reader(mutex);
109  return internal_acquire_writer(mutex); // always returns false
110 }
