173 void test_reset() {
174  // source_node -> function_node
175  tbb::flow::graph g;
176  tbb::atomic<int> counters3[N];
177  tbb::flow::source_node<int> src3(g, source_body<int>() );
178  tbb::flow::source_node<int> src_inactive(g, source_body<int>(), /*active*/ false );
179  function_body<int> b3( counters3 );
180  tbb::flow::function_node<int,bool> dest3(g, tbb::flow::unlimited, b3 );
181  tbb::flow::make_edge( src3, dest3 );
182  // source_node already in active state. Let the graph run,
183  g.wait_for_all();
184  // check the array for each value.
185  for (int i = 0; i < N; ++i ) {
186  int v = counters3[i];
187  ASSERT( v == 1, NULL );
188  counters3[i] = 0;
189  }
190  g.reset(tbb::flow::rf_reset_bodies); // <-- re-initializes the counts.
191  // and spawns task to run source
192  g.wait_for_all();
193  // check output queue again. Should be the same contents.
194  for (int i = 0; i < N; ++i ) {
195  int v = counters3[i];
196  ASSERT( v == 1, NULL );
197  counters3[i] = 0;
198  }
199  g.reset(); // doesn't reset the source_node_body to initial state, but does spawn a task
200  // to run the source_node.
201 
202  g.wait_for_all();
203  // array should be all zero
204  for (int i = 0; i < N; ++i ) {
205  int v = counters3[i];
206  ASSERT( v == 0, NULL );
207  }
208 
209  remove_edge(src3, dest3);
210  make_edge(src_inactive, dest3);
211 
212  // src_inactive doesn't run
213  g.wait_for_all();
214  for (int i = 0; i < N; ++i ) {
215  int v = counters3[i];
216  ASSERT( v == 0, NULL );
217  }
218 
219  // run graph
220  src_inactive.activate();
221  g.wait_for_all();
222  // check output
223  for (int i = 0; i < N; ++i ) {
224  int v = counters3[i];
225  ASSERT( v == 1, NULL );
226  counters3[i] = 0;
227  }
228  g.reset(tbb::flow::rf_reset_bodies); // <-- reinitializes the counts
229  // src_inactive doesn't run
230  g.wait_for_all();
231  for (int i = 0; i < N; ++i ) {
232  int v = counters3[i];
233  ASSERT( v == 0, NULL );
234  }
235 
236  // start it up
237  src_inactive.activate();
238  g.wait_for_all();
239  for (int i = 0; i < N; ++i ) {
240  int v = counters3[i];
241  ASSERT( v == 1, NULL );
242  counters3[i] = 0;
243  }
244  g.reset(); // doesn't reset the source_node_body to initial state, and doesn't
245  // spawn a task to run the source_node.
246 
247  g.wait_for_all();
248  // array should be all zero
249  for (int i = 0; i < N; ++i ) {
250  int v = counters3[i];
251  ASSERT( v == 0, NULL );
252  }
253  src_inactive.activate();
254  // source_node_body is already in final state, so source_node will not forward a message.
255  g.wait_for_all();
256  for (int i = 0; i < N; ++i ) {
257  int v = counters3[i];
258  ASSERT( v == 0, NULL );
259  }
260 }
