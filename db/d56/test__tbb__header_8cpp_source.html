139  // The assertion here serves to shut up warnings about "eid not used".
140  ASSERT( eid<tbb::internal::eid_max, NULL );
141 #if TBB_USE_EXCEPTIONS
142  for ( int i = 0; i < 2; ++i ) {
143  try {
144  if ( i == 0 )
145  throw exc;
146 #if !__TBB_THROW_ACROSS_MODULE_BOUNDARY_BROKEN
147  else
148  tbb::internal::throw_exception( eid );
149 #endif
150  }
151  catch ( E& e ) {
152  ASSERT ( e.what(), "Missing what() string" );
153  }
154  catch ( ... ) {
155  ASSERT ( __TBB_EXCEPTION_TYPE_INFO_BROKEN, "Unrecognized exception. Likely RTTI related exports are missing" );
156  }
157  }
158 #else /* TBB_USE_EXCEPTIONS */
159  (void)exc;
160 #endif /* TBB_USE_EXCEPTIONS */
