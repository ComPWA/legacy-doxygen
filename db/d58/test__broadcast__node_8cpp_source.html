88 template< typename T >

90 
91  tbb::flow::graph g;
92  tbb::flow::broadcast_node<T> b(g);
93 
94  for ( int num_receivers = 1; num_receivers < R; ++num_receivers ) {
95  std::vector< counting_array_receiver<T> > receivers(num_receivers, counting_array_receiver<T>(g));
96 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
97  ASSERT(b.successor_count() == 0, NULL);

99  typename tbb::flow::broadcast_node<T>::successor_list_type my_succs;
100  b.copy_successors(my_succs);
101  ASSERT(my_succs.size() == 0, NULL);
102  typename tbb::flow::broadcast_node<T>::predecessor_list_type my_preds;
103  b.copy_predecessors(my_preds);
104  ASSERT(my_preds.size() == 0, NULL);
105 #endif
106 
107  for ( int r = 0; r < num_receivers; ++r ) {
108  tbb::flow::make_edge( b, receivers[r] );
109  }
110 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
111  ASSERT( b.successor_count() == (size_t)num_receivers, NULL);
112 #endif
113 
114  for (int n = 0; n < N; ++n ) {
115  ASSERT( b.try_put( (T)n ), NULL );
116  }
117 
118  for ( int r = 0; r < num_receivers; ++r ) {
119  for (int n = 0; n < N; ++n ) {
120  ASSERT( receivers[r][n] == 1, NULL );
121  }
122  tbb::flow::remove_edge( b, receivers[r] );
123  }
124  ASSERT( b.try_put( (T)0 ), NULL );
125  for ( int r = 0; r < num_receivers; ++r )
126  ASSERT( receivers[0][0] == 1, NULL );
127  }

148 template< typename T >
149 void run_parallel_broadcasts(tbb::flow::graph& g, int p, tbb::flow::broadcast_node<T>& b) {
150  for ( int num_receivers = 1; num_receivers < R; ++num_receivers ) {
151  std::vector< counting_array_receiver<T> > receivers(num_receivers, counting_array_receiver<T>(g));
152 
153  for ( int r = 0; r < num_receivers; ++r ) {
154  tbb::flow::make_edge( b, receivers[r] );
155  }
156 
157  NativeParallelFor( p, native_body<T>( b ) );
158 
159  for ( int r = 0; r < num_receivers; ++r ) {
160  for (int n = 0; n < N; ++n ) {
161  ASSERT( (int)receivers[r][n] == p, NULL );
162  }
163  tbb::flow::remove_edge( b, receivers[r] );
164  }
165  ASSERT( b.try_put( (T)0 ), NULL );
166  for ( int r = 0; r < num_receivers; ++r )
167  ASSERT( (int)receivers[r][0] == p, NULL );
168  }
169 }

186 void test_resets() {
187  tbb::flow::graph g;
188  tbb::flow::broadcast_node<T> b0(g);
189  tbb::flow::broadcast_node<T> b1(g);
190  tbb::flow::queue_node<T> q0(g);
191  tbb::flow::make_edge(b0,b1);
192  tbb::flow::make_edge(b1,q0);
193  T j;
194 
195  // test standard reset
196  for(int testNo = 0; testNo < 2; ++testNo) {
197  for(T i= 0; i <= 3; i += 1) {
198  b0.try_put(i);
199  }
200  g.wait_for_all();
201  for(T i= 0; i <= 3; i += 1) {
202  ASSERT(q0.try_get(j) && j == i, "Bad value in queue");
203  }
204  ASSERT(!q0.try_get(j), "extra value in queue");
205 
206  // reset the graph. It should work as before.
207  if (testNo == 0) g.reset();
208  }
209 
210  g.reset(tbb::flow::rf_clear_edges);
211  for(T i= 0; i <= 3; i += 1) {
212  b0.try_put(i);
213  }
214  g.wait_for_all();
215  ASSERT(!q0.try_get(j), "edge between nodes not removed");
216  for(T i= 0; i <= 3; i += 1) {
217  b1.try_put(i);
218  }
219  g.wait_for_all();
220  ASSERT(!q0.try_get(j), "edge between nodes not removed");
221 }
