116 class MyObserver : public tbb::task_scheduler_observer, public ObserverStats {
117  uintptr_t m_flag;
118  tbb::atomic<int> m_leave_ticket;
119  tbb::atomic<bool> m_dying;

121  void on_scheduler_entry( bool is_worker ) __TBB_override {
122  ThreadState& state = ThreadState::get();
123  ASSERT( is_worker==!state.m_isMaster, NULL );
124  if ( theTestMode & tmLeavingControl )
125  ASSERT( m_leave_ticket, NULL );
126  if ( thePrevMode & tmSynchronized ) {
127  ASSERT( !(state.m_flags & m_flag), "Observer repeatedly invoked for the same thread" );
128  if ( theTestMode & tmLocalObservation )
129  ASSERT( !state.m_flags, "Observer locality breached" );
130  }
131  if ( m_dying && theTestMode & tmLocalObservation ) {
132  // In case of local observation a worker may enter the arena after
133  // the wait for lagging on_entry calls in the MyObserver destructor
134  // succeeds but before its base class tbb::task_scheduler_observer
135  // destructor removes it from the internal list maintained by the
136  // task scheduler. This will result in on_entry notification without,
137  // subsequent on_exit as the observer is likely to be destroyed before
138  // the worker discovers that the arena is empty and leaves it.
139  //
140  // To prevent statistics distortion, ignore the notifications for
141  // observers about to be destroyed.
142  ASSERT( !state.m_dyingObserver || state.m_dyingObserver != this || thePrevMode & tmSynchronized, NULL );
143  state.m_dyingObserver = this;
144  return;
145  }
146  state.m_dyingObserver = NULL;
147  ++m_entries;
148  state.m_flags |= m_flag;
149  if ( is_worker )
150  ++m_workerEntries;
151  }
152  void on_scheduler_exit( bool is_worker ) __TBB_override {
153  ThreadState& state = ThreadState::get();
154  ASSERT( is_worker==!state.m_isMaster, NULL );
155  if ( m_dying && state.m_dyingObserver ) {
156  ASSERT( state.m_dyingObserver == this, "Exit without entry (for a dying observer)" );
157  state.m_dyingObserver = NULL;
158  return;
159  }
160  ASSERT( state.m_flags & m_flag, "Exit without entry" );
161  state.m_flags &= ~m_flag;
162  ++m_exits;
163  if ( is_worker )
164  ++m_workerExits;
165  }
166  bool may_sleep() __TBB_override {
167  ThreadState& state = ThreadState::get();
168  ++state.m_maySleepCalls;
169  Harness::Sleep(10); // helps to reproduce the issues
170  ASSERT( !state.m_isMaster, NULL );
171  if( m_dying ) { // check the anti-starvation logic
172  return keep_awake; // thread should exit despite the return value
173  }
174  if( state.m_canSleep ) {// the permission for sleep was previously received
175  // though, it is an important check for the test, we still do not guarantee this condition
176  ASSERT_WARNING( !(theTestMode & tmLeavingControl), "may_sleep() called again after leaving permission was granted once, check if repeated");
177  return allow_sleep;
178  }
179  // note, may_sleep can be called before on_entry()
180  if( !(theTestMode & tmLeavingControl) || m_leave_ticket.fetch_and_store(-1) > 0 ) {
181  state.m_canSleep = true;
182  ++m_workerSleeps;
183  return allow_sleep;
184  }
185  return keep_awake;
186  }
187 public:
188  // the method is called before the work in new arena starts enabling the leaving test mode
189  // in this mode may_sleep() does not allow a thread to fall asleep unless permitted below
190  void enable_leaving_test() {
191  ASSERT(theTestMode & tmLeavingControl, NULL);
192  m_leave_ticket.store<tbb::relaxed>(-1);
193  ASSERT(!is_observing(), NULL);
194  observe(true);
195  }

197  // the work is just done in the only arena, assume workers start entering may_sleep
198  void test_leaving() {
199 #if TEST_SLEEP_PERMISSION
200  if( !(theTestMode & tmLeavingControl) )
201  return; // second call to the test TODO: extend the test for the second round as well
202  REMARK( "Testing may_sleep()\n");
203  ASSERT( !m_workerSleeps, "permission for sleep was given before the test starts?");
204  ASSERT( (theTestMode & tmSynchronized) && m_workerEntries >= P-1, "test_leaving assumes full subscription of the only arena");
205  for ( int j = 0; j < m_workerEntries; j++ ) {

207  ASSERT( m_leave_ticket == -1, "unexpected mode, signal was not consumed by a worker?" );
208  m_leave_ticket = 1; // dismiss one

258 class FibTask : public tbb::task {
259  const int N;
260  uintptr_t m_flag;
261  MyObserver &m_observer;
262 public:
263  FibTask( int n, uintptr_t flags, MyObserver &obs ) : N(n), m_flag(flags), m_observer(obs) {}
264 
265  tbb::task* execute() __TBB_override {
266  ThreadState& s = ThreadState::get();
267  ASSERT( !(~s.m_flags & m_flag), NULL );
268  if( N < 2 )
269  return NULL;
270  bool globalBarrierActive = false;
271  if ( s.m_isMaster ) {
272  if ( theGlobalBarrierActive ) {
273  // This is the root task. Its N is equal to the number of threads.
274  // Spawn a task for each worker.
275  set_ref_count(N);
276  for ( int i = 1; i < N; ++i )
277  spawn( *new( allocate_child() ) FibTask(20, m_flag, m_observer) );
278  if ( theTestMode & tmSynchronized ) {
279  theGlobalBarrier.wait();
280  ASSERT( m_observer.m_entries >= N, "Wrong number of on_entry calls after the first barrier" );
281  // All the spawned tasks have been stolen by workers.
282  // Now wait for workers to spawn some more tasks for this thread to steal back.
283  theGlobalBarrier.wait();
284  ASSERT( !theGlobalBarrierActive, "Workers are expected to have reset this flag" );
285  }
286  else
287  theGlobalBarrierActive = false;
288  wait_for_all();
289  return NULL;
290  }
291  }
292  else {
293  if ( theGlobalBarrierActive ) {
294  if ( theTestMode & tmSynchronized ) {
295  theGlobalBarrier.wait();
296  globalBarrierActive = true;
297  }
298  theGlobalBarrierActive = false;
299  }
300  }
301  set_ref_count(3);
302  spawn( *new( allocate_child() ) FibTask(N-1, m_flag, m_observer) );
303  spawn( *new( allocate_child() ) FibTask(N-2, m_flag, m_observer) );
304  if ( globalBarrierActive ) {
305  // It's the first task executed by a worker. Release the master thread.
306  theGlobalBarrier.wait();
307  }
308  wait_for_all();
309  return NULL;
310  }
311 }; // class FibTask

350 void TestObserver( int M, int T, uintptr_t testMode ) {
351  theLocalState.clear();
352  theStats.Reset();
353  theGlobalBarrierActive = true;
354  theTestMode = testMode;
355  NativeParallelFor( M, TestBody(T) );
356  // When T (number of threads in arena, i.e. master + workers) is less than P
357  // (hardware concurrency), more than T-1 workers can visit the same arena. This
358  // is possible in case of imbalance or when other arenas are activated/deactivated
359  // concurrently).
360  ASSERT( !theNumObservers, "Unexpected alive observer(s)" );
361  REMARK( "Entries %d / %d, exits %d\n", (int)theStats.m_entries, (int)theStats.m_workerEntries, (int)theStats.m_exits );
362  if ( testMode & tmSynchronized ) {
363  if ( testMode & tmLocalObservation ) {
364  ASSERT( theStats.m_entries >= M * T, "Too few on_entry calls" );
365  ASSERT( theStats.m_workerEntries >= M * (T - 1), "Too few worker entries" );
366  }
367  else {
368  ASSERT( theStats.m_entries >= M * M * T, "Too few on_entry calls" );
369  ASSERT( theStats.m_entries <= M * (P + 1), "Too many on_entry calls" );
370  ASSERT( theStats.m_workerEntries >= M * M * (T - 1), "Too few worker entries" );
371  ASSERT( theStats.m_workerEntries <= M * (P - 1), "Too many worker entries" );
372  }
373  ASSERT( theStats.m_entries == theStats.m_exits, "Entries/exits mismatch" );
374  }
375  else {
376  ASSERT( theStats.m_entries >= M, "Too few on_entry calls" );
377  ASSERT( theStats.m_exits >= M || (testMode & tmAutoinitialization), "Too few on_exit calls" );
378  if ( !(testMode & tmLocalObservation) ) {
379  ASSERT( theStats.m_entries <= M * M * P, "Too many on_entry calls" );
380  ASSERT( theStats.m_exits <= M * M * T, "Too many on_exit calls" );
381  }
382  ASSERT( theStats.m_entries >= theStats.m_exits, "More exits than entries" );
383  }
384 }
