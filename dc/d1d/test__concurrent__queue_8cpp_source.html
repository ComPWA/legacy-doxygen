163 struct Body: NoAssign {
164  CQ* queue;
165  const int nthread;
166  Body( int nthread_ ) : nthread(nthread_) {}
167  void operator()( int thread_id ) const {
168  long pop_kind[3] = {0,0,0};
169  int serial[MAXTHREAD+1];
170  memset( serial, 0, nthread*sizeof(int) );
171  ASSERT( thread_id<nthread, NULL );
172 
173  long sum = 0;
174  for( long j=0; j<M; ++j ) {
175  T f;
176  f.thread_id = DEAD;
177  f.serial = DEAD;
178  bool prepopped = false;
179  if( j&1 ) {
180  prepopped = queue->try_pop( f );
181  ++pop_kind[prepopped];
182  }
183  T g;
184  g.thread_id = thread_id;
185  g.serial = j+1;
186  push( *queue, g, j );
187  if( !prepopped ) {
188  while( !(queue)->try_pop(f) ) __TBB_Yield();
189  ++pop_kind[2];
190  }
191  ASSERT( f.thread_id<=nthread, NULL );
192  ASSERT( f.thread_id==nthread || serial[f.thread_id]<f.serial, "partial order violation" );
193  serial[f.thread_id] = f.serial;
194  sum += f.serial-1;
195  }
196  Sum[thread_id] = sum;
197  for( int k=0; k<3; ++k )
198  PopKind[k] += pop_kind[k];

374  if( signed_prefill+1>=capacity )

377  for( int k=0; k<3; ++k )
378  PopKind[k] = 0;
379  for( int trial=0; !success; ++trial ) {
380  T::clear_counters();
381  Body<CQ,T> body(nthread);
382  CQ queue;
383  queue.set_capacity( capacity );

386  T f;
387  f.thread_id = nthread;
388  f.serial = 1+int(i);
389  push(queue, f, i);
390  ASSERT( unsigned(queue.size())==i+1, NULL );
391  ASSERT( !queue.empty(), NULL );
392  }
393  tbb::tick_count t0 = tbb::tick_count::now();
394  NativeParallelFor( nthread, body );
395  tbb::tick_count t1 = tbb::tick_count::now();
396  double timing = (t1-t0).seconds();
397  REMARK("prefill=%d capacity=%d threads=%d time = %g = %g nsec/operation\n", int(prefill), int(capacity), nthread, timing, timing/(2*M*nthread)*1.E9);
398  int sum = 0;
399  for( int k=0; k<nthread; ++k )
400  sum += Sum[k];
401  int expected = int(nthread*((M-1)*M/2) + ((prefill-1)*prefill)/2);
402  for( int i=int(prefill); --i>=0; ) {
403  ASSERT( !queue.empty(), NULL );
404  T f;
405  bool result = queue.try_pop(f);
406  ASSERT( result, NULL );
407  ASSERT( int(queue.size())==i, NULL );
408  sum += f.serial-1;
409  }
410  ASSERT( queue.empty(), "The queue should be empty" );
411  ASSERT( queue.size()==0, "The queue should have zero size" );
412  if( sum!=expected )
