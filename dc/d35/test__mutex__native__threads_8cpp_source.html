46 {
47  if( mode&1 ) {
48  // Try implicit acquire and explicit release
49  typename mutex_type::scoped_lock lock(mutex);
50  value = value+1;
51  lock.release();

60 template<typename M, long N>
61 struct Invariant {
62  typedef M mutex_type;
63  M mutex;
64  const char* mutex_name;
65  volatile long value[N];
66  Invariant( const char* mutex_name_ ) :
67  mutex_name(mutex_name_)
68  {
69  for( long k=0; k<N; ++k )
70  value[k] = 0;
71  }
72  void update() {
73  for( long k=0; k<N; ++k )
74  ++value[k];
75  }
76  bool value_is( long expected_value ) const {
77  long tmp;
78  for( long k=0; k<N; ++k )

80  REPORT("ERROR: %ld!=%ld\n", tmp, expected_value);
81  return false;
82  }
83  return true;
84  }
85  bool is_okay() {

87  }

89 };

91 template<typename M, long N>
92 void Invariant<M,N>::flog_once( size_t mode )
93 {
95  bool write = (mode%8)==7;
96  bool okay = true;
97  bool lock_kept = true;
98  if( (mode/8)&1 ) {
99  // Try implicit acquire and explicit release
100  typename mutex_type::scoped_lock lock(mutex,write);
101  if( write ) {
102  long my_value = value[0];
103  update();
104  if( mode%16==7 ) {
105  lock_kept = lock.downgrade_to_reader();
106  if( !lock_kept )
107  my_value = value[0] - 1;
108  okay = value_is(my_value+1);
109  }
110  } else {
111  okay = is_okay();
112  if( mode%8==3 ) {
113  long my_value = value[0];
114  lock_kept = lock.upgrade_to_writer();
115  if( !lock_kept )
116  my_value = value[0];
117  update();
118  okay = value_is(my_value+1);
119  }
120  }
121  lock.release();
122  } else {
123  // Try explicit acquire and implicit release
124  typename mutex_type::scoped_lock lock;
125  lock.acquire(mutex,write);
126  if( write ) {
127  long my_value = value[0];
128  update();
129  if( mode%16==7 ) {
130  lock_kept = lock.downgrade_to_reader();
131  if( !lock_kept )
132  my_value = value[0] - 1;
133  okay = value_is(my_value+1);
134  }
135  } else {
136  okay = is_okay();
137  if( mode%8==3 ) {
138  long my_value = value[0];
139  lock_kept = lock.upgrade_to_writer();
140  if( !lock_kept )
141  my_value = value[0];
142  update();
143  okay = value_is(my_value+1);
144  }
145  }
146  }
147  if( !okay ) {
148  REPORT( "ERROR for %s at %ld: %s %s %s %s\n",mutex_name, long(mode),
149  write?"write,":"read,", write?(mode%16==7?"downgrade,":""):(mode%8==3?"upgrade,":""),
150  lock_kept?"lock kept,":"lock not kept,", (mode/8)&1?"imp/exp":"exp/imp" );
151  }
152 }
