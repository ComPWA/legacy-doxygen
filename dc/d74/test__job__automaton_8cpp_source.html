66 void State::exercise( bool is_owner ) {
67  barrier.wait();
68  if( is_owner ) {
69  Cover(0);
70  if( ja.try_acquire() ) {
71  Cover(1);
72  ++job_created;
73  ja.set_and_release(&job);
74  Cover(2);
75  if( ja.try_acquire() ) {
76  Cover(3);
77  ja.release();
78  Cover(4);
79  if( ja.try_acquire() ) {
80  Cover(5);
81  ja.release();
82  }
83  }
84  Cover(6);
85  } else {
86  Cover(7);
87  }
88  if( DelayMask&1<<N ) {
89  while( !job_received )
90  __TBB_Yield();
91  }
92  } else {
93  // Using extra bit of DelayMask for choosing whether to run wait_for_job or not.
94  if( DelayMask&1<<N ) {
95  rml::job* j= ja.wait_for_job();
96  if( j!=&job ) REPORT("%p\n",j);
97  ASSERT( j==&job, NULL );
98  job_received = true;
99  }
100  Cover(8);
101  }
102  rml::job* j;
103  if( ja.try_plug(j) ) {
104  ASSERT( j==&job || !j, NULL );
105  if( j ) {
106  Cover(9+is_owner);
107  ++job_destroyed;
108  } else {
109  __TBB_ASSERT( !is_owner, "owner failed to create job but plugged self" );
110  Cover(11);
111  }
112  } else {
113  Cover(12+is_owner);
114  }
115 }
