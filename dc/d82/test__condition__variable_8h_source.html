118 template<typename M>
119 void Counter<M>::flog_once_lock_guard(size_t mode)
121 {
122  if( mode&1 ) {
123  // Try acquire and release with implicit lock_guard
124  // precondition: if mutex_type is not a recursive mutex, the calling thread does not own the mutex m.
125  // if the precondition is not met, either dead-lock incorrect 'value' would result in.
126  lock_guard<M> lg(mutex);
127  value = value+1;
128  } else {
129  // Try acquire and release with adopt lock_quard
130  // precodition: the calling thread owns the mutex m.
131  // if the precondition is not met, incorrect 'value' would result in because the thread unlocks
132  // mutex that it does not own.
133  mutex.lock();
134  lock_guard<M> lg( mutex, adopt_lock );
135  value = value+1;
136  }
137 }

139 template<typename M>
140 void Counter<M>::flog_once_unique_lock(size_t mode)
142 {
143  switch( mode&7 ) {
144  case 0:
145  {// implicitly acquire and release mutex with unique_lock
146  unique_lock<M> ul( mutex );
147  value = value+1;
148  ASSERT( ul==true, NULL );
149  }
150  break;
151  case 1:
152  {// unique_lock with defer_lock
153  unique_lock<M> ul( mutex, defer_lock );
154  ASSERT( ul.owns_lock()==false, NULL );
155  ul.lock();
156  value = value+1;
157  ASSERT( ul.owns_lock()==true, NULL );
158  }
159  break;
160  case 2:
161  {// unique_lock::try_lock() with try_to_lock
162  unique_lock<M> ul( mutex, try_to_lock );
163  if( !ul )
164  while( !ul.try_lock() )
165  __TBB_Yield();
166  value = value+1;
167  }
168  break;
169  case 3:
170  {// unique_lock::try_lock_for() with try_to_lock
171  unique_lock<M> ul( mutex, defer_lock );
172  tbb::tick_count::interval_t i(1.0);
173  while( !ul.try_lock_for( i ) )
174  ;
175  value = value+1;
176  ASSERT( ul.owns_lock()==true, NULL );
177  }
178  break;
179  case 4:
180  {
181  unique_lock<M> ul_o4;
182  {// unique_lock with adopt_lock
183  mutex.lock();
184  unique_lock<M> ul( mutex, adopt_lock );
185  value = value+1;
186  ASSERT( ul.owns_lock()==true, NULL );
187  ASSERT( ul.mutex()==&mutex, NULL );
188  ASSERT( ul_o4.owns_lock()==false, NULL );
189  ASSERT( ul_o4.mutex()==NULL, NULL );
190  swap( ul, ul_o4 );
191  ASSERT( ul.owns_lock()==false, NULL );
192  ASSERT( ul.mutex()==NULL, NULL );
193  ASSERT( ul_o4.owns_lock()==true, NULL );
194  ASSERT( ul_o4.mutex()==&mutex, NULL );
195  ul_o4.unlock();
196  }
197  ASSERT( ul_o4.owns_lock()==false, NULL );
198  }
199  break;
200  case 5:
201  {
202  unique_lock<M> ul_o5;
203  {// unique_lock with adopt_lock
204  mutex.lock();
205  unique_lock<M> ul( mutex, adopt_lock );
206  value = value+1;
207  ASSERT( ul.owns_lock()==true, NULL );
208  ASSERT( ul.mutex()==&mutex, NULL );
209  ASSERT( ul_o5.owns_lock()==false, NULL );
210  ASSERT( ul_o5.mutex()==NULL, NULL );
211  ul_o5.swap( ul );
212  ASSERT( ul.owns_lock()==false, NULL );
213  ASSERT( ul.mutex()==NULL, NULL );
214  ASSERT( ul_o5.owns_lock()==true, NULL );
215  ASSERT( ul_o5.mutex()==&mutex, NULL );
216  ul_o5.unlock();
217  }
218  ASSERT( ul_o5.owns_lock()==false, NULL );
219  }
220  break;
221  default:
222  {// unique_lock with adopt_lock, and release()
223  mutex.lock();
224  unique_lock<M> ul( mutex, adopt_lock );
225  ASSERT( ul==true, NULL );
226  value = value+1;
227  M* old_m = ul.release();
228  old_m->unlock();
229  ASSERT( ul.owns_lock()==false, NULL );
230  }
231  break;
232  }
233 }

240  State& state;
241  WorkForLocks( State& state_ ) : state(state_) {}

243  size_t step;

498 template<typename M>
499 struct WorkForCondVarWaitAndNotifyOne: NoAssign {
500  int nthread;
501  condition_variable& test_cv;
502  M& my_mtx;
503  WorkForCondVarWaitAndNotifyOne( int n_, condition_variable& cv_, M& mtx_ ) : nthread(n_), test_cv(cv_), my_mtx(mtx_) {}
504  void operator()( int tid ) const {
505  if( tid&1 ) {
506  // exercise signal part
507  while( ticket_for_wakeup<max_ticket ) {
508  int my_ticket = ++ticket_for_wakeup; // atomically grab the next ticket
509  if( my_ticket>max_ticket )
510  break;
511 
512  for( ;; ) {
513  unique_lock<M> ul( my_mtx, defer_lock );
514  ul.lock();
515  if( n_waiters>0 && my_ticket<=ticket_for_sleep && my_ticket==(wokeup_ticket+1) ) {
516  signaled_ticket = my_ticket;
517  test_cv.notify_one();
518  ++n_signaled;
519  ul.unlock();
520  break;
521  }
522  ul.unlock();
523  __TBB_Yield();
524  }
525 
526  // give waiters time to go to sleep.
527  for( int m=0; m<short_delay; ++m )
528  __TBB_Yield();
529  }
530  } else {
531  while( ticket_for_sleep<max_ticket ) {
532  unique_lock<M> ul( my_mtx, defer_lock );
533  ul.lock();
534  // exercise wait part
535  int my_ticket = ++ticket_for_sleep; // grab my ticket
536  if( my_ticket>max_ticket ) break;
537 
538  // each waiter should go to sleep at least once
539  unsigned nw = ++n_waiters;
540  for( ;; ) {
541  // update to max_waitq_length
542  if( nw>max_waitq_length ) max_waitq_length = nw;
543  ++n_visit_to_waitq;
544  test_cv.wait( ul );
545  // if( ret==false ) ++n_timedout;
546  ASSERT( ul, "mutex should have been locked" );
547  --n_waiters;
548  if( signaled_ticket==my_ticket ) {
549  wokeup_ticket = my_ticket;
550  break;
551  }
552  if( n_waiters>0 )
553  test_cv.notify_one();
554  nw = ++n_waiters; // update to max_waitq_length occurs above
555  }
556 
557  ul.unlock();
558  __TBB_Yield(); // give other threads chance to run.
559  }
560  }
561  ++n_done;
562  spin_wait_until_eq( n_done, nthread );
563  ASSERT( n_signaled==max_ticket, "incorrect number of notifications sent" );
564  }
