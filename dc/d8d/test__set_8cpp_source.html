110 template <typename T1, typename T2, typename Compare>
111 void
112 test_set(Compare compare)
113 {
114 
115  const std::size_t n_max = 100000;
116 
117  // The rand()%(2*n+1) encourages generation of some duplicates.
118  std::srand(4200);
119 
120  for (std::size_t n = 0; n < n_max; n = n <= 16 ? n + 1 : size_t(3.1415 * n))
121  {
122  for (std::size_t m = 0; m < n_max; m = m <= 16 ? m + 1 : size_t(2.71828 * m))
123  {
124  //prepare the input ranges
125  Sequence<T1> in1(n, [n](std::size_t k) { return rand() % (2 * k + 1); });
126  Sequence<T2> in2(m, [m](std::size_t k) { return (m % 2) * rand() + rand() % (k + 1); });
127 
128  std::sort(in1.begin(), in1.end(), compare);
129  std::sort(in2.begin(), in2.end(), compare);
130 
131  invoke_on_all_policies(test_one_policy(), in1.begin(), in1.end(), in2.cbegin(), in2.cend(), compare);
132  }
133  }
134 }
