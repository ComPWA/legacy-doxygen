192 template< typename T >
193 void RunParallelScalarTests(const char *test_name) {
194 
195  tbb::task_scheduler_init init(tbb::task_scheduler_init::deferred);
196  for (int p = MinThread; p <= MaxThread; ++p) {
197 
198  if (p == 0) continue;
199  REMARK(" Testing parallel %s on %d thread(s)...\n", test_name, p);
200  init.initialize(p);
201 
202  tbb::tick_count t0;
203  T combine_sum(0);
204  T combine_ref_sum(0);
205  T combine_finit_sum(0);
206  T combine_each_sum(0);
207  T copy_construct_sum(0);
208  T copy_assign_sum(0);
209 #if __TBB_ETS_USE_CPP11
210  T move_construct_sum(0);
211  T move_assign_sum(0);
212 #endif
213  for (int t = -1; t < REPETITIONS; ++t) {
214  if (Verbose && t == 0) t0 = tbb::tick_count::now();
215 
216  // test uninitialized parallel combinable
217  tbb::combinable<T> sums;
218  tbb::parallel_for( tbb::blocked_range<int>( 0, N, 10000 ), ParallelScalarBody<T>( sums ) );
219  combine_sum += sums.combine(my_combine<T>);
220  combine_ref_sum += sums.combine(my_combine_ref<T>);
221 
222  // test parallel combinable preinitialized with a functor that returns 0
223  FunctorAddFinit<T> my_finit_decl;
224  tbb::combinable<T> finit_combinable(my_finit_decl);
225  tbb::parallel_for( tbb::blocked_range<int>( 0, N, 10000 ), ParallelScalarBodyNoInit<T>( finit_combinable ) );
226  combine_finit_sum += finit_combinable.combine(my_combine<T>);
227 
228  // test another way of combining the elements using CombineEachHelper<T> functor
229  CombineEachHelper<T> my_helper(combine_each_sum);
230  sums.combine_each(my_helper);
231 
232  // test copy constructor for parallel combinable
233  tbb::combinable<T> copy_constructed(sums);
234  copy_construct_sum += copy_constructed.combine(my_combine<T>);
235 
236  // test copy assignment for uninitialized parallel combinable
237  tbb::combinable<T> assigned;
238  assigned = sums;
239  copy_assign_sum += assigned.combine(my_combine<T>);
240 
241 #if __TBB_ETS_USE_CPP11
242  // test move constructor for parallel combinable
243  tbb::combinable<T> moved1(std::move(sums));
244  move_construct_sum += moved1.combine(my_combine<T>);
245 
246  // test move assignment for uninitialized parallel combinable
247  tbb::combinable<T> moved2;
248  moved2=std::move(finit_combinable);
249  move_assign_sum += moved2.combine(my_combine<T>);
250 #endif
251  }
252  // Here and below comparison for equality of float numbers succeeds
253  // as the rounding error doesn't accumulate and doesn't affect the comparison
254  ASSERT( EXPECTED_SUM == combine_sum, NULL);
255  ASSERT( EXPECTED_SUM == combine_ref_sum, NULL);
256  ASSERT( EXPECTED_SUM == combine_finit_sum, NULL);
257  ASSERT( EXPECTED_SUM == combine_each_sum, NULL);
258  ASSERT( EXPECTED_SUM == copy_construct_sum, NULL);
259  ASSERT( EXPECTED_SUM == copy_assign_sum, NULL);
260 #if __TBB_ETS_USE_CPP11
261  ASSERT( EXPECTED_SUM == move_construct_sum, NULL);
262  ASSERT( EXPECTED_SUM == move_assign_sum, NULL);
263 #endif
264  REMARK(" done parallel %s, %d, %g, %g\n", test_name, p, static_cast<double>(combine_sum),
265  ( tbb::tick_count::now() - t0).seconds());
266  init.terminate();
267  }
