106  void trace ( const char* fmt, ... ) {
107  char msg[MAX_TRACE_SIZE];
108  char msg_fmt_buf[MAX_TRACE_SIZE];
109  const char *msg_fmt = fmt;
110  if ( m_flags & prefix ) {
111  snprintf (msg_fmt_buf, MAX_TRACE_SIZE, "[%s] %s", m_func, fmt);
112  msg_fmt = msg_fmt_buf;
113  }
114  std::va_list argptr;
115  va_start (argptr, fmt);
116  int len = vsnprintf (msg, MAX_TRACE_SIZE, msg_fmt, argptr);
117  va_end (argptr);
118  if ( m_flags & need_lf &&
119  len < MAX_TRACE_SIZE - 1 && msg_fmt[len-1] != '\n' )
120  {
121  msg[len] = '\n';
122  msg[len + 1] = 0;
123  }
124  m_reporter.Report(msg);
125  }
126  }; // class Tracer
