00075     // Load the XML file into the property tree. If reading fails

00079     // Get the filename and store it in the m_file variable.
00080     // Note that we construct the path to the value by separating
00081     // the individual keys with dots. If dots appear in the keys,
00082     // a path type with a different separator can be used.
00083     // If the amplitude_setup.filename key is not found, an

00085     m_file = pt.get<std::string>("amplitude_setup.filename");
00086 
00087     // Iterate over the amplitude_setup.resonances section and
00088     // store all found resonance names in the m_name set.
00089     // The get_child() function returns a reference to the child
00090     // at the specified path; if there is no such child, it throws.
00091     // Property tree iterators are models of BidirectionalIterator.
00092     BOOST_FOREACH( ptree::value_type const& v, pt.get_child("amplitude_setup") ) {
00093         if( v.first == "resonance" ) {
00094             Resonance f;
00095             f.m_name = v.second.get<std::string>("name");
00096             f.m_mass = v.second.get<double>("mass");
00097             f.m_mass_min = v.second.get<double>("mass_min");;
00098             f.m_mass_max = v.second.get<double>("mass_max");;

00101             f.m_width_min = v.second.get<double>("width_min");;
00102             f.m_width_max = v.second.get<double>("width_max");;
00103             f.m_strength = v.second.get<double>("strength");;
