67 template<typename C>
68 struct AddOne: NoAssign {
69  C& counter;
71  void operator()( tbb::blocked_range<size_t>& range ) const {
72  for( size_t i=range.begin(); i!=range.end(); ++i ) {
73  if( i&1 ) {
74  // Try implicit acquire and explicit release
75  typename C::mutex_type::scoped_lock lock(counter.mutex);
76  counter.value = counter.value+1;
77  lock.release();
78  } else {
79  // Try explicit acquire and implicit release
80  typename C::mutex_type::scoped_lock lock;
81  lock.acquire(counter.mutex);
82  counter.value = counter.value+1;
83  }
84  }
85  }
86  AddOne( C& counter_ ) : counter(counter_) {}
87 };

91 class TBB_MutexFromISO_Mutex {
92  M my_iso_mutex;
93 public:
94  typedef TBB_MutexFromISO_Mutex mutex_type;

96  class scoped_lock;
97  friend class scoped_lock;

99  class scoped_lock {
100  mutex_type* my_mutex;
101  public:
102  scoped_lock() : my_mutex(NULL) {}
103  scoped_lock( mutex_type& m ) : my_mutex(NULL) {
104  acquire(m);
105  }
106  scoped_lock( mutex_type& m, bool is_writer ) : my_mutex(NULL) {
107  acquire(m,is_writer);
108  }
109  void acquire( mutex_type& m ) {
110  m.my_iso_mutex.lock();
111  my_mutex = &m;
112  }
113  bool try_acquire( mutex_type& m ) {
114  if( m.my_iso_mutex.try_lock() ) {
115  my_mutex = &m;
116  return true;
117  } else {
118  return false;
119  }
120  }
121  void release() {
122  my_mutex->my_iso_mutex.unlock();
123  my_mutex = NULL;
124  }
125 
126  // Methods for reader-writer mutex
127  // These methods can be instantiated only if M supports lock_read() and try_lock_read().
128 
129  void acquire( mutex_type& m, bool is_writer ) {
130  if( is_writer ) m.my_iso_mutex.lock();
131  else m.my_iso_mutex.lock_read();
132  my_mutex = &m;
133  }
134  bool try_acquire( mutex_type& m, bool is_writer ) {
135  if( is_writer ? m.my_iso_mutex.try_lock() : m.my_iso_mutex.try_lock_read() ) {
136  my_mutex = &m;
137  return true;
138  } else {
139  return false;
140  }
141  }
142  bool upgrade_to_writer() {
143  my_mutex->my_iso_mutex.unlock();
144  my_mutex->my_iso_mutex.lock();
145  return false;
146  }
147  bool downgrade_to_reader() {
148  my_mutex->my_iso_mutex.unlock();
149  my_mutex->my_iso_mutex.lock_read();
150  return false;
151  }
152  ~scoped_lock() {
153  if( my_mutex )
154  release();
155  }
156  };

158  static const bool is_recursive_mutex = M::is_recursive_mutex;
159  static const bool is_rw_mutex = M::is_rw_mutex;
160 };

190 template<typename M, size_t N>
191 struct Invariant {
192  typedef M mutex_type;
193  M mutex;
194  const char* mutex_name;
195  volatile long value[N];
196  Invariant( const char* mutex_name_ ) :
197  mutex_name(mutex_name_)
198  {
199  for( size_t k=0; k<N; ++k )
200  value[k] = 0;
201  tbb::profiling::set_name(mutex, mutex_name_);
202  }
203  ~Invariant() {
204  }
205  void update() {
206  for( size_t k=0; k<N; ++k )
207  ++value[k];
208  }
209  bool value_is( long expected_value ) const {
210  long tmp;
211  for( size_t k=0; k<N; ++k )
212  if( (tmp=value[k])!=expected_value ) {
213  REPORT("ERROR: %ld!=%ld\n", tmp, expected_value);
214  return false;
215  }
216  return true;
217  }
218  bool is_okay() {
219  return value_is( value[0] );
220  }
221 };

224 template<typename I>
225 struct TwiddleInvariant: NoAssign {
226  I& invariant;
227  TwiddleInvariant( I& invariant_ ) : invariant(invariant_) {}
228 
230  void operator()( tbb::blocked_range<size_t>& range ) const {
231  for( size_t i=range.begin(); i!=range.end(); ++i ) {
233  const bool write = (i%8)==7;
234  bool okay = true;
235  bool lock_kept = true;
236  if( (i/8)&1 ) {
237  // Try implicit acquire and explicit release
238  typename I::mutex_type::scoped_lock lock(invariant.mutex,write);
239  execute_aux(lock, i, write, /*ref*/okay, /*ref*/lock_kept);
240  lock.release();
241  } else {
242  // Try explicit acquire and implicit release
243  typename I::mutex_type::scoped_lock lock;
244  lock.acquire(invariant.mutex,write);
245  execute_aux(lock, i, write, /*ref*/okay, /*ref*/lock_kept);
246  }
247  if( !okay ) {
248  REPORT( "ERROR for %s at %ld: %s %s %s %s\n",invariant.mutex_name, long(i),
249  write ? "write," : "read,",
250  write ? (i%16==7?"downgrade,":"") : (i%8==3?"upgrade,":""),
251  lock_kept ? "lock kept," : "lock not kept,", // TODO: only if downgrade/upgrade
252  (i/8)&1 ? "impl/expl" : "expl/impl" );
253  }
254  }
255  }
256 private:
257  void execute_aux(typename I::mutex_type::scoped_lock & lock, const size_t i, const bool write, bool & okay, bool & lock_kept) const {
258  if( write ) {
259  long my_value = invariant.value[0];
260  invariant.update();
261  if( i%16==7 ) {
262  lock_kept = lock.downgrade_to_reader();
263  if( !lock_kept )
264  my_value = invariant.value[0] - 1;
265  okay = invariant.value_is(my_value+1);
266  }
267  } else {
268  okay = invariant.is_okay();
269  if( i%8==3 ) {
270  long my_value = invariant.value[0];
271  lock_kept = lock.upgrade_to_writer();
272  if( !lock_kept )
273  my_value = invariant.value[0];
274  invariant.update();
275  okay = invariant.value_is(my_value+1);
276  }
277  }
278  }
279 };

366 const int RecurN = 4;
367 int RecurArray[ RecurN ];
368 tbb::recursive_mutex RecurMutex[ RecurN ];
369 
370 struct RecursiveAcquisition {
374  void Body( size_t x, int max_lock=-1, unsigned int mask=0 ) const
375  {
376  int i = (int) (x % RecurN);
377  bool first = (mask&1U<<i)==0;
378  if( first ) {
379  // first time to acquire lock
380  if( i<max_lock )
381  // out of order acquisition might lead to deadlock, so stop
382  return;
383  max_lock = i;
384  }
385 
386  if( (i&1)!=0 ) {
387  // acquire lock on location RecurArray[i] using explicit acquire
388  tbb::recursive_mutex::scoped_lock r_lock;
389  r_lock.acquire( RecurMutex[i] );
390  int a = RecurArray[i];
391  ASSERT( (a==0)==first, "should be either a==0 if it is the first time to acquire the lock or a!=0 otherwise" );
392  ++RecurArray[i];
393  if( x )
394  Body( x/RecurN, max_lock, mask|1U<<i );
395  --RecurArray[i];
396  ASSERT( a==RecurArray[i], "a is not equal to RecurArray[i]" );
397 
398  // release lock on location RecurArray[i] using explicit release; otherwise, use implicit one
399  if( (i&2)!=0 ) r_lock.release();
400  } else {
401  // acquire lock on location RecurArray[i] using implicit acquire
402  tbb::recursive_mutex::scoped_lock r_lock( RecurMutex[i] );
403  int a = RecurArray[i];
404 
405  ASSERT( (a==0)==first, "should be either a==0 if it is the first time to acquire the lock or a!=0 otherwise" );
406 
407  ++RecurArray[i];
408  if( x )
409  Body( x/RecurN, max_lock, mask|1U<<i );
410  --RecurArray[i];
411 
412  ASSERT( a==RecurArray[i], "a is not equal to RecurArray[i]" );
413 
414  // release lock on location RecurArray[i] using explicit release; otherwise, use implicit one
415  if( (i&2)!=0 ) r_lock.release();
416  }
417  }
418 
419  void operator()( const tbb::blocked_range<size_t> &r ) const
420  {
421  for( size_t x=r.begin(); x<r.end(); x++ ) {
422  Body( x );
423  }
424  }
425 };

440 template<typename C>
441 struct NullRecursive: NoAssign {
442  void recurse_till( size_t i, size_t till ) const {
443  if( i==till ) {
444  counter.value = counter.value+1;
445  return;
446  }
447  if( i&1 ) {
448  typename C::mutex_type::scoped_lock lock2(counter.mutex);
449  recurse_till( i+1, till );
450  lock2.release();
451  } else {
452  typename C::mutex_type::scoped_lock lock2;
453  lock2.acquire(counter.mutex);
454  recurse_till( i+1, till );
455  }
456  }

459  typename C::mutex_type::scoped_lock lock(counter.mutex);
460  recurse_till( range.begin(), range.end() );
461  }
462  NullRecursive( C& counter_ ) : counter(counter_) {
463  ASSERT( C::mutex_type::is_recursive_mutex, "Null mutex should be a recursive mutex." );
464  }
