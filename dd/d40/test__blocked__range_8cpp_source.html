133 void TestProportionalSplitOverflow()
134 {
135  REMARK("Testing overflow during proportional split - ");
136  using tbb::blocked_range;
137  using tbb::proportional_split;
138 
139  blocked_range<size_t> r1(0, size_t(-1) / 2);
140  size_t size = r1.size();
141  size_t begin = r1.begin();
142  size_t end = r1.end();
143 
144  proportional_split p(1, 3);
145  blocked_range<size_t> r2(r1, p);
146 
147  // overflow-free computation
148  size_t parts = p.left() + p.right();
149  size_t int_part = size / parts;
150  size_t fraction = size - int_part * parts; // fraction < parts
151  size_t right_idx = int_part * p.right() + fraction * p.right() / parts + 1;
152  size_t newRangeBegin = end - right_idx;
153 
154  // Division in 'right_idx' very likely is inexact also.
155  size_t tolerance = 1;
156  size_t diff = (r2.begin() < newRangeBegin) ? (newRangeBegin - r2.begin()) : (r2.begin() - newRangeBegin);
157  bool is_split_correct = diff <= tolerance;
158  bool test_passed = (r1.begin() == begin && r1.end() == r2.begin() && is_split_correct &&
159  r2.end() == end);
160  if (!test_passed) {
161  REPORT("Incorrect split of blocked range[%lu, %lu) into r1[%lu, %lu) and r2[%lu, %lu), "
162  "must be r1[%lu, %lu) and r2[%lu, %lu)\n", begin, end, r1.begin(), r1.end(), r2.begin(), r2.end(), begin, newRangeBegin, newRangeBegin, end);
163  ASSERT(test_passed, NULL);
164  }
165  REMARK("OK\n");
166 }
167 #endif /* __TBB_USE_PROPORTIONAL_SPLIT_IN_BLOCKED_RANGES */
