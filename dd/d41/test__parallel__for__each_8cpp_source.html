45 // Tests tbb::parallel_for_each functionality
46 template <typename Iterator>
47 void RunPForEachTests()
48 {
49  size_t test_vector[NUMBER_OF_ELEMENTS + 1];
50 
51  sum = 0;
52  size_t test_sum = 0;
53 
54  for (size_t i =0; i < NUMBER_OF_ELEMENTS; i++) {
55  test_vector[i] = i;
56  test_sum += i;
57  }
58  test_vector[NUMBER_OF_ELEMENTS] = 1000000; // parallel_for_each shouldn't touch this element
59 
60  Iterator begin(&test_vector[0]);
61  Iterator end(&test_vector[NUMBER_OF_ELEMENTS]);
62 
63  tbb::parallel_for_each(begin, end, (TestFunctionType)TestFunction);
64  ASSERT(sum == test_sum, "Not all items of test vector were processed by parallel_for_each");
65  ASSERT(test_vector[NUMBER_OF_ELEMENTS] == 1000000, "parallel_for_each processed an extra element");
66 }
