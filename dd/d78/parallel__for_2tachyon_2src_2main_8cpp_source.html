139  tachyon_video tachyon;

142 
143  // always using window even if(!global_usegraphics)
144  global_usegraphics =
145  tachyon.init_window(global_xwinsize, global_ywinsize);
146  if(!tachyon.running)
147  exit(-1);
148 
149  video = &tachyon;
150 
151  for(;;) {
152  global_elapsedTime = 0;
153  global_startTime=(long) time(NULL);
154  global_isCancelled=false;
155  if (video)video->running = true;
156  tbb::task_scheduler_init init (global_number_of_threads);
157  memset(g_pImg, 0, sizeof(unsigned int) * global_xsize * global_ysize);
