65 void InitializeAndTerminate( int maxthread ) {
66  __TBB_TRY {
67  for( int i=0; i<256; ++i ) {
68  int threads = (std::rand() % maxthread) + 1;
69  switch( i&3 ) {
70  default: {
71  tbb::task_scheduler_init init( threads );
72  ASSERT(init.is_active(), NULL);
73  ASSERT(ArenaConcurrency()==(threads==1?2:threads), NULL);
74  if (test_mandatory_parallelism)
75  Harness::ExactConcurrencyLevel::check(threads, Harness::ExactConcurrencyLevel::Serialize);
76  if(i&0x20) tbb::task::enqueue( (*new( tbb::task::allocate_root() ) TaskGenerator(2,6)) ); // a work deferred to workers
77  break;
78  }
79  case 0: {
80  tbb::task_scheduler_init init;
81  ASSERT(init.is_active(), NULL);
82  ASSERT(ArenaConcurrency()==(DefaultThreads==1?2:init.default_num_threads()), NULL);
83  if (test_mandatory_parallelism)
84  Harness::ExactConcurrencyLevel::check(init.default_num_threads(), Harness::ExactConcurrencyLevel::Serialize);
85  if(i&0x40) tbb::task::enqueue( (*new( tbb::task::allocate_root() ) TaskGenerator(3,5)) ); // a work deferred to workers
86  break;
87  }
88  case 1: {
89  tbb::task_scheduler_init init( tbb::task_scheduler_init::deferred );
90  ASSERT(!init.is_active(), "init should not be active; initialization was deferred");
91  init.initialize( threads );
92  ASSERT(init.is_active(), NULL);
93  ASSERT(ArenaConcurrency()==(threads==1?2:threads), NULL);
94  if (test_mandatory_parallelism)
95  Harness::ExactConcurrencyLevel::check(threads, Harness::ExactConcurrencyLevel::Serialize);
96  init.terminate();
97  ASSERT(!init.is_active(), "init should not be active; it was terminated");
98  break;
99  }
100  case 2: {
101  tbb::task_scheduler_init init( tbb::task_scheduler_init::automatic );
102  ASSERT(init.is_active(), NULL);
103  ASSERT(ArenaConcurrency()==(DefaultThreads==1?2:init.default_num_threads()), NULL);
104  if (test_mandatory_parallelism)
105  Harness::ExactConcurrencyLevel::check(init.default_num_threads(), Harness::ExactConcurrencyLevel::Serialize);
106  break;
107  }
108  }
109  }
110  } __TBB_CATCH( std::runtime_error& error ) {
111 #if TBB_USE_EXCEPTIONS
112  REPORT("ERROR: %s\n", error.what() );
113 #endif /* TBB_USE_EXCEPTIONS */
114  }

180  class TestMultpleWaitBody {
181  bool myAutoInit;
182  public:
183  TestMultpleWaitBody( bool autoInit = false ) : myAutoInit( autoInit ) {}
184  void operator()( int ) const {
185  tbb::task_scheduler_init init( tbb::task_scheduler_init::deferred );
186  if ( !myAutoInit )
187  init.initialize( tbb::task_scheduler_init::automatic );
188  Harness::FastRandom rnd( ++gSeed );
189  // In case of auto init sub-tests we skip
190  // - case #4 to avoid recursion
191  // - case #5 because it is explicit initialization
192  const int numCases = myAutoInit ? 4 : 6;
193  switch ( rnd.get() % numCases ) {
194  case 0: {
195  tbb::task_arena a;
196  a.enqueue( EmptyBody() );
197  break;
198  }
199  case 1: {
200  tbb::task_group tg;
201  tg.run( EmptyBody() );
202  tg.wait();
203  break;
204  }
205  case 2:
206  tbb::parallel_for( 0, 100, EmptyBody() );
207  break;
208  case 3:
209  /* do nothing */
210  break;
211  case 4:
212  // Create and join several threads with auto initialized scheduler.
213  NativeParallelFor( rnd.get() % 5 + 1, TestMultpleWaitBody( true ) );
214  break;
215  case 5:
216  {
217  tbb::task_scheduler_init init2;
218  bool res = init2.blocking_terminate( std::nothrow );
219  ASSERT( !res, NULL );
220  }
221  break;
222  }
223  if ( !myAutoInit && init.blocking_terminate( std::nothrow ) )
224  ++gNumSuccesses;
225  }
226  };

346 #if RML_USE_WCRM
347  REPORT("Known issue: shared RML for ConcRT does not support oversubscription\n");

349 #else
350  REPORT("Known issue: machine is heavy loaded or shared RML which does not support oversubscription is loaded\n");
351 #endif
352  }
353  }
354  InitializeAndTerminate( threads ); // test initialization of more than default number of threads
355  for( int p=MinThread; p<=MaxThread; ++p ) {
356  REMARK("testing with %d threads\n", p );
357  // protect market with excess threads from default initializations
358  // TODO IDEA: enhance task_scheduler_init to serve as global_control setting so that
359  // number of threads > default concurrency will be requested from market.
360  // Such settings must be aggregated via 'max' function and 'max_allowed_parallelism' control
361  // (which has 'min' aggregation) will have precedence over it.
362  tbb::task_scheduler_init init( tbb::task_scheduler_init::deferred );
363  if( MaxThread > DefaultThreads ) init.initialize( MaxThread );
364  NativeParallelFor( p, ThreadedInit() );
365  }
366  AssertExplicitInitIsNotSupplanted();
367 #if TBB_PREVIEW_WAITING_FOR_WORKERS
368  TestBlockingTerminate();
