104 void TestWorkers(size_t curr_par)
105 {
106  const size_t max_parallelism =
107  tbb::global_control::active_value(tbb::global_control::max_allowed_parallelism);
108  ASSERT(max(2U, tbb::tbb_thread::hardware_concurrency()) == max_parallelism, NULL);
109  {
110  const unsigned h_c = tbb::tbb_thread::hardware_concurrency();
111  tbb::global_control c(tbb::global_control::max_allowed_parallelism, curr_par);
112  size_t v = tbb::global_control::active_value(tbb::global_control::max_allowed_parallelism);
113  ASSERT(!curr_par || max((size_t)2, curr_par) == v, NULL);
114  if (h_c > 1)
115  TSI_and_RunWorkers(tbb::task_scheduler_init::automatic, min(h_c, curr_par), curr_par);
116  if (curr_par) // do not call task_scheduler_init t(0);
117  TSI_and_RunWorkers((int)curr_par, curr_par, curr_par);
118  if (curr_par > 2) { // check that min(tsi, parallelism) is active
119  TSI_and_RunWorkers((int)curr_par-1, curr_par, curr_par);
120  TSI_and_RunWorkers((int)curr_par, curr_par-1, curr_par);
121  }
122  // check constrains on control's value: it can't be increased
123  tbb::global_control c1(tbb::global_control::max_allowed_parallelism, curr_par+1);
124  v = tbb::global_control::active_value(tbb::global_control::max_allowed_parallelism);
125  if (curr_par)
126  ASSERT(max(2U, curr_par) == v, "It's impossible to increase maximal parallelism.");
127  else
128  ASSERT(2 == v, NULL);
129  }
130  ASSERT(tbb::global_control::active_value(tbb::global_control::max_allowed_parallelism)
131  == max_parallelism,
132  "max parallelism has been restored successfully after decreasing/increasing");
133 }
