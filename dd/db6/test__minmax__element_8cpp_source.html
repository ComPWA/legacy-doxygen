74 template <typename T>
75 struct sequence_wrapper
76 {
77  TestUtils::Sequence<T> seq;
78  const T min_value;
79  const T max_value;
80  static const std::size_t bits = 30; // We assume that T can handle signed 2^bits+1 value
81 
82  // TestUtils::HashBits returns value between 0 and (1<<bits)-1,
83  // therefore we could threat 1<<bits as maximum and -(1<<bits) as a minimum
84  sequence_wrapper(std::size_t n) : seq(n), min_value(-(1 << bits)), max_value(1 << bits) {}
85 
86  void
87  pattern_fill()
88  {
89  seq.fill([](std::size_t i) -> T { return T(TestUtils::HashBits(i, bits)); });
90  }
91 
92  // sets first one at position `at` and bunch of them farther
93  void
94  set_desired_value(std::size_t at, T value)
95  {
96  if (seq.size() == 0)
97  return;
98  seq[at] = value;
99 
100  //Producing serveral red herrings

102  seq[i] = value;
103  }

106 template <typename T>
107 void
108 test_by_type(std::size_t n)
109 {
110  sequence_wrapper<T> wseq(n);
111 
112  // to avoid overtesing we use std::set to leave only unique indexes
113  std::set<std::size_t> targets{0};
114  if (n > 1)
115  {
116  targets.insert(1);
117  targets.insert(2.718282 * n / 3);
118  targets.insert(n / 2);
119  targets.insert(n / 7.389056);
120  targets.insert(n - 1); // last
121  }
122 
123  for (std::set<std::size_t>::iterator it = targets.begin(); it != targets.end(); ++it)
124  {
125  wseq.pattern_fill();
126  wseq.set_desired_value(*it, wseq.min_value);
127  TestUtils::invoke_on_all_policies(check_minelement(), wseq.seq.cbegin(), wseq.seq.cend());
128  TestUtils::invoke_on_all_policies(check_minelement(), wseq.seq.begin(), wseq.seq.end());
129 
130  wseq.set_desired_value(*it, wseq.max_value);
131  TestUtils::invoke_on_all_policies(check_maxelement(), wseq.seq.cbegin(), wseq.seq.cend());
132  TestUtils::invoke_on_all_policies(check_maxelement(), wseq.seq.begin(), wseq.seq.end());
133 
134  if (targets.size() > 1)
135  {
136  for (std::set<std::size_t>::reverse_iterator rit = targets.rbegin(); rit != targets.rend(); ++rit)
137  {
138  if (*rit == *it) // we requires at least 2 unique indexes in targets
139  break;
140  wseq.pattern_fill();
141  wseq.set_desired_value(*it, wseq.min_value); // setting minimum element
142  wseq.set_desired_value(*rit, wseq.max_value); // setting maximum element
143  TestUtils::invoke_on_all_policies(check_minmaxelement(), wseq.seq.cbegin(), wseq.seq.cend());
144  TestUtils::invoke_on_all_policies(check_minmaxelement(), wseq.seq.begin(), wseq.seq.end());
145  }
146  }
147  else
148  { // we must check this corner case; it can not be tested in loop above
149  TestUtils::invoke_on_all_policies(check_minmaxelement(), wseq.seq.cbegin(), wseq.seq.cend());
150  TestUtils::invoke_on_all_policies(check_minmaxelement(), wseq.seq.begin(), wseq.seq.end());
151  }
152  }
153 }
