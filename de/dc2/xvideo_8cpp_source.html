21 // Uncomment next line to disable shared memory features if you do not have libXext
22 // (http://www.xfree86.org/current/mit-shm.html)
23 //#define X_NOSHMEM
24 
25 // Note that it may happen that the build environment supports the shared-memory extension
26 // (so there's no build-time reason to disable the relevant code by defining X_NOSHMEM),
27 // but that using shared memory still fails at run time.
28 // This situation will (ultimately) cause the error handler set by XSetErrorHandler()
29 // to be invoked with XErrorEvent::minor_code==X_ShmAttach. The code below tries to make
30 // such a determination at XShmAttach() time, which seems plausible, but unfortunately
31 // it has also been observed in a specific environment that the error may be reported
32 // at a later time instead, even after video::init_window() has returned.
33 // It is not clear whether this may happen in that way in any environment where it might
34 // depend on the kind of display, e.g., local vs. over "ssh -X", so #define'ing X_NOSHMEM
35 // may not always be the appropriate solution, therefore an environment variable
36 // has been introduced to disable shared memory at run time.
37 // A diagnostic has been added to advise the user about possible workarounds.
38 // X_ShmAttach macro was changed to 1 due to recent changes to X11/extensions/XShm.h header.

92 inline void mask2bits(unsigned int mask, unsigned int &save, depth_t &shift)
93 {
94  save = mask; if(!mask) { shift = dispdepth/3; return; }
95  shift = 0; while(!(mask&1)) ++shift, mask >>= 1;
96  int bits = 0; while(mask&1) ++bits, mask >>= 1;
97  shift += bits - 8;
98 }

122  { //enclose local variables before fail label
123  g_sizex = xsize; g_sizey = ysize;
124 
125  // Open the display
126  if (!dpy) {
127  dpy = XOpenDisplay(display_name);
128  if (!dpy) {
129  fprintf(stderr, "Can't open X11 display %s\n", XDisplayName(display_name));

131  }
132  }
133  int theScreen = DefaultScreen(dpy);
134  scrn = ScreenOfDisplay(dpy, theScreen);
135  dispdepth = DefaultDepth(dpy, theScreen);
136  XVisualInfo vinfo;
137  if (!( (dispdepth >= 15 && dispdepth <= 32 && XMatchVisualInfo(dpy, theScreen, dispdepth, TrueColor, &vinfo) )
138  || XMatchVisualInfo(dpy, theScreen, 24, TrueColor, &vinfo)
139  || XMatchVisualInfo(dpy, theScreen, 32, TrueColor, &vinfo)
140  || XMatchVisualInfo(dpy, theScreen, 16, TrueColor, &vinfo)
141  || XMatchVisualInfo(dpy, theScreen, 15, TrueColor, &vinfo)
142  )) {
143  fprintf(stderr, "Display has no appropriate True Color visual\n");
144  goto fail;
145  }
146  vis = vinfo.visual;
147  depth = dispdepth = vinfo.depth;
148  mask2bits(vinfo.red_mask, red_mask, red_shift);
149  mask2bits(vinfo.green_mask, green_mask, green_shift);
150  mask2bits(vinfo.blue_mask, blue_mask, blue_shift);
151  rootW = RootWindow(dpy, theScreen);
152  cmap = XCreateColormap(dpy, rootW, vis, AllocNone);
153  XSetWindowAttributes attrs;
154  attrs.backing_store = Always;
155  attrs.colormap = cmap;
156  attrs.event_mask = StructureNotifyMask|KeyPressMask|ButtonPressMask|ButtonReleaseMask;
157  attrs.background_pixel = BlackPixelOfScreen(scrn);
158  attrs.border_pixel = WhitePixelOfScreen(scrn);
159  win = XCreateWindow(dpy, rootW,
160  0, 0, xsize, ysize, 2,
161  dispdepth, InputOutput, vis,
162  CWBackingStore | CWColormap | CWEventMask |
163  CWBackPixel | CWBorderPixel,
164  &attrs);
165  if(!win) {
166  fprintf(stderr, "Can't create the window\n");
167  goto fail;
168  }
169  XSizeHints sh;
170  sh.flags = PSize | PMinSize | PMaxSize;
171  sh.width = sh.min_width = sh.max_width = xsize;
172  sh.height = sh.min_height = sh.max_height = ysize;
173  XSetStandardProperties( dpy, win, g_video->title, g_video->title, None, NULL, 0, &sh );
174  _XA_WM_DELETE_WINDOW = XInternAtom(dpy, "WM_DELETE_WINDOW", false);
175  XSetWMProtocols(dpy, win, &_XA_WM_DELETE_WINDOW, 1);
176  gc = XCreateGC(dpy, win, 0L, &xgcv);
177  XMapRaised(dpy, win);
178  XFlush(dpy);
179 #ifdef X_FULLSYNC
180  XSynchronize(dpy, true);
181 #endif
182  XSetErrorHandler(xerr_handler);
183 
184  int imgbytes = xsize*ysize*(dispdepth<=16?2:4);
185  const char *vidstr;
186 #ifndef X_NOSHMEM
187  int major, minor, pixmaps;
188  if(XShmQueryExtension(dpy) &&
189  XShmQueryVersion(dpy, &major, &minor, &pixmaps))
190  { // Shared memory
191  if(NULL!=getenv(NOSHMEM_env_var_name) && 0!=strcmp("0",getenv(NOSHMEM_env_var_name))) {
192  goto generic;
193  }
194  shmseginfo.shmid = shmget(IPC_PRIVATE, imgbytes, IPC_CREAT|0777);
195  if(shmseginfo.shmid < 0) {
196  fprintf(stderr, "Warning: Can't get shared memory: %s\n", strerror(errno));
197  goto generic;
198  }
199  g_pImg = (unsigned int*)(shmseginfo.shmaddr = (char*)shmat(shmseginfo.shmid, 0, 0));
200  if(g_pImg == (unsigned int*)-1) {
201  fprintf(stderr, "Warning: Can't attach to shared memory: %s\n", strerror(errno));
202  shmctl(shmseginfo.shmid, IPC_RMID, NULL);
203  goto generic;
204  }
205  shmseginfo.readOnly = false;
206  if(!XShmAttach(dpy, &shmseginfo) || x_error) {
207  char err[256]; XGetErrorText(dpy, x_error, err, 255);
208  fprintf(stderr, "Warning: Can't attach shared memory to display: %s (%d)\n", err, x_error);
209  shmdt(shmseginfo.shmaddr); shmctl(shmseginfo.shmid, IPC_RMID, NULL);
210  goto generic;
211  }
212  already_called_X_ShmAttach = true;
213 
214 #ifndef X_NOSHMPIX
215  if(pixmaps && XShmPixmapFormat(dpy) == ZPixmap)
216  { // Pixmaps
217  vidtype = 2; vidstr = "X11 shared memory pixmap";
218  pixmap = XShmCreatePixmap(dpy, win, (char*)g_pImg, &shmseginfo, xsize, ysize, dispdepth);
219  XSetWindowBackgroundPixmap(dpy, win, pixmap);
220  } else
221 #endif
222  { // Standard
223  vidtype = 1; vidstr = "X11 shared memory";
224  ximage = XShmCreateImage(dpy, vis, dispdepth,
225  ZPixmap, 0, &shmseginfo, xsize, ysize);
226  if(!ximage) {
227  fprintf(stderr, "Can't create the shared image\n");
228  goto fail;
229  }
230  assert(ximage->bytes_per_line == xsize*(dispdepth<=16?2:4));
231  ximage->data = shmseginfo.shmaddr;
232  }
233  } else
234 #endif
235  {
236 #ifndef X_NOSHMEM
237 generic:
238 #endif
239  vidtype = 0; vidstr = "generic X11";
240  g_pImg = new unsigned int[imgbytes/sizeof(int)];
241  ximage = XCreateImage(dpy, vis, dispdepth, ZPixmap, 0, (char*)g_pImg, xsize, ysize, 32, imgbytes/ysize);
242  if(!ximage) {
243  fprintf(stderr, "Can't create the image\n");
244  goto fail;
245  }
246  }
247  if( ximage ) {
248  // Note: It may be more efficient to adopt the server's byte order
249  // and swap once per get_color() call instead of once per pixel.
250  const uint32_t probe = 0x03020100;
251  const bool big_endian = (((const char*)(&probe))[0]==0x03);
252  ximage->byte_order = big_endian ? MSBFirst : LSBFirst;
253  }
254  printf("Note: using %s with %s visual for %d-bit color depth\n", vidstr, vis==DefaultVisual(dpy, theScreen)?"default":"non-default", dispdepth);
255  running = true;
256  return true;
257  } // end of enclosing local variables
258 fail:
259  terminate(); init_console();
260  return false;
261 }

273 void video::terminate()
274 {
275  running = false;
276  if(dpy) {
277  vidtype = 3; // stop video
278  if(threaded) { pthread_mutex_lock(&g_mutex); pthread_mutex_unlock(&g_mutex); }
279  if(ximage) { XDestroyImage(ximage); ximage = 0; g_pImg = 0; } // it frees g_pImg for vidtype == 0
280 #ifndef X_NOSHMEM
281  if(pixmap) XFreePixmap(dpy, pixmap);

286  if(win) XDestroyWindow(dpy, win);
287  XCloseDisplay(dpy); dpy = 0;
288  }
289  if(g_pImg) { delete[] g_pImg; g_pImg = 0; } // if was allocated for console mode
290 }

306 bool video::next_frame()
307 {
308  if(!running) return false;
310  if(vidtype == 3 || threaded && pthread_mutex_trylock(&g_mutex))
311  return running;
313  g_fps++;
314 #ifndef X_NOSHMPIX
315  if(vidtype == 2 && updating) XClearWindow(dpy, win);
316 #endif
317  while( XPending(dpy) ) {
318  XEvent report; XNextEvent(dpy, &report);
319  switch( report.type ) {
320  case ClientMessage:
321  if(report.xclient.format != 32 || report.xclient.data.l[0] != _XA_WM_DELETE_WINDOW) break;
322  case DestroyNotify:
323  running = false;
324  case KeyPress:
325  on_key( XLookupKeysym(&report.xkey, 0) ); break;
326  case ButtonPress:
327  on_mouse( report.xbutton.x, report.xbutton.y, report.xbutton.button ); break;
328  case ButtonRelease:
329  on_mouse( report.xbutton.x, report.xbutton.y, -report.xbutton.button ); break;
330  }
331  }
332  struct timezone tz; struct timeval now_time; gettimeofday(&now_time, &tz);
333  double sec = (now_time.tv_sec+1.0*now_time.tv_usec/1000000.0) - (g_time.tv_sec+1.0*g_time.tv_usec/1000000.0);
334  if(sec > 1) {
335  memcpy(&g_time, &now_time, sizeof(g_time));
336  if(calc_fps) {
337  double fps = g_fps; g_fps = 0;
338  char buffer[256]; snprintf(buffer, 256, "%s%s: %d fps", title, updating?"":" (no updating)", int(fps/sec));
339  XStoreName(dpy, win, buffer);
340  }
341 #ifndef X_FULLSYNC
342  XSync(dpy, false); // It is often better then using XSynchronize(dpy, true)
343 #endif//X_FULLSYNC
344  }
345  if(threaded) pthread_mutex_unlock(&g_mutex);
346  return true;
347 }
