30 bool CheckSignatures() {

32  THREAD::id id1, id2;

34  result |= id1 != id2;

128 void CheckRelations( const THREAD::id ids[], int n, bool duplicates_allowed ) {
129  for( int i=0; i<n; ++i ) {
130  const THREAD::id x = ids[i];
131  for( int j=0; j<n; ++j ) {
132  const THREAD::id y = ids[j];
133  ASSERT( (x==y)==!(x!=y), NULL );
134  ASSERT( (x<y)==!(x>=y), NULL );
135  ASSERT( (x>y)==!(x<=y), NULL );
136  ASSERT( (x<y)+(x==y)+(x>y)==1, NULL );
137  ASSERT( x!=y || i==j || duplicates_allowed, NULL );
138  for( int k=0; k<n; ++k ) {
139  const THREAD::id z = ids[j];
140  ASSERT( !(x<y && y<z) || x<z, "< is not transitive" );
141  }
142  }
143  }

159  // Try all combinations of throw/nothrow for f, x, and y's copy constructor.
160  for( int i=0; i<8; ++i ) {
161  {
162  const AnotherThreadFunc f = AnotherThreadFunc();
163  if( i&1 ) f.copy_throws = d;
164  const Data<1> x(0);
165  if( i&2 ) x.copy_throws = d;
166  const Data<2> y(0);

168  bool exception_caught = false;
169  for( int j=0; j<3; ++j ) {
170  try {
171  switch(j) {

174  case 2: {THREAD t(f,x,y); t.join();} break;
