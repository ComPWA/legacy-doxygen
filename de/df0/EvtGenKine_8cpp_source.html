33 double EvtPawt(double a,double b,double c)
34 {
35  double temp=(a*a-(b+c)*(b+c))*(a*a-(b-c)*(b-c));
36 
37  if (temp<=0) {
38  return 0.0;
39  }
40 
41  return sqrt(temp)/(2.0*a);
42 }
