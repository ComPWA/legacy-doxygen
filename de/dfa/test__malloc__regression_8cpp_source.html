46 // The regression test against bug #1518 where thread bootstrap allocations "leaked"

48 #include "harness_memory.h"

50 bool TestBootstrapLeak() {
51  /* In the bug 1518, each thread leaked ~384 bytes.
52  Initially, scalable allocator maps 1MB. Thus it is necessary to take out most of this space.
53  1MB is chunked into 16K blocks; of those, one block is for thread bootstrap, and one more
54  should be reserved for the test body. 62 blocks left, each can serve 15 objects of 1024 bytes.
55  */
56  const int alloc_size = 1024;
57  const int take_out_count = 15*62;
58 
59  tbb::scalable_allocator<char> a;
60  char* array[take_out_count];
61  for( int i=0; i<take_out_count; ++i )
62  array[i] = a.allocate( alloc_size );
63 
64  RunThread( minimalAllocFree(), alloc_size ); // for threading library to take some memory
65  size_t memory_in_use = GetMemoryUsage();
66  // Wait for memory usage data to "stabilize". The test number (1000) has nothing underneath.
67  for( int i=0; i<1000; i++) {
68  if( GetMemoryUsage()!=memory_in_use ) {
69  memory_in_use = GetMemoryUsage();
70  i = -1;
71  }
72  }
73 
74  ptrdiff_t memory_leak = 0;
75  // Note that 16K bootstrap memory block is enough to serve 42 threads.
76  const int num_thread_runs = 200;
77  for (int run=0; run<3; run++) {
78  memory_in_use = GetMemoryUsage();
79  for( int i=0; i<num_thread_runs; ++i )
80  RunThread( minimalAllocFree(), alloc_size );
81 
82  memory_leak = GetMemoryUsage() - memory_in_use;
83  if (!memory_leak)
84  break;
85  }
86  if( memory_leak>0 ) { // possibly too strong?
87  REPORT( "Error: memory leak of up to %ld bytes\n", static_cast<long>(memory_leak));
88  }
89 
90  for( int i=0; i<take_out_count; ++i )
91  a.deallocate( array[i], alloc_size );
92 
93  return memory_leak<=0;
94 }
