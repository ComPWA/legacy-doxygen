74 template< typename OutputType >
75 void run_continue_nodes( int p, tbb::flow::graph& g, tbb::flow::continue_node< OutputType >& n ) {
76  fake_continue_sender fake_sender;
77  for (size_t i = 0; i < N; ++i) {
78  n.register_predecessor( fake_sender );
79  }
80 
81  for (size_t num_receivers = 1; num_receivers <= MAX_NODES; ++num_receivers ) {
82  std::vector< harness_counting_receiver<OutputType> > receivers(num_receivers, harness_counting_receiver<OutputType>(g));
83  harness_graph_executor<tbb::flow::continue_msg, OutputType>::execute_count = 0;
84 
85  for (size_t r = 0; r < num_receivers; ++r ) {
86  tbb::flow::make_edge( n, receivers[r] );
87  }
88 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
89  ASSERT(n.successor_count() == (size_t)num_receivers, NULL);
90  ASSERT(n.predecessor_count() == 0, NULL);
91  typename tbb::flow::continue_node<OutputType>::successor_list_type my_succs;
92  typedef typename tbb::flow::continue_node<OutputType>::successor_list_type::iterator sv_iter_type;
93  n.copy_successors(my_succs);
94  ASSERT(my_succs.size() == num_receivers, NULL);
95 #endif
96 
97  NativeParallelFor( p, parallel_puts<tbb::flow::continue_msg>(n) );
98  g.wait_for_all();
99 
100  // 2) the nodes will receive puts from multiple predecessors simultaneously,
101  size_t ec = harness_graph_executor<tbb::flow::continue_msg, OutputType>::execute_count;
102  ASSERT( (int)ec == p, NULL );
103  for (size_t r = 0; r < num_receivers; ++r ) {
104  size_t c = receivers[r].my_count;
105  // 3) the nodes will send to multiple successors.
106  ASSERT( (int)c == p, NULL );
107  }
108 
109 #if TBB_DEPRECATED_FLOW_NODE_EXTRACTION
110  for(sv_iter_type si=my_succs.begin(); si != my_succs.end(); ++si) {
111  tbb::flow::remove_edge( n, **si );
112  }
113 #else
114  for (size_t r = 0; r < num_receivers; ++r ) {
115  tbb::flow::remove_edge( n, receivers[r] );
116  }
117 #endif
118  }
119 }

168  std::vector< harness_counting_receiver<OutputType> > receivers(num_receivers, harness_counting_receiver<OutputType>(g));
169 
170  for (size_t r = 0; r < num_receivers; ++r ) {
171  tbb::flow::make_edge( exe_node, receivers[r] );
172  }
173 
174  NativeParallelFor( p, parallel_puts<tbb::flow::continue_msg>(exe_node) );
175  g.wait_for_all();

179  size_t c = receivers[r].my_count;
180  // 3) the nodes will send to multiple successors.
181  ASSERT( (int)c == p, NULL );
182  }
183  for (size_t r = 0; r < num_receivers; ++r ) {
184  tbb::flow::remove_edge( exe_node, receivers[r] );
185  }
186  }
187 
188  // validate that the local body matches the global execute_count and both are correct
189  inc_functor<OutputType> body_copy = tbb::flow::copy_body< inc_functor<OutputType> >( exe_node );
190  const size_t expected_count = p*MAX_NODES + Offset;
191  size_t global_count = global_execute_count;
192  size_t inc_count = body_copy.local_execute_count;
193  ASSERT( global_count == expected_count && global_count == inc_count, NULL );
194  g.reset(tbb::flow::rf_reset_bodies);
195  body_copy = tbb::flow::copy_body< inc_functor<OutputType> >( exe_node );
196  inc_count = body_copy.local_execute_count;
197  ASSERT( Offset == inc_count, "reset(rf_reset_bodies) did not reset functor" );
198 
199  }
