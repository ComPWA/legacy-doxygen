263 void hash_buffer_test(const char *sname) {
264  typedef typename K_deref<K>::type KnoR;
265  tbb::flow::interface10::internal::hash_buffer<
266  K,
267  V,
268  typename VtoKFB<K, V>::type,
269  tbb::tbb_hash_compare<KnoR>
270  > my_hash_buffer;
271  const bool k_is_ref = is_ref<K>::value;
272  typedef tbb::flow::interface10::internal::type_to_key_function_body_leaf<
273  V, K, my_struct_key<K, V> > my_func_body_type;
274  typename VtoKFB<K, V>::type *kp = new my_func_body_type(my_struct_key<K, V>());
275  my_hash_buffer.set_key_func(kp);
276  REMARK("Running hash_buffer test on %s; is ref == %s\n", sname, k_is_ref ? "true" : "false");
277  V mv1, mv0;
278  bool res;
279  for(int cnt = 0; cnt < 2; ++cnt) {
280  // insert 50 items after checking they are not already in the table
281  for(int i = 0; i < 50; ++i) {
282  KnoR kk = index_to_key<KnoR>()(i);
283  mv1.my_key = kk;
284  mv1.my_value = 0.5*i;
285  res = my_hash_buffer.find_with_key(kk, mv0);
286  ASSERT(!res, "Found non-inserted item");
287  res = my_hash_buffer.insert_with_key(mv1);
288  ASSERT(res, "insert failed");
289  res = my_hash_buffer.find_with_key(kk, mv0);
290  ASSERT(res, "not found after insert");
291  ASSERT(mv0.my_value==mv1.my_value, "result not correct");
292  }
293  // go backwards checking they are still there.
294  for(int i = 49; i>=0; --i) {
295  KnoR kk = index_to_key<KnoR>()(i);
296  double value = 0.5*i;
297  res = my_hash_buffer.find_with_key(kk, mv0);
298  ASSERT(res, "find failed");
299  ASSERT(mv0.my_value==value, "result not correct");
300  }
301  // delete every third item, check they are gone
302  for(int i = 0; i < 50; i += 3) {
303  KnoR kk = index_to_key<KnoR>()(i);
304  my_hash_buffer.delete_with_key(kk);
305  res = my_hash_buffer.find_with_key(kk, mv0);
306  ASSERT(!res, "Found deleted item");
307  }
308  // check the deleted items are gone, the non-deleted items are there.
309  for(int i = 0; i < 50; ++i) {
310  KnoR kk = index_to_key<KnoR>()(i);
311  double value = 0.5*i;
312  if(i%3==0) {
313  res = my_hash_buffer.find_with_key(kk, mv0);
314  ASSERT(!res, "found an item that was previously deleted");
315  }
316  else {
317  res = my_hash_buffer.find_with_key(kk, mv0);
318  ASSERT(res, "find failed");
319  ASSERT(mv0.my_value==value, "result not correct");
320  }
321  }
322  // insert new items, check the deleted items return true, the non-deleted items return false.
323  for(int i = 0; i < 50; ++i) {
324  KnoR kk = index_to_key<KnoR>()(i);
325  double value = 1.5*i;
326  mv1.my_key = kk;
327  mv1.my_value = value;
328  res = my_hash_buffer.insert_with_key(mv1);
329  if(i%3==0) {
330  ASSERT(res, "didn't insert in empty slot");
331  }
332  else {
333  ASSERT(!res, "slot was empty on insert");
334  }
335  }
336  // delete all items
337  for(int i = 0; i < 50; ++i) {
338  KnoR kk = index_to_key<KnoR>()(i);
339  my_hash_buffer.delete_with_key(kk);
340  res = my_hash_buffer.find_with_key(kk, mv0);
341  ASSERT(!res, "Found deleted item");
342  }
343  } // perform tasks twice
344 }
