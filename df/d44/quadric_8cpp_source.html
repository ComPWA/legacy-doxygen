86 void quadric_intersect(quadric * q, ray * ry) {
87  flt Aq, Bq, Cq;
88  flt t1, t2;
89  flt disc;
90  vector rd;
91  vector ro;
92 
93  rd=ry->d;
94  VNorm(&rd);
95 
96  ro.x = ry->o.x - q->ctr.x;
97  ro.y = ry->o.y - q->ctr.y;
98  ro.z = ry->o.z - q->ctr.z;
99 
100 
101  Aq = (q->mat.a*(rd.x * rd.x)) +
102  (2.0 * q->mat.b * rd.x * rd.y) +
103  (2.0 * q->mat.c * rd.x * rd.z) +
104  (q->mat.e * (rd.y * rd.y)) +
105  (2.0 * q->mat.f * rd.y * rd.z) +
106  (q->mat.h * (rd.z * rd.z));
107 
108  Bq = 2.0 * (
109  (q->mat.a * ro.x * rd.x) +
110  (q->mat.b * ((ro.x * rd.y) + (rd.x * ro.y))) +
111  (q->mat.c * ((ro.x * rd.z) + (rd.x * ro.z))) +
112  (q->mat.d * rd.x) +
113  (q->mat.e * ro.y * rd.y) +
114  (q->mat.f * ((ro.y * rd.z) + (rd.y * ro.z))) +
115  (q->mat.g * rd.y) +
116  (q->mat.h * ro.z * rd.z) +
117  (q->mat.i * rd.z)
118  );
119 
120  Cq = (q->mat.a * (ro.x * ro.x)) +
121  (2.0 * q->mat.b * ro.x * ro.y) +
122  (2.0 * q->mat.c * ro.x * ro.z) +
123  (2.0 * q->mat.d * ro.x) +
124  (q->mat.e * (ro.y * ro.y)) +
125  (2.0 * q->mat.f * ro.y * ro.z) +
126  (2.0 * q->mat.g * ro.y) +
127  (q->mat.h * (ro.z * ro.z)) +
128  (2.0 * q->mat.i * ro.z) +
129  q->mat.j;
130 
131  if (Aq == 0.0) {
132  t1 = - Cq / Bq;
133  add_intersection(t1, (object *) q, ry);
134  }
135  else {
136  disc=(Bq*Bq - 4.0 * Aq * Cq);
137  if (disc > 0.0) {
138  disc=sqrt(disc);
139  t1 = (-Bq + disc) / (2.0 * Aq);
140  t2 = (-Bq - disc) / (2.0 * Aq);
141  add_intersection(t1, (object *) q, ry);
142  add_intersection(t2, (object *) q, ry);
143  }
144  }
145 }

147 void quadric_normal(quadric * q, vector * pnt, ray * incident, vector * N) {
148 
149  N->x = (q->mat.a*(pnt->x - q->ctr.x) +
150  q->mat.b*(pnt->y - q->ctr.y) +
151  q->mat.c*(pnt->z - q->ctr.z) + q->mat.d);
152 
153  N->y = (q->mat.b*(pnt->x - q->ctr.x) +
154  q->mat.e*(pnt->y - q->ctr.y) +
155  q->mat.f*(pnt->z - q->ctr.z) + q->mat.g);
156 
157  N->z = (q->mat.c*(pnt->x - q->ctr.x) +
158  q->mat.f*(pnt->y - q->ctr.y) +
159  q->mat.h*(pnt->z - q->ctr.z) + q->mat.i);
160 
161  VNorm(N);
162 
163  if (VDot(N, &(incident->d)) > 0.0) {
164  N->x=-N->x;
165  N->y=-N->y;
166  N->z=-N->z;
167  }
168 }
