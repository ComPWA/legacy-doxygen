325 void checkMatrix(const std::vector<std::vector<double>> &CovFit,
326  const std::vector<std::vector<double>> &CovLoad,
327  const std::string MatrixName) {
328  BOOST_CHECK_EQUAL(CovFit.size(), CovLoad.size());
329  if (CovFit.size() != CovLoad.size())
330  return;
331  for (std::size_t i = 0; i < CovFit.size(); ++i) {
332  BOOST_CHECK_EQUAL(CovFit.at(i).size(), CovLoad.at(i).size());
333  if (CovFit.at(i).size() != CovLoad.at(i).size())
334  return;
335  for (std::size_t j = 0; j < CovLoad.size(); ++j) {
336  LOG(INFO) << "Check " << MatrixName << ": ";
337  LOG(INFO) << "i = " << i << " j = " << j;
338  LOG(INFO) << " Value from fit:\t" << CovFit.at(i).at(j);
339  LOG(INFO) << " Value from load:\t" << CovLoad.at(i).at(j);
340  BOOST_CHECK_EQUAL(CovFit.at(i).at(j), CovLoad.at(i).at(j));
341  }
342  }
