29  std::vector<double>
30  evaluate(const std::vector<std::vector<double>> &data) noexcept {
31  auto const &xvals = data[0];
32  std::vector<double> result(xvals.size());
33  std::transform(xvals.begin(), xvals.end(), result.begin(), [&](double x) {
34  return Strength.Value * std::exp(-0.5 * std::pow(x - Mean.Value, 2) /

126 PullInfo calculatePull(const std::vector<std::pair<double, double>> &FitValues,
127  double TrueValue) {
128 
129  std::vector<double> PullDist;
130 
131  for (auto const x : FitValues) {
132  PullDist.push_back((x.first - TrueValue) / x.second);
133  }
134 
135  PullInfo PInfo;
136  // determine mean and sigma of the pull distribution
137  double tempsum(0.0);
138  for (auto const x : PullDist) {
139  tempsum += x;
140  }
141  unsigned int N(PullDist.size());
142  PInfo.Mean = tempsum / N;
143 
144  tempsum = 0.0;
145  for (auto const x : PullDist) {
146  tempsum += std::pow(x - PInfo.Mean, 2);
147  }
148  PInfo.Width = std::sqrt(tempsum / (N - 1));
149 
150  PInfo.MeanError = PInfo.Width / std::sqrt(N);
151  PInfo.WidthError = PInfo.Width / std::sqrt(2.0 * N); // taken from ROOT
152 
153  return PInfo;
154 }
