165 {
166  MALLOC_STATIC_ASSERT(sizeof(Arg) == sizeof(intptr_t),
167  "Type of argument must match AtomicCompareExchange type.");

177 // TODO: make BitMaskBasic more general
178 // (currently, it fits BitMaskMin well, but not as suitable for BitMaskMax)
179 template<unsigned NUM>
180 class BitMaskBasic {
181  static const unsigned SZ = (NUM-1)/(CHAR_BIT*sizeof(uintptr_t))+1;
182  static const unsigned WORD_LEN = CHAR_BIT*sizeof(uintptr_t);
183  uintptr_t mask[SZ];
184 protected:
185  void set(size_t idx, bool val) {
186  MALLOC_ASSERT(idx<NUM, ASSERT_TEXT);
187 
188  size_t i = idx / WORD_LEN;
189  int pos = WORD_LEN - idx % WORD_LEN - 1;
190  if (val)
191  AtomicOr(&mask[i], 1ULL << pos);
192  else
193  AtomicAnd(&mask[i], ~(1ULL << pos));
194  }
195  int getMinTrue(unsigned startIdx) const {
196  unsigned idx = startIdx / WORD_LEN;
197  int pos;
198 
199  if (startIdx % WORD_LEN) {
200  // only interested in part of a word, clear bits before startIdx
201  pos = WORD_LEN - startIdx % WORD_LEN;
202  uintptr_t actualMask = mask[idx] & (((uintptr_t)1<<pos) - 1);
203  idx++;
204  if (-1 != (pos = BitScanRev(actualMask)))
205  return idx*WORD_LEN - pos - 1;
206  }
207 
208  while (idx<SZ)
209  if (-1 != (pos = BitScanRev(mask[idx++])))
210  return idx*WORD_LEN - pos - 1;
211  return -1;
212  }
213 public:
214  void reset() { for (unsigned i=0; i<SZ; i++) mask[i] = 0; }
215 };

596 public:
