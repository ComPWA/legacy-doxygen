42 class TextSlice {
44  char* logical_end;
46  char* physical_end;
47 public:
49  static TextSlice* allocate( size_t max_size ) {
50  // +1 leaves room for a terminating null character.
51  TextSlice* t = (TextSlice*)tbb::tbb_allocator<char>().allocate( sizeof(TextSlice)+max_size+1 );
52  t->logical_end = t->begin();
53  t->physical_end = t->begin()+max_size;
54  return t;
55  }
57  void free() {
58  tbb::tbb_allocator<char>().deallocate((char*)this,sizeof(TextSlice)+(physical_end-begin())+1);
59  }
61  char* begin() {return (char*)(this+1);}
63  char* end() {return logical_end;}
65  size_t size() const {return logical_end-(char*)(this+1);}
67  size_t avail() const {return physical_end-logical_end;}
69  void append( char* first, char* last ) {
70  memcpy( logical_end, first, last-first );
71  logical_end += last-first;
72  }
74  void set_end( char* p ) {logical_end=p;}
75 };

102 void* MyInputFilter::operator()(void*) {
103  // Read characters into space that is available in the next slice.
104  size_t m = next_slice->avail();
105  size_t n = fread( next_slice->end(), 1, m, input_file );
106  if( !n && next_slice->size()==0 ) {
107  // No more characters to process
108  return NULL;
109  } else {
110  // Have more characters to process.
111  TextSlice& t = *next_slice;
112  next_slice = TextSlice::allocate( MAX_CHAR_PER_INPUT_SLICE );
113  char* p = t.end()+n;
114  if( n==m ) {
115  // Might have read partial number. If so, transfer characters of partial number to next slice.
116  while( p>t.begin() && isdigit(p[-1]) )
117  --p;
118  next_slice->append( p, t.end()+n );
119  }
120  t.set_end(p);
121  return &t;
122  }
123 }

136 void* MyTransformFilter::operator()( void* item ) {
137  TextSlice& input = *static_cast<TextSlice*>(item);
138  // Add terminating null so that strtol works right even if number is at end of the input.
139  *input.end() = '\0';
140  char* p = input.begin();
141  TextSlice& out = *TextSlice::allocate( 2*MAX_CHAR_PER_INPUT_SLICE );
142  char* q = out.begin();
143  for(;;) {
144  while( p<input.end() && !isdigit(*p) )
145  *q++ = *p++;
146  if( p==input.end() )
147  break;
148  long x = strtol( p, &p, 10 );
149  // Note: no overflow checking is needed here, as we have twice the
150  // input string length, but the square of a non-negative integer n
151  // cannot have more than twice as many digits as n.
152  long y = x*x;
153  sprintf(q,"%ld",y);
154  q = strchr(q,0);
155  }
156  out.set_end(q);
157  input.free();
158  return &out;
159 }

216  // Run the pipeline
217  tbb::tick_count t0 = tbb::tick_count::now();
218  // Need more than one token in flight per thread to keep all threads
219  // busy; 2-4 works
220  pipeline.run( nthreads*4 );
221  tbb::tick_count t1 = tbb::tick_count::now();
222 
223  fclose( output_file );
224  fclose( input_file );
225 
226  if ( !silent ) printf("time = %g\n", (t1-t0).seconds());
227
