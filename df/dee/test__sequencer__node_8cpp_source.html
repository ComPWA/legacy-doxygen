80  touches( int num_threads ) : my_num_threads(num_threads) {

82  my_touches = new bool* [my_num_threads];
83  for ( int p = 0; p < my_num_threads; ++p) {
84  my_last_touch[p] = T(-1);
85  my_touches[p] = new bool[N];
86  for ( int n = 0; n < N; ++n)
87  my_touches[p][n] = false;
88  }
89  }
90 
91  ~touches() {
92  for ( int p = 0; p < my_num_threads; ++p) {
93  delete [] my_touches[p];
94  }
95  delete [] my_touches;
96  delete [] my_last_touch;
97  }
98 
99  bool check( int tid, T v ) {
100  if ( my_touches[tid][v] != false ) {
101  printf("Error: value seen twice by local thread\n");
102  return false;
103  }
104  if ( v <= my_last_touch[tid] ) {
105  printf("Error: value seen in wrong order by local thread\n");
106  return false;
107  }
108  my_last_touch[tid] = v;
109  my_touches[tid][v] = true;
110  return true;
111  }
112 
113  bool validate_touches() {
114  bool *all_touches = new bool[N];
115  for ( int n = 0; n < N; ++n)
116  all_touches[n] = false;
117 
118  for ( int p = 0; p < my_num_threads; ++p) {
119  for ( int n = 0; n < N; ++n) {
120  if ( my_touches[p][n] == true ) {
121  ASSERT( all_touches[n] == false, "value see by more than one thread\n" );
122  all_touches[n] = true;
123  }
124  }
125  }
126  for ( int n = 0; n < N; ++n) {
127  if ( !all_touches[n] )
128  printf("No touch at %d, my_num_threads = %d\n", n, my_num_threads);
129  //ASSERT( all_touches[n] == true, "value not seen by any thread\n" );
130  }
131  delete [] all_touches;
132  return true;
