107 static void bndbox_intersect(bndbox * bx, ray * ry) {
108  flt a, tx1, tx2, ty1, ty2, tz1, tz2;
109  flt tnear, tfar;
110  object * obj;
111  ray newray;
112 
113  /* eliminate bounded rays whose bounds do not intersect */
114  /* the bounds of the box.. */
115  if (ry->flags & RT_RAY_BOUNDED) {
116  if ((ry->s.x > bx->max.x) && (ry->e.x > bx->max.x)) return;
117  if ((ry->s.x < bx->min.x) && (ry->e.x < bx->min.x)) return;
118 
119  if ((ry->s.y > bx->max.y) && (ry->e.y > bx->max.y)) return;
120  if ((ry->s.y < bx->min.y) && (ry->e.y < bx->min.y)) return;
121 
122  if ((ry->s.z > bx->max.z) && (ry->e.z > bx->max.z)) return;
123  if ((ry->s.z < bx->min.z) && (ry->e.z < bx->min.z)) return;
124  }
125 
126  tnear= -FHUGE;
127  tfar= FHUGE;
128 
129  if (ry->d.x == 0.0) {
130  if ((ry->o.x < bx->min.x) || (ry->o.x > bx->max.x)) return;
131  }
132  else {
133  tx1 = (bx->min.x - ry->o.x) / ry->d.x;
134  tx2 = (bx->max.x - ry->o.x) / ry->d.x;
135  if (tx1 > tx2) { a=tx1; tx1=tx2; tx2=a; }
136  if (tx1 > tnear) tnear=tx1;
137  if (tx2 < tfar) tfar=tx2;
138  }
139  if (tnear > tfar) return;
140  if (tfar < 0.0) return;
141 
142  if (ry->d.y == 0.0) {
143  if ((ry->o.y < bx->min.y) || (ry->o.y > bx->max.y)) return;
144  }
145  else {
146  ty1 = (bx->min.y - ry->o.y) / ry->d.y;
147  ty2 = (bx->max.y - ry->o.y) / ry->d.y;
148  if (ty1 > ty2) { a=ty1; ty1=ty2; ty2=a; }
149  if (ty1 > tnear) tnear=ty1;
150  if (ty2 < tfar) tfar=ty2;
151  }
152  if (tnear > tfar) return;
153  if (tfar < 0.0) return;
154 
155  if (ry->d.z == 0.0) {
156  if ((ry->o.z < bx->min.z) || (ry->o.z > bx->max.z)) return;
157  }
158  else {
159  tz1 = (bx->min.z - ry->o.z) / ry->d.z;
160  tz2 = (bx->max.z - ry->o.z) / ry->d.z;
161  if (tz1 > tz2) { a=tz1; tz1=tz2; tz2=a; }
162  if (tz1 > tnear) tnear=tz1;
163  if (tz2 < tfar) tfar=tz2;
164  }
165  if (tnear > tfar) return;
166  if (tfar < 0.0) return;
167 
168 
169  /* intersect all of the enclosed objects */
170  newray=*ry;
171  newray.flags |= RT_RAY_BOUNDED;
172 
173  RAYPNT(newray.s , (*ry) , tnear);
174  RAYPNT(newray.e , (*ry) , (tfar + EPSILON));
175 
176  obj = bx->objlist;
177  while (obj != NULL) {
178  obj->methods->intersect(obj, &newray);
179  obj = (object *)obj->nextobj;
180  }
181 }
