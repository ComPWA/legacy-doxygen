37 // Each philosopher is an object, and is invoked in the think() function_node, the
38 // eat() function_node and forward() multifunction_node.
39 //
40 // The graph is constructed, and each think() function_node is started with a continue_msg.
41 //
42 // The philosopher will think, then gather two chopsticks, eat, place the chopsticks back,
43 // and if they have not completed the required number of cycles, will start to think() again
44 // by sending a continue_msg to their corresponding think() function_node.
45 //
46 // The reserving join has as its inputs the left and right chopstick queues an a queue
47 // that stores the continue_msg emitted by the function_node after think()ing is done.
48 // When all three inputs are available, a tuple of the inputs will be forwarded to the
49 // eat() function_node. The output of the eat() function_node is sent to the forward()
50 // multifunction_node.

185 void philosopher::forward( const continue_msg &/*in*/, forward_node_type::output_ports_type &out_ports ) {
186  if(my_count < 0) abort();
187  --my_count;
188  (void)tbb::flow::get<1>(out_ports).try_put(chopstick());
189  (void)tbb::flow::get<2>(out_ports).try_put(chopstick());
190  if (my_count > 0) {
191  (void)tbb::flow::get<0>(out_ports).try_put(continue_msg()); //start thinking again
192  } else {
193  if(verbose) {
194  tbb::spin_mutex::scoped_lock lock(my_mutex);
195  std::printf("%s has left the building\n", name());
196  }
197  }

226 int main(int argc, char *argv[]) {
227  try {
228  tbb::tick_count main_time = tbb::tick_count::now();
229  int num_threads;
230  int num_philosophers;
231 
232  RunOptions options = ParseCommandLine(argc, argv);
233  num_philosophers = options.number_of_philosophers;
234  verbose = !options.silent;
235 
236  for(num_threads = options.threads.first; num_threads <= options.threads.last; num_threads = options.threads.step(num_threads)) {
237 
238  tbb::task_scheduler_init init(num_threads);
239 
240  graph g;
241 
242  if(verbose) std::cout << std::endl << num_philosophers << " philosophers with "
243  << num_threads << " threads" << std::endl << std::endl;
244  t0 = tbb::tick_count::now();
245 
246  std::vector<queue_node<chopstick> > places(num_philosophers, queue_node<chopstick>(g));
247  std::vector<philosopher> philosophers;
248  philosophers.reserve(num_philosophers);
249  std::vector<think_node_type *> think_nodes;
250  think_nodes.reserve(num_philosophers);
251  std::vector<thinking_done_type> done_vector(num_philosophers, thinking_done_type(g));
252  std::vector<join_node_type> join_vector(num_philosophers,join_node_type(g));
253  std::vector<eat_node_type *> eat_nodes;
254  eat_nodes.reserve(num_philosophers);
255  std::vector<forward_node_type *> forward_nodes;
256  forward_nodes.reserve(num_philosophers);
257  for ( int i = 0; i < num_philosophers; ++i ) {
258  places[i].try_put(chopstick());
259  philosophers.push_back( philosopher( names[i] ) ); // allowed because of default generated assignment
260  if(verbose) {
261  tbb::spin_mutex::scoped_lock lock(my_mutex);
262  std::cout << "Built philosopher " << philosophers[i] << std::endl;
263  }
264  think_nodes.push_back(new think_node_type(g, unlimited, think_node_body(philosophers[i])));
265  eat_nodes.push_back( new eat_node_type(g, unlimited, eat_node_body(philosophers[i])));
266  forward_nodes.push_back( new forward_node_type(g, unlimited, forward_node_body(philosophers[i])));
267  }
268 
269  // attach chopstick buffers and think function_nodes to joins
270  for(int i = 0; i < num_philosophers; ++i) {
271  make_edge( *think_nodes[i], done_vector[i] );
272  make_edge( done_vector[i], input_port<0>(join_vector[i]) );
273  make_edge( places[i], input_port<1>(join_vector[i]) ); // left chopstick
274  make_edge( places[(i+1) % num_philosophers], input_port<2>(join_vector[i]) ); // right chopstick
275  make_edge( join_vector[i], *eat_nodes[i] );
276  make_edge( *eat_nodes[i], *forward_nodes[i] );
277  make_edge( output_port<0>(*forward_nodes[i]), *think_nodes[i] );
278  make_edge( output_port<1>(*forward_nodes[i]), places[i] );
279  make_edge( output_port<2>(*forward_nodes[i]), places[(i+1) % num_philosophers] );
280  }
281 
282  // start all the philosophers thinking
283  for(int i = 0; i < num_philosophers; ++i) think_nodes[i]->try_put(continue_msg());
284 
285  g.wait_for_all();
286 
287  tbb::tick_count t1 = tbb::tick_count::now();
288  if(verbose) std::cout << std::endl << num_philosophers << " philosophers with "
289  << num_threads << " threads have taken " << (t1-t0).seconds() << "seconds" << std::endl;
290 
291  for ( int i = 0; i < num_philosophers; ++i ) philosophers[i].check();
292 
293  for(int i = 0; i < num_philosophers; ++i) {
294  delete think_nodes[i];
295  delete eat_nodes[i];
296  delete forward_nodes[i];
297  }
298  }
299 
300  utility::report_elapsed_time((tbb::tick_count::now() - main_time).seconds());
301  return 0;
302  } catch(std::exception& e) {
303  std::cerr<<"error occurred. error text is :\"" <<e.what()<<"\"\n";
304  return 1;
305  }
