131 color ExtVoxelColor(flt scalar) {
132  color col;
133 
134  if (scalar > 1.0)
135  scalar = 1.0;
136 
137  if (scalar < 0.0)
138  scalar = 0.0;
139 
140  if (scalar < 0.5) {
141  col.g = 0.0;
142  }
143  else {
144  col.g = (scalar - 0.5) * 2.0;
145  }
146 
147  col.r = scalar;
148  col.b = 1.0 - (scalar / 2.0);
149 
150  return col;
151 }

153 color ext_volume_texture(vector * hit, texture * tex, ray * ry) {
154  color col, col2;
155  box * bx;
156  extvol * xvol;
157  flt a, tx1, tx2, ty1, ty2, tz1, tz2;
158  flt tnear, tfar;
159  flt t, tdist, dt, ddt, sum, tt;
160  vector pnt, bln;
161  flt scalar, transval;
162  int i;
163  point_light * li;
164  color diffint;
165  vector N, L;
166  flt inten;
167 
168  col.r = 0.0;
169  col.g = 0.0;
170  col.b = 0.0;
171 
172  bx = (box *) tex->obj;
173  xvol = (extvol *) tex->obj;
174 
175  tnear= -FHUGE;
176  tfar= FHUGE;
177 
178  if (ry->d.x == 0.0) {
179  if ((ry->o.x < bx->min.x) || (ry->o.x > bx->max.x)) return col;
180  }
181  else {
182  tx1 = (bx->min.x - ry->o.x) / ry->d.x;
183  tx2 = (bx->max.x - ry->o.x) / ry->d.x;
184  if (tx1 > tx2) { a=tx1; tx1=tx2; tx2=a; }
185  if (tx1 > tnear) tnear=tx1;
186  if (tx2 < tfar) tfar=tx2;
187  }
188  if (tnear > tfar) return col;
189  if (tfar < 0.0) return col;
190 
191  if (ry->d.y == 0.0) {
192  if ((ry->o.y < bx->min.y) || (ry->o.y > bx->max.y)) return col;
193  }
194  else {
195  ty1 = (bx->min.y - ry->o.y) / ry->d.y;
196  ty2 = (bx->max.y - ry->o.y) / ry->d.y;
197  if (ty1 > ty2) { a=ty1; ty1=ty2; ty2=a; }
198  if (ty1 > tnear) tnear=ty1;
199  if (ty2 < tfar) tfar=ty2;
200  }
201  if (tnear > tfar) return col;
202  if (tfar < 0.0) return col;
203 
204  if (ry->d.z == 0.0) {
205  if ((ry->o.z < bx->min.z) || (ry->o.z > bx->max.z)) return col;
206  }
207  else {
208  tz1 = (bx->min.z - ry->o.z) / ry->d.z;
209  tz2 = (bx->max.z - ry->o.z) / ry->d.z;
210  if (tz1 > tz2) { a=tz1; tz1=tz2; tz2=a; }
211  if (tz1 > tnear) tnear=tz1;
212  if (tz2 < tfar) tfar=tz2;
213  }
214  if (tnear > tfar) return col;
215  if (tfar < 0.0) return col;
216 
217  if (tnear < 0.0) tnear=0.0;
218 
219  tdist = xvol->samples;
220 
221  tt = (xvol->opacity / tdist);
222 
223  bln.x=fabs(bx->min.x - bx->max.x);
224  bln.y=fabs(bx->min.y - bx->max.y);
225  bln.z=fabs(bx->min.z - bx->max.z);
226 
227  dt = 1.0 / tdist;
228  sum = 0.0;
229 
230 /* Accumulate color as the ray passes through the voxels */
231  for (t=tnear; t<=tfar; t+=dt) {
232  if (sum < 1.0) {
233  pnt.x=((ry->o.x + (ry->d.x * t)) - bx->min.x) / bln.x;
234  pnt.y=((ry->o.y + (ry->d.y * t)) - bx->min.y) / bln.y;
235  pnt.z=((ry->o.z + (ry->d.z * t)) - bx->min.z) / bln.z;
236 
237  /* call external evaluator assume 0.0 -> 1.0 range.. */
238  scalar = xvol->evaluator(pnt.x, pnt.y, pnt.z);
239 
240  transval = tt * scalar;
241  sum += transval;
242 
243  col2 = ExtVoxelColor(scalar);
244 
245  col.r += transval * col2.r * xvol->ambient;
246  col.g += transval * col2.g * xvol->ambient;
247  col.b += transval * col2.b * xvol->ambient;
248 
249  ddt = dt;
250 
251  /* Add in diffuse shaded light sources (no shadows) */
252  if (xvol->diffuse > 0.0) {
253 
254  /* Calculate the Volume gradient at the voxel */
255  N.x = (xvol->evaluator(pnt.x - ddt, pnt.y, pnt.z) -
256  xvol->evaluator(pnt.x + ddt, pnt.y, pnt.z)) * 8.0 * tt;
257 
258  N.y = (xvol->evaluator(pnt.x, pnt.y - ddt, pnt.z) -
259  xvol->evaluator(pnt.x, pnt.y + ddt, pnt.z)) * 8.0 * tt;
260 
261  N.z = (xvol->evaluator(pnt.x, pnt.y, pnt.z - ddt) -
262  xvol->evaluator(pnt.x, pnt.y, pnt.z + ddt)) * 8.0 * tt;
263 
264  /* only light surfaces with enough of a normal.. */
265  if ((N.x*N.x + N.y*N.y + N.z*N.z) > 0.0) {
266  diffint.r = 0.0;
267  diffint.g = 0.0;
268  diffint.b = 0.0;
269 
270  /* add the contribution of each of the lights.. */
271  for (i=0; i<numlights; i++) {
272  li=lightlist[i];
273  VSUB(li->ctr, (*hit), L)
274  VNorm(&L);
275  VDOT(inten, N, L)
276 
277  /* only add light if its from the front of the surface */
278  /* could add back-lighting if we wanted to later.. */
279  if (inten > 0.0) {
280  diffint.r += inten*li->tex->col.r;
281  diffint.g += inten*li->tex->col.g;
282  diffint.b += inten*li->tex->col.b;
283  }
284  }
285  col.r += col2.r * diffint.r * xvol->diffuse;
286  col.g += col2.g * diffint.g * xvol->diffuse;
287  col.b += col2.b * diffint.b * xvol->diffuse;
288  }
289  }
290  }
291  else {
292  sum=1.0;
293  }
294  }
295 
296  /* Add in transmitted ray from outside environment */
297  if (sum < 1.0) { /* spawn transmission rays / refraction */
298  color transcol;
299 
300  transcol = shade_transmission(ry, hit, 1.0 - sum);
301 
302  col.r += transcol.r; /* add the transmitted ray */
303  col.g += transcol.g; /* to the diffuse and */
304  col.b += transcol.b; /* transmission total.. */
305  }
306 
307  return col;
308 }
