201  tbb::flow::graph g;
202 
203  AsyncNodeActivity asyncNodeActivity(io);
204 
205  async_file_reader_node file_reader(g, tbb::flow::unlimited, [&asyncNodeActivity](const tbb::flow::continue_msg& msg, async_file_reader_node::gateway_type& gateway) {
206  asyncNodeActivity.submitRead(gateway);
207  });
208 
209  tbb::flow::function_node< BufferMsg, BufferMsg > compressor(g, tbb::flow::unlimited, BufferCompressor(blockSizeIn100KB));
210 
211  tbb::flow::sequencer_node< BufferMsg > ordering(g, [](const BufferMsg& bufferMsg)->size_t {
212  return bufferMsg.seqId;
213  });
214 
215  // The node is serial to preserve the right order of buffers set by the preceding sequencer_node
216  async_file_writer_node output_writer(g, tbb::flow::serial, [&asyncNodeActivity](const BufferMsg& bufferMsg, async_file_writer_node::gateway_type& gateway) {
217  asyncNodeActivity.submitWrite(bufferMsg);
218  });
219 
220  make_edge(file_reader, compressor);
221  make_edge(compressor, ordering);
222  make_edge(ordering, output_writer);
223 
224  file_reader.try_put(tbb::flow::continue_msg());
225 
226  g.wait_for_all();

235 class AsyncMsgActivity {
236 public:
237 
238  AsyncMsgActivity(tbb::flow::graph& g, IOOperations& io)
239  : m_io(io), m_graph(g), m_fileReaderThread(&AsyncMsgActivity::readingLoop, this),
240  m_fileWriterThread(&AsyncMsgActivity::writingLoop, this)
241  {
242  // Graph synchronization starts here and ends
243  // when the last buffer was written in "writing thread"
244  m_graph.increment_wait_count();
245  }
246 
247  ~AsyncMsgActivity() {
248  m_fileReaderThread.join();
249  m_fileWriterThread.join();
250 
251  // Lets release resources that async
252  // activity and graph were acquired
253  freeBuffers();
254  }

256  async_msg_type submitRead(BufferMsg& bufferMsg) {
257  async_msg_type msg;
258  work_type readWork = { bufferMsg, msg };
259  m_readQueue.push(readWork);
260  return msg;
261  }
262 
263  async_msg_type submitWrite(const BufferMsg& bufferMsg) {
264  async_msg_type msg;
265  work_type writeWork = { bufferMsg, msg };
266  m_writeQueue.push(writeWork);
267  return msg;
268  }

272  struct work_type {
273  BufferMsg bufferMsg;
274  async_msg_type msg;
275  };
276 
277  void readingLoop() {
278  work_type readWork;
279  m_readQueue.pop(readWork);
280 
281  // Reading thread waits for buffers to be received
282  // (the graph reuses limitted number of buffers)
283  // and reads the file while there is something to read
284  while (m_io.hasDataToRead()) {
285  readWork.bufferMsg.seqId = m_io.chunksRead();
286  m_io.readChunk(readWork.bufferMsg.inputBuffer);
287  readWork.msg.set(readWork.bufferMsg);
288  m_readQueue.pop(readWork);
289  }
290 
291  // Pass message with an end flag to the graph
292  sendLastMessage(readWork);
293  }
294 
295  void sendLastMessage(work_type& work) {
296  work.bufferMsg.markLast(m_io.chunksRead());
297  work.msg.set(work.bufferMsg);
298  }
299 
300  void writingLoop() {
301  work_type writeWork;
302  m_writeQueue.pop(writeWork);
303 
304  // Writing thread writes all buffers that it gets
305  // and reuses them. At the end all reusing buffers
306  // is stored in read queue
307  while (!writeWork.bufferMsg.isLast) {
308  m_io.writeChunk(writeWork.bufferMsg.outputBuffer);
309  writeWork.msg.set(writeWork.bufferMsg);
310  m_writeQueue.pop(writeWork);
311  }
312 
313  // Store last message to the reading queue to free resources later
314  writeWork.msg.set(writeWork.bufferMsg);
315 
316  // After all buffers have been written
317  // the synchronization ends
318  m_graph.decrement_wait_count();
319  }
320 
321  void freeBuffers() {
322  int buffersNumber = m_readQueue.size();
323  for (int i = 0; i < buffersNumber; i++) {
324  work_type workToDelete;
325  m_readQueue.pop(workToDelete);
326  BufferMsg::destroyBufferMsg(workToDelete.bufferMsg);
327  }
328  }

341 void fgCompressionAsyncMsg(IOOperations& io, int blockSizeIn100KB, size_t memoryLimitIn1MB) {
342  // Memory limit sets the number of buffers that can be reused
343  int buffersNumber = memoryLimitIn1MB * 1000 * 1024 / io.chunkSize();
344 
345  tbb::flow::graph g;
346 
347  AsyncMsgActivity asyncMsgActivity(g, io);
348 
349  tbb::flow::function_node< BufferMsg, async_msg_type > file_reader(g, tbb::flow::unlimited, [&asyncMsgActivity](BufferMsg bufferMsg) -> async_msg_type {
350  return asyncMsgActivity.submitRead(bufferMsg);
351  });
352 
353  tbb::flow::function_node< BufferMsg, BufferMsg > compressor(g, tbb::flow::unlimited, BufferCompressor(blockSizeIn100KB));
354 
355  tbb::flow::sequencer_node< BufferMsg > ordering(g, [](const BufferMsg& bufferMsg) -> size_t {
356  return bufferMsg.seqId;
357  });
358 
359  // The node is serial to preserve the right order of buffers set by the preceding sequencer_node
360  tbb::flow::function_node< BufferMsg, async_msg_type > output_writer(g, tbb::flow::serial, [&asyncMsgActivity](const BufferMsg& bufferMsg) -> async_msg_type {
361  return asyncMsgActivity.submitWrite(bufferMsg);
362  });
363 
364  make_edge(file_reader, compressor);
365  make_edge(compressor, ordering);
366  make_edge(ordering, output_writer);
367  make_edge(output_writer, file_reader);
368 
369  // Creating buffers to be reused in read/compress/write graph loop
370  for (int i = 0; i < buffersNumber; i++) {
371  BufferMsg reuseBufferMsg = BufferMsg::createBufferMsg(0, io.chunkSize());
372  file_reader.try_put(reuseBufferMsg);
373  }
374 
375  g.wait_for_all();
376 }
