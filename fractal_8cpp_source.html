41 color_t fractal::calc_one_pixel( int x0, int y0 ) const {
42  unsigned int iter;
43  double fx0, fy0, xtemp, x, y, mu;
44 
45  color_t color;
46 
47  fx0 = (double)x0 - (double) size_x / 2.0;
48  fy0 = (double)y0 - (double) size_y / 2.0;
49  fx0 = fx0 / magn + cx;
50  fy0 = fy0 / magn + cy;
51 
52  iter = 0; x = 0; y = 0;
53  mu = 0;
54 
55  while (((x*x + y*y) <= 4) && (iter < max_iterations)) {
56  xtemp = x*x - y*y + fx0;
57  y = 2*x*y + fy0;
58  x = xtemp;
59  mu += exp(-sqrt(x*x+y*y));
60  iter++;
61  }
62 
63  if (iter == max_iterations) {
64  // point corresponds to the mandelbrot set
65  color = v->get_color(255, 255, 255);
66  return color;
67  }
68 
69  int b = (int)(256*mu);
70  int g = (b/8);
71  int r = (g/16);
72 
73  b = b>255 ? 255 : b;
74  g = g>255 ? 255 : g;
75  r = r>255 ? 255 : r;
76 
77  color = v->get_color(r, g, b);
78  return color;
79 }

138 void fractal::render( tbb::task_group_context &context ) {
139  // Make copy of fractal object and render fractal with parallel_for with
140  // the provided context and partitioner chosen by schedule_auto.
141  // Updates to fractal are not reflected in the render.
142  fractal f = *this;
143  fractal_body body(f);
144 
145  if( schedule_auto )
146  tbb::parallel_for( tbb::blocked_range2d<int>(0, size_y, grain_size, 0, size_x, grain_size ),
147  body, tbb::auto_partitioner(), context);
148  else
149  tbb::parallel_for( tbb::blocked_range2d<int>(0, size_y, grain_size, 0, size_x, grain_size ),
150  body, tbb::simple_partitioner(), context);
151 }

164 void fractal_group::calc_fractal( int num ) {
165  // calculate the fractal
166  fractal &f = num ? f1 : f0;
167 
168  tbb::tick_count t0 = tbb::tick_count::now();
169  while ( v->next_frame() && num_frames[num] != 0 ) {
170  f.run( context[num] );
171  if ( num_frames[num]>0 ) num_frames[num] -= 1;
172  }
173  tbb::tick_count t1 = tbb::tick_count::now();
174 
175  if ( !silent ) {
176  printf(" %s fractal finished. Time: %g\n", num ? "Second" : "First", (t1-t0).seconds());
177  }
178 }

226 void fractal_group::run( bool create_second_fractal ) {
227  // initialize task scheduler
228  tbb::task_scheduler_init init( num_threads );
229 
230  // create contexts to manage fractal priorities
231  context = new tbb::task_group_context[2];
232 
233  set_priorities();
234  draw_borders();
235 
236  tbb::task_arena arena;
237  tbb::task_group gr;
238 
239  // the second fractal is calculating on separated thread
240  if ( create_second_fractal ) {
241 #if __TBB_CPP11_LAMBDAS_PRESENT
242  arena.execute( [&] {
243  gr.run( [&] { calc_fractal( 1 ); } );
244  } );
245 #else
246  task_group_body tg_body( *this );
247  arena_body a_body( tg_body, gr );
248  arena.execute( a_body );
249 #endif
250  }
251 
252  // calculate the first fractal
253  calc_fractal( 0 );
254 
255  if ( create_second_fractal ) {
256 #if __TBB_CPP11_LAMBDAS_PRESENT
257  // wait for second fractal
258  arena.execute( [&] { gr.wait(); } );
259 #else
260  arena.execute( arena_body_wait( gr ) );
261 #endif
262  }
263 
264  delete[] context;
265 }
