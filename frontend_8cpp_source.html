737  * Given a number return the highest non-zero bit in it. It is intended to work with 32-bit values only.
738  * Moreover, on IPF, for sake of simplicity and performance, it is narrowed to only serve for 64 to 1023.
739  * This is enough for current algorithm of distribution of sizes among bins.
740  * __TBB_Log2 is not used here to minimize dependencies on TBB specific sources.

752 # if __linux__||__APPLE__||__FreeBSD__||__NetBSD__||__OpenBSD__||__sun||__MINGW32__

758  mov pos, eax

760 # elif _WIN64 && _MSC_VER>=1400

764 # endif
765 #elif __arm__

794  * for objects of this size, or the actual size of objects in this bin.

982 /* Return an empty uninitialized block in a non-blocking fashion. */
983 Block *MemoryPool::getEmptyBlock(size_t size)
984 {
985  TLSData* tls = extMemPool.tlsPointerKey.getThreadMallocTLS();
986  // try to use per-thread cache, if TLS available
987  FreeBlockPool::ResOfGet resOfGet = tls?
988  tls->freeSlabBlocks.getBlock() : FreeBlockPool::ResOfGet(NULL, false);
989  Block *result = resOfGet.block;
990 
991  if (!result) { // not found in local cache, asks backend for slabs
992  int num = resOfGet.lastAccMiss? Backend::numOfSlabAllocOnMiss : 1;
993  BackRefIdx backRefIdx[Backend::numOfSlabAllocOnMiss];
994 
995  result = static_cast<Block*>(extMemPool.backend.getSlabBlock(num));
996  if (!result) return NULL;
997 
998  if (!extMemPool.userPool())
999  for (int i=0; i<num; i++) {
1000  backRefIdx[i] = BackRefIdx::newBackRef(/*largeObj=*/false);
1001  if (backRefIdx[i].isInvalid()) {
1002  // roll back resource allocation
1003  for (int j=0; j<i; j++)
1004  removeBackRef(backRefIdx[j]);
1005  Block *b = result;
1006  for (int j=0; j<num; b=(Block*)((uintptr_t)b+slabSize), j++)
1007  extMemPool.backend.putSlabBlock(b);
1008  return NULL;
1009  }
1010  }
1011  // resources were allocated, register blocks
1012  Block *b = result;
1013  for (int i=0; i<num; b=(Block*)((uintptr_t)b+slabSize), i++) {
1014  // slab block in user's pool must have invalid backRefIdx
1015  if (extMemPool.userPool()) {
1016  new (&b->backRefIdx) BackRefIdx();
1017  } else {
1018  setBackRef(backRefIdx[i], b);
1019  b->backRefIdx = backRefIdx[i];
1020  }
1021  b->tlsPtr = tls;
1022  b->poolPtr = this;
1023  // all but first one go to per-thread pool
1024  if (i > 0) {
1025  MALLOC_ASSERT(tls, ASSERT_TEXT);
1026  tls->freeSlabBlocks.returnBlock(b);
1027  }
1028  }
1029  }
1030  MALLOC_ASSERT(result, ASSERT_TEXT);
1031  result->initEmptyBlock(tls, size);
1032  STAT_increment(getThreadId(), getIndex(result->objectSize), allocBlockNew);
1033  return result;
1034 }

1376 void Block::privatizePublicFreeList( bool cleanup )
1377 {
1378  FreeObject *temp, *localPublicFreeList;
1379  const intptr_t endMarker = cleanup? UNUSABLE : 0;
1380 
1381  // During cleanup of orphaned blocks, the calling thread is not registered as the owner
1382  MALLOC_ASSERT( cleanup || isOwnedByCurrentThread(), ASSERT_TEXT );
1383 #if FREELIST_NONBLOCKING
1384  temp = publicFreeList;
1385  do {
1386  localPublicFreeList = temp;
1387  temp = (FreeObject*)AtomicCompareExchange( (intptr_t&)publicFreeList,
1388  endMarker, (intptr_t)localPublicFreeList);
1389  // no backoff necessary because trying to make change, not waiting for a change
1390  } while( temp != localPublicFreeList );
1391 #else

1393  {
1394  MallocMutex::scoped_lock scoped_cs(publicFreeListLock);
1395  localPublicFreeList = publicFreeList;
1396  publicFreeList = endMarker;
1397  }
1398  temp = localPublicFreeList;
1399 #endif
1400  MALLOC_ITT_SYNC_ACQUIRED(&publicFreeList);
1401

1403  MALLOC_ASSERT( localPublicFreeList!=NULL, ASSERT_TEXT );

1408  allocatedCount--;
1409  while( isSolidPtr(temp->next) ){ // the list will end with either NULL or UNUSABLE
1410  temp = temp->next;
1411  allocatedCount--;
1412  MALLOC_ASSERT( allocatedCount < (slabSize-sizeof(Block))/objectSize, ASSERT_TEXT );
1413  }
1414  /* merge with local freeList */
1415  temp->next = freeList;
1416  freeList = localPublicFreeList;
