42 double fractal::calcOnePixel(int x0, int y0) {
43  double fx0 = double(x0) - double(my_size[0]) / 2;
44  double fy0 = double(y0) - double(my_size[1]) / 2;
45  fx0 = fx0 / magn + cx;
46  fy0 = fy0 / magn + cy;
47 
48  double res = 0, x = 0, y = 0;
49  for(int iter = 0; x*x + y*y <= 4 && iter < max_iter; ++iter) {
50  const double val = x*x - y*y + fx0;
51  y = 2*x*y + fy0, x = val;
52  res += exp(-sqrt(x*x+y*y));
53  }
54 
55  return res;
56 }

58 template<typename Rows>
59 void applyGamma(Rows& image, double g) {
60  typedef decltype(image[0]) Row;
61  typedef decltype(image[0][0]) Pixel;
62  const int w = image[1] - image[0];
63 
64  //execution STL algorithms with execution policies - pstl::execution::par and pstl::execution::unseq
65  std::for_each(pstl::execution::par, image.begin(), image.end(), [g, w](Row& r) {
66  std::transform(pstl::execution::unseq, r, r+w, r, [g](Pixel& p) {
67  double v = 0.3*p.bgra[2] + 0.59*p.bgra[1] + 0.11*p.bgra[0]; //RGB Luminance value
68  assert(v > 0);
69  auto res = static_cast<std::uint8_t>(255.0 * pow(v / 255.0, g));
70  return image::pixel(res, res, res);
71  });
72  });
73 }
