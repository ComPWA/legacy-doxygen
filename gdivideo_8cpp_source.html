56 LRESULT CALLBACK InternalWndProc(HWND hwnd, UINT iMsg, WPARAM wParam, LPARAM lParam)
57 {
58  switch (iMsg)
59  {
60  case WM_MOVE:
61  // Check to make sure our window exists before we tell it to repaint.
62  // This will fail the first time (while the window is being created).
63  if (hwnd) {
64  InvalidateRect(hwnd, NULL, FALSE);
65  UpdateWindow(hwnd);
66  }
67  return 0L;
68 
69  case WM_PAINT:
70  {
71  PAINTSTRUCT ps;
72  Gdiplus::Graphics graphics( BeginPaint(hwnd, &ps) );
73  // redraw just requested area. This call is as fast as simple DrawImage() call.
74  if(g_video->updating) graphics.DrawImage(g_pBitmap, ps.rcPaint.left, ps.rcPaint.top, ps.rcPaint.left, ps.rcPaint.top,
75  ps.rcPaint.right, ps.rcPaint.bottom, Gdiplus::UnitPixel);
76  EndPaint(hwnd, &ps);
77  }
78  return 0L;
79 
80  // Process all mouse and keyboard events
81  case WM_LBUTTONDOWN: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), 1); break;
82  case WM_LBUTTONUP: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), -1); break;
83  case WM_RBUTTONDOWN: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), 2); break;
84  case WM_RBUTTONUP: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), -2); break;
85  case WM_MBUTTONDOWN: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), 3); break;
86  case WM_MBUTTONUP: g_video->on_mouse( (int)LOWORD(lParam), (int)HIWORD(lParam), -3); break;
87  case WM_CHAR: g_video->on_key( (int)wParam); break;
88 
89  // some useless stuff
90  case WM_ERASEBKGND: return 1; // keeps erase-background events from happening, reduces chop
91  case WM_DISPLAYCHANGE: return 0;
92 
93  // Now, shut down the window...
94  case WM_DESTROY: PostQuitMessage(0); return 0;
95  }
96  // call user defined proc, if exists
97  return g_pUserProc? g_pUserProc(hwnd, iMsg, wParam, lParam) : DefWindowProc(hwnd, iMsg, wParam, lParam);
98 }
