82 object * newgrid(int xsize, int ysize, int zsize, vector min, vector max) {
83  grid * g;
84

86  memset(g, 0, sizeof(grid));

89  g->id = new_objectid();
90 
91  g->xsize = xsize;
92  g->ysize = ysize;

94 
95  g->min = min;
96  g->max = max;
97 
98  VSub(&g->max, &g->min, &g->voxsize);
99  g->voxsize.x /= (flt) g->xsize;
100  g->voxsize.y /= (flt) g->ysize;
101  g->voxsize.z /= (flt) g->zsize;

104  memset(g->cells, 0, xsize*ysize*zsize * sizeof(objectlist *));
105 
106 /* fprintf(stderr, "New grid, size: %8d %8d %8d\n", g->xsize, g->ysize, g->zsize); */
107 
108  return (object *) g;

120 static void grid_free(void * v) {
121  int i, numvoxels;
122  grid * g = (grid *) v;
123 
124  /* loop through all voxels and free the object lists */
125  numvoxels = g->xsize * g->ysize * g->zsize;
126  for (i=0; i<numvoxels; i++) {
127  objectlist * lcur, * lnext;
128 
129  lcur = g->cells[i];
130  while (lcur != NULL) {
131  lnext = lcur->next;
132  free(lcur);

136  /* free the grid cells */
137  free(g->cells);

145 static void globalbound(object ** rootlist, vector * gmin, vector * gmax) {
146  vector min, max;
147  object * cur;
148 
149  if (*rootlist == NULL) /* don't bound non-existant objects */
150  return;
151 
152  gmin->x = FHUGE; gmin->y = FHUGE; gmin->z = FHUGE;
153  gmax->x = -FHUGE; gmax->y = -FHUGE; gmax->z = -FHUGE;
154 
155  cur=*rootlist;
156  while (cur != NULL) { /* Go! */
157  min.x = -FHUGE; min.y = -FHUGE; min.z = -FHUGE;
158  max.x = FHUGE; max.y = FHUGE; max.z = FHUGE;
159 
160  if (cur->methods->bbox((void *) cur, &min, &max)) {
161  gmin->x = MYMIN( gmin->x , min.x);
162  gmin->y = MYMIN( gmin->y , min.y);
163  gmin->z = MYMIN( gmin->z , min.z);
164 
165  gmax->x = MYMAX( gmax->x , max.x);
166  gmax->y = MYMAX( gmax->y , max.y);
167  gmax->z = MYMAX( gmax->z , max.z);
168  }
169 
170  cur=(object *)cur->nextobj;
171  }
172 }

175 static int cellbound(grid *g, gridindex *index, vector * cmin, vector * cmax) {
176  vector min, max, cellmin, cellmax;
177  objectlist * cur;
178  int numinbounds = 0;
179 
180  cur = g->cells[index->z*g->xsize*g->ysize + index->y*g->xsize + index->x];
181 
182  if (cur == NULL) /* don't bound non-existant objects */
183  return 0;
184 
185  cellmin.x = voxel2x(g, index->x);
186  cellmin.y = voxel2y(g, index->y);
187  cellmin.z = voxel2z(g, index->z);
188 
189  cellmax.x = cellmin.x + g->voxsize.x;
190  cellmax.y = cellmin.y + g->voxsize.y;
191  cellmax.z = cellmin.z + g->voxsize.z;
192 
193  cmin->x = FHUGE; cmin->y = FHUGE; cmin->z = FHUGE;
194  cmax->x = -FHUGE; cmax->y = -FHUGE; cmax->z = -FHUGE;
195 
196  while (cur != NULL) { /* Go! */
197  min.x = -FHUGE; min.y = -FHUGE; min.z = -FHUGE;
198  max.x = FHUGE; max.y = FHUGE; max.z = FHUGE;
199 
200  if (cur->obj->methods->bbox((void *) cur->obj, &min, &max)) {
201  if ((min.x >= cellmin.x) && (max.x <= cellmax.x) &&
202  (min.y >= cellmin.y) && (max.y <= cellmax.y) &&
203  (min.z >= cellmin.z) && (max.z <= cellmax.z)) {
204 
205  cmin->x = MYMIN( cmin->x , min.x);
206  cmin->y = MYMIN( cmin->y , min.y);
207  cmin->z = MYMIN( cmin->z , min.z);
208 
209  cmax->x = MYMAX( cmax->x , max.x);
210  cmax->y = MYMAX( cmax->y , max.y);
211  cmax->z = MYMAX( cmax->z , max.z);
212 
213  numinbounds++;
214  }
215  }
216 
217  cur=cur->next;
218  }
219 
220  /* in case we get a 0.0 sized axis on the cell bounds, we'll */
221  /* use the original cell bounds */
222  if ((cmax->x - cmin->x) < EPSILON) {
223  cmax->x += EPSILON;
224  cmin->x -= EPSILON;
225  }
226  if ((cmax->y - cmin->y) < EPSILON) {
227  cmax->y += EPSILON;
228  cmin->y -= EPSILON;
229  }
230  if ((cmax->z - cmin->z) < EPSILON) {
231  cmax->z += EPSILON;
232  cmin->z -= EPSILON;
233  }
234 
235  return numinbounds;
236 }

273  return 0;

296  }

298  }

326 static int engrid_cell(grid * gold, gridindex *index) {
327  vector gmin, gmax, gsize;
328  flt len;
329  int numobj, numcbrt, xs, ys, zs;
330  grid * g;
331  objectlist **list;
332  objectlist * newobj;
333 
334  list = &gold->cells[index->z*gold->xsize*gold->ysize +
335  index->y*gold->xsize + index->x];
336 
337  if (*list == NULL)
338  return 0;
339 
340  numobj = cellbound(gold, index, &gmin, &gmax);
341 
342  VSub(&gmax, &gmin, &gsize);
343  len = 1.0 / (MYMAX( MYMAX(gsize.x, gsize.y), gsize.z ));
344  gsize.x *= len;
345  gsize.y *= len;
346  gsize.z *= len;
347 
348  if (numobj > 16) {
349  numcbrt = (int) cbrt(2*numobj);
350 
351  xs = (int) ((flt) numcbrt * gsize.x);
352  if (xs < 1) xs = 1;
353  ys = (int) ((flt) numcbrt * gsize.y);
354  if (ys < 1) ys = 1;
355  zs = (int) ((flt) numcbrt * gsize.z);
356  if (zs < 1) zs = 1;
357 
358  g = (grid *) newgrid(xs, ys, zs, gmin, gmax);
359  engrid_objectlist(g, list);
360 
361  newobj = (objectlist *) rt_getmem(sizeof(objectlist));
362  newobj->obj = (object *) g;
363  newobj->next = *list;
364  *list = newobj;
365 
366  g->nextobj = gold->objects;
367  gold->objects = (object *) g;
368  }
369 
370  return 1;
371 }

374  objectlist * cur, * next, **prev;
375  int numsucceeded = 0;
376

403 static int engrid_object(grid * g, object * obj) {
404  vector omin, omax;
405  gridindex low, high;
406  int x, y, z, zindex, yindex, voxindex;
407  objectlist * tmp;
408 
409  if (obj->methods->bbox(obj, &omin, &omax)) {
410  if (!pos2grid(g, &omin, &low) || !pos2grid(g, &omax, &high)) {
411  return 0; /* object is not wholly contained in the grid */
412  }
413  }
414  else {
415  return 0; /* object is unbounded */
416  }
417 
418  /* add the object to the complete list of objects in the grid */
419  obj->nextobj = g->objects;
420  g->objects = obj;
421 
422  /* add this object to all voxels it inhabits */
423  for (z=low.z; z<=high.z; z++) {
424  zindex = z * g->xsize * g->ysize;
425  for (y=low.y; y<=high.y; y++) {
426  yindex = y * g->xsize;
427  for (x=low.x; x<=high.x; x++) {
428  voxindex = x + yindex + zindex;
429  tmp = (objectlist *) rt_getmem(sizeof(objectlist));
430  tmp->next = g->cells[voxindex];
431  tmp->obj = obj;
432  g->cells[voxindex] = tmp;
433  }
434  }
435  }
436 
437  return 1;
438 }

440 static int pos2grid(grid * g, vector * pos, gridindex * index) {
441  index->x = (int) ((pos->x - g->min.x) / g->voxsize.x);
442  index->y = (int) ((pos->y - g->min.y) / g->voxsize.y);
443  index->z = (int) ((pos->z - g->min.z) / g->voxsize.z);
444 
445  if (index->x == g->xsize)
446  index->x--;
447  if (index->y == g->ysize)
448  index->y--;
449  if (index->z == g->zsize)
450  index->z--;
451 
452  if (index->x < 0 || index->x > g->xsize ||
453  index->y < 0 || index->y > g->ysize ||
454  index->z < 0 || index->z > g->zsize)
455  return 0;
456 
457  if (pos->x < g->min.x || pos->x > g->max.x ||
458  pos->y < g->min.y || pos->y > g->max.y ||
459  pos->z < g->min.z || pos->z > g->max.z)
460  return 0;
461 
462  return 1;
463 }

467 static void grid_intersect(grid * g, ray * ry) {
468  flt tnear, tfar, offset;
469  vector curpos, tmax, tdelta, pdeltaX, pdeltaY, pdeltaZ, nXp, nYp, nZp;
470  gridindex curvox, step, out;
471  int voxindex;
472  objectlist * cur;
473 
474  if (ry->flags & RT_RAY_FINISHED)
475  return;
476 
477  if (!grid_bounds_intersect(g, ry, &tnear, &tfar))
478  return;
479 
480  if (ry->maxdist < tnear)
481  return;
482 
483  curpos = Raypnt(ry, tnear);
484  pos2grid(g, &curpos, &curvox);
485  offset = tnear;
486 
487  /* Setup X iterator stuff */
488  if (fabs(ry->d.x) < EPSILON) {
489  tmax.x = FHUGE;
490  tdelta.x = 0.0;
491  step.x = 0;
492  out.x = 0; /* never goes out of bounds on this axis */
493  }
494  else if (ry->d.x < 0.0) {
495  tmax.x = offset + ((voxel2x(g, curvox.x) - curpos.x) / ry->d.x);
496  tdelta.x = g->voxsize.x / - ry->d.x;
497  step.x = out.x = -1;
498  }
499  else {
500  tmax.x = offset + ((voxel2x(g, curvox.x + 1) - curpos.x) / ry->d.x);
501  tdelta.x = g->voxsize.x / ry->d.x;
502  step.x = 1;
503  out.x = g->xsize;
504  }
505 
506  /* Setup Y iterator stuff */
507  if (fabs(ry->d.y) < EPSILON) {
508  tmax.y = FHUGE;
509  tdelta.y = 0.0;
510  step.y = 0;
511  out.y = 0; /* never goes out of bounds on this axis */
512  }
513  else if (ry->d.y < 0.0) {
514  tmax.y = offset + ((voxel2y(g, curvox.y) - curpos.y) / ry->d.y);
515  tdelta.y = g->voxsize.y / - ry->d.y;
516  step.y = out.y = -1;
517  }
518  else {
519  tmax.y = offset + ((voxel2y(g, curvox.y + 1) - curpos.y) / ry->d.y);
520  tdelta.y = g->voxsize.y / ry->d.y;
521  step.y = 1;
522  out.y = g->ysize;
523  }
524 
525  /* Setup Z iterator stuff */
526  if (fabs(ry->d.z) < EPSILON) {
527  tmax.z = FHUGE;
528  tdelta.z = 0.0;
529  step.z = 0;
530  out.z = 0; /* never goes out of bounds on this axis */
531  }
532  else if (ry->d.z < 0.0) {
533  tmax.z = offset + ((voxel2z(g, curvox.z) - curpos.z) / ry->d.z);
534  tdelta.z = g->voxsize.z / - ry->d.z;
535  step.z = out.z = -1;
536  }
537  else {
538  tmax.z = offset + ((voxel2z(g, curvox.z + 1) - curpos.z) / ry->d.z);
539  tdelta.z = g->voxsize.z / ry->d.z;
540  step.z = 1;
541  out.z = g->zsize;
542  }
543 
544  pdeltaX = ry->d;
545  VScale(&pdeltaX, tdelta.x);
546  pdeltaY = ry->d;
547  VScale(&pdeltaY, tdelta.y);
548  pdeltaZ = ry->d;
549  VScale(&pdeltaZ, tdelta.z);
550 
551  nXp = Raypnt(ry, tmax.x);
552  nYp = Raypnt(ry, tmax.y);
553  nZp = Raypnt(ry, tmax.z);
554 
555  voxindex = curvox.z*g->xsize*g->ysize + curvox.y*g->xsize + curvox.x;
556  while (1) {
557  if (tmax.x < tmax.y && tmax.x < tmax.z) {
558  cur = g->cells[voxindex];
559  while (cur != NULL) {
560  if (ry->mbox[cur->obj->id] != ry->serial) {
561  ry->mbox[cur->obj->id] = ry->serial;
562  cur->obj->methods->intersect(cur->obj, ry);
563  }
564  cur = cur->next;
565  }
566  curvox.x += step.x;
567  if (ry->maxdist < tmax.x || curvox.x == out.x)
568  break;
569  voxindex += step.x;
570  tmax.x += tdelta.x;
571  curpos = nXp;
572  nXp.x += pdeltaX.x;
573  nXp.y += pdeltaX.y;
574  nXp.z += pdeltaX.z;
575  }
576  else if (tmax.z < tmax.y) {
577  cur = g->cells[voxindex];
578  while (cur != NULL) {
579  if (ry->mbox[cur->obj->id] != ry->serial) {
580  ry->mbox[cur->obj->id] = ry->serial;
581  cur->obj->methods->intersect(cur->obj, ry);
582  }
583  cur = cur->next;
584  }
585  curvox.z += step.z;
586  if (ry->maxdist < tmax.z || curvox.z == out.z)
587  break;
588  voxindex += step.z*g->xsize*g->ysize;
589  tmax.z += tdelta.z;
590  curpos = nZp;
591  nZp.x += pdeltaZ.x;
592  nZp.y += pdeltaZ.y;
593  nZp.z += pdeltaZ.z;
594  }
595  else {
596  cur = g->cells[voxindex];
597  while (cur != NULL) {
598  if (ry->mbox[cur->obj->id] != ry->serial) {
599  ry->mbox[cur->obj->id] = ry->serial;
600  cur->obj->methods->intersect(cur->obj, ry);
601  }
602  cur = cur->next;
603  }
604  curvox.y += step.y;
605  if (ry->maxdist < tmax.y || curvox.y == out.y)
606  break;
607  voxindex += step.y*g->xsize;
608  tmax.y += tdelta.y;
609  curpos = nYp;
610  nYp.x += pdeltaY.x;
611  nYp.y += pdeltaY.y;
612  nYp.z += pdeltaY.z;
613  }
614 
615  if (ry->flags & RT_RAY_FINISHED)
616  break;
617  }
618 }

630 static int grid_bounds_intersect(grid * g, ray * ry, flt *nr, flt *fr) {
631  flt a, tx1, tx2, ty1, ty2, tz1, tz2;
632  flt tnear, tfar;
633 
634  tnear= -FHUGE;
635  tfar= FHUGE;
636 
637  if (ry->d.x == 0.0) {
638  if ((ry->o.x < g->min.x) || (ry->o.x > g->max.x)) return 0;
639  }
640  else {
641  tx1 = (g->min.x - ry->o.x) / ry->d.x;
642  tx2 = (g->max.x - ry->o.x) / ry->d.x;
643  if (tx1 > tx2) { a=tx1; tx1=tx2; tx2=a; }
644  if (tx1 > tnear) tnear=tx1;
645  if (tx2 < tfar) tfar=tx2;
646  }
647  if (tnear > tfar) return 0;
648  if (tfar < 0.0) return 0;
649 
650  if (ry->d.y == 0.0) {
651  if ((ry->o.y < g->min.y) || (ry->o.y > g->max.y)) return 0;
652  }
