68 int fakeimage(char * name, int * xres, int * yres, unsigned char ** imgdata) {
69  int i, imgsize;
70 
71  fprintf(stderr, "Error loading image %s. Faking it.\n", name);
72 
73  *xres = 2;
74  *yres = 2;

76  *imgdata = (unsigned char *)rt_getmem(imgsize);

85 int readimage(rawimage * img) {
86  int rc;
87  int xres, yres;
88  unsigned char * imgdata = NULL;
89  char * name = img->name;
90 
91  if (strstr(name, ".ppm")) {
92  rc = readppm(name, &xres, &yres, &imgdata);
93  }
94  else if (strstr(name, ".tga")) {
95  rc = readtga(name, &xres, &yres, &imgdata);
96  }
97  else if (strstr(name, ".jpg")) {
98  rc = readjpeg(name, &xres, &yres, &imgdata);
99  }
100  else if (strstr(name, ".gif")) {
101  rc = IMAGEUNSUP;
102  }
103  else if (strstr(name, ".png")) {
104  rc = IMAGEUNSUP;
105  }
106  else if (strstr(name, ".tiff")) {
107  rc = IMAGEUNSUP;
108  }
109  else if (strstr(name, ".rgb")) {
110  rc = IMAGEUNSUP;
111  }
112  else if (strstr(name, ".xpm")) {
113  rc = IMAGEUNSUP;
114  }
115  else {
116  rc = readppm(name, &xres, &yres, &imgdata);
117  }
118 
119  switch (rc) {
120  case IMAGEREADERR:
121  fprintf(stderr, "Short read encountered while loading image %s\n", name);
122  rc = IMAGENOERR; /* remap to non-fatal error */
123  break;
124 
125  case IMAGEUNSUP:
126  fprintf(stderr, "Cannot read unsupported image format for image %s\n", name);
127  break;
128  }
129 
130  /* If the image load failed, create a tiny white colored image to fake it */
131  /* this allows a scene to render even when a file can't be loaded */
132  if (rc != IMAGENOERR) {
133  rc = fakeimage(name, &xres, &yres, &imgdata);
134  }
135 
136  /* If we succeeded in loading the image, return it. */
137  if (rc == IMAGENOERR) {
138  img->xres = xres;
139  img->yres = yres;
140  img->bpp = 3;
141  img->data = imgdata;
142  }
143 
144  return rc;
145 }
