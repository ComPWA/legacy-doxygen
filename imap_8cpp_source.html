78 color ImageMap(rawimage * image, flt u, flt v) {
79  color col, colx, colx2;
80  flt x,y, px, py;
81  int x1, x2, y1, y2;
82  unsigned char * ptr;
83  unsigned char * ptr2;
84 
85  if (!image->loaded) {
86  LoadImage(image);
87  image->loaded=1;
88  }
89 
90  if ((u <= 1.0) && (u >=0.0) && (v <= 1.0) && (v >= 0.0)) {
91  x=(image->xres - 1.0) * u; /* floating point X location */
92  y=(image->yres - 1.0) * v; /* floating point Y location */
93 
94  px = x - ((int) x);
95  py = y - ((int) y);
96 
97  x1 = (int) x;
98  x2 = x1 + 1;
99 
100  y1 = (int) y;
101  y2 = y1 + 1;
102 
103  ptr = image->data + ((image->xres * y1) + x1) * 3;
104  ptr2 = image->data + ((image->xres * y1) + x2) * 3;
105 
106  colx.r = (flt) ((flt)ptr[0] + px*((flt)ptr2[0] - (flt) ptr[0])) / 255.0;
107  colx.g = (flt) ((flt)ptr[1] + px*((flt)ptr2[1] - (flt) ptr[1])) / 255.0;
108  colx.b = (flt) ((flt)ptr[2] + px*((flt)ptr2[2] - (flt) ptr[2])) / 255.0;
109 
110  ptr = image->data + ((image->xres * y2) + x1) * 3;
111  ptr2 = image->data + ((image->xres * y2) + x2) * 3;
112 
113  colx2.r = ((flt)ptr[0] + px*((flt)ptr2[0] - (flt)ptr[0])) / 255.0;
114  colx2.g = ((flt)ptr[1] + px*((flt)ptr2[1] - (flt)ptr[1])) / 255.0;
115  colx2.b = ((flt)ptr[2] + px*((flt)ptr2[2] - (flt)ptr[2])) / 255.0;
116 
117  col.r = colx.r + py*(colx2.r - colx.r);
118  col.g = colx.g + py*(colx2.g - colx.g);
119  col.b = colx.b + py*(colx2.b - colx.b);
120 
121  }
122  else {
123  col.r=0.0;
124  col.g=0.0;
125  col.b=0.0;
126  }
127  return col;
128 }

130 rawimage * AllocateImage(char * filename) {
131  rawimage * newimage = NULL;
132  int i, intable;
133  size_t len;
134 
135  intable=0;
136  if (numimages!=0) {

139  newimage=imagelist[i];
140  intable=1;

142  }
143  }
144 
145  if (!intable) {
146  newimage=(rawimage *)rt_getmem(sizeof(rawimage));
147  newimage->loaded=0;
148  newimage->xres=0;
149  newimage->yres=0;
150  newimage->bpp=0;
151  newimage->data=NULL;
152  len=strlen(filename);
153  if (len > 80) rtbomb("Filename too long in image map!!");
154  strcpy(newimage->name, filename);
155 
156  imagelist[numimages]=newimage; /* add new one to the table */
157  numimages++; /* increment the number of images */
158  }
159 
160  return newimage;
