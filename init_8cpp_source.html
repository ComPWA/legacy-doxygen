27 static void posdef_gen( double * A, int n )
28 {
29  /* Allocate memory for the matrix and its transpose */
30  double *L = (double *)calloc( sizeof( double ), n*n );
31  assert( L );
32 
33  double *LT = (double *)calloc( sizeof( double ), n*n) ;
34  assert( LT );
35 
36  memset( A, 0, sizeof( double )*n*n );
37 
38  /* Generate a conditioned matrix and fill it with random numbers */
39  for ( int j = 0; j < n; ++j ) {
40  for ( int k = 0; k < j; ++k ) {
41  // The initial value has to be between [0,1].
42  L[k*n+j] = ( ( (j*k) / ((double)(j+1)) / ((double)(k+2)) * 2.0) - 1.0 ) / ((double)n);
43  }
44 
45  L[j*n+j] = 1;
46  }
47 
48  /* Compute transpose of the matrix */
49  for ( int i = 0; i < n; ++i ) {
50  for ( int j = 0; j < n; ++j ) {
51  LT[j*n+i] = L[i*n+j];
52  }
53  }
54  cblas_dgemm( CblasColMajor, CblasNoTrans, CblasNoTrans, n, n, n, 1, L, n, LT, n, 0, A, n );
55 
56  free( L );
57  free( LT );
58 }
