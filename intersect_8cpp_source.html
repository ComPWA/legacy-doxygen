123 void add_intersection(flt t, object * obj, ray * ry) {
124  intersectstruct * intstruct = ry->intstruct;
125 
126  if (t > EPSILON) {
127 
128  /* if we hit something before maxdist update maxdist */
129  if (t < ry->maxdist) {
130  ry->maxdist = t;
131 
132  /* if we hit *anything* before maxdist, and we're firing a */
133  /* shadow ray, then we are finished ray tracing the shadow */
134  if (ry->flags & RT_RAY_SHADOW)
135  ry->flags |= RT_RAY_FINISHED;
136  }
137 
138  intstruct->num++;
139  intstruct->list[intstruct->num].obj = obj;
140  intstruct->list[intstruct->num].t = t;
141  }
142 }

145 int closest_intersection(flt * t, object ** obj, intersectstruct * intstruct) {
146  int i;
147  *t=FHUGE;
148 
149  for (i=1; i<=intstruct->num; i++) {
150  if (intstruct->list[i].t < *t) {
151  *t=intstruct->list[i].t;
152  *obj=intstruct->list[i].obj;
153  }
154  }
155 
156  return intstruct->num;
157 }
