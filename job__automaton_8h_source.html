68  bool try_acquire() {
69  intptr_t snapshot = my_job;
70  if( snapshot==-1 ) {
71  return false;
72  } else {

107  bool try_plug( rml::job*&j ) {
108  for(;;) {
109  intptr_t snapshot = my_job;
110  if( snapshot&1 ) {
111  j = NULL;
112  return false;
113  }
114  // Not busy
115  if( my_job.compare_and_swap( -1, snapshot )==snapshot ) {
116  j = reinterpret_cast<rml::job*>(snapshot);
117  return true;
118  }
119  // Need to retry, because current thread may be non-owner that read a 0, and owner might have
120  // caused transition 0->1->ptr after we took our snapshot.
121  }
122  }
