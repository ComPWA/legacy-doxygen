30 /* The functor called by the aggregator for the operation list */
31 template<typename Props>
32 class CacheBinFunctor {
33  typename LargeObjectCacheImpl<Props>::CacheBin *const bin;
34  ExtMemoryPool *const extMemPool;
35  typename LargeObjectCacheImpl<Props>::BinBitMask *const bitMask;
36  const int idx;
37 
38  LargeMemoryBlock *toRelease;
39  bool needCleanup;
40  uintptr_t currTime;
41 
42  /* Do preprocessing under the operation list. */
43  /* All the OP_PUT_LIST operations are merged in the one operation.
44  All OP_GET operations are merged with the OP_PUT_LIST operations but
45  it demands the update of the moving average value in the bin.
46  Only the last OP_CLEAN_TO_THRESHOLD operation has sense.
47  The OP_CLEAN_ALL operation also should be performed only once.
48  Moreover it cancels the OP_CLEAN_TO_THRESHOLD operation. */
49  class OperationPreprocessor {
50  // TODO: remove the dependency on CacheBin.
51  typename LargeObjectCacheImpl<Props>::CacheBin *const bin;
52 
53  /* Contains the relative time in the operation list.
54  It counts in the reverse order since the aggregator also
55  provides operations in the reverse order. */
56  uintptr_t lclTime;
57 
58  /* opGet contains only OP_GET operations which cannot be merge with OP_PUT operations
59  opClean contains all OP_CLEAN_TO_THRESHOLD and OP_CLEAN_ALL operations. */
60  CacheBinOperation *opGet, *opClean;
61  /* The time of the last OP_CLEAN_TO_THRESHOLD operations */
62  uintptr_t cleanTime;
63 
64  /* lastGetOpTime - the time of the last OP_GET operation.
65  lastGet - the same meaning as CacheBin::lastGet */
66  uintptr_t lastGetOpTime, lastGet;
67 
68  /* The total sum of all usedSize changes requested with CBOP_UPDATE_USED_SIZE operations. */
69  size_t updateUsedSize;
70 
71  /* The list of blocks for the OP_PUT_LIST operation. */
72  LargeMemoryBlock *head, *tail;
73  int putListNum;
74 
75  /* if the OP_CLEAN_ALL is requested. */
76  bool isCleanAll;
77 
78  inline void commitOperation(CacheBinOperation *op) const;
79  inline void addOpToOpList(CacheBinOperation *op, CacheBinOperation **opList) const;
80  bool getFromPutList(CacheBinOperation* opGet, uintptr_t currTime);
81  void addToPutList( LargeMemoryBlock *head, LargeMemoryBlock *tail, int num );
82 
83  public:
84  OperationPreprocessor(typename LargeObjectCacheImpl<Props>::CacheBin *bin) :
85  bin(bin), lclTime(0), opGet(NULL), opClean(NULL), cleanTime(0),
86  lastGetOpTime(0), updateUsedSize(0), head(NULL), isCleanAll(false) {}
87  void operator()(CacheBinOperation* opList);
88  uintptr_t getTimeRange() const { return -lclTime; }
89 
90  friend class CacheBinFunctor;
91  };
92 
93 public:
94  CacheBinFunctor(typename LargeObjectCacheImpl<Props>::CacheBin *bin, ExtMemoryPool *extMemPool,
95  typename LargeObjectCacheImpl<Props>::BinBitMask *bitMask, int idx) :
96  bin(bin), extMemPool(extMemPool), bitMask(bitMask), idx(idx), toRelease(NULL), needCleanup(false) {}
97  void operator()(CacheBinOperation* opList);
98 
99  bool isCleanupNeeded() const { return needCleanup; }
100  LargeMemoryBlock *getToRelease() const { return toRelease; }
101  uintptr_t getCurrTime() const { return currTime; }
102 };

180 // The opCast function can be the member of CacheBinOperation but it will have
181 // small stylistic ambiguity: it will look like a getter (with a cast) for the
182 // CacheBinOperation::data data member but it should return a reference to
183 // simplify the code from a lot of getter/setter calls. So the global cast in
184 // the style of static_cast (or reinterpret_cast) seems to be more readable and
185 // have more explicit semantic.
186 template <typename OpTypeData>
187 OpTypeData& opCast(CacheBinOperation &op) {
188  return *reinterpret_cast<OpTypeData*>(&op.data);
189 }

218 template<typename Props> bool CacheBinFunctor<Props>::
219  OperationPreprocessor::getFromPutList(CacheBinOperation *opGet, uintptr_t currTime)
220 {
221  if ( head ) {
222  uintptr_t age = head->age;
223  LargeMemoryBlock *next = head->next;
224  *opCast<OpGet>(*opGet).res = head;
225  commitOperation( opGet );
226  head = next;
227  putListNum--;
228  MALLOC_ASSERT( putListNum>=0, ASSERT_TEXT );
229 
230  // use moving average with current hit interval
231  bin->updateMeanHitRange( currTime - age );
232  return true;
233  }
234  return false;
235 }

237 template<typename Props> void CacheBinFunctor<Props>::
238  OperationPreprocessor::addToPutList(LargeMemoryBlock *h, LargeMemoryBlock *t, int num)
239 {
240  if ( head ) {
241  MALLOC_ASSERT( tail, ASSERT_TEXT );
242  tail->next = h;

246  } else {
247  head = h;
248  tail = t;
249  putListNum = num;

253 template<typename Props> void CacheBinFunctor<Props>::
254  OperationPreprocessor::operator()(CacheBinOperation* opList)
255 {
256  for ( CacheBinOperation *op = opList, *opNext; op; op = opNext ) {
257  opNext = op->next;
258  switch ( op->type ) {
259  case CBOP_GET:
260  {
261  lclTime--;
262  if ( !lastGetOpTime ) {
263  lastGetOpTime = lclTime;
264  lastGet = 0;
265  } else if ( !lastGet ) lastGet = lclTime;
266 
267  if ( !getFromPutList(op,lclTime) ) {
268  opCast<OpGet>(*op).currTime = lclTime;
269  addOpToOpList( op, &opGet );
270  }
271  }
272  break;
273 
274  case CBOP_PUT_LIST:
275  {
276  LargeMemoryBlock *head = opCast<OpPutList>(*op).head;
277  LargeMemoryBlock *curr = head, *prev = NULL;
278 
279  int num = 0;
280  do {
281  // we do not kept prev pointers during assigning blocks to bins, set them now
282  curr->prev = prev;
283 
284  // Save the local times to the memory blocks. Local times are necessary
285  // for the getFromPutList function which updates the hit range value in
286  // CacheBin when OP_GET and OP_PUT_LIST operations are merged successfully.
287  // The age will be updated to the correct global time after preprocessing
288  // when global cache time is updated.
289  curr->age = --lclTime;
290 
291  prev = curr;
292  num += 1;
293 
294  STAT_increment(getThreadId(), ThreadCommonCounters, cacheLargeObj);
295  } while ((curr = curr->next) != NULL);
296 
297  LargeMemoryBlock *tail = prev;
298  addToPutList(head, tail, num);
299 
300  while ( opGet ) {
301  CacheBinOperation *next = opGet->next;
302  if ( !getFromPutList(opGet, opCast<OpGet>(*opGet).currTime) )
303  break;
304  opGet = next;
305  }
306  }
307  break;
308 
309  case CBOP_UPDATE_USED_SIZE:
310  updateUsedSize += opCast<OpUpdateUsedSize>(*op).size;
311  commitOperation( op );
312  break;
313 
314  case CBOP_CLEAN_ALL:
315  isCleanAll = true;
316  addOpToOpList( op, &opClean );
317  break;
318 
319  case CBOP_CLEAN_TO_THRESHOLD:
320  {
321  uintptr_t currTime = opCast<OpCleanToThreshold>(*op).currTime;
322  // We don't worry about currTime overflow since it is a rare
323  // occurrence and doesn't affect correctness
324  cleanTime = cleanTime < currTime ? currTime : cleanTime;
325  addOpToOpList( op, &opClean );
326  }
327  break;
328 
329  default:
330  MALLOC_ASSERT( false, "Unknown operation." );
331  }
332  }
333  MALLOC_ASSERT( !( opGet && head ), "Not all put/get pairs are processed!" );
334 }

336 template<typename Props> void CacheBinFunctor<Props>::operator()(CacheBinOperation* opList)
337 {
338  MALLOC_ASSERT( opList, "Empty operation list is passed into operation handler." );
339 
340  OperationPreprocessor prep(bin);
341  prep(opList);
342 
343  if ( uintptr_t timeRange = prep.getTimeRange() ) {
344  uintptr_t startTime = extMemPool->loc.getCurrTimeRange(timeRange);
345  // endTime is used as the current (base) time since the local time is negative.
346  uintptr_t endTime = startTime + timeRange;
347 
348  if ( prep.lastGetOpTime && prep.lastGet ) bin->setLastGet(prep.lastGet+endTime);
349 
350  if ( CacheBinOperation *opGet = prep.opGet ) {
351  bool isEmpty = false;
352  do {
353 #if __TBB_MALLOC_WHITEBOX_TEST
354  tbbmalloc_whitebox::locGetProcessed++;
355 #endif
356  const OpGet &opGetData = opCast<OpGet>(*opGet);
357  if ( !isEmpty ) {
358  if ( LargeMemoryBlock *res = bin->get() ) {
359  uintptr_t getTime = opGetData.currTime + endTime;
360  // use moving average with current hit interval
361  bin->updateMeanHitRange( getTime - res->age);
362  bin->updateCachedSize( -opGetData.size );
363  *opGetData.res = res;
364  } else {
365  isEmpty = true;
366  uintptr_t lastGetOpTime = prep.lastGetOpTime+endTime;
367  bin->forgetOutdatedState(lastGetOpTime);
368  bin->updateAgeThreshold(lastGetOpTime);
369  }
370  }
371 
372  CacheBinOperation *opNext = opGet->next;
373  bin->updateUsedSize( opGetData.size, bitMask, idx );
374  prep.commitOperation( opGet );
375  opGet = opNext;
376  } while ( opGet );
377  if ( prep.lastGetOpTime )
378  bin->setLastGet( prep.lastGetOpTime + endTime );
379  } else if ( LargeMemoryBlock *curr = prep.head ) {
380  curr->prev = NULL;
381  while ( curr ) {
382  // Update local times to global times
383  curr->age += endTime;
384  curr=curr->next;
385  }
386 #if __TBB_MALLOC_WHITEBOX_TEST
387  tbbmalloc_whitebox::locPutProcessed+=prep.putListNum;
388 #endif
389  toRelease = bin->putList(prep.head, prep.tail, bitMask, idx, prep.putListNum);
390  }
391  needCleanup = extMemPool->loc.isCleanupNeededOnRange(timeRange, startTime);
392  currTime = endTime - 1;
393  }
394 
395  if ( CacheBinOperation *opClean = prep.opClean ) {
396  if ( prep.isCleanAll )
397  *opCast<OpCleanAll>(*opClean).res = bin->cleanAll(bitMask, idx);
398  else
399  *opCast<OpCleanToThreshold>(*opClean).res = bin->cleanToThreshold(prep.cleanTime, bitMask, idx);
400 
401  CacheBinOperation *opNext = opClean->next;
402  prep.commitOperation( opClean );
403 
404  while ((opClean = opNext) != NULL) {
405  opNext = opClean->next;
406  prep.commitOperation(opClean);
407  }
408  }
409 
410  if ( size_t size = prep.updateUsedSize )
411  bin->updateUsedSize(size, bitMask, idx);
412 }

448 template<typename Props> bool LargeObjectCacheImpl<Props>::
449  CacheBin::cleanToThreshold(ExtMemoryPool *extMemPool, BinBitMask *bitMask, uintptr_t currTime, int idx)
450 {
451  LargeMemoryBlock *toRelease = NULL;
452 
453  /* oldest may be more recent then age, that's why cast to signed type
454  was used. age overflow is also processed correctly. */
455  if (last && (intptr_t)(currTime - oldest) > ageThreshold) {
456  OpCleanToThreshold data = {&toRelease, currTime};
457  CacheBinOperation op(data);
458  ExecuteOperation( &op, extMemPool, bitMask, idx );

460  bool released = toRelease;
461 
462  Backend *backend = &extMemPool->backend;
463  while ( toRelease ) {
464  LargeMemoryBlock *helper = toRelease->next;
465  backend->returnLargeObject(toRelease);
466  toRelease = helper;
467  }
468  return released;

501 /* ------------------------------ Unsafe methods used with the aggregator ------------------------------ */
502 template<typename Props> LargeMemoryBlock *LargeObjectCacheImpl<Props>::
503  CacheBin::putList(LargeMemoryBlock *head, LargeMemoryBlock *tail, BinBitMask *bitMask, int idx, int num)
504 {
505  size_t size = head->unalignedSize;
506  usedSize -= num*size;
507  MALLOC_ASSERT( !last || (last->age != 0 && last->age != -1U), ASSERT_TEXT );
508  MALLOC_ASSERT( (tail==head && num==1) || (tail!=head && num>1), ASSERT_TEXT );
509  LargeMemoryBlock *toRelease = NULL;
510  if (!lastCleanedAge) {
511  // 1st object of such size was released.
512  // Not cache it, and remember when this occurs
513  // to take into account during cache miss.
514  lastCleanedAge = tail->age;
515  toRelease = tail;
516  tail = tail->prev;
517  if (tail)
518  tail->next = NULL;
519  else
520  head = NULL;
521  num--;
522  }
523  if (num) {
524  // add [head;tail] list to cache
525  MALLOC_ASSERT( tail, ASSERT_TEXT );
526  tail->next = first;
527  if (first)
528  first->prev = tail;
529  first = head;
530  if (!last) {
531  MALLOC_ASSERT(0 == oldest, ASSERT_TEXT);
532  oldest = tail->age;
533  last = tail;
534  }
535 
536  cachedSize += num*size;
537  }
538 
539  // No used object, and nothing in the bin, mark the bin as empty
540  if (!usedSize && !first)
541  bitMask->set(idx, false);
542 
543  return toRelease;
544 }

563 // forget the history for the bin if it was unused for long time
564 template<typename Props> void LargeObjectCacheImpl<Props>::
565  CacheBin::forgetOutdatedState(uintptr_t currTime)
566 {
567  // If the time since the last get is LongWaitFactor times more than ageThreshold
568  // for the bin, treat the bin as rarely-used and forget everything we know
569  // about it.
570  // If LongWaitFactor is too small, we forget too early and
571  // so prevents good caching, while if too high, caching blocks
572  // with unrelated usage pattern occurs.
573  const uintptr_t sinceLastGet = currTime - lastGet;
574  bool doCleanup = false;
575 
576  if (ageThreshold)
577  doCleanup = sinceLastGet > Props::LongWaitFactor*ageThreshold;
578  else if (lastCleanedAge)
579  doCleanup = sinceLastGet > Props::LongWaitFactor*(lastCleanedAge - lastGet);
580 
581  if (doCleanup) {
582  lastCleanedAge = 0;
583  ageThreshold = 0;
584  }
585 
586 }

588 template<typename Props> LargeMemoryBlock *LargeObjectCacheImpl<Props>::
589  CacheBin::cleanToThreshold(uintptr_t currTime, BinBitMask *bitMask, int idx)
590 {
591  /* oldest may be more recent then age, that's why cast to signed type
592  was used. age overflow is also processed correctly. */
593  if ( !last || (intptr_t)(currTime - last->age) < ageThreshold ) return NULL;
594 
595 #if MALLOC_DEBUG
596  uintptr_t nextAge = 0;
597 #endif
598  do {
599 #if MALLOC_DEBUG
600  // check that list ordered
601  MALLOC_ASSERT(!nextAge || lessThanWithOverflow(nextAge, last->age),
602  ASSERT_TEXT);
603  nextAge = last->age;
604 #endif
605  cachedSize -= last->unalignedSize;
606  last = last->prev;
607  } while (last && (intptr_t)(currTime - last->age) > ageThreshold);
608 
609  LargeMemoryBlock *toRelease = NULL;
610  if (last) {
611  toRelease = last->next;
612  oldest = last->age;
613  last->next = NULL;
614  } else {
615  toRelease = first;
616  first = NULL;
617  oldest = 0;
618  if (!usedSize)
619  bitMask->set(idx, false);
620  }
621  MALLOC_ASSERT( toRelease, ASSERT_TEXT );
622  lastCleanedAge = toRelease->age;
623 
624  return toRelease;
625 }

659 // release from cache blocks that are older than ageThreshold
660 template<typename Props>
661 bool LargeObjectCacheImpl<Props>::regularCleanup(ExtMemoryPool *extMemPool, uintptr_t currTime, bool doThreshDecr)
662 {
663  bool released = false;
664  BinsSummary binsSummary;
665 
666  for (int i = bitMask.getMaxTrue(numBins-1); i >= 0;
667  i = bitMask.getMaxTrue(i-1)) {
668  bin[i].updateBinsSummary(&binsSummary);
669  if (!doThreshDecr && tooLargeLOC>2 && binsSummary.isLOCTooLarge()) {
670  // if LOC is too large for quite long time, decrease the threshold
671  // based on bin hit statistics.
672  // For this, redo cleanup from the beginning.
673  // Note: on this iteration total usedSz can be not too large
674  // in comparison to total cachedSz, as we calculated it only
675  // partially. We are ok with it.
676  i = bitMask.getMaxTrue(numBins-1)+1;
677  doThreshDecr = true;
678  binsSummary.reset();
679  continue;
680  }
681  if (doThreshDecr)
682  bin[i].decreaseThreshold();
683  if (bin[i].cleanToThreshold(extMemPool, &bitMask, currTime, i))
684  released = true;
685  }
686 
687  // We want to find if LOC was too large for some time continuously,
688  // so OK with races between incrementing and zeroing, but incrementing
689  // must be atomic.
690  if (binsSummary.isLOCTooLarge())
691  AtomicIncrement(tooLargeLOC);
692  else
693  tooLargeLOC = 0;
694  return released;
695 }

840 void LargeObjectCache::putList(LargeMemoryBlock *list)
841 {
842  LargeMemoryBlock *toProcess, *n;
843 
844  for (LargeMemoryBlock *curr = list; curr; curr = toProcess) {
845  LargeMemoryBlock *tail = curr;
846  toProcess = curr->next;
847  if (curr->unalignedSize >= maxHugeSize) {
848  extMemPool->backend.returnLargeObject(curr);
849  continue;
850  }
851  int currIdx = sizeToIdx(curr->unalignedSize);
852 
853  // Find all blocks fitting to same bin. Not use more efficient sorting
854  // algorithm because list is short (commonly,
855  // LocalLOC's HIGH_MARK-LOW_MARK, i.e. 24 items).
856  for (LargeMemoryBlock *b = toProcess; b; b = n) {
857  n = b->next;
858  if (sizeToIdx(b->unalignedSize) == currIdx) {
859  tail->next = b;
860  tail = b;
861  if (toProcess == b)
862  toProcess = toProcess->next;
863  else {
864  b->prev->next = b->next;
865  if (b->next)
866  b->next->prev = b->prev;
867  }
868  }
869  }
870  tail->next = NULL;
871  if (curr->unalignedSize < maxLargeSize)
872  largeCache.putList(extMemPool, curr);
873  else
874  hugeCache.putList(extMemPool, curr);
875  }
876 }
