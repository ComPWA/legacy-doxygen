69 // The naive straightforward algorithm used to obtain reference results on CPU
70 void matmult_naive(const float* A, const float* B, float* C) {
71  for (int y = 0; y < SIZE_Y; y++) {
72  for (int x = 0; x < SIZE_X; x++) {
73  C[y * SIZE_Y + x] = (float)0;
74 
75  for (int k = 0; k < SIZE_K; k++) {
76  C[y * SIZE_Y + x] += A[y * SIZE_K + k] * B[k * SIZE_K + x];
77  }
78  }
79  }
80 }

82 // Shared Local Memory based algorithm
83 __declspec(target(gfx_kernel))
84 void matmult_tiled_slm(const float A[][SIZE_K], const float B[][SIZE_X], float C[][SIZE_X]) {
85  // The parallel loop nest below iterates over "supertiles" in the resulting
86  // matrix C and it is parallelized across thread groups, 1 iteration per
87  // group, which effectively means that the loop nest is peeled off.
88  // This kernel is programmed so that each thread group calculates one
89  // resulting supertile in matrix C.
90  _Cilk_for _Thread_group(int tg_y = 0; tg_y < SIZE_Y; tg_y += SLM_TILE_Y) {
91  _Cilk_for _Thread_group(int tg_x = 0; tg_x < SIZE_X; tg_x += SLM_TILE_X) {
92  // declare "supertiles" of each matrix to be allocated in SLM
93  __thread_group_local float slm_atile[SLM_TILE_Y][SLM_TILE_K];
94  __thread_group_local float slm_btile[SLM_TILE_K][SLM_TILE_X];
95  __thread_group_local float slm_ctile[SLM_TILE_Y][SLM_TILE_X];
96 
97  // initialize the result supertile (in parallel)
98  //slm_ctile[:][:] = (float)0;
99  _Cilk_for(int i0 = 0; i0 < SLM_TILE_Y; i0++)
100  _Cilk_for(int i1 = 0; i1 < SLM_TILE_X; i1++)
101  slm_ctile[i0][i1] = (float)0;
102 
103  // calculate the dot product of supertiles:
104  for (int super_k = 0; super_k < SIZE_K; super_k += SLM_TILE_K) {
105  // cache A's and B's "supertiles" in SLM (in parallel)
106  //slm_atile[:][:] = A[tg_y:SLM_TILE_Y][super_k:SLM_TILE_K];
107  _Cilk_for(int i0 = 0; i0 < SLM_TILE_Y; i0++)
108  _Cilk_for(int i1 = 0; i1 < SLM_TILE_K; i1++)
109  slm_atile[i0][i1] = A[tg_y + i0][super_k + i1];
110 
111  //slm_btile[:][:] = B[super_k:SLM_TILE_K][tg_x:SLM_TILE_X];
112  _Cilk_for(int i0 = 0; i0 < SLM_TILE_K; i0++)
113  _Cilk_for(int i1 = 0; i1 < SLM_TILE_X; i1++)
114  slm_btile[i0][i1] = B[super_k + i0][tg_x + i1];
115 
116  // need a barrier, since every tile in tiles are used by
117  // multiple threads in the group
118  BARRIER;
119 
120  // now multiply the supertiles as usual matrices (in parallel)
121  // ...
122  // ... using the most effective tiled algorithm:
123  _Cilk_for(int t_y = 0; t_y < SLM_TILE_Y; t_y += TILE_Y) {
124  _Cilk_for(int t_x = 0; t_x < SLM_TILE_X; t_x += TILE_X) {
125  // allocate tiles in registers
126  float atile[TILE_Y][TILE_K], btile[TILE_X];
127  float ctile[TILE_Y][TILE_X];
128 
129  // ... and initialize ctile to zero
130  ctile[:][:] = (float)0;
131 
132  // calculate the dot product of the tiles
133  for (int k = 0; k < SLM_TILE_K; k += TILE_K) {
134  atile[:][:] = slm_atile[t_y:TILE_Y][k:TILE_K];
135 
136  for (int k_ind = 0; k_ind < TILE_K; k_ind++) {
137  btile[:] = slm_btile[k + k_ind][t_x:TILE_X];
138 
139  // multiply current btile row by atile's
140  // current element and add up to corresponding
141  // ctile row
142  for (int y_ind = 0; y_ind < TILE_Y; y_ind++) {
143  ctile[y_ind][:] += atile[y_ind][k_ind] *
144  btile[:];
145  }
146  }
147  }
148  // flush the thread-local ctile (registers) into the
149  // thread group-local supertile (SLM) adding up
150  // elements
151  slm_ctile[t_y:TILE_Y][t_x:TILE_X] += ctile[:][:];
152  }
153  }
154 
155  // barrier to make sure
156  // (1) next iteration of the loop does not overwrite a and b
157  // SLM tiles used in the above calculation of slm_ctile
158  // (2) on the last iteration of the loop, all threads wait
159  // for the SLM ctile calculation to be completed before
160  // writing it back to memory below this loop
161  BARRIER;
162  }
163 
164  // write (in parallel) the result supertile back to memory:
165  //C[tg_y:SLM_TILE_Y][tg_x:SLM_TILE_X] = slm_ctile[:][:];
166  _Cilk_for(int i0 = 0; i0 < SLM_TILE_Y; i0++)
167  _Cilk_for(int i1 = 0; i1 < SLM_TILE_X; i1++)
168  C[tg_y + i0][tg_x + i1] = slm_ctile[i0][i1];
169 
170  // next iteration of the loop zeroes out slm_ctile - make sure this
171  // always happens after slm_ctile has been dumped to memory (above
172  // loop nest completes):
173  BARRIER;
174  }
175  }
176 }

187 // Validate matrix with reference values
188 bool validate_mat(char *title, int size_y, int size_x, const float* mat, const float* ref_mat) {
189  int err_cnt = 0;
190  printf("verifying %s...", title);
191  fflush(stdout);
192 
193  for (int y = 0; y < size_y; y++) {
194  for (int x = 0; x < size_x; x++) {
195  float val = mat[y * size_y + x];
196  float gold_val = ref_mat[y * size_y + x];
197 
198  if (val != gold_val) {
199  if (err_cnt == 0) {
200  printf("\n");
201  }
202  if (++err_cnt < 10) {
203  std::cout <<
204  " ERROR at [" << y << "][" << x << "]: " <<
205  val << "(should be " << gold_val << ")" << std::endl;
206  }
207  }
208  }
209  }
210  if (err_cnt == 0) {
211  printf(" ok\n");
212  } else {
213  printf("FAILED\n");
214  }
215  return err_cnt == 0;
216 }
