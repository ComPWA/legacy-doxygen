346 // Adapted from Intel(R) Cilk(TM) version from cilkpub.
347 // Let i:len denote a counted interval of length n starting at i. s denotes a generalized-sum value.
348 // Expected actions of the functors are:
349 // reduce(i,len) -> s -- return reduction value of i:len.
350 // combine(s1,s2) -> s -- return merged sum
351 // apex(s) -- do any processing necessary between reduce and scan.
352 // scan(i,len,initial) -- perform scan over i:len starting with initial.
353 // The initial range 0:n is partitioned into consecutive subranges.
354 // reduce and scan are each called exactly once per subrange.
355 // Thus callers can rely upon side effects in reduce.
356 // combine must not throw an exception.
357 // apex is called exactly once, after all calls to reduce and before all calls to scan.
358 // For example, it's useful for allocating a buffer used by scan but whose size is the sum of all reduction values.
359 // T must have a trivial constructor and destructor.
360 template <class _ExecutionPolicy, typename _Index, typename _Tp, typename _Rp, typename _Cp, typename _Sp, typename _Ap>
361 void
362 parallel_strict_scan(_ExecutionPolicy&& __exec, _Index __n, _Tp __initial, _Rp __reduce, _Cp __combine, _Sp __scan,
363  _Ap __apex)
364 {
365  tbb::this_task_arena::isolate([=, &__exec, &__combine]() {
366  if (__n > 1)
367  {
368  _Index __p = tbb::this_task_arena::max_concurrency();
369  const _Index __slack = 4;
370  _Index __tilesize = (__n - 1) / (__slack * __p) + 1;
371  _Index __m = (__n - 1) / __tilesize;
372  buffer<_Tp> __buf(std::forward<_ExecutionPolicy>(__exec), __m + 1);
373  _Tp* __r = __buf.get();
374  par_backend::upsweep(_Index(0), _Index(__m + 1), __tilesize, __r, __n - __m * __tilesize, __reduce,
375  __combine);
376  // When __apex is a no-op and __combine has no side effects, a good optimizer
377  // should be able to eliminate all code between here and __apex.
378  // Alternatively, provide a default value for __apex that can be
379  // recognized by metaprogramming that conditionlly executes the following.
380  size_t __k = __m + 1;
381  _Tp __t = __r[__k - 1];
382  while ((__k &= __k - 1))
383  __t = __combine(__r[__k - 1], __t);
384  __apex(__combine(__initial, __t));
385  par_backend::downsweep(_Index(0), _Index(__m + 1), __tilesize, __r, __n - __m * __tilesize, __initial,
386  __combine, __scan);
387  return;
388  }
389  // Fewer than 2 elements in sequence, or out of memory. Handle has single block.
390  _Tp __sum = __initial;
391  if (__n)
392  __sum = __combine(__sum, __reduce(_Index(0), __n));
393  __apex(__sum);
394  if (__n)
395  __scan(_Index(0), __n, __initial);
396  });
397 }
