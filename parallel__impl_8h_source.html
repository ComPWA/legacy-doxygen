33 template <class _ExecutionPolicy, class _Index, class _Brick, class _Compare>
34 _Index
35 parallel_find(_ExecutionPolicy&& __exec, _Index __first, _Index __last, _Brick __f, _Compare __comp, bool __b_first)
36 {
37  typedef typename std::iterator_traits<_Index>::difference_type _DifferenceType;
38  const _DifferenceType __n = __last - __first;
39  _DifferenceType __initial_dist = __b_first ? __n : -1;
40  std::atomic<_DifferenceType> __extremum(__initial_dist);
41  // TODO: find out what is better here: parallel_for or parallel_reduce
42  par_backend::parallel_for(std::forward<_ExecutionPolicy>(__exec), __first, __last,
43  [__comp, __f, __first, &__extremum](_Index __i, _Index __j) {
44  // See "Reducing Contention Through Priority Updates", PPoPP '13, for discussion of
45  // why using a shared variable scales fairly well in this situation.
46  if (__comp(__i - __first, __extremum))
47  {
48  _Index __res = __f(__i, __j);
49  // If not '__last' returned then we found what we want so put this to extremum
50  if (__res != __j)
51  {
52  const _DifferenceType __k = __res - __first;
53  for (_DifferenceType __old = __extremum; __comp(__k, __old);
54  __old = __extremum)
55  {
56  __extremum.compare_exchange_weak(__old, __k);
57  }
58  }
59  }
60  });
61  return __extremum != __initial_dist ? __first + __extremum : __last;
62 }

68 template <class _ExecutionPolicy, class _Index, class _Brick>
69 bool
70 parallel_or(_ExecutionPolicy&& __exec, _Index __first, _Index __last, _Brick __f)
71 {
72  std::atomic<bool> __found(false);
73  par_backend::parallel_for(std::forward<_ExecutionPolicy>(__exec), __first, __last,
74  [__f, &__found](_Index __i, _Index __j) {
75  if (!__found.load(std::memory_order_relaxed) && __f(__i, __j))
76  {
77  __found.store(true, std::memory_order_relaxed);
78  par_backend::cancel_execution();
79  }
80  });
81  return __found;
82 }
