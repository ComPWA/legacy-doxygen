100  double StandardDeviation ( double avg, const durations_t& d ) {
101  double std_dev = 0;
102  for ( uintptr_t i = 0; i < d.size(); ++i ) {
103  double dev = fabs(d[i] - avg);
104  std_dev += dev * dev;
105  }
106  std_dev = sqrt(std_dev / d.size());
107  return std_dev / avg * 100;
108  }

110  void Statistics ( const durations_t& d,
111  duration_t& avgTime, double& stdDev,
112  duration_t& minTime, duration_t& maxTime )
113  {
114  minTime = maxTime = avgTime = d[0];
115  for ( size_t i = 1; i < d.size(); ++i ) {
116  avgTime += d[i];
117  if ( minTime > d[i] )
118  minTime = d[i];
119  else if ( maxTime < d[i] )
120  maxTime = d[i];
121  }
122  avgTime = avgTime / d.size();
123  stdDev = StandardDeviation( avgTime, d );

241  void TraceHistogram ( const durations_t& t, const char* histogramFileName ) {
242  FILE* f = histogramFileName ? fopen(histogramFileName, "wt") : stdout;
243  uintptr_t n = t.size();
244  const uintptr_t num_buckets = 100;
245  double min_val = *std::min_element(t.begin(), t.end()),
246  max_val = *std::max_element(t.begin(), t.end()),
247  bucket_size = (max_val - min_val) / num_buckets;
248  std::vector<uintptr_t> hist(num_buckets + 1, 0);
249  for ( uintptr_t i = 0; i < n; ++i )
250  ++hist[uintptr_t((t[i]-min_val)/bucket_size)];
251  ASSERT (hist[num_buckets] == 1, "");
252  ++hist[num_buckets - 1];
253  hist.resize(num_buckets);
254  fprintf (f, "Histogram: nvals = %u, min = %g, max = %g, nbuckets = %u\n", (unsigned)n, min_val, max_val, (unsigned)num_buckets);
255  double bucket = min_val;
256  for ( uintptr_t i = 0; i < num_buckets; ++i, bucket+=bucket_size )
257  fprintf (f, "%12g\t%u\n", bucket, (unsigned)hist[i]);
258  fclose(f);

347  bool PinTheThread ( int cpu_idx, tbb::atomic<int>& nThreads ) {
348  #if _MSC_VER || __linux__
349  cpu_set_t orig_mask, target_mask;
350  CPU_ZERO( &target_mask );
351  CPU_SET( cpu_idx, &target_mask );
352  ASSERT ( CPU_ISSET(cpu_idx, &target_mask), "CPU_SET failed" );
353  #endif
354  #if _MSC_VER
355  orig_mask = SetThreadAffinityMask( GetCurrentThread(), target_mask );
356  if ( !orig_mask )
357  return false;
358  #elif __linux__
359  CPU_ZERO( &orig_mask );
360  int res = sched_getaffinity( gettid(), sizeof(cpu_set_t), &orig_mask );
361  ASSERT ( res == 0, "sched_getaffinity failed" );
362  res = sched_setaffinity( gettid(), sizeof(cpu_set_t), &target_mask );
363  ASSERT ( res == 0, "sched_setaffinity failed" );
364  #endif /* _MSC_VER */
365  --nThreads;
366  while ( nThreads )
367  __TBB_Yield();
368  #if _MSC_VER
369  SetThreadPriority (GetCurrentThread(), THREAD_PRIORITY_HIGHEST);
370  #endif
371  return true;
372  }

524  void operator()( int tid ) const {
525  Test::ThreadInfo ti = { tid, NULL };
526  durations_t &d = TlsTimings[tid].my_durations;
527  bool singleMaster = my_cfg->my_numMasters == 1;
528  START_WORKERS( (!singleMaster || (singleMaster && StatisticsMode)) && my_fnRun != &Test::RunSerial,
529  my_cfg->my_numThreads, my_cfg->my_affinityMode, singleMaster, singleMaster );
530  for ( uintptr_t k = 0; k < my_numRuns; ++k ) {
531  if ( my_numRepeats > 1 ) {
532  d[k] = 0;

535  d[k] += TimeSingleRun(ti);
536  }

538  multipleMastersBarrier.wait();
539  tbb::tick_count t0 = tbb::tick_count::now();
540  for ( uintptr_t i = 0; i < my_numRepeats; ++i )
541  (my_test->*my_fnRun)(ti);
542  d[k] = (tbb::tick_count::now() - t0).seconds();
543  }
544  d[k] /= my_numRepeats;
545  }
546  else
547  d[k] = TimeSingleRun(ti);
548  }
549  STOP_WORKERS();
550  TlsTimings[tid].CalculateStatistics();
551  }

583  void RunTestImpl ( TestResults& tr, RunConfig &cfg, RunMemFnPtr pfnTest, TimingSeries& ts ) {
584  // nRepeats is a number of repeated calls to the test function made as
585  // part of the same run. It is determined experimentally by the following
586  // calibration process so that the total run time was approx. RunDuration.
587  // This is helpful to increase the measurement precision in case of very
588  // short tests.
589  size_t nRepeats = 1;
590  // A minimal stats is enough when doing calibration
591  CalibrationTiming.my_durations.resize( (NumRuns < 4 ? NumRuns : 3) * cfg.NumMasters() );
592  // There's no need to be too precise when calculating nRepeats. And reasonably
593  // far extrapolation can speed up the process significantly.
594  for (;;) {
595  DoTiming( tr, cfg, pfnTest, nRepeats, CalibrationTiming );
596  if ( CalibrationTiming.my_avgTime * nRepeats > 1e-4 )
597  break;
598  nRepeats *= 2;
599  }
600  nRepeats *= (uintptr_t)ceil( RunDuration / (CalibrationTiming.my_avgTime * nRepeats) );
601 
602  DoTiming(tr, cfg, pfnTest, nRepeats, ts);
603 
604  // No histogram for baseline measurements
605  if ( pfnTest != &Test::RunSerial && pfnTest != &Test::Baseline ) {
606  const char* histogramName = theSettings.my_histogramName;
607  if ( histogramName != NoHistogram && tr.my_test->HistogramName() != DefaultHistogram )
608  histogramName = tr.my_test->HistogramName();
609  if ( histogramName != NoHistogram )
610  TraceHistogram( ts.my_durations, histogramName );
611  }
612  } // RunTestImpl

635  NumThreads = p;

696  const char* testName = t.Name();

702  tr.my_results.resize( numConfigs );

722  if ( !(g_absentMethods & idOnStart) )
723  tr.my_availableMethods |= idOnStart;

743  }

766  void PrintResults () {

769  Report( "%-*s %-*s %s", TitleFieldLen, "Test-name", WorkloadFieldLen, "Workload",
770  MaxTbbMasters > 1 ? "W M " : "T " );
771  if ( theSettings.my_opts & UseAffinityModes )
772  Report( "Aff " );
773  Report( "%-*s SD,%% %-*s %-*s %-*s ",
774  RateFieldLen, "Avg.time", OvhdFieldLen, "Par.ovhd,%",
775  RateFieldLen, "Min.time", RateFieldLen, "Max.time" );

777  for ( size_t i = 0; i < theSession.size(); ++i ) {
778  TestResults &tr = theSession[i];

781  RunConfig &rc = rr.my_config;
782  int w = rc.my_workloadID;
783  TimingSeries &ts = rr.my_timing;
784  duration_t baselineTime = tr.my_baselines[w].my_avgTime,
785  cleanTime = ts.my_avgTime - baselineTime;
786  Report( "%-*s %-*s ", TitleFieldLen, tr.my_testName, WorkloadFieldLen, tr.my_workloadNames[w] );
787  if ( MaxTbbMasters > 1 )
788  Report( "%-4d %-4d ", rc.my_numThreads - 1, rc.my_numMasters );
789  else
790  Report( "%-4d ", rc.my_numThreads );
791  if ( theSettings.my_opts & UseAffinityModes )
792  Report( "%%-8s ", AffinitizationModeNames[rc.my_affinityMode] );
793  Report( "%-*.2e %-6.1f ", RateFieldLen, cleanTime, ts.my_stdDev);
794  if ( tr.my_availableMethods & idRunSerial ) {
795  duration_t serialTime = (tr.my_serialBaselines[w].my_avgTime - baselineTime) / rc.my_maxConcurrency;
796  Report( "%-*.1f ", OvhdFieldLen, 100*(cleanTime - serialTime)/serialTime );
797  }
798  else
799  Report( "%*s%*s ", OvhdFieldLen/2, "-", OvhdFieldLen - OvhdFieldLen/2, "" );
800  Report( "%-*.2e %-*.2e ", RateFieldLen, ts.my_minTime - baselineTime, RateFieldLen, ts.my_maxTime - baselineTime);
801  Report( "\n" );
802  }
803  }
804  delete [] TlsTimings;
805  if ( ResFile )
806  fclose(ResFile);
807  }
