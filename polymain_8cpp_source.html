294 // create a polygon map with at least gNPolygons polygons.
295 // Usually more than gNPolygons polygons will be generated, because the
296 // process of growing the polygons results in holes.
297 bool GenerateMap(Polygon_map_t **newMap, int xSize, int ySize, int gNPolygons, colorcomp_t maxR, colorcomp_t maxG, colorcomp_t maxB) {
298  bool error_found = false;
299  int *validPolys;
300  int *validSide;
301  int maxSides;
302  RPolygon *newPoly;
303 
304  if(xSize <= 0) {
305  cout << "xSize (" << xSize << ") should be > 0." << std::endl;
306  error_found = true;
307  }
308  if(ySize <= 0) {
309  cout << "ySize (" << ySize << ") should be > 0." << std::endl;
310  error_found = true;
311  }
312  if(gNPolygons > (xSize * ySize)) {
313  cout << "gNPolygons (" << gNPolygons << ") should be less than " << (xSize * ySize) << std::endl;
314  error_found = true;
315  }
316  if(error_found) return false;
317  // the whole map is [xSize x ySize] squares
318  // the way we create the map is to
319  // 1) pick nPolygon discrete squares on an [xSize x ySize] grid
320  // 2) while there are unused squares on the grid
321  // 3) pick a polygon with a side that has unused squares on a side
322  // 4) expand the polygon by 1 to occupy the unused squares
323  //
324  // Continue until every square on the grid is occupied by a polygon
325  int *tempMap;
326  tempMap = (int *)malloc(xSize * ySize * sizeof(int));
327  for(int i=0;i < xSize; i++) {
328  for(int j=0;j < ySize; j++) {
329  tempMap[i*ySize + j] = 0;
330  }
331  }
332 
333  // *newMap = new vector<RPolygon>;
334  *newMap = new Polygon_map_t;
335  (*newMap)->reserve(gNPolygons + 1); // how much bigger does this need to be on average?
336  (*newMap)->push_back(RPolygon(0,0,xSize-1, ySize-1));
337  for(int i=0; i < gNPolygons; i++) {
338  int nX;
339  int nY;
340  do { // look for an empty square.
341  nX = NextRan(xSize);
342  nY = NextRan(ySize);
343  } while(tempMap[nX * ySize + nY] != 0);
344  int nR = (maxR * NextRan(1000)) / 999;
345  int nG = (maxG * NextRan(1000)) / 999;
346  int nB = (maxB * NextRan(1000)) / 999;
347  (*newMap)->push_back(RPolygon(nX,nY,nX,nY,nR,nG,nB));
348  tempMap[nX * ySize + nY] = i+1; // index of this polygon + 1
349  }
350  // now have to grow polygons to fill the space.
351  validPolys = (int *)malloc(4*gNPolygons * sizeof(int));
352  validSide = (int *)malloc(4*gNPolygons * sizeof(int));
353  for(int i=0;i<gNPolygons;i++) {
354  validPolys[4*i] = validPolys[4*i + 1] = validPolys[4*i + 2] = validPolys[4*i + 3] = i + 1;
355  validSide[4*i] = NORTH_SIDE;
356  validSide[4*i+1] = EAST_SIDE;
357  validSide[4*i+2] = SOUTH_SIDE;
358  validSide[4*i+3] = WEST_SIDE;
359  }
360  maxSides = 4*gNPolygons;
361  while(maxSides > 0) {
362  int indx = NextRan(maxSides);
363  int polyIndx = validPolys[indx];
364  int checkSide = validSide[indx];
365  int xlow, xhigh, ylow, yhigh;
366  int xlnew, xhnew, ylnew, yhnew;
367  (**newMap)[polyIndx].get(&xlow,&ylow,&xhigh,&yhigh);
368  xlnew = xlow;
369  xhnew = xhigh;
370  ylnew = ylow;
371  yhnew = yhigh;
372  // can this polygon be expanded along the chosen side?
373  switch(checkSide) {
374  case NORTH_SIDE:
375  // y-1 from xlow to xhigh
376  ylow = yhigh = (ylow - 1);
377  ylnew--;
378  break;
379  case EAST_SIDE:
380  // x+1 from ylow to yhigh
381  xlow = xhigh = (xhigh + 1);
382  xhnew++;
383  break;
384  case SOUTH_SIDE:
385  // y+1 from xlow to xhigh
386  ylow = yhigh = (yhigh+1);
387  yhnew++;
388  break;
389  case WEST_SIDE:
390  // x-1 from ylow to yhigh
391  xlow = xhigh = (xlow - 1);
392  xlnew--;
393  break;
394  }
395  bool okay_to_extend = !(((xlow < 0) || (xlow >= xSize)) || ((ylow < 0) || (ylow >= ySize)));
396  for(int ii = xlow; (ii <= xhigh) && okay_to_extend; ii++) {
397  for(int jj=ylow; (jj <= yhigh) && okay_to_extend; jj++) {
398  okay_to_extend = tempMap[ii*ySize + jj] == 0;
399  }
400  }
401  if(okay_to_extend) {
402  (**newMap)[polyIndx].set(xlnew,ylnew,xhnew,yhnew);
403  for(int ii = xlow; ii <= xhigh; ii++) {
404  for(int jj=ylow; jj <= yhigh && okay_to_extend; jj++) {
405  tempMap[ii*ySize + jj] = polyIndx;
406  }
407  }
408  }
409  else {
410  // once we cannot expand along a side, we will never be able to; remove from the list.
411  for(int i=indx + 1; i < maxSides; i++) {
412  validPolys[i-1] = validPolys[i];
413  validSide[i-1] = validSide[i];
414  }
415  maxSides--;
416  }
417  }
418 
419  // Once no polygons can be grown, look for unused squares, and fill them with polygons.
420  for(int j=0;j<ySize;j++) {
421  for(int i=0;i<xSize;i++) {
422  if(tempMap[i*ySize+j] == 0) {
423  // try to grow in the x direction, then the y direction
424  int ilen = i;
425  int jlen = j;
426  while(ilen < (xSize - 1) && tempMap[(ilen+1)*ySize + jlen] == 0) {
427  ilen++;
428  }
429  bool yok = true;
430  while(yok && jlen < (ySize - 1)) {
431  for(int ii = i; ii <= ilen && yok; ii++) {
432  yok = (tempMap[ii*ySize + jlen + 1] == 0);
433  }
434  if(yok) {
435  jlen++;
436  }
437  }
438 
439  // create new polygon and push it on our list.
440  int nR = (maxR * NextRan(1000)) / 999;
441  int nG = (maxG * NextRan(1000)) / 999;
442  int nB = (maxB * NextRan(1000)) / 999;
443  (*newMap)->push_back(RPolygon(i,j,ilen,jlen,nR,nG,nB));
444  gNPolygons++;
445  for(int ii=i; ii<=ilen;ii++) {
446  for(int jj=j;jj<=jlen;jj++) {
447  tempMap[ii*ySize + jj] = gNPolygons;
448  }
449  }
450  }
451  }
452  }
453 
454 #if _DEBUG
455  if(!gIsGraphicalVersion) {
456  cout << std::endl << "Final Map:" << std::endl;
457  for(int j=0; j < ySize; j++ ) {
458  cout << "Row " << setw(2) << j << ":";
459  for(int i=0;i<xSize;i++) {
460  int it = tempMap[i*ySize + j];
461  if(it<10) {
462  cout << setw(2) << it;
463  }
464  else {
465  char ct = (int)'a' + it - 10;
466  cout << " " << ct;
467  }
468  }
469  cout << std::endl;
470  }
471  }
472 #endif // _DEBUG
473  free(tempMap);
474  free(validPolys);
475  free(validSide);
476  return true;
477 }

479 void CheckPolygonMap(Polygon_map_t *checkMap) {
480 #define indx(i,j) (i*gMapYSize + j)
481 #define rangeCheck(str,n,limit) if(((n)<0)||((n)>=limit)) {cout << "checkMap error: " << str << " out of range (" << n << ")" << std::endl;anError=true;}
482 #define xRangeCheck(str,n) rangeCheck(str,n,gMapXSize)
483 #define yRangeCheck(str,n) rangeCheck(str,n,gMapYSize)
484  // The first polygon is the whole map.
485  bool anError = false;
486  int *cArray;
487  if(checkMap->size() <= 0) {
488  cout << "checkMap error: no polygons in map" << std::endl;
489  return;
490  }
491  // mapXhigh and mapYhigh are inclusive, that is, if the map is 5x5, those values would be 4.
492  int mapXhigh, mapYhigh, mapLowX, mapLowY;
493  int gMapXSize, gMapYSize;
494  (*checkMap)[0].get(&mapLowX, &mapLowY, &mapXhigh, &mapYhigh);
495  if((mapLowX !=0) || (mapLowY != 0)) {
496  cout << "checkMap error: map origin not (0,0) (X=" << mapLowX << ", Y=" << mapLowY << ")" << std::endl;
497  anError = true;
498  }
499  if((mapXhigh < 0) || (mapYhigh < 0)) {
500  cout << "checkMap error: no area in map (X=" << mapXhigh << ", Y=" << mapYhigh << ")" << std::endl;
501  anError = true;
502  }
503  if(anError) return;
504  // bounds for array.
505  gMapXSize = mapXhigh + 1;
506  gMapYSize = mapYhigh + 1;
507  cArray = (int *)malloc(sizeof(int)*(gMapXSize*gMapYSize));
508 
509  for(int i=0; i<gMapXSize; i++) {
510  for(int j=0; j< gMapYSize; j++) {
511  cArray[indx(i,j)] = 0;
512  }
513  }
514 
515  int xlow, xhigh, ylow, yhigh;
516  for(int p=1; p < int(checkMap->size()) && !anError; p++) {
517  (*checkMap)[p].get(&xlow, &ylow, &xhigh, &yhigh);
518  xRangeCheck("xlow", xlow);
519  yRangeCheck("ylow", ylow);
520  xRangeCheck("xhigh", xhigh);
521  yRangeCheck("yhigh", yhigh);
522  if(xlow>xhigh) {
523  cout << "checkMap error: xlow > xhigh (" << xlow << "," << xhigh << ")" << std::endl;
524  anError = true;
525  }
526  if(ylow>yhigh) {
527  cout << "checkMap error: ylow > yhigh (" << ylow << "," << yhigh << ")" << std::endl;
528  anError = true;
529  }
530  for(int ii = xlow; ii <= xhigh; ii++) {
531  for(int jj = ylow; jj <= yhigh; jj++) {
532  if(cArray[indx(ii,jj)] != 0) {
533  cout << "checkMap error: polygons " << cArray[indx(ii,jj)] << " and " << p << " intersect" << std::endl;
534  anError = true;
535  }
536  cArray[indx(ii,jj)] = p;
537  }
538  }
539  }
540  for(int ii=0; ii < gMapXSize; ii++) {
541  for(int jj=0; jj < gMapYSize; jj++) {
542  if(cArray[indx(ii,jj)] == 0) {
543  cout << "checkMap error: block(" << ii << ", " << jj << ") not in any polygon" << std::endl;
544  anError = true;
545  }
546  }
547  }
548  free(cArray);
549 }

569 bool ComparePolygonMaps(Polygon_map_t *map1, Polygon_map_t *map2) {
570  // create two new polygon maps, copy the pointers from the original to these.
