49 void OverlayOnePolygonWithMap(Polygon_map_t *resultMap, RPolygon *myPoly, Polygon_map_t *map2, tbb::spin_mutex *rMutex) {
50  int r1, g1, b1, r2, g2, b2;
51  int myr=0;
52  int myg=0;
53  int myb=0;
54  int p1Area = myPoly->area();
55  for(unsigned int j=1; (j < map2->size()) && (p1Area > 0); j++) {
56  RPolygon *p2 = &((*map2)[j]);
57  RPolygon *pnew;
58  int newxMin, newxMax, newyMin, newyMax;
59  myPoly->getColor(&r1, &g1, &b1);
60  if(PolygonsOverlap(myPoly, p2, newxMin, newyMin, newxMax, newyMax)) {
61  p2->getColor(&r2, &g2, &b2);
62  myr = r1 + r2;
63  myg = g1 + g2;
64  myb = b1 + b2;
65  p1Area -= (newxMax-newxMin+1)*(newyMax - newyMin + 1);
66  if(rMutex) {
67  tbb::spin_mutex::scoped_lock lock(*rMutex);
68  resultMap->push_back(RPolygon(newxMin, newyMin, newxMax, newyMax, myr, myg, myb));
69  }
70  else {
71  resultMap->push_back(RPolygon(newxMin, newyMin, newxMax, newyMax, myr, myg, myb));
72  }
73  }

128 void NaiveParallelOverlay(Polygon_map_t *&result_map, Polygon_map_t &polymap1, Polygon_map_t &polymap2) {
129 // -----------------------------------
130  bool automatic_threadcount = false;
131 
132  if(gThreadsLow == THREADS_UNSET || gThreadsLow == tbb::task_scheduler_init::automatic) {
133  gThreadsLow = gThreadsHigh = tbb::task_scheduler_init::automatic;
134  automatic_threadcount = true;

136  result_map = new Polygon_map_t;
137 
138  RPolygon *p0 = &(polymap1[0]);
139  int mapxSize, mapySize, ignore1, ignore2;
140  p0->get(&ignore1, &ignore2, &mapxSize, &mapySize);
141  result_map->reserve(mapxSize*mapySize); // can't be any bigger than this
142  // push the map size as the first polygon,
143  tbb::spin_mutex *resultMutex = new tbb::spin_mutex();
144  int grain_size = gGrainSize;
145 
146  for(int nthreads = gThreadsLow; nthreads <= gThreadsHigh; nthreads++) {
147  tbb::task_scheduler_init init(nthreads);
148  if(gIsGraphicalVersion) {
149  RPolygon *xp = new RPolygon(0, 0, gMapXSize-1, gMapYSize-1, 0, 0, 0); // Clear the output space
150  delete xp;
151  }
152  // put size polygon in result map
153  result_map->push_back(RPolygon(0,0,mapxSize, mapySize));
154 
155  tbb::tick_count t0 = tbb::tick_count::now();
156  tbb::parallel_for (tbb::blocked_range<int>(1,(int)(polymap1.size()),grain_size), ApplyOverlay(result_map, &polymap1, &polymap2, resultMutex));

182 void split_at( Flagged_map_t& in_map, Flagged_map_t &left_out, Flagged_map_t &right_out, const T median) {
183  left_out.reserve(in_map.size());
184  right_out.reserve(in_map.size());
185  for(Flagged_map_t::iterator i = in_map.begin(); i != in_map.end(); ++i ) {
186  RPolygon *p = i->p();
187  if(p->xmax() < median) {
188  // in left map
189  left_out.push_back(*i);
190  }
191  else if(p->xmin() >= median) {
192  right_out.push_back(*i);
193  // in right map
194  }
195  else {
196  // in both maps.
197  left_out.push_back(*i);
198  right_out.push_back(RPolygon_flagged(p, true));
199  }
200  }

206 class blocked_range_with_maps {
207 
208  typedef blocked_range<T> my_range_type;
209

212  my_range_type my_range;

216 public:
217 
218  blocked_range_with_maps(
219  T begin, T end, typename my_range_type::size_type my_grainsize,
220  Polygon_map_t *p1, Polygon_map_t *p2
221  )
222  : my_range(begin, end, my_grainsize)

300 #ifdef _DEBUG

302  // see what is happening in each strip without the interleaving
303  // confusing things.
304  tbb::spin_mutex::scoped_lock lock(*m_rMutex);
305  cout << unitbuf << "From " << r.range().begin() << " to " << r.range().end()-1 << std::endl;
