48 class Multiples {
49  inline NumberType strike( NumberType start, NumberType limit, NumberType stride ) {
50  // Hoist "my_is_composite" into register for sake of speed.
51  bool* is_composite = my_is_composite;
52  assert( stride>=2 );
53  for( ;start<limit; start+=stride )
54  is_composite[start] = true;
55  return start;
56  }
58  bool* my_is_composite;
59 
61 
63  NumberType* my_striker;
64 
66  NumberType* my_factor;
67 public:
69  NumberType n_factor;
70  NumberType m;
71  Multiples( NumberType n ) {
72  m = NumberType(sqrt(double(n)));
73  // Round up to even
74  m += m&1;
75  my_is_composite = new bool[m/2];
76  my_striker = new NumberType[m/2];
77  my_factor = new NumberType[m/2];
78  n_factor = 0;
79  memset( my_is_composite, 0, m/2 );
80  for( NumberType i=3; i<m; i+=2 ) {
81  if( !my_is_composite[i/2] ) {
82  if( printPrimes )
83  printf("%d\n",(int)i);
84  my_striker[n_factor] = strike( i/2, m/2, i );
85  my_factor[n_factor++] = i;
86  }
87  }
88  }
89 
91 
92  NumberType find_primes_in_window( NumberType start, NumberType window_size ) {
93  bool* is_composite = my_is_composite;
94  memset( is_composite, 0, window_size/2 );
95  for( size_t k=0; k<n_factor; ++k )
96  my_striker[k] = strike( my_striker[k]-m/2, window_size/2, my_factor[k] );
97  NumberType count = 0;
98  for( NumberType k=0; k<window_size/2; ++k ) {
99  if( !is_composite[k] ) {
100  if( printPrimes )
101  printf("%ld\n",long(start+2*k+1));
102  ++count;
103  }
104  }
105  return count;
106  }

108  ~Multiples() {
109  delete[] my_factor;
110  delete[] my_striker;
111  delete[] my_is_composite;
112  }

131  void initialize( NumberType start ) {
132  assert( start>=1 );
133  my_is_composite = new bool[m/2];
134  my_striker = new NumberType[m/2];
135  for( size_t k=0; k<n_factor; ++k ) {
136  NumberType f = my_factor[k];
137  NumberType p = (start-1)/f*f % m;
138  my_striker[k] = (p&1 ? p+2*f : p+f)/2;
139  assert( m/2<=my_striker[k] );
140  }
141  }
142 
143  // Move other to *this.
144  void move( Multiples& other ) {
145  // The swap moves the contents of other to *this and causes the old contents
146  // of *this to be deleted later when other is destroyed.
147  std::swap( my_striker, other.my_striker );
148  std::swap( my_is_composite, other.my_is_composite );
149  // other.my_factor is a shared pointer that was copied by the splitting constructor.
150  // Set it to NULL to prevent premature deletion by the destructor of ~other.
151  assert(my_factor==other.my_factor);
152  other.my_factor = NULL;
153  }

156  // End extra methods required by parallel version

238 class Sieve {
239 public:
241  ::Multiples multiples;
242 
244  NumberType count;
245 
247  Sieve( NumberType n ) :
248  multiples(n),
249  count(0)
250  {}
251 
252  //------------------------------------------------------------------------
253  // Begin signatures required by parallel_reduce
254  //------------------------------------------------------------------------
255  void operator()( const SieveRange& r ) {
256  NumberType m = multiples.m;
257  if( multiples.is_initialized() ) {
258  // Simply reuse "Multiples" structure from previous window
259  // This works because parallel_reduce always applies
260  // *this from left to right.
261  } else {
262  // Need to initialize "Multiples" because *this is a forked copy
263  // that needs to be set up to start at r.begin().
264  multiples.initialize( r.begin() );
265  }
266  NumberType window_size = m;
267  for( NumberType j=r.begin(); j<r.end(); j+=window_size ) {
268  assert( j%multiples.m==0 );
269  if( j+window_size>r.end() )
270  window_size = r.end()-j;
271  count += multiples.find_primes_in_window( j, window_size );
272  }
273  }
274  void join( Sieve& other ) {
275  count += other.count;
276  // Final value of multiples needs to final value of other multiples,
277  // so that *this can correctly process next window to right.
278  multiples.move( other.multiples );
279  }
280  Sieve( Sieve& other, tbb::split ) :
281  multiples(other.multiples,tbb::split()),
282  count(0)
283  {}
284  //------------------------------------------------------------------------
285  // End of signatures required by parallel_reduce
286  //------------------------------------------------------------------------
287 };

292  tbb::task_scheduler_init init(number_of_threads);

294  // Two is special case

304  // Doing so improves odds that the working set fits in cache when evaluating Sieve::operator().
