100 static inline void* InternalOperatorNew(size_t sz) {
101  void* res = scalable_malloc(sz);

112  }
113 #endif

262 /* pvalloc allocates smallest set of complete pages which can hold
263  the requested number of bytes. Result is aligned on page boundary. */
264 void *pvalloc(size_t size) __THROW
265 {
266  if (! memoryPageSize) initPageSize();
267  // align size up to the page size,
268  // pvalloc(0) returns 1 page, see man libmpatrol

270

703 {

718 void doMallocReplacement()
719 {
720  // Replace functions and keep backup of original code (separate for each runtime)
721 #if __TBB_OVERLOAD_OLD_MSVCR
722  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr70)
723  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr71)
724  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr80)
725  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr90)
726 #endif
727  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr100)
728  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr110)
729  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL(msvcr120)
730  __TBB_ORIG_ALLOCATOR_REPLACEMENT_CALL_RELEASE(ucrtbase)
731 
732  // Replace functions without storing original code
733  for (size_t j = 0; j < arrayLength(modules_to_replace); j++) {
734  if (!modules_to_replace[j].doFuncReplacement)
735  continue;
736  for (size_t i = 0; i < arrayLength(c_routines_to_replace); i++)
737  {
738  ReplaceFunctionWithStore( modules_to_replace[j].name, c_routines_to_replace[i]._func, c_routines_to_replace[i]._fptr, NULL, NULL, c_routines_to_replace[i]._on_error );
739  }
740  if ( strcmp(modules_to_replace[j].name, "ucrtbase.dll") == 0 ) {
741  // If _o_free function is present and patchable, redirect it to tbbmalloc as well
742  // This prevents issues with other _o_* functions which might allocate memory with malloc
743  if ( IsPrologueKnown(GetModuleHandle("ucrtbase.dll"), "_o_free", known_bytecodes) ) {
744  ReplaceFunctionWithStore( "ucrtbase.dll", "_o_free", (FUNCPTR)__TBB_malloc__o_free, known_bytecodes, (FUNCPTR*)&orig__o_free, FRR_FAIL );
745  }
746  // Similarly for _free_base
747  if (IsPrologueKnown(GetModuleHandle("ucrtbase.dll"), "_free_base", known_bytecodes)) {
748  ReplaceFunctionWithStore("ucrtbase.dll", "_free_base", (FUNCPTR)__TBB_malloc__free_base, known_bytecodes, (FUNCPTR*)&orig__free_base, FRR_FAIL);
749  }
750  // ucrtbase.dll does not export operator new/delete, so skip the rest of the loop.
751  continue;
752  }
753 
754  for (size_t i = 0; i < arrayLength(cxx_routines_to_replace); i++)
755  {
756 #if !_WIN64
757  // in Microsoft* Visual Studio* 2012 and 2013 32-bit operator delete consists of 2 bytes only: short jump to free(ptr);
758  // replacement should be skipped for this particular case.
759  if ( ((strcmp(modules_to_replace[j].name, "msvcr110.dll") == 0) || (strcmp(modules_to_replace[j].name, "msvcr120.dll") == 0)) && (strcmp(cxx_routines_to_replace[i]._func, "??3@YAXPAX@Z") == 0) ) continue;
760  // in Microsoft* Visual Studio* 2013 32-bit operator delete[] consists of 2 bytes only: short jump to free(ptr);
761  // replacement should be skipped for this particular case.
762  if ( (strcmp(modules_to_replace[j].name, "msvcr120.dll") == 0) && (strcmp(cxx_routines_to_replace[i]._func, "??_V@YAXPAX@Z") == 0) ) continue;
763 #endif
764  ReplaceFunctionWithStore( modules_to_replace[j].name, cxx_routines_to_replace[i]._func, cxx_routines_to_replace[i]._fptr, NULL, NULL, cxx_routines_to_replace[i]._on_error );
765  }
766  }
767 }
