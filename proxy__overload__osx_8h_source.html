151  zone.realloc = &impl_realloc;
152  zone.destroy = &impl_zone_destroy;
153  zone.zone_name = "tbbmalloc";
154  zone.introspect = &introspect;
155  zone.version = 8;
156  zone.memalign = impl_memalign;
157  zone.free_definite_size = &impl_free_definite_size;
158  zone.pressure_relief = &impl_pressure_relief;
159 
160  // make sure that default purgeable zone is initialized
161  malloc_default_purgeable_zone();
162  void* ptr = malloc(1);
163  // get all registered memory zones
164  unsigned zcount = 0;
165  malloc_zone_t** zone_array = NULL;
166  kern_return_t errorcode = malloc_get_all_zones(mach_task_self(),NULL,(vm_address_t**)&zone_array,&zcount);
167  if (!errorcode && zone_array && zcount>0) {
168  // find the zone that allocated ptr
169  for (unsigned i=0; i<zcount; ++i) {
170  malloc_zone_t* z = zone_array[i];
171  if (z && z->size(z,ptr)>0) { // the right one is found
172  system_zone = z;
173  break;
174  }
175  }
176  }
177  free(ptr);
178 
179  malloc_zone_register(&zone);
180  if (system_zone) {
181  // after unregistration of the system zone, the last registered (i.e. our) zone becomes the default
182  malloc_zone_unregister(system_zone);
183  // register the system zone back
184  malloc_zone_register(system_zone);
185  }
186  }
187 };
