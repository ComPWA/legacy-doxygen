136 // Declaration of reduction functor, where
137 // NAME - the name of the functor
138 // OP - type of the callable object with the reduction operation
139 // omp_in - refers to the local partial result
140 // omp_out - refers to the final value of the combiner operator
141 // omp_priv - refers to the private copy of the initial value
142 // omp_orig - refers to the original variable to be reduced
143 #define __PSTL_PRAGMA_DECLARE_REDUCTION(NAME, OP) \
