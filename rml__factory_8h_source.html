53 ::rml::factory::status_type FACTORY::open() {
54  // Failure of following assertion indicates that factory is already open, or not zero-inited.
55  LIBRARY_ASSERT( !library_handle, NULL );
56  status_type (*open_factory_routine)( factory&, version_type&, version_type );
57  dynamic_link_descriptor server_link_table[4] = {
58  DLD(__RML_open_factory,open_factory_routine),
59  MAKE_SERVER(my_make_server_routine),
60  DLD(__RML_close_factory,my_wait_to_close_routine),
61  GET_INFO(my_call_with_server_info_routine),

63  status_type result;
64  if( dynamic_link( RML_SERVER_NAME, server_link_table, 4, &library_handle ) ) {
65  version_type server_version;
66  result = (*open_factory_routine)( *this, server_version, CLIENT_VERSION );
67  // server_version can be checked here for incompatibility if necessary.
68  } else {
69  library_handle = NULL;
70  result = st_not_found;
71  }
72  return result;
