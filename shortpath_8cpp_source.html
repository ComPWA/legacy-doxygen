65 bool die_toss(size_t a, size_t b, utility::FastRandom& mr) {
66  int node_diff = std::abs((int)(a-b));
67  // near nodes
68  if (node_diff < 16) return true;
69  // mid nodes
70  if (node_diff < 64) return ((int)mr.get() % 8 == 0);
71  // far nodes
72  if (node_diff < 512) return ((int)mr.get() % 16 == 0);
73  return false;

119 void shortpath() {
120  sp_group = new task_group;
121  g_distance[src] = 0.0; // src's distance from src is zero
122  f_distance[src] = get_distance(vertices[src], vertices[dst]); // estimate distance from src to dst
123  open_set.push(make_pair(src,f_distance[src])); // push src into open_set
124 #if __TBB_CPP11_LAMBDAS_PRESENT
125  sp_group->run([](){ shortpath_helper(); });
126 #else
127  sp_group->run( shortpath_helper_functor() );
128 #endif
129  sp_group->wait();
130  delete sp_group;
131 }

133 void shortpath_helper() {
134  vertex_rec u_rec;
135  while (open_set.try_pop(u_rec)) {
136  vertex_id u = u_rec.first;
137  if (u==dst) continue;
138  double f = u_rec.second;
139  double old_g_u = 0.0;
140  {
141  spin_mutex::scoped_lock l(locks[u]);
142  if (f > f_distance[u]) continue; // prune search space
143  old_g_u = g_distance[u];
144  }
145  for (size_t i=0; i<edges[u].size(); ++i) {
146  vertex_id v = edges[u][i];
147  double new_g_v = old_g_u + get_distance(vertices[u], vertices[v]);
148  double new_f_v = 0.0;
149  // the push flag lets us move some work out of the critical section below
150  bool push = false;
151  {
152  spin_mutex::scoped_lock l(locks[v]);
153  if (new_g_v < g_distance[v]) {
154  predecessor[v] = u;
155  g_distance[v] = new_g_v;
156  new_f_v = f_distance[v] = g_distance[v] + get_distance(vertices[v], vertices[dst]);
157  push = true;
158  }
159  }
160  if (push) {
161  open_set.push(make_pair(v,new_f_v));
162  size_t n_spawn = ++num_spawn;
163  if (n_spawn < max_spawn) {
164 #if __TBB_CPP11_LAMBDAS_PRESENT
165  sp_group->run([]{ shortpath_helper(); });
166 #else
167  sp_group->run( shortpath_helper_functor() );
168 #endif
169  }
170  else --num_spawn;
171  }
172  }
173  }
174  --num_spawn;
175 }

177 void make_path(vertex_id src, vertex_id dst, vector<vertex_id>& path) {
178  vertex_id at = predecessor[dst];
179  if (at == N) path.push_back(src);
180  else if (at == src) { path.push_back(src); path.push_back(dst); }
181  else { make_path(src, at, path); path.push_back(dst); }
182 }
