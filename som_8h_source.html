79 // all SOM_element vectors are the same length (nElements), so we do not have

81 class SOM_element {
82  std::vector<double> w;
83 public:
84  friend std::ostream& operator<<( std::ostream &out, const SOM_element &s);
85  friend void remark_SOM_element(const SOM_element &s);
86  SOM_element() : w(nElements,0.0) {}
87  double &operator[](int indx) { return w.at(indx); }
88  const double &operator[](int indx) const { return w.at(indx); }
89  bool operator==(SOM_element const &other) const {
90  for(size_t i=0;i<size();++i) {
91  if(w[i] != other.w[i]) {
92  return false;
93  }
94  }
95  return true;
96  }
97  bool operator!=(SOM_element const &other) const { return !operator==(other); }
98  void elementwise_max(SOM_element const &other) {
99  for(size_t i = 0; i < w.size(); ++i) if(w[i] < other.w[i]) w[i] = other.w[i];
100  }
101  void elementwise_min(SOM_element const &other) {
102  for(size_t i = 0; i < w.size(); ++i) if(w[i] > other.w[i]) w[i] = other.w[i];
103  }
104  size_t size() const { return w.size(); }
105 };

138  void epoch_update( SOM_element const &s, int epoch, int min_x, int min_y, double radius, double learning_rate) {
139  int min_xiter = (int)((double)min_x - radius);
140  if(min_xiter < 0) min_xiter = 0;
141  int max_xiter = (int)((double)min_x + radius);
142  if(max_xiter > (int)my_map.size()-1) max_xiter = (int)(my_map.size()-1);
143  blocked_range<int> br1(min_xiter, max_xiter, 1);
144  epoch_update_range(s, epoch, min_x, min_y, radius, learning_rate, br1);
