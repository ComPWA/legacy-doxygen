48 void applyLeftImageEffect(utils::image_buffer& image) {
49  const int heighBase = channelsPerPixel * image.width;
50  std::vector<unsigned char>& buffer = *image.buffer;
51 
52  // Increase the Red channel of left image by 10
53  for (unsigned int y = 0; y < image.height; y++) {
54  const int heightOffset = heighBase * y;
55  for (unsigned int x = 0; x < image.width; x++) {
56  int pixelOffset = heightOffset + channelsPerPixel * x + redChannelOffset;
57  buffer[pixelOffset] += channelIncreaseValue;
58  }
59  }
60 }

62 void applyRightImageEffect(utils::image_buffer& image) {
63  const int heighBase = channelsPerPixel * image.width;
64  std::vector<unsigned char>& buffer = *image.buffer;
65 
66  // Increase the Blue channel of left image by 10
67  for (unsigned int y = 0; y < image.height; y++) {
68  const int heightOffset = heighBase * y;
69  for (unsigned int x = 0; x < image.width; x++) {
70  const int pixelOffset = heightOffset + channelsPerPixel * x + blueChannelOffset;

76 // This function merges to image buffers into the first buffer (leftImageBuffer as a destination)
77 void mergeImageBuffers(utils::image_buffer& leftImage, const utils::image_buffer& rightImage) {
78  const int heighBase = channelsPerPixel * leftImage.width;
79  std::vector<unsigned char>& leftImageBuffer = *leftImage.buffer;
80  std::vector<unsigned char>& rightImageBuffer = *rightImage.buffer;
81 
82  // Apply stereoscopic merge using algorithm: R: left image, G: left and right images (middle value), B: right image
83  for (unsigned int y = 0; y < leftImage.height; y++) {
84  const int heightOffset = heighBase * y;
85  for (unsigned int x = 0; x < leftImage.width; x++) {
86  const int pixelOffset = heightOffset + channelsPerPixel * x;
87  const int greenChannelIndex = pixelOffset + greenChannelOffset;
88  const int blueChannelIndex = pixelOffset + blueChannelOffset;
89  const int middleGreenChannel = (leftImageBuffer[greenChannelIndex] + rightImageBuffer[greenChannelIndex]);
90  leftImageBuffer[greenChannelIndex] = middleGreenChannel / 2;
91  leftImageBuffer[blueChannelIndex] = rightImageBuffer[blueChannelIndex];
92  }
93  }
94 }

201  function_node< std::string, OpenclImageTuple > fileReaderOne(g, serial, [&g](const std::string& fileToRead) -> OpenclImageTuple {
202  utils::image_buffer src = utils::getOrGenerateImage(fileToRead);
203 
204  // Create and initialize opencl_buffer in order to pass it to kernel
205  OpenclImageBuffer oclImage(src.buffer->size());
206  fillOpenclBuffer(oclImage, *src.buffer);
207 
208  NDRange rangeList = { src.width, src.height };
209  return std::make_tuple(oclImage, src.width, rangeList);

216  // Kernel should be in the current folder

231  // the second parameter - image size, can be taken by const reference
232  OpenclImageBuffer imageBuffer = std::get<0>(image);
233  const NDRange& imageSize = std::get<1>(image);

255

258  make_edge(output_port<0>(rightImageEffect), input_port<1>(mergeImages));
259  make_edge(output_port<1>(leftImageEffect), input_port<2>(mergeImages));
260 
261  // Set OpenCL NDRange here (because the values may vary, depending on input data)
262  make_edge(output_port<2>(leftImageEffect), input_port<3>(mergeImages));
263 
264  // Write image to PNG
265  make_edge(output_port<0>(mergeImages), input_port<0>(joinTupleNode));
266  make_edge(output_port<3>(mergeImages), input_port<1>(joinTupleNode));
267  make_edge(joinTupleNode, outputWriter);
268 
269  // Define where to get ndrange and kernel arguments

271  leftImageEffect.set_range(port_ref<2>());
272
