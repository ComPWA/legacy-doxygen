35  void operator() ( const tbb::blocked_range<std::size_t> &r ) const {
36  for (std::size_t i = r.begin(); i != r.end(); ++i) {
37  std::size_t max_size = 0, max_pos = 0;
38  for (std::size_t j = 0; j < str.size(); ++j) {
39  if (j != i) {
40  std::size_t limit = str.size()-(std::max)(i,j);
41  for (std::size_t k = 0; k < limit; ++k) {
42  if (str[i + k] != str[j + k])
43  break;
44  if (k > max_size) {
45  max_size = k;
46  max_pos = j;
47  }
48  }
49  }
50  }
51  max_array[i] = max_size;
52  pos_array[i] = max_pos;
53  }
54  }
