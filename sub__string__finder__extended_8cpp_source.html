42 void SerialSubStringFinder ( const std::string &str, std::vector<std::size_t> &max_array, std::vector<std::size_t> &pos_array ) {
43  for (std::size_t i = 0; i < str.size(); ++i) {
44  std::size_t max_size = 0, max_pos = 0;
45  for (std::size_t j = 0; j < str.size(); ++j)
46  if (j != i) {
47  std::size_t limit = str.size()-(std::max)(i,j);
48  for (std::size_t k = 0; k < limit; ++k) {
49  if (str[i + k] != str[j + k])
50  break;
51  if (k > max_size) {
52  max_size = k;
53  max_pos = j;
54  }
55  }
56  }
57  max_array[i] = max_size;
58  pos_array[i] = max_pos;
59  }
60 }
