149 void calculate_potentials(board_element *b) {
150  for (unsigned i=0; i<BOARD_SIZE; ++i) {
151  b[i].potential_set = 0;
152  if (!b[i].solved_element) { // element is not yet fixed
153  unsigned row = i/BOARD_DIM, col = i%BOARD_DIM;
154  for (unsigned potential=1; potential<=BOARD_DIM; ++potential) {
155  if (!in_row(b, row, col, potential) && !in_col(b, row, col, potential)
156  && !in_block(b, row, col, potential))
157  b[i].potential_set |= 1<<(potential-1);
158  }
159  }
160  }
161 }

175 bool examine_potentials(board_element *b, bool *progress) {
176  bool singletons = false;
177  for (unsigned i=0; i<BOARD_SIZE; ++i) {
178  if (b[i].solved_element==0 && b[i].potential_set==0) // empty set
179  return false;
180  switch (b[i].potential_set) {
181  case 1: { b[i].solved_element = 1; singletons=true; break; }
182  case 2: { b[i].solved_element = 2; singletons=true; break; }
183  case 4: { b[i].solved_element = 3; singletons=true; break; }
184  case 8: { b[i].solved_element = 4; singletons=true; break; }
185  case 16: { b[i].solved_element = 5; singletons=true; break; }
186  case 32: { b[i].solved_element = 6; singletons=true; break; }
187  case 64: { b[i].solved_element = 7; singletons=true; break; }
188  case 128: { b[i].solved_element = 8; singletons=true; break; }
189  case 256: { b[i].solved_element = 9; singletons=true; break; }
190  }
191  }
192  *progress = singletons;
193  return valid_board(b);

211 void partial_solve(board_element *b, unsigned first_potential_set) {
212  if (fixed_board(b)) {
213  if ( find_one )
214  g->cancel();
215  if (++nSols==1 && verbose) {
216  print_board(b);
217  }
218  free(b);
219  return;
220  }
221  calculate_potentials(b);
222  bool progress=true;
223  bool success = examine_potentials(b, &progress);
224  if (success && progress) {
225  partial_solve(b, first_potential_set);
226  } else if (success && !progress) {
227  board_element *new_board;
228  while (b[first_potential_set].solved_element!=0) ++first_potential_set;
229  for (unsigned short potential=1; potential<=BOARD_DIM; ++potential) {
230  if (1<<(potential-1) & b[first_potential_set].potential_set) {
231  new_board = (board_element *)malloc(BOARD_SIZE*sizeof(board_element));
232  copy_board(b, new_board);
233  new_board[first_potential_set].solved_element = potential;
234 #if __TBB_CPP11_LAMBDAS_PRESENT
235  g->run( [=]{ partial_solve(new_board, first_potential_set); } );
236 #else
237  g->run(PartialSolveBoard(new_board, first_potential_set));
238 #endif
239  }
240  }
241  free(b);
242  }
243  else {
244  free(b);
245  }
246 }

248 unsigned solve(int p) {
249  task_scheduler_init init(p);
250  nSols = 0;
251  board_element *start_board = (board_element *)malloc(BOARD_SIZE*sizeof(board_element));
252  init_board(start_board, init_values);
253  g = new task_group;
254  tick_count t0 = tick_count::now();
255  partial_solve(start_board, 0);
256  g->wait();
257  solve_time = (tick_count::now() - t0).seconds();
258  delete g;
259  return nSols;
260 }
