64 void tachyon_video::on_process() {
65  char buf[8192];
66  flt runtime;
67  scenedef *scene = (scenedef *) global_scene;
68  updating_mode = scene->displaymode == RT_DISPLAY_ENABLED;
69  recycling = false;
70  pausing = false;
71  do {
72  updating = updating_mode;
73  timer start_timer = gettimer();
74  rt_renderscene(global_scene);
75  timer end_timer = gettimer();
76  runtime = timertime(start_timer, end_timer);
77  sprintf(buf, "%s: %.3f seconds", global_window_title, runtime);
78  rt_ui_message(MSG_0, buf);
79  title = buf; show_title(); // show time spent for rendering
80  if(!updating) {
81  updating = true;
82  drawing_memory dm = get_drawing_memory();
83  drawing_area drawing(0, 0, dm.sizex, dm.sizey);// invalidate whole screen
84  }
85  rt_finalize();
86  title = global_window_title; show_title(); // reset title to default
87  } while(recycling && running);
88 }

90 void tachyon_video::on_key(int key) {

92  recycling = true;

107  }
108  }
