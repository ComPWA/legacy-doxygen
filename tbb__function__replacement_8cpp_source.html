58 /*
59  * When inserting a probe in 64 bits process the distance between the insertion
60  * point and the target may be bigger than 2^32. In this case we are using
61  * indirect jump through memory where the offset to this memory location
62  * is smaller than 2^32 and it contains the absolute address (8 bytes).
63  *
64  * This class is used to hold the pages used for the above trampolines.
65  * Since this utility will be used to replace malloc functions this implementation
66  * doesn't allocate memory dynamically.
67  *
68  * The struct MemoryBuffer holds the data about a page in the memory used for
69  * replacing functions in 64-bit code where the target is too far to be replaced
70  * with a short jump. All the calculations of m_base and m_next are in a multiple
71  * of SIZE_OF_ADDRESS (which is 8 in Win64).
72  */

92 MemoryBuffer *CreateBuffer(UINT_PTR addr)
93  {
94  // No more room in the pages database
95  if (m_lastBuffer - m_pages == MAX_NUM_BUFFERS)
96  return 0;
97 
98  void *newAddr = Addrint2Ptr(addr);
99  // Get information for the region which the given address belongs to
100  MEMORY_BASIC_INFORMATION memInfo;
101  if (VirtualQuery(newAddr, &memInfo, sizeof(memInfo)) != sizeof(memInfo))
102  return 0;
103 
104  for(;;) {
105  // The new address to check is beyond the current region and aligned to allocation size
106  newAddr = Addrint2Ptr( (Ptr2Addrint(memInfo.BaseAddress) + memInfo.RegionSize + m_allocSize) & ~(UINT_PTR)(m_allocSize-1) );
107 
108  // Check that the address is in the right distance.
109  // VirtualAlloc can only round the address down; so it will remain in the right distance
110  if (!IsInDistance(addr, Ptr2Addrint(newAddr), MAX_DISTANCE))
111  break;
112 
113  if (VirtualQuery(newAddr, &memInfo, sizeof(memInfo)) != sizeof(memInfo))
114  break;
115 
116  if (memInfo.State == MEM_FREE && memInfo.RegionSize >= m_allocSize)
117  {
118  // Found a free region, try to allocate a page in this region
119  void *newPage = VirtualAlloc(newAddr, m_allocSize, MEM_COMMIT|MEM_RESERVE, PAGE_READWRITE);
120  if (!newPage)
121  break;
122 
123  // Add the new page to the pages database
124  MemoryBuffer *pBuff = new (m_lastBuffer) MemoryBuffer(newPage, m_allocSize);
125  ++m_lastBuffer;
126  return pBuff;
127  }
128  }
129 
130  // Failed to find a buffer in the distance
131  return 0;
132  }

150  // Return a memory location in distance less than 2^31 from input address
151  UINT_PTR GetLocation(UINT_PTR addr)
152  {
153  MemoryBuffer *pBuff = m_pages;
154  for (; pBuff<m_lastBuffer && IsInDistance(pBuff->m_next, addr, MAX_DISTANCE); ++pBuff)
155  {
156  if (pBuff->m_next < pBuff->m_base + pBuff->m_size)

192 // Check function prologue with known prologues from the dictionary
193 // opcodes - dictionary
194 // inpAddr - pointer to function prologue
195 // Dictionary contains opcodes for several full asm instructions
196 // + one opcode byte for the next asm instruction for safe address processing
197 // RETURN: 1 + the index of the matched pattern, or 0 if no match found.
198 static UINT CheckOpcodes( const char ** opcodes, void *inpAddr, bool abortOnError )
199 {
200  static size_t opcodesStringsCount = 0;
201  static size_t maxOpcodesLength = 0;
202  static size_t opcodes_pointer = (size_t)opcodes;
203  char opcodeString[2*MAX_PATTERN_SIZE+1];
204  size_t i;
205  size_t result;
206 
207  // Get the values for static variables
208  // max length and number of patterns
209  if( !opcodesStringsCount || opcodes_pointer != (size_t)opcodes ){
210  while( *(opcodes + opcodesStringsCount)!= NULL ){
211  if( (i=strlen(*(opcodes + opcodesStringsCount))) > maxOpcodesLength )
212  maxOpcodesLength = i;
213  opcodesStringsCount++;
214  }
215  opcodes_pointer = (size_t)opcodes;
216  __TBB_ASSERT( maxOpcodesLength/2 <= MAX_PATTERN_SIZE, "Pattern exceeded the limit of 28 opcodes/56 symbols" );
217  }
218 
219  // Translate prologue opcodes to string format to compare
220  for( i=0; i<maxOpcodesLength/2 && i<MAX_PATTERN_SIZE; ++i ){
221  sprintf( opcodeString + 2*i, "%.2X", *((unsigned char*)inpAddr+i) );
222  }
223  opcodeString[2*i] = 0;
224 
225  // Compare translated opcodes with patterns
226  for( UINT idx=0; idx<opcodesStringsCount; ++idx ){
227  result = compareStrings( opcodes[idx],opcodeString );
228  if( result )
229  return idx+1; // avoid 0 which indicates a failure
230  }
231  if (abortOnError) {
232  // Impossibility to find opcodes in the dictionary is a serious issue,
233  // as if we unable to call original function, leak or crash is expected result.
234  __TBB_ASSERT_RELEASE( false, "CheckOpcodes failed" );
235  }
236  return 0;
237 }

239 // Modify offsets in original code after moving it to a trampoline.
240 // We do not have more than one offset to correct in existing opcode patterns.
241 static void CorrectOffset( UINT_PTR address, const char* pattern, UINT distance )
242 {
243  const char* pos = strstr(pattern, "#*******");
244  if( pos ) {
245  address += (pos - pattern)/2; // compute the offset position
246  UINT value;
247  // UINT assignment is not used to avoid potential alignment issues
248  memcpy(&value, Addrint2Ptr(address), sizeof(value));
249  value += distance;
250  memcpy(Addrint2Ptr(address), &value, sizeof(value));
251  }
252 }

254 // Insert jump relative instruction to the input address
255 // RETURN: the size of the trampoline or 0 on failure
256 static DWORD InsertTrampoline32(void *inpAddr, void *targetAddr, const char* pattern, void** storedAddr)
257 {
258  size_t bytesToMove = SIZE_OF_RELJUMP;
259  UINT_PTR srcAddr = Ptr2Addrint(inpAddr);
260  UINT_PTR tgtAddr = Ptr2Addrint(targetAddr);
261  // Check that the target fits in 32 bits
262  if (!IsInDistance(srcAddr, tgtAddr, MAX_DISTANCE))
263  return 0;
264 
265  UINT_PTR offset;
266  UINT offset32;
267  UCHAR *codePtr = (UCHAR *)inpAddr;
268 
269  if ( storedAddr ){ // If requested, store original function code
270  bytesToMove = strlen(pattern)/2-1; // The last byte matching the pattern must not be copied
271  __TBB_ASSERT_RELEASE( bytesToMove >= SIZE_OF_RELJUMP, "Incorrect bytecode pattern?" );
272  UINT_PTR trampAddr = memProvider.GetLocation(srcAddr);
273  if (!trampAddr)
274  return 0;
275  *storedAddr = Addrint2Ptr(trampAddr);
276  // Set 'executable' flag for original instructions in the new place
277  DWORD pageFlags = PAGE_EXECUTE_READWRITE;
278  if (!VirtualProtect(*storedAddr, MAX_PROBE_SIZE, pageFlags, &pageFlags)) return 0;
279  // Copy original instructions to the new place
280  memcpy(*storedAddr, codePtr, bytesToMove);
281  offset = srcAddr - trampAddr;
282  offset32 = (UINT)(offset & 0xFFFFFFFF);
283  CorrectOffset( trampAddr, pattern, offset32 );
284  // Set jump to the code after replacement
285  offset32 -= SIZE_OF_RELJUMP;
286  *(UCHAR*)(trampAddr+bytesToMove) = 0xE9;
287  memcpy((UCHAR*)(trampAddr+bytesToMove+1), &offset32, sizeof(offset32));
288  }
289 
290  // The following will work correctly even if srcAddr>tgtAddr, as long as
291  // address difference is less than 2^31, which is guaranteed by IsInDistance.
292  offset = tgtAddr - srcAddr - SIZE_OF_RELJUMP;
293  offset32 = (UINT)(offset & 0xFFFFFFFF);
294  // Insert the jump to the new code
295  *codePtr = 0xE9;
296  memcpy(codePtr+1, &offset32, sizeof(offset32));
297 
298  // Fill the rest with NOPs to correctly see disassembler of old code in debugger.
299  for( unsigned i=SIZE_OF_RELJUMP; i<bytesToMove; i++ ){
300  *(codePtr+i) = 0x90;
301  }
302 
303  return SIZE_OF_RELJUMP;
304 }

306 // This function is called when the offset doesn't fit in 32 bits
307 // 1 Find and allocate a page in the small distance (<2^31) from input address
308 // 2 Put jump RIP relative indirect through the address in the close page
309 // 3 Put the absolute address of the target in the allocated location
310 // RETURN: the size of the trampoline or 0 on failure
311 static DWORD InsertTrampoline64(void *inpAddr, void *targetAddr, const char* pattern, void** storedAddr)
312 {
313  size_t bytesToMove = SIZE_OF_INDJUMP;
314 
315  UINT_PTR srcAddr = Ptr2Addrint(inpAddr);
316  UINT_PTR tgtAddr = Ptr2Addrint(targetAddr);
317 
318  // Get a location close to the source address
319  UINT_PTR location = memProvider.GetLocation(srcAddr);
320  if (!location)
321  return 0;
322 
323  UINT_PTR offset;
324  UINT offset32;
325  UCHAR *codePtr = (UCHAR *)inpAddr;
326 
327  // Fill the location
328  UINT_PTR *locPtr = (UINT_PTR *)Addrint2Ptr(location);
329  *locPtr = tgtAddr;
330 
331  if ( storedAddr ){ // If requested, store original function code
332  bytesToMove = strlen(pattern)/2-1; // The last byte matching the pattern must not be copied
333  __TBB_ASSERT_RELEASE( bytesToMove >= SIZE_OF_INDJUMP, "Incorrect bytecode pattern?" );
334  UINT_PTR trampAddr = memProvider.GetLocation(srcAddr);
335  if (!trampAddr)
336  return 0;
337  *storedAddr = Addrint2Ptr(trampAddr);
338  // Set 'executable' flag for original instructions in the new place
339  DWORD pageFlags = PAGE_EXECUTE_READWRITE;
340  if (!VirtualProtect(*storedAddr, MAX_PROBE_SIZE, pageFlags, &pageFlags)) return 0;
341  // Copy original instructions to the new place
342  memcpy(*storedAddr, codePtr, bytesToMove);
343  offset = srcAddr - trampAddr;
344  offset32 = (UINT)(offset & 0xFFFFFFFF);
345  CorrectOffset( trampAddr, pattern, offset32 );
346  // Set jump to the code after replacement. It is within the distance of relative jump!
347  offset32 -= SIZE_OF_RELJUMP;
348  *(UCHAR*)(trampAddr+bytesToMove) = 0xE9;
349  memcpy((UCHAR*)(trampAddr+bytesToMove+1), &offset32, sizeof(offset32));
350  }
351 
352  // Fill the buffer
353  offset = location - srcAddr - SIZE_OF_INDJUMP;
354  offset32 = (UINT)(offset & 0xFFFFFFFF);
355  *(codePtr) = 0xFF;
356  *(codePtr+1) = 0x25;
357  memcpy(codePtr+2, &offset32, sizeof(offset32));
358 
359  // Fill the rest with NOPs to correctly see disassembler of old code in debugger.
360  for( unsigned i=SIZE_OF_INDJUMP; i<bytesToMove; i++ ){
361  *(codePtr+i) = 0x90;
362  }
363 
364  return SIZE_OF_INDJUMP;
365 }

367 // Insert a jump instruction in the inpAddr to the targetAddr
368 // 1. Get the memory protection of the page containing the input address
369 // 2. Change the memory protection to writable
370 // 3. Call InsertTrampoline32 or InsertTrampoline64
371 // 4. Restore memory protection
372 // RETURN: FALSE on failure, TRUE on success
373 static bool InsertTrampoline(void *inpAddr, void *targetAddr, const char ** opcodes, void** origFunc)
374 {
375  DWORD probeSize;
376  // Change page protection to EXECUTE+WRITE
377  DWORD origProt = 0;
378  if (!VirtualProtect(inpAddr, MAX_PROBE_SIZE, PAGE_EXECUTE_WRITECOPY, &origProt))
379  return FALSE;
380 
381  const char* pattern = NULL;
382  if ( origFunc ){ // Need to store original function code
383  UCHAR * const codePtr = (UCHAR *)inpAddr;
384  if ( *codePtr == 0xE9 ){ // JMP relative instruction
385  // For the special case when a system function consists of a single near jump,
386  // instead of moving it somewhere we use the target of the jump as the original function.
387  unsigned offsetInJmp = *(unsigned*)(codePtr + 1);
388  *origFunc = (void*)(Ptr2Addrint(inpAddr) + offsetInJmp + SIZE_OF_RELJUMP);
389  origFunc = NULL; // now it must be ignored by InsertTrampoline32/64
390  } else {
391  // find the right opcode pattern
392  UINT opcodeIdx = CheckOpcodes( opcodes, inpAddr, /*abortOnError=*/true );
393  __TBB_ASSERT( opcodeIdx > 0, "abortOnError ignored in CheckOpcodes?" );
394  pattern = opcodes[opcodeIdx-1]; // -1 compensates for +1 in CheckOpcodes
395  }
396  }
397 
398  probeSize = InsertTrampoline32(inpAddr, targetAddr, pattern, origFunc);
399  if (!probeSize)
400  probeSize = InsertTrampoline64(inpAddr, targetAddr, pattern, origFunc);
401 
402  // Restore original protection
403  VirtualProtect(inpAddr, MAX_PROBE_SIZE, origProt, &origProt);
404 
405  if (!probeSize)
406  return FALSE;
407 
408  FlushInstructionCache(GetCurrentProcess(), inpAddr, probeSize);
409  FlushInstructionCache(GetCurrentProcess(), origFunc, probeSize);
410 
411  return TRUE;
412 }

417 {
418  // Cache the results of the last search for the module
419  // Assume that there was no DLL unload between
420  static char cachedName[MAX_PATH+1];
421  static HMODULE cachedHM = 0;
422 
423  if (!dllName || !*dllName)
424  return FRR_NODLL;
425 
426  if (!cachedHM || strncmp(dllName, cachedName, MAX_PATH) != 0)
427  {
428  // Find the module handle for the input dll
429  HMODULE hModule = GetModuleHandleA(dllName);
430  if (hModule == 0)
431  {
432  // Couldn't find the module with the input name
433  cachedHM = 0;
434  return FRR_NODLL;
435  }
436 
437  cachedHM = hModule;
438  strncpy(cachedName, dllName, MAX_PATH);
