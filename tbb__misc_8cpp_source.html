74 /* The "what" should be fairly short, not more than about 128 characters.
75  Because we control all the call sites to handle_perror, it is pointless
76  to bullet-proof it for very long strings.
77 
78  Design note: ADR put this routine off to the side in tbb_misc.cpp instead of
79  Task.cpp because the throw generates a pathetic lot of code, and ADR wanted
80  this large chunk of code to be placed on a cold page. */
81 void handle_perror( int error_code, const char* what ) {
82  char buf[256];
83 #if _MSC_VER
84  #define snprintf _snprintf
85 #endif
86  int written = snprintf(buf, sizeof(buf), "%s: %s", what, strerror( error_code ));
87  // On overflow, the returned value exceeds sizeof(buf) (for GLIBC) or is negative (for MSVC).
88  __TBB_ASSERT_EX( written>0 && written<(int)sizeof(buf), "Error description is too long" );
89  // Ensure that buffer ends in terminator.
90  buf[sizeof(buf)-1] = 0;
91 #if TBB_USE_EXCEPTIONS
92  throw std::runtime_error(buf);
93 #else
94  PRINT_ERROR_AND_ABORT( "runtime_error", buf);
95 #endif /* !TBB_USE_EXCEPTIONS */
96 }

120 void throw_exception_v4 ( exception_id eid ) {
121  __TBB_ASSERT ( eid > 0 && eid < eid_max, "Unknown exception ID" );
122  switch ( eid ) {
123  case eid_bad_alloc: DO_THROW(std::bad_alloc, () );
124  case eid_bad_last_alloc: DO_THROW( bad_last_alloc, () );
125  case eid_nonpositive_step: DO_THROW(std::invalid_argument, ("Step must be positive") );
126  case eid_out_of_range: DO_THROW(std::out_of_range, ("Index out of requested size range") );
127  case eid_segment_range_error: DO_THROW(std::range_error, ("Index out of allocated segment slots") );
128  case eid_index_range_error: DO_THROW(std::range_error, ("Index is not allocated") );
129  case eid_missing_wait: DO_THROW( missing_wait, () );
130  case eid_invalid_multiple_scheduling: DO_THROW( invalid_multiple_scheduling, () );
131  case eid_improper_lock: DO_THROW( improper_lock, () );
132  case eid_possible_deadlock: DO_THROW(std::runtime_error, ("Resource deadlock would occur") );
133  case eid_operation_not_permitted: DO_THROW(std::runtime_error, ("Operation not permitted") );
134  case eid_condvar_wait_failed: DO_THROW(std::runtime_error, ("Wait on condition variable failed") );
135  case eid_invalid_load_factor: DO_THROW(std::out_of_range, ("Invalid hash load factor") );
136  case eid_reserved: DO_THROW(std::out_of_range, ("[backward compatibility] Invalid number of buckets") );
137  case eid_invalid_swap: DO_THROW(std::invalid_argument, ("swap() is invalid on non-equal allocators") );
138  case eid_reservation_length_error: DO_THROW(std::length_error, ("reservation size exceeds permitted max size") );
139  case eid_invalid_key: DO_THROW(std::out_of_range, ("invalid key") );
140  case eid_user_abort: DO_THROW( user_abort, () );
141  case eid_bad_tagged_msg_cast: DO_THROW(std::runtime_error, ("Illegal tagged_msg cast") );
142 #if __TBB_SUPPORTS_WORKERS_WAITING_IN_TERMINATE
143  case eid_blocking_thread_join_impossible: DO_THROW(std::runtime_error, ("Blocking terminate failed") );
144 #endif
145  default: break;
146  }

162 void fix_broken_rethrow() {
163  struct gcc_eh_data {
164  void * caughtExceptions;
165  unsigned int uncaughtExceptions;
166  };
167  gcc_eh_data* eh_data = punned_cast<gcc_eh_data*>( abi::__cxa_get_globals() );

171 bool gcc_rethrow_exception_broken() {
172  bool is_broken;
173  __TBB_ASSERT( !std::uncaught_exception(),
174  "gcc_rethrow_exception_broken() must not be called when an exception is active" );
175  try {
176  // Throw, catch, and rethrow an exception

178  throw __TBB_GLIBCXX_VERSION;
179  } catch(...) {
180  std::rethrow_exception( std::current_exception() );

234 bool cpu_has_speculation() {
235 #if __TBB_TSX_AVAILABLE
236 #if (__INTEL_COMPILER || __GNUC__ || _MSC_VER || __SUNPRO_CC)
237  bool result = false;
238  const int rtm_ebx_mask = 1<<11;
239 #if _MSC_VER
240  int info[4] = {0,0,0,0};
241  const int reg_ebx = 1;
242  __cpuidex(info, 7, 0);
243  result = (info[reg_ebx] & rtm_ebx_mask)!=0;
244 #elif __GNUC__ || __SUNPRO_CC
245  int32_t reg_ebx = 0;
246  int32_t reg_eax = 7;
247  int32_t reg_ecx = 0;
248  __asm__ __volatile__ ( "movl %%ebx, %%esi\n"
249  "cpuid\n"
250  "movl %%ebx, %0\n"
251  "movl %%esi, %%ebx\n"
252  : "=a"(reg_ebx) : "0" (reg_eax), "c" (reg_ecx) : "esi",
253 #if __TBB_x86_64
254  "ebx",
255 #endif
256  "edx"
257  );
258  result = (reg_ebx & rtm_ebx_mask)!=0 ;
259 #endif
260  return result;
261 #else
262  #error Speculation detection not enabled for compiler
263 #endif /* __INTEL_COMPILER || __GNUC__ || _MSC_VER */
264 #else /* __TBB_TSX_AVAILABLE */
265  return false;
266 #endif /* __TBB_TSX_AVAILABLE */
267 }

288 extern "C" void __TBB_machine_store8_slow_perf_warning( volatile void *ptr ) {
289  // Report run-time warning unless we have already recently reported warning for that address.
290  const unsigned n = 4;
291  static tbb::atomic<void*> cache[n];
292  static tbb::atomic<unsigned> k;

294  if( ptr==cache[i] )
295  goto done;

297  tbb::internal::runtime_warning( "atomic store on misaligned 8-byte location %p is slow", ptr );
298 done:;
