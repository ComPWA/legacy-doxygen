77 // Print message to stderr. Do not call it directly, use say() or tell() instead.
78 static void _say( char const * format, va_list args ) {
79  /*
80  On 64-bit Linux* OS, vsnprintf() modifies args argument,
81  so vsnprintf() crashes if it is called for the second time with the same args.
82  To prevent the crash, we have to pass a fresh intact copy of args to vsnprintf() each time.
83 
84  On Windows* OS, unfortunately, standard va_copy() macro is not available. However, it
85  seems vsnprintf() does not modify args argument.
86  */
87  #if ! ( _WIN32 || _WIN64 )
88  va_list _args;
89  __va_copy( _args, args ); // Make copy of args.
90  #define args _args // Substitute args with its copy, _args.
91  #endif
92  int len = vsnprintf( NULL, 0, format, args );
93  #if ! ( _WIN32 || _WIN64 )
94  #undef args // Remove substitution.
95  va_end( _args );
96  #endif
97  char * buf = reinterpret_cast< char * >( malloc( len + 1 ) );
98  if ( buf != NULL ) {
99  vsnprintf( buf, len + 1, format, args );
100  fprintf( stderr, "TBB: %s\n", buf );
101  free( buf );
102  } else {
103  fprintf( stderr, "TBB: Not enough memory for message: %s\n", format );
104  }
105 } // _say

139 static tbb::runtime_loader::error_code error( tbb::runtime_loader::error_mode mode, tbb::runtime_loader::error_code err, char const * format, ... ) {
140  va_list args;
141  va_start( args, format );
142  if ( mode == tbb::runtime_loader::em_abort ) {
143  // In em_abort mode error message printed unconditionally.
144  _say( format, args );
145  } else {
146  // In other modes printing depends on TBB_VERSION environment variable.
147  _tell( format, args );
148  } // if
149  va_end( args );
150  switch ( mode ) {
151  case tbb::runtime_loader::em_abort : {
152  say( "Aborting..." );
153  #if TBB_USE_DEBUG && ( _WIN32 || _WIN64 )
154  DebugBreak();
155  #endif
156  abort();
157  } break;
158  case tbb::runtime_loader::em_throw : {
159  throw err;
160  } break;
161  case tbb::runtime_loader::em_status : {
162  // Do nothing.
163  } break;
164  } // switch
165  return err;
166 } // error

177 static void strip( char * str, char ch ) {
178  int in = 0; // Input character index.
179  int out = 0; // Output character index.
180  for ( ; ; ) {
181  if ( str[ in ] != ch ) {
182  str[ out ] = str[ in ];
183  ++ out;
184  } // if
185  if ( str[ in ] == 0 ) {
186  break;
187  } // if
188  ++ in;
189  } // forever

213 // Concatenate name of directory and name of file.
214 void cat_file( char const * dir, char const * file, char * buffer, size_t len ) {
215  size_t i = 0;
216  // Copy directory name
217  for( ; i<len && *dir; ++i, ++dir ) {
218  buffer[i] = char_or_slash(*dir);
219  }
220  // Append trailing slash if missed.
221  if( i>0 && i<len && buffer[i-1]!=proper_slash ) {
222  buffer[i++] = proper_slash;
223  }
224  // Copy file name
225  __TBB_ASSERT( char_or_slash(*file)!=proper_slash, "File name starts with a slash" );
226  for( ; i<len && *file; ++i, ++file ) {
227  buffer[i] = *file;
228  }
229  // Append null terminator
230  buffer[ i<len? i: len-1 ] = '\0';
231 } // cat_file

268  // The function mimics Unix dlerror() function.
269  // Note: Not thread-safe due to statically allocated buffer.
270  static char * dlerror() {
271 
272  static char buffer[ 2048 ]; // Note: statically allocated buffer.
273 
274  DWORD err = GetLastError();
275  if ( err == ERROR_SUCCESS ) {
276  return NULL;
277  } // if
278 
279  DWORD rc;
280  rc =
281  FormatMessage(
282  FORMAT_MESSAGE_FROM_SYSTEM,
283  NULL,
284  err,
285  MAKELANGID( LANG_NEUTRAL, SUBLANG_DEFAULT ), // Default language.
286  reinterpret_cast< LPTSTR >( & buffer ),
287  sizeof( buffer ),
288  NULL
289  );
290  if ( rc == 0 ) {
291  // FormatMessage() failed to format system error message. Buffer to short or another issue.
292  snprintf( buffer, sizeof( buffer ), "System error %u.", err );
293  } else {
294  /*
295  FormatMessage() returns Windows-style end-of-lines, "\r\n". When string is printed,
296  printf() also replaces all the occurrences of "\n" with "\r\n" (again!), so sequences
297  like "\r\r\r\n" appear in output. It is not too good. Stripping all "\r" normalizes
298  string and returns it to canonical form, so printf() will produce correct end-of-line
299  sequences.
300  */
301  strip( buffer, '\r' ); // Delete carriage returns if any.
302  trim( buffer ); // Delete trailing newlines and spaces.
303  } // if
304 
305  return buffer;
306 
307  } // dlerror

348 // One attempt to load library (dll_name can be a full path or just a file name).
349 static tbb::runtime_loader::error_code _load( char const * dll_name, int min_ver, int max_ver ) {
350 
351  tbb::runtime_loader::error_mode mode = tbb::runtime_loader::em_status;
352  tbb::runtime_loader::error_code code = tbb::runtime_loader::ec_ok;
353 
354  /*
355  If these variables declared at the first usage, Intel C++ Compiler may issue warning(s):
356  transfer of control [goto error] bypasses initialization of: ...
357  Declaring variables at the beginning of the function eliminates warnings.
358  */
359  typedef int (*int_func_t)( void );
360  char const * get_ver_name = "TBB_runtime_interface_version"; // Name of function.
361  int_func_t get_ver_func = NULL; // Pointer to function.
362  handle_t _handle = NULL;
363  int _version = 0;
364  int total = 0;
365  int not_found = 0;
366 
367  // This function should be called iff there is no loaded library.
368  __TBB_ASSERT( handle == NULL, "Handle is invalid; " ANOTHER_RTL );
369  __TBB_ASSERT( version == 0, "Version is invalid; " ANOTHER_RTL );
370  __TBB_ASSERT( counter == 0, "Counter is invalid; " ANOTHER_RTL );
371 
372  tell( "Loading \"%s\"...", dll_name );
373 
374  // First load the library.
375  _handle = dlopen( dll_name, RTLD_NOW );
376  if ( _handle == NULL ) {
377  const char * msg = dlerror();
378  code = error( mode, tbb::runtime_loader::ec_no_lib, "Loading \"%s\" failed; system error: %s", dll_name, msg );
379  goto error;
380  } // if
381 
382  // Then try to find out its version.
383  /*
384  g++ 3.4 issues error:
385  ISO C++ forbids casting between pointer-to-function and pointer-to-object
386  on reinterpret_cast<>. Thus, we have no choice but using C-style type cast.
387  */
388  get_ver_func = (int_func_t) dlsym( _handle, get_ver_name );
389  if ( get_ver_func == NULL ) {
390  code = error( mode, tbb::runtime_loader::ec_bad_lib, "Symbol \"%s\" not found; library rejected.", get_ver_name );
391  goto error;
392  } // if
393  _version = get_ver_func();
394  if ( ! ( min_ver <= _version && _version <= max_ver ) ) {
395  code = error( mode, tbb::runtime_loader::ec_bad_ver, "Version %d is out of requested range; library rejected.", _version );
396  goto error;
397  } // if
398 
399  // Library is suitable. Mark it as loaded.
400  handle = _handle;
401  version = _version;
402  counter += 1;
403  __TBB_ASSERT( counter == 1, "Counter is invalid; " ANOTHER_RTL );
404 
405  // Now search for all known symbols.
406  for ( int i = 0; __tbb_internal_runtime_loader_symbols[ i ].name != NULL; ++ i ) {
407  symbol_t & symbol = __tbb_internal_runtime_loader_symbols[ i ];
408  // Verify symbol descriptor.
409  __TBB_ASSERT( symbol.type == st_object || symbol.type == st_function, "Invalid symbol type" );
410  #if _WIN32 || _WIN64
411  __TBB_ASSERT( symbol.type == st_function, "Should not be symbols of object type on Windows" );
412  #endif
413  if ( symbol.type == st_object ) {
414  __TBB_ASSERT( symbol.addr != NULL, "Object address invalid" );
415  __TBB_ASSERT( symbol.size > 0, "Symbol size must be > 0" );
416  __TBB_ASSERT( symbol.size <= 0x1000, "Symbol size too big" );
417  } else { // Function
418  // __TBB_ASSERT( symbol.addr == reinterpret_cast< void * >( & stub ), "Invalid symbol address" );
419  __TBB_ASSERT( symbol.size == sizeof( void * ), "Invalid symbol size" );
420  } // if
421  void * addr = dlsym( _handle, symbol.name );
422  if ( addr != NULL ) {
423  if ( symbol.type == st_object ) {
424  if ( strncmp( symbol.name, "_ZTS", 4 ) == 0 ) {
425  // If object name begins with "_ZTS", it is a string, mangled type name.
426  // Its value must equal to name of symbol without "_ZTS" prefix.
427  char const * name = static_cast< char const * >( addr );
428  __TBB_ASSERT( strlen( name ) + 1 == size_t( symbol.size ), "Unexpected size of typeinfo name" );
429  __TBB_ASSERT( strcmp( symbol.name + 4, name ) == 0, "Unexpected content of typeinfo name" );
430  strncpy( reinterpret_cast< char * >( symbol.addr ), name, symbol.size );
431  reinterpret_cast< char * >( symbol.addr )[ symbol.size - 1 ] = 0;
432  } else {
433  #if TBB_USE_ASSERT
434  // If object name begins with "_ZTI", it is an object of std::type_info class.
435  // Its protected value must equal to name of symbol without "_ZTI" prefix.
436  if ( strncmp( symbol.name, "_ZTI", 4 ) == 0 ) {
437  std::type_info const * info = static_cast< std::type_info const * >( addr );
438  __TBB_ASSERT( size_t( symbol.size ) >= sizeof( std::type_info ), "typeinfo size is too small" );
439  // std::type_info::name is not a virtual method, it is safe to call it.
440  __TBB_ASSERT( strcmp( symbol.name + 4, info->name() ) == 0, "Unexpected content of typeinfo" );
441  } // if
442  #endif
443  // Copy object content from libtbb into runtime_loader.
444  memcpy( symbol.addr, addr, symbol.size );
445  }; // if
446  } else { // Function
447  symbol.addr = addr;
448  } // if
449  } else {
450  char const * msg = dlerror();
451  tell( "Symbol \"%s\" not found; system error: %s", symbol.name, msg );
452  ++ not_found;
453  } // if
454  ++ total;
455  } // for i
456 
457  if ( not_found > 0 ) {
458  tell( "%d of %d symbols not found.", not_found, total );
459  } // if
460 
461  tell( "The library successfully loaded." );
462  return code;
463 
464  error:
465  if ( _handle != NULL ) {
466  int rc = dlclose( _handle );
467  if ( rc != 0 ) {
468  // Error occurred.
469  __TBB_ASSERT( rc != 0, "Unexpected error: dlclose() failed" );
470  } // if
471  } // if
472  _handle = NULL;
473  return code;
474 
475 } // _load

478 static tbb::runtime_loader::error_code load( tbb::runtime_loader::error_mode mode, char const * path[], int min_ver, int max_ver ) {
479  // Check arguments first.
480  if ( min_ver <= 0 ) {
481  return error( mode, tbb::runtime_loader::ec_bad_arg, "tbb::runtime_loader::load(): Invalid value of min_ver argument: %d.", min_ver );
482  } // if
483  if ( max_ver <= 0 ) {
484  return error( mode, tbb::runtime_loader::ec_bad_arg, "tbb::runtime_loader::load(): Invalid value of max_ver argument: %d.", max_ver );
485  } // if
486  if ( min_ver > max_ver ) {
487  return error( mode, tbb::runtime_loader::ec_bad_arg, "tbb::runtime_loader::load(): min_ver and max_ver specify empty range: [%d, %d].", min_ver, max_ver );
488  } // if
489  if ( min_ver == max_ver ) {
490  tell( "Searching for \"%s\" version %d...", tbb_dll_name, min_ver );
491  } else if ( max_ver == INT_MAX ) {
492  tell( "Searching for \"%s\" version %d+...", tbb_dll_name, min_ver );
493  } else {
494  tell( "Searching for \"%s\" version in range [%d, %d]...", tbb_dll_name, min_ver, max_ver );
495  } // if
496  // Then check whether a library already loaded.
497  if ( handle != NULL ) {
498  // Library already loaded. Check whether the version is compatible.
499  __TBB_ASSERT( version > 0, "Version is invalid; " ANOTHER_RTL );
500  __TBB_ASSERT( counter > 0, "Counter is invalid; " ANOTHER_RTL );
501  if ( min_ver <= version && version <= max_ver ) {
502  // Version is ok, let us use this library.
503  tell( "Library version %d is already loaded.", version );
504  counter += 1;
505  return tbb::runtime_loader::ec_ok;
506  } else {
507  // Version is not suitable.
508  return error( mode, tbb::runtime_loader::ec_bad_ver, "Library version %d is already loaded.", version );
509  } // if
510  } // if
511  // There is no loaded library, try to load it using provided directories.
512  __TBB_ASSERT( version == 0, "Version is invalid; " ANOTHER_RTL );
513  __TBB_ASSERT( counter == 0, "Counter is invalid; " ANOTHER_RTL );
514  size_t namelen = strlen(tbb_dll_name);
515  size_t buflen = 0;
516  char * buffer = NULL;
517  for ( int i = 0; path[i] != NULL; ++ i ) {
518  size_t len = strlen(path[i]) + namelen + 2; // 1 for slash and 1 for null terminator
519  if( buflen<len ) {
520  free( buffer );
521  buflen = len;
522  buffer = (char*)malloc( buflen );
523  if( !buffer )
524  return error( mode, tbb::runtime_loader::ec_no_lib, "Not enough memory." );
525  }
526  cat_file( path[i], tbb_dll_name, buffer, buflen );
527  __TBB_ASSERT(strstr(buffer,tbb_dll_name), "Name concatenation error");
528  tbb::runtime_loader::error_code ec = _load( buffer, min_ver, max_ver );
529  if ( ec == tbb::runtime_loader::ec_ok ) {
530  return ec; // Success. Exiting...
531  } // if
532  } // for i
533  free( buffer );
534  return error( mode, tbb::runtime_loader::ec_no_lib, "No suitable library found." );
535 } // load
