77 const int M = 10000;

79 struct Body: NoAssign {
80  tbb::concurrent_queue<Foo>* queue;
81  const int nthread;
82  Body( int nthread_ ) : nthread(nthread_) {}
83  void operator()( long thread_id ) const {
84  long pop_kind[3] = {0,0,0};
85  int serial[MAXTHREAD+1];
86  memset( serial, 0, nthread*sizeof(unsigned) );
87  ASSERT( thread_id<nthread, NULL );
88 
89  long sum = 0;
90  for( long j=0; j<M; ++j ) {
91  Foo f;
92  f.thread_id = DEAD;
93  f.serial = DEAD;
94  bool prepopped = false;
95  if( j&1 ) {
96  prepopped = queue->pop_if_present(f);
97  ++pop_kind[prepopped];
98  }
99  Foo g;
100  g.thread_id = thread_id;
101  g.serial = j+1;
102  queue->push( g );
103  if( !prepopped ) {
104  queue->pop(f);
105  ++pop_kind[2];
106  }
107  ASSERT( f.thread_id<=nthread, NULL );
108  ASSERT( f.thread_id==nthread || serial[f.thread_id]<f.serial, "partial order violation" );
109  serial[f.thread_id] = f.serial;
110  sum += f.serial-1;
111  }
112  Sum[thread_id] = sum;
113  for( int k=0; k<3; ++k )
114  PopKind[k] += pop_kind[k];
115  }
116 };

118 void TestPushPop( int prefill, ptrdiff_t capacity, int nthread ) {
119  ASSERT( nthread>0, "nthread must be positive" );
120  if( prefill+1>=capacity )
121  return;
122  bool success = false;
123  for( int k=0; k<3; ++k )
124  PopKind[k] = 0;
125  for( int trial=0; !success; ++trial ) {
126  FooConstructed = 0;
127  FooDestroyed = 0;
128  Body body(nthread);
129  tbb::concurrent_queue<Foo> queue;
130  queue.set_capacity( capacity );
131  body.queue = &queue;
132  for( int i=0; i<prefill; ++i ) {
133  Foo f;
134  f.thread_id = nthread;
135  f.serial = 1+i;
136  queue.push(f);
137  ASSERT( queue.size()==i+1, NULL );
138  ASSERT( !queue.empty(), NULL );
139  }
140  tbb::tick_count t0 = tbb::tick_count::now();
141  NativeParallelFor( nthread, body );
142  tbb::tick_count t1 = tbb::tick_count::now();
143  double timing = (t1-t0).seconds();
144  if( Verbose )
145  printf("prefill=%d capacity=%d time = %g = %g nsec/operation\n", prefill, int(capacity), timing, timing/(2*M*nthread)*1.E9);
146  int sum = 0;
147  for( int k=0; k<nthread; ++k )
148  sum += Sum[k];
149  int expected = nthread*((M-1)*M/2) + ((prefill-1)*prefill)/2;
150  for( int i=prefill; --i>=0; ) {
151  ASSERT( !queue.empty(), NULL );
152  Foo f;
153  queue.pop(f);
154  ASSERT( queue.size()==i, NULL );
155  sum += f.serial-1;
156  }
157  ASSERT( queue.empty(), NULL );
158  ASSERT( queue.size()==0, NULL );
159  if( sum!=expected )
160  printf("sum=%d expected=%d\n",sum,expected);
161  ASSERT( FooConstructed==FooDestroyed, NULL );
162 
163  success = true;
164  if( nthread>1 && prefill==0 ) {
165  // Check that pop_if_present got sufficient exercise
166  for( int k=0; k<2; ++k ) {
167 #if (_WIN32||_WIN64)
168  // The TBB library on Windows seems to have a tough time generating
169  // the desired interleavings for pop_if_present, so the code tries longer, and settles
170  // for fewer desired interleavings.
171  const int max_trial = 100;
172  const int min_requirement = 20;
173 #else
174  const int min_requirement = 100;
175  const int max_trial = 20;
176 #endif /* _WIN32||_WIN64 */
177  if( PopKind[k]<min_requirement ) {
178  if( trial>=max_trial ) {
179  if( Verbose )
180  printf("Warning: %d threads had only %ld pop_if_present operations %s after %d trials (expected at least %d). "
181  "This problem may merely be unlucky scheduling. "
182  "Investigate only if it happens repeatedly.\n",
183  nthread, long(PopKind[k]), k==0?"failed":"succeeded", max_trial, min_requirement);
184  else
185  printf("Warning: the number of %s pop_if_present operations is less than expected for %d threads. Investigate if it happens repeatedly.\n",
186  k==0?"failed":"succeeded", nthread );
187  } else {
188  success = false;
189  }
190  }
191  }
192  }
193  }
194 }
