54 template<typename C>
55 struct AddOne: NoAssign {
56  C& counter;
58  void operator()( tbb::blocked_range<size_t>& range ) const {
59  for( size_t i=range.begin(); i!=range.end(); ++i ) {
60  if( i&1 ) {
61  // Try implicit acquire and explicit release
62  typename C::mutex_type::scoped_lock lock(counter.mutex);
63  counter.value = counter.value+1;
64  lock.release();
65  } else {
66  // Try explicit acquire and implicit release
67  typename C::mutex_type::scoped_lock lock;
68  lock.acquire(counter.mutex);
69  counter.value = counter.value+1;
70  }
71  }
72  }
73  AddOne( C& counter_ ) : counter(counter_) {}
74 };

96 template<typename M, size_t N>
97 struct Invariant {
98  typedef M mutex_type;
99  M mutex;
100  const char* mutex_name;
101  volatile long value[N];
102  Invariant( const char* mutex_name_ ) :
103  mutex_name(mutex_name_)
104  {
105  for( size_t k=0; k<N; ++k )
106  value[k] = 0;
107  }
108  void update() {
109  for( size_t k=0; k<N; ++k )
110  ++value[k];
111  }
112  bool value_is( long expected_value ) const {
113  long tmp;
114  for( size_t k=0; k<N; ++k )
115  if( (tmp=value[k])!=expected_value ) {
116  printf("ERROR: %ld!=%ld\n", tmp, expected_value);
117  return false;
118  }
119  return true;
120  }
121  bool is_okay() {
122  return value_is( value[0] );
123  }
124 };

127 template<typename I>
128 struct TwiddleInvariant: NoAssign {
129  I& invariant;
130  TwiddleInvariant( I& invariant_ ) : invariant(invariant_) {}
131 
133  void operator()( tbb::blocked_range<size_t>& range ) const {
134  for( size_t i=range.begin(); i!=range.end(); ++i ) {
136  const bool write = (i%8)==7;
137  bool okay = true;
138  bool lock_kept = true;
139  if( (i/8)&1 ) {
140  // Try implicit acquire and explicit release
141  typename I::mutex_type::scoped_lock lock(invariant.mutex,write);
142  execute_aux(lock, i, write, okay, lock_kept);
143  lock.release();
144  } else {
145  // Try explicit acquire and implicit release
146  typename I::mutex_type::scoped_lock lock;
147  lock.acquire(invariant.mutex,write);
148  execute_aux(lock, i, write, okay, lock_kept);
149  }
150  if( !okay ) {
151  printf( "ERROR for %s at %ld: %s %s %s %s\n",invariant.mutex_name, long(i),
152  write ? "write," : "read,",
153  write ? (i%16==7?"downgrade,":"") : (i%8==3?"upgrade,":""),
154  lock_kept ? "lock kept," : "lock not kept,", // TODO: only if downgrade/upgrade
155  (i/8)&1 ? "impl/expl" : "expl/impl" );
156  }
157  }
158  }
159 private:
160  void execute_aux(typename I::mutex_type::scoped_lock & lock, const size_t i, const bool write, bool & okay, bool & lock_kept) const {
161  if( write ) {
162  long my_value = invariant.value[0];
163  invariant.update();
164  if( i%16==7 ) {
165  lock_kept = lock.downgrade_to_reader();
166  if( !lock_kept )
167  my_value = invariant.value[0] - 1;
168  okay = invariant.value_is(my_value+1);
169  }
170  } else {
171  okay = invariant.is_okay();
172  if( i%8==3 ) {
173  long my_value = invariant.value[0];
174  lock_kept = lock.upgrade_to_writer();
175  if( !lock_kept )
176  my_value = invariant.value[0];
177  invariant.update();
178  okay = invariant.value_is(my_value+1);
179  }
180  }
181  }
182 };
