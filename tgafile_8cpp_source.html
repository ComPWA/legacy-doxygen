61 void createtgafile(char *name, unsigned short width, unsigned short height) {
62  int filesize;
63  FILE * ofp;
64 
65  filesize = 3*width*height + 18 - 10;
66 
67  if (name==NULL)
68  exit(1);
69  else {
70  ofp=fopen(name, "w+b");
71  if (ofp == NULL) {
72  char msgtxt[2048];
73  sprintf(msgtxt, "Cannot create %s for output!", name);
74  rt_ui_message(MSG_ERR, msgtxt);
75  rt_ui_message(MSG_ABORT, "Rendering Aborted.");
76  exit(1);
77  }
78 
79  fputc(0, ofp); /* IdLength */
80  fputc(0, ofp); /* ColorMapType */
81  fputc(2, ofp); /* ImageTypeCode */
82  fputc(0, ofp); /* ColorMapOrigin, low byte */
83  fputc(0, ofp); /* ColorMapOrigin, high byte */
84  fputc(0, ofp); /* ColorMapLength, low byte */
85  fputc(0, ofp); /* ColorMapLength, high byte */
86  fputc(0, ofp); /* ColorMapEntrySize */
87  fputc(0, ofp); /* XOrigin, low byte */
88  fputc(0, ofp); /* XOrigin, high byte */
89  fputc(0, ofp); /* YOrigin, low byte */
90  fputc(0, ofp); /* YOrigin, high byte */
91  fputc((width & 0xff), ofp); /* Width, low byte */
92  fputc(((width >> 8) & 0xff), ofp); /* Width, high byte */
93  fputc((height & 0xff), ofp); /* Height, low byte */
94  fputc(((height >> 8) & 0xff), ofp); /* Height, high byte */
95  fputc(24, ofp); /* ImagePixelSize */
96  fputc(0x20, ofp); /* ImageDescriptorByte 0x20 == flip vertically */
97 
98  fseek(ofp, filesize, 0);
99  fprintf(ofp, "9876543210");
100 
101  fclose(ofp);
102  }
103 }

155 int readtga(char * name, int * xres, int * yres, unsigned char **imgdata) {
156  int format, width, height, w1, w2, h1, h2, depth, flags;
157  int imgsize, i, tmp;
158  size_t bytesread;
159  FILE * ifp;
160 
161  ifp=fopen(name, "r");
162  if (ifp==NULL) {
163  return IMAGEBADFILE; /* couldn't open the file */
164  }
165 
166  /* read the targa header */
167  getc(ifp); /* ID length */
168  getc(ifp); /* colormap type */
169  format = getc(ifp); /* image type */
170  getc(ifp); /* color map origin */
171  getc(ifp); /* color map origin */
172  getc(ifp); /* color map length */
173  getc(ifp); /* color map length */
174  getc(ifp); /* color map entry size */
175  getc(ifp); /* x origin */
176  getc(ifp); /* x origin */
177  getc(ifp); /* y origin */
178  getc(ifp); /* y origin */
179  w1 = getc(ifp); /* width (low) */
180  w2 = getc(ifp); /* width (hi) */
181  h1 = getc(ifp); /* height (low) */
182  h2 = getc(ifp); /* height (hi) */
183  depth = getc(ifp); /* image pixel size */
184  flags = getc(ifp); /* image descriptor byte */
185 
186  if ((format != 2) || (depth != 24)) {
187  fclose(ifp);
188  return IMAGEUNSUP; /* unsupported targa format */
189  }
190 
191 
192  width = ((w2 << 8) | w1);
193  height = ((h2 << 8) | h1);
194 
195  imgsize = 3 * width * height;
196  *imgdata = (unsigned char *)rt_getmem(imgsize);
197  bytesread = fread(*imgdata, 1, imgsize, ifp);
198  fclose(ifp);
199 
200  /* flip image vertically */
201  if (flags == 0x20) {
202  int rowsize = 3 * width;
203  unsigned char * copytmp;
204 
205  copytmp = (unsigned char *)malloc(rowsize);
206 
207  for (i=0; i<height / 2; i++) {
208  memcpy(copytmp, &((*imgdata)[rowsize*i]), rowsize);
209  memcpy(&(*imgdata)[rowsize*i], &(*imgdata)[rowsize*(height - 1 - i)], rowsize);
210  memcpy(&(*imgdata)[rowsize*(height - 1 - i)], copytmp, rowsize);
211  }
212 
213  free(copytmp);
214  }
215 
216 
217  /* convert from BGR order to RGB order */
218  for (i=0; i<imgsize; i+=3) {
219  tmp = (*imgdata)[i]; /* Blue */
220  (*imgdata)[i] = (*imgdata)[i+2]; /* Red */
221  (*imgdata)[i+2] = tmp; /* Blue */
222  }
223 
224  *xres = width;
225  *yres = height;
226 
227  if (bytesread != imgsize)
228  return IMAGEREADERR;
229 
230  return IMAGENOERR;
231 }
