61 void ThreadLevelRecorder::change_level(int delta, client_t whichClient) {
62  int tox=tbb_outer_level, tix=tbb_inner_level, oox=omp_outer_level, oix=omp_inner_level;
63  if (whichClient == tbb_outer) {
64  tox = tbb_outer_level+=delta;
65  } else if (whichClient == tbb_inner) {
66  tix = tbb_inner_level+=delta;
67  } else if (whichClient == omp_outer) {
68  oox = omp_outer_level+=delta;
69  } else if (whichClient == omp_inner) {
70  oix = omp_inner_level+=delta;
71  } else {
72  printf("WARNING: Bad client type; ignoring.\n");
73  return;
74  }
75  // log non-negative entries
76  tbb::tick_count t = tbb::tick_count::now();
77  unsigned k = next++;
78  if (k<max_record_count) {
79  record& r = array[k];
80  r.time = t;
81  r.n_tbb_outer_thread = tox>=0?tox:0;
82  r.n_omp_outer_thread = oox>=0?oox:0;
83  r.n_tbb_inner_thread = tix>=0?tix:0;
84  r.n_omp_inner_thread = oix>=0?oix:0;
85  }
86  char errStr[100];
87  int tot_threads;
88  tot_threads = tox+tix+oox+oix;
89  sprintf(errStr, "ERROR: Number of threads (%d+%d+%d+%d=%d) in use exceeds maximum (%d).\n",
90  tox, tix, oox, oix, tot_threads, max_threads);
91  if (tot_threads > max_threads) {
92 #ifdef NO_BAIL_OUT
93  if (!fail) {
94  printf("%sContinuing...\n", errStr);
95  fail = true;
96  }
97 #else
98  dump();
99  printf("%s\n", errStr);
100  assert(tot_threads <= max_threads);
101 #endif
102  }
103 }

105 void ThreadLevelRecorder::dump() {
106  FILE* f = fopen("time.txt","w");
107  if (!f) {
108  perror("fopen(time.txt)\n");
109  exit(1);
110  }
111  unsigned limit = next;
112  if (limit>max_record_count) { // Clip
113  limit = max_record_count;
114  }
115  for (unsigned i=0; i<limit; ++i) {
116  fprintf(f,"%f\t%d\t%d\t%d\t%d\n",(array[i].time-array[0].time).seconds(), array[i].n_tbb_outer_thread,
117  array[i].n_tbb_inner_thread, array[i].n_omp_outer_thread, array[i].n_omp_inner_thread);
118  }
119  fclose(f);
120  int tox=tbb_outer_level, tix=tbb_inner_level, oox=omp_outer_level, oix=omp_inner_level;
121  int tot_threads;
122  tot_threads = tox+tix+oox+oix;
123  if (!fail) printf("INFO: Passed.\n");
124  else printf("INFO: Failed.\n");
125 }
