140 my_data_type do_pop(int nThr, int impl) {
141  my_data_type elem;
142  if (impl == IMPL_SERIAL) {
143  if (!serial_cpq->empty()) {
144  elem = serial_cpq->top();
145  serial_cpq->pop();
146  return elem;
147  }
148  }
149  else if (impl == IMPL_STL) {
150  tbb::spin_mutex::scoped_lock myLock(*my_mutex);
151  if (!stl_cpq->empty()) {
152  elem = stl_cpq->top();
153  stl_cpq->pop();
154  return elem;
155  }
156  }
157  else if (impl == IMPL_CPQ) {
158  if (agg_cpq->try_pop(elem)) {
159  return elem;
160  }
161  }
162  return elem;

166 struct TestThroughputBody : NoAssign {

173  void operator()(const int threadID) const {
174  tbb::tick_count now;
175  size_t pos_in = threadID, pos_out = threadID;
176  my_data_type elem;
177  while (1) {
178  for (int i=0; i<sample_operations; i+=ops_per_iteration) {
179  // do pushes
180  for (int j=0; j<pushes_per_iter; ++j) {
181  elem = input_data[pos_in];
182  do_push(elem, nThread, implementation);
183  busy_wait(contention);
184  pos_in += nThread;
185  if (pos_in >= arrsz) pos_in = pos_in % arrsz;
186  }
187  // do pops
188  for (int j=0; j<pops_per_iter; ++j) {
189  output_data[pos_out] = do_pop(nThread, implementation);
190  busy_wait(contention);
191  pos_out += nThread;
192  if (pos_out >= arrsz) pos_out = pos_out % arrsz;
193  }
194  }
195  now = tbb::tick_count::now();
196  operation_count += sample_operations;
197  if ((now-start).seconds() >= throughput_window) break;
198  }
199  }
200 };
