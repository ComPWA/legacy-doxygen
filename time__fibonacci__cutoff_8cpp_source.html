53 void find_cutoff(const int P, const long n, const int T, const double serial_elapsed) {
54  long lo = 1, hi = n;
55  double elapsed = 0, lo_elapsed = 0, hi_elapsed = 0;
56  long final_cutoff = -1;
57 
58  tbb::task_scheduler_init init(P);
59 
60  while(true) {
61  CutOff = calculate_new_cutoff(lo, hi);
62  long result = 0;
63  tbb::tick_count t0;
64  for (int t = -1; t < T; ++t) {
65  if (t == 0) t0 = tbb::tick_count::now();
66  result += ParallelFib(n);
67  }
68  elapsed = (tbb::tick_count::now() - t0).seconds();
69  output(P,n,CutOff,T,serial_elapsed,elapsed,result);
70 
71  if (serial_elapsed / elapsed >= P/2.0) {
72  final_cutoff = CutOff;
73  if (hi == CutOff) {
74  if (hi == lo) {
75  // we have had this value at both above and below 50%
76  lo = 1; lo_elapsed = 0;
77  } else {
78  break;
79  }
80  }
81  hi = CutOff;
82  hi_elapsed = elapsed;
83  } else {
84  if (lo == CutOff) break;
85  lo = CutOff;
86  lo_elapsed = elapsed;
87  }
88  }
89 
90  double interpolated_cutoff = lo + ( P/2.0 - serial_elapsed/lo_elapsed ) * ( (hi - lo) / ( serial_elapsed/hi_elapsed - serial_elapsed/lo_elapsed ));
91 
92  if (final_cutoff != -1) {
93  printf("50%% efficiency cutoff is %ld ( linearly interpolated cutoff is %g )\n", final_cutoff, interpolated_cutoff);
94  } else {
95  printf("Cannot achieve 50%% efficiency\n");
96  }
97 
98  return;
99 }
