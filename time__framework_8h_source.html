183  template<typename Test>
184  TestRunner(const char *name, Test *test)
185  : tester_name(name), tester(*static_cast<TesterBase*>(test))
186  {
187  test->tester_name = name;
188  }
189

191 
192  void init(arg_t value, int threads, Harness::SpinBarrier &barrier, StatisticsCollector *s) {
193  tester.base_init(value, threads, barrier);
194  stat = s;
195  keys.resize(tester.tests_count);
196  for(int testn = 0; testn < tester.tests_count; testn++) {
197  keys[testn].resize(threads);
198  std::string test_name(tester.get_name(testn));
199  for(int threadn = 0; threadn < threads; threadn++)
200  keys[testn][threadn] = stat->SetTestCase(tester_name, test_name.c_str(), threadn);
201  }
202  }
203 
204  void run_test(int threadn) {
205  for(int testn = 0; testn < tester.tests_count; testn++) {
206  tester.test_prefix(testn, threadn);
207  tester.barrier->wait(); // <<<<<<<<<<<<<<<<< Barrier before running test mode
208  value_t result = tester.test(testn, threadn);
209  stat->AddRoundResult(keys[testn][threadn], result);
210  }
211  }
212 
213  void post_process(StatisticsCollector &report) {
214  const int threads = tester.threads_count;
215  for(int testn = 0; testn < tester.tests_count; testn++) {
216  size_t coln = keys[testn][0].getResults().size()-1;
217  value_t rsum = keys[testn][0].getResults()[coln];
218  value_t rmin = rsum, rmax = rsum;
219  for(int threadn = 1; threadn < threads; threadn++) {
220  value_t result = keys[testn][threadn].getResults()[coln];
221  rsum += result; // for both SUM or AVG
222  if(rmin > result) rmin = result;
223  if(rmax < result) rmax = result;
224  }
225  std::string test_name(tester.get_name(testn));
226  const char *rname = tester.get_result_type(testn, TesterBase::SUM);
227  if( rname ) {
228  report.SetTestCase(tester_name, (test_name+rname).c_str(), threads);
229  report.AddRoundResult(rsum);
230  }
231  rname = tester.get_result_type(testn, TesterBase::MIN);
232  if( rname ) {
233  report.SetTestCase(tester_name, (test_name+rname).c_str(), threads);
234  report.AddRoundResult(rmin);
235  }
236  rname = tester.get_result_type(testn, TesterBase::AVG);
237  if( rname ) {
238  report.SetTestCase(tester_name, (test_name+rname).c_str(), threads);
239  report.AddRoundResult(rsum / threads);
240  }
241  rname = tester.get_result_type(testn, TesterBase::MAX);
242  if( rname ) {
243  report.SetTestCase(tester_name, (test_name+rname).c_str(), threads);
244  report.AddRoundResult(rmax);
245  }
246  }
247  }
248 };

303  // iteration processing
304  void process(arg_t value, int threads, ...) {
305  // prepare items
306  stat = stat_by_threads[threads];
307  if(!stat) {
308  stat_by_threads[threads] = stat = new StatisticsCollector((collection_name + Format("@%d", threads)).c_str(), StatisticsCollector::ByAlg);
309  stat->SetTitle("Detailed log of %s running with %d threads.", collection_name, threads);
310  }
311  Harness::SpinBarrier barrier(threads);
312  // init args
313  va_list args; va_start(args, threads);
314  vector<TestRunner*> run_list; run_list.reserve(16);
315  while(true) {
316  TestRunner *item = va_arg(args, TestRunner*);
317  if( !item ) break;
318  item->init(value, threads, barrier, stat);
319  run_list.push_back(item);
320  }
321  va_end(args);
322  std::ostringstream buf;
323  buf << value;
324  const size_t round_number = stat->GetRoundsCount();
325  stat->SetRoundTitle(round_number, buf.str().c_str());
326  report.SetRoundTitle(round_number, buf.str().c_str());
327  // run them
328 #ifndef __TBB_parallel_for_H
329  NativeParallelFor(threads, RunArgsBody(run_list));
330 #else
331  tbb::parallel_for(tbb::blocked_range<int>(0,threads,1), RunArgsBody(run_list));
332 #endif
333  // destroy args
334  for(size_t i = 0; i < run_list.size(); i++) {
335  run_list[i]->post_process(report);
336  delete run_list[i];
337  }
338  }
339 
340 public:
341  TestProcessor(const char *name, StatisticsCollector::Sorting sort_by = StatisticsCollector::ByAlg)
342  : collection_name(name), stat(NULL), end(0), report(collection_name, sort_by)
343  { }
344 
345  ~TestProcessor() {
346  for(statistics_collection::iterator i = stat_by_threads.begin(); i != stat_by_threads.end(); i++)
347  delete i->second;
348  }
