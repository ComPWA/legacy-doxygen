149  struct _{ static void retrieve_from_cache(self_type* _this, size_t thread_index){
150  parameter_pack& p = _this->m_parameter_pack;
151  access_sequence_type::iterator const begin_it =_this->m_access_sequence.begin()+ thread_index * _this->per_thread_sample_size;
152  access_sequence_type::iterator const end_it = begin_it + _this->per_thread_sample_size;
153 
154  _this->m_barrier.wait();
155  tbb::tick_count start = tbb::tick_count::now();
156 
157  size_t local_loops_count =0;
158  do {
159  size_t part_of_the_sample_so_far = (local_loops_count * p.time_check_granularity_ops) % _this->per_thread_sample_size;
160  access_sequence_type::iterator const iteration_begin_it = begin_it + part_of_the_sample_so_far;
161  access_sequence_type::iterator const iteration_end_it = iteration_begin_it +
162  (std::min)(p.time_check_granularity_ops, _this->per_thread_sample_size - part_of_the_sample_so_far);
163 
164  for (access_sequence_type::iterator it = iteration_begin_it; it < iteration_end_it; ++it){
165  typename cache_type::handle h = _this->m_cache[*it];
166  micro_benchmarking::utils::busy_wait(p.time_of_item_use_usec);
167  micro_benchmarking::utils::disable_elimination(h.value());
168  }
169  ++local_loops_count;
170  }while((tbb::tick_count::now()-start).seconds() < p.time_window_sec);
171  _this->loops_count+=local_loops_count;
172  }};
173  m_barrier.initialize(m_parameter_pack.threads_number);
174 
175  NativeParallelFor(m_parameter_pack.threads_number,std::bind1st(std::ptr_fun(&_::retrieve_from_cache),this));
176 
177  return loops_count * m_parameter_pack.time_check_granularity_ops;
178  }
