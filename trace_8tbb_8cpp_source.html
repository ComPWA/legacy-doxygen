104 static color_t render_one_pixel (int x, int y, unsigned int *local_mbox, unsigned int &serial,
105  int startx, int stopx, int starty, int stopy
106 #ifdef MARK_RENDERING_AREA
107  , int *blend, float alpha
108 #endif
109 )
110 {
111  /* private vars moved inside loop */
112  ray primary, sample;
113  color col, avcol;
114  int R,G,B;
115  intersectstruct local_intersections;
116  int alias;
117  /* end private */
118 
119  primary=camray(&scene, x, y);
120  primary.intstruct = &local_intersections;
121  primary.flags = RT_RAY_REGULAR;
122 
123  serial++;
124  primary.serial = serial;
125  primary.mbox = local_mbox;
126  primary.maxdist = FHUGE;
127  primary.scene = &scene;
128  col=trace(&primary);
129 
130  serial = primary.serial;
131 
132  /* perform antialiasing if enabled.. */
133  if (scene.antialiasing > 0) {
134  for (alias=0; alias < scene.antialiasing; alias++) {
135 
136  serial++; /* increment serial number */
137  sample=primary; /* copy the regular primary ray to start with */
138  sample.serial = serial;
139 
140  {
141  tbb::spin_mutex::scoped_lock lock (MyMutex);
142  sample.d.x+=((rand() % 100) - 50) / jitterscale;
143  sample.d.y+=((rand() % 100) - 50) / jitterscale;
144  sample.d.z+=((rand() % 100) - 50) / jitterscale;
145  }
146 
147  avcol=trace(&sample);
148 
149  serial = sample.serial; /* update our overall serial # */
150 
151  col.r += avcol.r;
152  col.g += avcol.g;
153  col.b += avcol.b;
154  }
155 
156  col.r /= (scene.antialiasing + 1.0);
157  col.g /= (scene.antialiasing + 1.0);
158  col.b /= (scene.antialiasing + 1.0);
159  }
160 
161  /* Handle overexposure and underexposure here... */
162  R=(int) (col.r*255);
163  if (R > 255) R = 255;
164  else if (R < 0) R = 0;
165 
166  G=(int) (col.g*255);
167  if (G > 255) G = 255;
168  else if (G < 0) G = 0;
169 
170  B=(int) (col.b*255);
171  if (B > 255) B = 255;
172  else if (B < 0) B = 0;
173 
174 #ifdef MARK_RENDERING_AREA
175  R = int((1.0 - alpha) * R + alpha * blend[0]);
176  G = int((1.0 - alpha) * G + alpha * blend[1]);
177  B = int((1.0 - alpha) * B + alpha * blend[2]);
178 #endif
179 
180  return video->get_color(R, G, B);
181 }

183 class parallel_task {
184 public:
185  void operator() (const tbb::blocked_range2d<int> &r) const
186  {
187  // task-local storage
188  unsigned int serial = 1;
189  unsigned int mboxsize = sizeof(unsigned int)*(max_objectid() + 20);
190  unsigned int * local_mbox = (unsigned int *) alloca(mboxsize);
191  memset(local_mbox,0,mboxsize);
192 #ifdef MARK_RENDERING_AREA
193  // compute thread number while first task run
194  thread_id_t::reference thread_id = thread_ids.local();
195  if (thread_id == -1) thread_id = thread_number++;
196  // choose thread color
197  int pos = thread_id % NUM_COLORS;
198  if(video->running) {
199  drawing_area drawing(r.cols().begin(), totaly-r.rows().end(), r.cols().end() - r.cols().begin(), r.rows().end()-r.rows().begin());
200  for (int i = 1, y = r.rows().begin(); y != r.rows().end(); ++y, i++) {
201  drawing.set_pos(0, drawing.size_y-i);
202  for (int x = r.cols().begin(); x != r.cols().end(); x++) {
203  int d = (y % 3 == 0) ? 2 : 1;
204  drawing.put_pixel(video->get_color(colors[pos][0]/d, colors[pos][1]/d, colors[pos][2]/d));
205  }
206  }
207  }
208 #endif
209  if(video->next_frame()) {
210  drawing_area drawing(r.cols().begin(), totaly-r.rows().end(), r.cols().end() - r.cols().begin(), r.rows().end()-r.rows().begin());
211  for (int i = 1, y = r.rows().begin(); y != r.rows().end(); ++y, i++) {
212  drawing.set_pos(0, drawing.size_y-i);
213  for (int x = r.cols().begin(); x != r.cols().end(); x++) {
214 #ifdef MARK_RENDERING_AREA
215  float alpha = y==r.rows().begin()||y==r.rows().end()-1||x==r.cols().begin()||x==r.cols().end()-1
216  ? border_alpha : inner_alpha;
217  color_t c = render_one_pixel (x, y, local_mbox, serial, startx, stopx, starty, stopy, colors[pos], alpha);
218 #else
219  color_t c = render_one_pixel (x, y, local_mbox, serial, startx, stopx, starty, stopy);
220 #endif
221  drawing.put_pixel(c);
222  }
223  }
224  }
225  }
