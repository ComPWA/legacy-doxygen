44 void Universe::InitializeUniverse(video const& colorizer) {
45 
46  pulseCounter = pulseTime = 100;
47  pulseX = UniverseWidth/3;
48  pulseY = UniverseHeight/4;
49  // Initialize V, S, and T to slightly non-zero values, in order to avoid denormal waves.
50  for( int i=0; i<UniverseHeight; ++i )
51 #pragma ivdep
52  for( int j=0; j<UniverseWidth; ++j ) {
53  T[i][j] = S[i][j] = V[i][j] = ValueType(1.0E-6);
54  }
55  for( int i=1; i<UniverseHeight-1; ++i ) {
56  for( int j=1; j<UniverseWidth-1; ++j ) {
57  float x = float(j-UniverseWidth/2)/(UniverseWidth/2);
58  ValueType t = (ValueType)i/UniverseHeight;
59  MaterialType m;
60  D[i][j] = 1.0;
61  // Coefficient values are fictitious, and chosen to visually exaggerate
62  // physical effects such as Rayleigh waves. The fabs/exp line generates
63  // a shale layer with a gentle upwards slope and an anticline.
64  if( t<0.3f ) {
65  m = WATER;
66  M[i][j] = 0.125;
67  L[i][j] = 0.125;
68  } else if( fabs(t-0.7+0.2*exp(-8*x*x)+0.025*x)<=0.1 ) {
69  m = SHALE;
70  M[i][j] = 0.5;
71  L[i][j] = 0.6;
72  } else {
73  m = SANDSTONE;
74  M[i][j] = 0.3;
75  L[i][j] = 0.4;
76  }
77  material[i][j] = m;
78  }
79  }
80  ValueType scale = 2.0f/ColorMapSize;
81  for( int k=0; k<4; ++k ) {
82  for( int i=0; i<ColorMapSize; ++i ) {
83  colorcomp_t c[3];
84  ValueType t = (i-ColorMapSize/2)*scale;
85  ValueType r = t>0 ? t : 0;
86  ValueType b = t<0 ? -t : 0;
87  ValueType g = 0.5f*fabs(t);
88  memcpy(c, MaterialColor[k], sizeof(c));
89  c[2] = colorcomp_t(r*(255-c[2])+c[2]);
90  c[1] = colorcomp_t(g*(255-c[1])+c[1]);
91  c[0] = colorcomp_t(b*(255-c[0])+c[0]);
92  ColorMap[k][i] = colorizer.get_color(c[2], c[1], c[0]);
93  }
94  }
95  // Set damping coefficients around border to reduce reflections from boundaries.
96  ValueType d = 1.0;
97  for( int k=DamperSize-1; k>0; --k ) {
98  d *= 1-1.0f/(DamperSize*DamperSize);
99  for( int j=1; j<UniverseWidth-1; ++j ) {
100  D[k][j] *= d;
101  D[UniverseHeight-1-k][j] *= d;
102  }
103  for( int i=1; i<UniverseHeight-1; ++i ) {
104  D[i][k] *= d;
105  D[i][UniverseWidth-1-k] *= d;
106  }
107  }
108  drawingMemory = colorizer.get_drawing_memory();
109 }

111  if( pulseCounter>0 ) {
112  ValueType t = (pulseCounter-pulseTime/2)*0.05f;
113  V[pulseY][pulseX] += 64*sqrt(M[pulseY][pulseX])*exp(-t*t);
114  --pulseCounter;
115  }
116 }

131 void Universe::UpdateStress(Rectangle const& r ) {
132  drawing_area drawing(r.StartX(),r.StartY(),r.Width(),r.Height(),drawingMemory);
133  for( int i=r.StartY(); i<r.EndY() ; ++i ) {
134  drawing.set_pos(1, i-r.StartY());
135 #pragma ivdep
136  for( int j=r.StartX(); j<r.EndX() ; ++j ) {
137  S[i][j] += M[i][j]*(V[i][j+1]-V[i][j]);
138  T[i][j] += M[i][j]*(V[i+1][j]-V[i][j]);
139  int index = (int)(V[i][j]*(ColorMapSize/2)) + ColorMapSize/2;
140  if( index<0 ) index = 0;
141  if( index>=ColorMapSize ) index = ColorMapSize-1;
142  color_t* c = ColorMap[material[i][j]];
143  drawing.put_pixel(c[index]);
144  }
145  }
146 }
