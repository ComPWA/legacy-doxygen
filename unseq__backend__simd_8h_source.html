66 template <class _Index, class _DifferenceType, class _Pred>
67 bool
68 simd_or(_Index __first, _DifferenceType __n, _Pred __pred) noexcept
69 {
70 #if __PSTL_EARLYEXIT_PRESENT
71  _DifferenceType __i;
72  __PSTL_PRAGMA_VECTOR_UNALIGNED
73  __PSTL_PRAGMA_SIMD_EARLYEXIT
74  for (__i = 0; __i < __n; ++__i)
75  if (__pred(__first[__i]))
76  break;
77  return __i < __n;
78 #else
79  _DifferenceType __block_size = 4 < __n ? 4 : __n;
80  const _Index __last = __first + __n;
81  while (__last != __first)
82  {
83  int32_t __flag = 1;
84  __PSTL_PRAGMA_SIMD_REDUCTION(& : __flag)
85  for (_DifferenceType __i = 0; __i < __block_size; ++__i)
86  if (__pred(*(__first + __i)))
87  __flag = 0;
88  if (!__flag)
89  return true;
90 
91  __first += __block_size;
92  if (__last - __first >= __block_size << 1)
93  {
94  // Double the block _Size. Any unnecessary iterations can be amortized against work done so far.
95  __block_size <<= 1;
96  }
97  else
98  {
99  __block_size = __last - __first;
100  }
101  }
102  return false;
103 #endif
104 }

106 template <class _Index, class _DifferenceType, class _Compare>
107 _Index
108 simd_first(_Index __first, _DifferenceType __begin, _DifferenceType __end, _Compare __comp) noexcept
109 {
110 #if __PSTL_EARLYEXIT_PRESENT
111  _DifferenceType __i = __begin;
112  __PSTL_PRAGMA_VECTOR_UNALIGNED // Do not generate peel loop part
113  __PSTL_PRAGMA_SIMD_EARLYEXIT for (; __i < __end; ++__i)
114  {
115  if (__comp(__first, __i))
116  {
117  break;
118  }
119  }
120  return __first + __i;
121 #else
122  // Experiments show good block sizes like this
123  const _DifferenceType __block_size = 8;
124  alignas(__lane_size) _DifferenceType __lane[__block_size] = {0};
125  while (__end - __begin >= __block_size)
126  {
127  _DifferenceType __found = 0;
128  __PSTL_PRAGMA_VECTOR_UNALIGNED // Do not generate peel loop part
129  __PSTL_PRAGMA_SIMD_REDUCTION(|
130  : __found) for (_DifferenceType __i = __begin; __i < __begin + __block_size;
131  ++__i)
132  {
133  const _DifferenceType __t = __comp(__first, __i);
134  __lane[__i - __begin] = __t;
135  __found |= __t;
136  }
137  if (__found)
138  {
139  _DifferenceType __i;
140  // This will vectorize
141  for (__i = 0; __i < __block_size; ++__i)
142  {
143  if (__lane[__i])
144  {
145  break;
146  }
147  }
148  return __first + __begin + __i;
149  }
150  __begin += __block_size;
151  }
152 
153  //Keep remainder scalar
154  while (__begin != __end)
155  {
156  if (__comp(__first, __begin))
157  {
158  return __first + __begin;
159  }
160  ++__begin;
161  }
162  return __first + __end;
163 #endif //__PSTL_EARLYEXIT_PRESENT
164 }

168 simd_first(_Index1 __first1, _DifferenceType __n, _Index2 __first2, _Pred __pred) noexcept
169 {

172  __PSTL_PRAGMA_VECTOR_UNALIGNED
173  __PSTL_PRAGMA_SIMD_EARLYEXIT
174  for (; __i < __n; ++__i)
175  if (__pred(__first1[__i], __first2[__i]))
176  break;
177  return std::make_pair(__first1 + __i, __first2 + __i);
178 #else
179  const _Index1 __last1 = __first1 + __n;
180  const _Index2 __last2 = __first2 + __n;
181  // Experiments show good block sizes like this
182  const _DifferenceType __block_size = 8;
183  alignas(__lane_size) _DifferenceType __lane[__block_size] = {0};
184  while (__last1 - __first1 >= __block_size)
185  {
186  _DifferenceType __found = 0;
187  _DifferenceType __i;
188  __PSTL_PRAGMA_VECTOR_UNALIGNED // Do not generate peel loop part
189  __PSTL_PRAGMA_SIMD_REDUCTION(|
190  : __found) for (__i = 0; __i < __block_size; ++__i)
191  {
192  const _DifferenceType __t = __pred(__first1[__i], __first2[__i]);
193  __lane[__i] = __t;
194  __found |= __t;
195  }
196  if (__found)
197  {
198  _DifferenceType __i;
199  // This will vectorize
200  for (__i = 0; __i < __block_size; ++__i)
201  {
202  if (__lane[__i])
203  break;
204  }
205  return std::make_pair(__first1 + __i, __first2 + __i);
206  }
207  __first1 += __block_size;
208  __first2 += __block_size;
209  }
210

212  for (; __last1 != __first1; ++__first1, ++__first2)
213  if (__pred(*(__first1), *(__first2)))

382 template <class _Index, class _BinaryPredicate>
383 _Index
384 simd_adjacent_find(_Index __first, _Index __last, _BinaryPredicate __pred, bool __or_semantic) noexcept
385 {
386  if (__last - __first < 2)
387  return __last;
388 
389  typedef typename std::iterator_traits<_Index>::difference_type _DifferenceType;
390  _DifferenceType __i = 0;
391 
392 #if __PSTL_EARLYEXIT_PRESENT
393  //Some compiler versions fail to compile the following loop when iterators are used. Indices are used instead
394  const _DifferenceType __n = __last - __first - 1;
395  __PSTL_PRAGMA_VECTOR_UNALIGNED
396  __PSTL_PRAGMA_SIMD_EARLYEXIT
397  for (; __i < __n; ++__i)
398  if (__pred(__first[__i], __first[__i + 1]))
399  break;
400 
401  return __i < __n ? __first + __i : __last;
402 #else
403  // Experiments show good block sizes like this
404  //TODO: to consider tuning block_size for various data types
405  const _DifferenceType __block_size = 8;
406  alignas(__lane_size) _DifferenceType __lane[__block_size] = {0};
407  while (__last - __first >= __block_size)
408  {
409  _DifferenceType __found = 0;
410  __PSTL_PRAGMA_VECTOR_UNALIGNED // Do not generate peel loop part
411  __PSTL_PRAGMA_SIMD_REDUCTION(|
412  : __found) for (__i = 0; __i < __block_size - 1; ++__i)
413  {
414  //TODO: to improve SIMD vectorization
415  const _DifferenceType __t = __pred(*(__first + __i), *(__first + __i + 1));
416  __lane[__i] = __t;
417  __found |= __t;
418  }
419 
420  //Process a pair of elements on a boundary of a data block
421  if (__first + __block_size < __last && __pred(*(__first + __i), *(__first + __i + 1)))
422  __lane[__i] = __found = 1;
423 
424  if (__found)
425  {
426  if (__or_semantic)
427  return __first;
428 
429  // This will vectorize
430  for (__i = 0; __i < __block_size; ++__i)
431  if (__lane[__i])
432  break;
433  return __first + __i; //As far as found is true a __result (__lane[__i] is true) is guaranteed
434  }
435  __first += __block_size;
436  }
437  //Process the rest elements
438  for (; __last - __first > 1; ++__first)
439  if (__pred(*__first, *(__first + 1)))
440  return __first;
441 
442  return __last;
443 #endif
444 }

461 template <typename _Size, typename _Tp, typename _BinaryOperation, typename _UnaryOperation>
462 typename std::enable_if<!is_arithmetic_plus<_Tp, _BinaryOperation>::value, _Tp>::type
463 simd_transform_reduce(_Size __n, _Tp __init, _BinaryOperation __binary_op, _UnaryOperation __f) noexcept
464 {
465  const std::size_t __block_size = __lane_size / sizeof(_Tp);
466  if (__n > 2 * __block_size && __block_size > 1)
467  {
468  alignas(__lane_size) char __lane_[__lane_size];
469  _Tp* __lane = reinterpret_cast<_Tp*>(__lane_);
470 
471  // initializer
472  __PSTL_PRAGMA_SIMD
473  for (_Size __i = 0; __i < __block_size; ++__i)
474  {
475  ::new (__lane + __i) _Tp(__binary_op(__f(__i), __f(__block_size + __i)));
476  }
477  // main loop
478  _Size __i = 2 * __block_size;
479  const _Size last_iteration = __block_size * (__n / __block_size);
480  for (; __i < last_iteration; __i += __block_size)
481  {
482  __PSTL_PRAGMA_SIMD
483  for (_Size __j = 0; __j < __block_size; ++__j)
484  {
485  __lane[__j] = __binary_op(__lane[__j], __f(__i + __j));
486  }
487  }
488  // remainder
489  __PSTL_PRAGMA_SIMD
490  for (_Size __j = 0; __j < __n - last_iteration; ++__j)
491  {
492  __lane[__j] = __binary_op(__lane[__j], __f(last_iteration + __j));
493  }
494  // combiner
495  for (_Size __i = 0; __i < __block_size; ++__i)
496  {
497  __init = __binary_op(__init, __lane[__i]);
498  }
499  // destroyer
500  __PSTL_PRAGMA_SIMD
501  for (_Size __i = 0; __i < __block_size; ++__i)
502  {
503  __lane[__i].~_Tp();
504  }
505  }
506  else
507  {
508  for (_Size __i = 0; __i < __n; ++__i)
509  {
510  __init = __binary_op(__init, __f(__i));
511  }
512  }
513  return __init;
514 }
