161 inline color_t video::get_color(colorcomp_t red, colorcomp_t green, colorcomp_t blue) const
162 {
163  if(red_shift == 16) // only for depth == 24 && red_shift > blue_shift
164  return (red<<16) | (green<<8) | blue;
165  else if(depth >= 24)
166  return
167 #if __ANDROID__
168  // Setting Alpha to 0xFF
169  0xFF000000 |
170 #endif
171  (red<<red_shift) | (green<<green_shift) | (blue<<blue_shift);
172  else if(depth > 0) {
173  depth_t bs = blue_shift, rs = red_shift;
174  if(blue_shift < 0) blue >>= -bs, bs = 0;
175  else /*red_shift < 0*/ red >>= -rs, rs = 0;
176  return ((red<<rs)&red_mask) | ((green<<green_shift)&green_mask) | ((blue<<bs)&blue_mask);
177  } else { // UYVY colorspace
178  unsigned y, u, v;
179  y = red * 77 + green * 150 + blue * 29; // sum(77+150+29=256) * max(=255): limit->2^16
180  u = (2048 + (blue << 3) - (y >> 5)) >> 4; // (limit->2^12)>>4
181  v = (2048 + (red << 3) - (y >> 5)) >> 4;
182  y = y >> 8;
183  return u | (y << 8) | (v << 16) | (y << 24);
184  }
185 }

203 inline void drawing_area::put_pixel(color_t color)
204 {
205  assert(index < max_index);
206  if(pixel_depth > 16) ptr32[index++] = color;
207  else if(pixel_depth > 0)
208  ((unsigned short*)ptr32)[index++] = (unsigned short)color;
209  else { // UYVY colorspace
210  if(index&1) color >>= 16;
211  ((unsigned short*)ptr32)[index++] = (unsigned short)color;
