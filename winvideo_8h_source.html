107 static bool RedirectIOToConsole(void)
108 {
109  int hConHandle; size_t lStdHandle;
110  CONSOLE_SCREEN_BUFFER_INFO coninfo;
111  FILE *fp;
112  // allocate a console for this app
113  AllocConsole();
114 
115  // set the screen buffer to be big enough to let us scroll text
116  GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &coninfo);
117  coninfo.dwSize.Y = MAX_CONSOLE_LINES;
118  SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE), coninfo.dwSize);
119 
120  // redirect unbuffered STDOUT to the console
121  lStdHandle = (size_t)GetStdHandle(STD_OUTPUT_HANDLE);
122  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
123  if(hConHandle <= 0) return false;
124  fp = _fdopen( hConHandle, "w" );
125  *stdout = *fp;
126  setvbuf( stdout, NULL, _IONBF, 0 );
127 
128  // redirect unbuffered STDERR to the console
129  lStdHandle = (size_t)GetStdHandle(STD_ERROR_HANDLE);
130  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
131  if(hConHandle > 0) {
132  fp = _fdopen( hConHandle, "w" );
133  *stderr = *fp;
134  setvbuf( stderr, NULL, _IONBF, 0 );
135  }
136 
137  // redirect unbuffered STDIN to the console
138  lStdHandle = (size_t)GetStdHandle(STD_INPUT_HANDLE);
139  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
140  if(hConHandle > 0) {
141  fp = _fdopen( hConHandle, "r" );
142  *stdin = *fp;
143  setvbuf( stdin, NULL, _IONBF, 0 );
144  }
145 
146  // make cout, wcout, cin, wcin, wcerr, cerr, wclog and clog
147  // point to console as well
148  std::ios::sync_with_stdio();
149  return true;

195 static bool loop_once(video *v)
196 {
197  // screen update notify
198  if(int updates = g_updates) {
199  g_updates = 0;
200  if(g_video->updating) { g_skips += updates-1; g_fps++; }
201  else g_skips += updates;
202  UpdateWindow(g_hAppWnd);
203  }
204  // update fps
205  DWORD msec = GetTickCount();
206  if(v->calc_fps && msec >= g_msec+1000) {
207  double sec = (msec - g_msec)/1000.0;
208  char buffer[256], n = _snprintf(buffer, 128, "%s: %d fps", v->title, int(double(g_fps + g_skips)/sec));
209  if(g_skips) _snprintf(buffer+n, 128, " - %d skipped = %d updates", int(g_skips/sec), int(g_fps/sec));
210  SetWindowTextA(g_hAppWnd, buffer);
211  g_msec = msec; g_skips = g_fps = 0;
212  }
213  // event processing, including painting
214  MSG msg;
215  if(PeekMessage(&msg, NULL, 0, 0, PM_REMOVE)){
216  if( msg.message == WM_QUIT ) { v->running = false; return false; }
217  if( !hAccelTable || !TranslateAccelerator(msg.hwnd, hAccelTable, &msg) ){
218  TranslateMessage(&msg);
219  DispatchMessage(&msg);
220  }
221  return true; // try again
222  }
223  return false;
224 }

227 void video::main_loop()
228 {
229  // let Windows draw and unroll the window
230  InvalidateRect(g_hAppWnd, 0, false);
231  g_msec = GetTickCount(); // let's stay for 0,5 sec
232  while(g_msec + 500 > GetTickCount()) { loop_once(this); Sleep(1); }
233  g_msec = GetTickCount();
234  // now, start main process
235  if(threaded) {
236  g_handles[0] = CreateThread (
237  NULL, // LPSECURITY_ATTRIBUTES security_attrs
238  0, // SIZE_T stacksize
239  (LPTHREAD_START_ROUTINE) thread_video,
240  this, // argument
241  0, 0);
242  if(!g_handles[0]) { DisplayError("Can't create thread"); return; }
243  else // harmless race is possible here
244  g_handles[1] = CreateEvent(NULL, false, false, NULL);
245  while(running) {
246  while(loop_once(this));
247  YIELD_TO_THREAD(); // give time for processing when running on single CPU
248  DWORD r = MsgWaitForMultipleObjects(2, g_handles, false, INFINITE, QS_ALLINPUT^QS_MOUSEMOVE);
249  if(r == WAIT_OBJECT_0) break; // thread terminated
250  }
251  running = false;
252  if(WaitForSingleObject(g_handles[0], 3000) == WAIT_TIMEOUT){
253  // there was not enough time for graceful shutdown, killing the example with code 1.
254  exit(1);
255  }
256  if(g_handles[0]) CloseHandle(g_handles[0]);
257  if(g_handles[1]) CloseHandle(g_handles[1]);
258  g_handles[0] = g_handles[1] = 0;
259  }
260  else on_process();
261 }
